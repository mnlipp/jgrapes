053/**
054 * Provides resources using {@link Request}/{@link Response}
055 * events. Some resource requests (page resource, portlet resource)
056 * are forwarded via the {@link Portal} component to the portlets.
057 */
058@SuppressWarnings({ "PMD.ExcessiveImports", "PMD.NcssCount",
059    "PMD.TooManyMethods" })
060public class JQueryUiWeblet extends FreeMarkerPortalWeblet {
061
062    private ServiceLoader<ThemeProvider> themeLoader;
063    private final ThemeProvider baseTheme;
064    private BiFunction<ThemeProvider, String, URL> fallbackResourceSupplier
065        = (themeProvider, resource) -> {
066            return null;
067        };
068
069    /**
070     * Instantiates a new jQuery UI weblet.
071     *
072     * @param webletChannel the weblet channel
073     * @param portalChannel the portal channel
074     * @param portalPrefix the portal prefix
075     */
076    public JQueryUiWeblet(Channel webletChannel, Channel portalChannel,
077            URI portalPrefix) {
078        super(webletChannel, portalChannel, portalPrefix);
079        baseTheme = new Provider();
080    }
081
082    @Override
083    public String styling() {
084        return "jqueryui";
085    }
086
087    /**
088     * The service loader must be created lazily, else the OSGi
089     * service mediator doesn't work properly.
090     * 
091     * @return
092     */
093    private ServiceLoader<ThemeProvider> themeLoader() {
094        if (themeLoader != null) {
095            return themeLoader;
096        }
097        return themeLoader = ServiceLoader.load(ThemeProvider.class);
098    }
099
100    @Override
101    protected Map<String, Object> createPortalBaseModel() {
102        Map<String, Object> model = super.createPortalBaseModel();
103        model.put("themeInfos",
104            StreamSupport.stream(themeLoader().spliterator(), false)
105                .map(thi -> new ThemeInfo(thi.themeId(), thi.themeName()))
106                .sorted().toArray(size -> new ThemeInfo[size]));
107        return model;
108    }
109
110    /**
111     * Sets a function for obtaining a fallback resource bundle for
112     * a given theme provider and locale.
113     * 
114     * @param supplier the function
115     * @return the portal fo reasy chaining
116     */
117    public JQueryUiWeblet setFallbackResourceSupplier(
118            BiFunction<ThemeProvider, String, URL> supplier) {
119        fallbackResourceSupplier = supplier;
120        return this;
121    }
122
123    @Override
124    protected void renderPortal(GetRequest event, IOSubchannel channel,
125            UUID portalSessionId) throws IOException, InterruptedException {
126        // Reloading themes on every reload allows themes
127        // to be added dynamically. Note that we must load again
128        // (not reload) in order for this to work in an OSGi environment.
129        themeLoader = null;
130        super.renderPortal(event, channel, portalSessionId);
131    }
132
133    @Override
134    protected void providePortalResource(GetRequest event,
135            String requestPath, IOSubchannel channel) {
136        String[] requestParts = ResourcePattern.split(requestPath, 1);
137        if (requestParts.length == 2 && requestParts[0].equals("theme")) {
138            sendThemeResource(event, channel, requestParts[1]);
139            return;
140        }
141        super.providePortalResource(event, requestPath, channel);
142    }
143
144    @SuppressWarnings("PMD.DataflowAnomalyAnalysis")
145    private void sendThemeResource(GetRequest event, IOSubchannel channel,
146            String resource) {
147        // Get resource
148        ThemeProvider themeProvider = event.associated(Session.class).flatMap(
149            session -> Optional.ofNullable(session.get("themeProvider"))
150                .flatMap(
151                    themeId -> StreamSupport
152                        .stream(themeLoader().spliterator(), false)
153                        .filter(thi -> thi.themeId().equals(themeId))
154                        .findFirst()))
155            .orElse(baseTheme);
156        URL resourceUrl;
157        try {
158            resourceUrl = themeProvider.getResource(resource);
159        } catch (ResourceNotFoundException e) {
160            try {
161                resourceUrl = baseTheme.getResource(resource);
162            } catch (ResourceNotFoundException e1) {
163                resourceUrl
164                    = fallbackResourceSupplier.apply(themeProvider, resource);
165                if (resourceUrl == null) {
166                    return;
167                }
168            }
169        }
170        final URL resUrl = resourceUrl;
171        ResponseCreationSupport.sendStaticContent(event, channel,
172            path -> resUrl, null);
173    }

175    /**
176     * Handle JSON input.
177     *
178     * @param event the event
179     * @param channel the channel
180     * @throws InterruptedException the interrupted exception
181     * @throws IOException Signals that an I/O exception has occurred.
182     */
183    @SuppressWarnings("PMD.DataflowAnomalyAnalysis")
184    @Handler(channels = PortalChannel.class)
185    public void onJsonInput(JsonInput event, PortalSession channel)
186            throws InterruptedException, IOException {
187        // Send events to portlets on portal's channel
188        JsonArray params = event.request().params();
189        switch (event.request().method()) { // NOPMD
190        case "setTheme": {
191            fire(new SetTheme(params.asString(0)), channel);
192            break;
193        }
194        default:
195            // Ignore unknown
196            break;
197        }
198    }
199
200    /**
201     * Handles a change of theme.
202     *
203     * @param event the event
204     * @param channel the channel
205     * @throws InterruptedException the interrupted exception
206     * @throws IOException Signals that an I/O exception has occurred.
207     */
208    @Handler(channels = PortalChannel.class)
209    public void onSetTheme(SetTheme event, PortalSession channel)
210            throws InterruptedException, IOException {
211        ThemeProvider themeProvider = StreamSupport
212            .stream(themeLoader().spliterator(), false)
213            .filter(thi -> thi.themeId().equals(event.theme())).findFirst()
214            .orElse(baseTheme);
215        channel.browserSession().put("themeProvider", themeProvider.themeId());
216        channel.respond(new KeyValueStoreUpdate().update(
217            "/" + PortalUtils.userFromSession(channel.browserSession())
218                .map(UserPrincipal::toString).orElse("")
219                + "/themeProvider",
220            themeProvider.themeId())).get();
221        channel.respond(new SimplePortalCommand("reload"));
222    }

224    /**
225     * Holds the information about a theme.
226     */
227    public static class ThemeInfo implements Comparable<ThemeInfo> {
228        private final String id;
229        private final String name;
230
231        /**
232         * Instantiates a new theme info.
233         *
234         * @param id the id
235         * @param name the name
236         */
237        public ThemeInfo(String id, String name) {
238            super();
239            this.id = id;
240            this.name = name;
241        }
242
243        /**
244         * Returns the id.
245         *
246         * @return the id
247         */
248        @SuppressWarnings("PMD.ShortMethodName")
249        public String id() {
250            return id;
251        }
252
253        /**
254         * Returns the name.
255         *
256         * @return the name
257         */
258        public String name() {
259            return name;
260        }
261
262        /*
263         * (non-Javadoc)
264         * 
265         * @see java.lang.Comparable#compareTo(java.lang.Object)
266         */
267        @Override
268        public int compareTo(ThemeInfo other) {
269            return name().compareToIgnoreCase(other.name());
270        }
271    }
