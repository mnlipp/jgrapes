019package org.jgrapes.portal.providers.datatables;
020
021import freemarker.core.ParseException;
022import freemarker.template.MalformedTemplateNameException;
023import freemarker.template.TemplateNotFoundException;
024
025import java.io.IOException;
026import java.util.Locale;
027import java.util.ResourceBundle;
028
029import org.jgrapes.core.Channel;
030import org.jgrapes.core.Event;
031import org.jgrapes.core.Manager;
032import org.jgrapes.core.annotation.Handler;
033import org.jgrapes.portal.PageResourceProvider;
034import org.jgrapes.portal.PortalSession;
035import org.jgrapes.portal.events.AddPageResources;
036import org.jgrapes.portal.events.AddPageResources.ScriptResource;
037import org.jgrapes.portal.events.PortalReady;
038
039/**
040 * Provider for the [Datatables](https://datatables.net/) library.
041 */
042public class DatatablesProvider extends PageResourceProvider {
043
044        /**
045         * Creates a new component with its channel set to the given 
046         * channel.
047         * 
048         * @param componentChannel the channel that the component's 
049         * handlers listen on by default and that 
050         * {@link Manager#fire(Event, Channel...)} sends the event to 
051         */
052        public DatatablesProvider(Channel componentChannel) {
053                super(componentChannel);
054        }
055
056        /**
057         * Provides a resource bundle for localization.
058         * The default implementation looks up a bundle using the
059         * package name plus "l10n" as base name.
060         * 
061         * @return the resource bundle
062         */
063        protected ResourceBundle resourceBundle(Locale locale) {
064                return ResourceBundle.getBundle(
065                        getClass().getPackage().getName() + ".datatables.l10n", locale, 
066                        getClass().getClassLoader(),
067                                ResourceBundle.Control.getNoFallbackControl(
068                                                ResourceBundle.Control.FORMAT_DEFAULT));
069        }
070        
071        /**
072         * On {@link PortalReady}, fire the appropriate {@link AddPageResources}.
073         *
074         * @param event the event
075         * @param portalSession the portal session
076         * @throws TemplateNotFoundException the template not found exception
077         * @throws MalformedTemplateNameException the malformed template name exception
078         * @throws ParseException the parse exception
079         * @throws IOException Signals that an I/O exception has occurred.
080         */
081        @Handler(priority=100)
082        public void onPortalReady(PortalReady event, PortalSession portalSession) 
083                        throws TemplateNotFoundException, MalformedTemplateNameException, 
084                        ParseException, IOException {
085                String minExt = event.renderSupport()
086                                .useMinifiedResources() ? ".min" : "";
087                ResourceBundle bundle = resourceBundle(portalSession.locale()); 
088                String script = 
089                                "$.fn.dataTable.defaults.oLanguage._hungarianMap"
090                                + "[\"lengthAll\"] = \"sLengthAll\";\n"
091                                + "$.extend( $.fn.dataTable.defaults.oLanguage, {\n"
092                                + "     'sLengthAll': 'all',\n"
093                                + "} );\n"
094                                + "$.extend( $.fn.dataTable.defaults.oLanguage, "
095                                + bundle.getString("DataTablesL10n") +  ");\n";
096                portalSession.respond(new AddPageResources()
097                                .addCss(event.renderSupport().pageResource(
098                                                "datatables/datatables" + minExt + ".css"))
099                                .addCss(event.renderSupport().pageResource(
100                                                "datatables/overrides-1.0.0.css"))
101                                .addScriptResource(new ScriptResource()
102                                                .setProvides(new String[] {"datatables.net"})
103                                                .setScriptUri(event.renderSupport().pageResource(
104                                                                "datatables/datatables" + minExt + ".js")))
105                                .addScriptResource(new ScriptResource()
106                                                .setRequires(new String[] {"datatables.net"})
107                                                .setScriptUri(event.renderSupport().pageResource(
108                                                                "datatables/processing().js")))
109                                .addScriptResource(new ScriptResource()
110                                                .setRequires(new String[] {"datatables.net"})
111                                                .setScriptSource(script)));
112        }
113
114}
