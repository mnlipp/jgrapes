019package org.jgrapes.portal;
020
021import java.io.IOException;
022import java.util.ArrayList;
023import java.util.HashMap;
024import java.util.List;
025import java.util.Map;
026
027import javax.json.JsonArray;
028
029import org.jgrapes.core.Channel;
030import org.jgrapes.core.Component;
031import org.jgrapes.core.Event;
032import org.jgrapes.core.Manager;
033import org.jgrapes.core.annotation.Handler;
034import org.jgrapes.portal.events.JsonInput;
035import org.jgrapes.portal.events.JsonOutput;
036import org.jgrapes.portal.events.PortalReady;
037import org.jgrapes.util.events.KeyValueStoreQuery;
038import org.jgrapes.util.events.KeyValueStoreUpdate;
039import org.jgrapes.util.events.KeyValueStoreUpdate.Action;
040import org.jgrapes.util.events.KeyValueStoreUpdate.Deletion;
041import org.jgrapes.util.events.KeyValueStoreUpdate.Update;
042
043/**
044 * 
045 */
046public class PortalLocalBackedKVStore extends Component {
047
048        private String portalPrefix;
049        
050        /**
051         * Create a new key/value store that uses the browser's local storage
052         * for persisting the values.
053         * 
054         * @param componentChannel the channel that the component's 
055         * handlers listen on by default and that 
056         * {@link Manager#fire(Event, Channel...)} sends the event to 
057         * @param portalPrefix the portal's prefix as returned by
058         * {@link Portal#prefix()}, i.e. staring and ending with a slash
059         */
060        public PortalLocalBackedKVStore(
061                        Channel componentChannel, String portalPrefix) {
062                super(componentChannel);
063                this.portalPrefix = portalPrefix;
064        }
065
066        @Handler(priority=1000)
067        public void onPortalReady(PortalReady event, PortalSession channel) 
068                        throws InterruptedException {
069                if (channel.browserSession().containsKey(
070                                PortalLocalBackedKVStore.class)) {
071                        return;
072                }
073                // Remove portal ready event from queue and save it
074                event.cancel(false);
075                channel.setAssociated(PortalLocalBackedKVStore.class, event);
076                String keyStart = portalPrefix 
077                                + PortalLocalBackedKVStore.class.getName() + "/";
078                channel.respond(new JsonOutput("retrieveLocalData", keyStart));
079        }
080        
081        @Handler
082        public void onJsonInput(JsonInput event, PortalSession channel) 
083                        throws InterruptedException, IOException {
084                if (!event.method().equals("retrievedLocalData")) {
085                        return;
086                }
087                @SuppressWarnings("unchecked")
088                Map<String,String> data = (Map<String,String>)channel
089                        .browserSession().computeIfAbsent(
090                                        PortalLocalBackedKVStore.class, k -> new HashMap<>());
091                final String keyStart = portalPrefix 
092                                + PortalLocalBackedKVStore.class.getName() + "/";
093                channel.associated(PortalLocalBackedKVStore.class, PortalReady.class)
094                        .ifPresent(origEvent -> {
095                                JsonArray params = (JsonArray)event.params();
096                                params.getJsonArray(0).forEach(item -> {
097                                        JsonArray kvPair = (JsonArray)item;
098                                        String key = kvPair.getString(0);
099                                        if (key.startsWith(keyStart)) {
100                                                data.put(kvPair.getString(0).substring(
101                                                                keyStart.length() - 1), kvPair.getString(1));
102                                        }
103                                });
104                                channel.setAssociated(PortalLocalBackedKVStore.class, null);
105                                // Let others process the portal ready event
106                                fire(new PortalReady(origEvent.renderSupport()), channel);
107                        });
108        }
109                
110        @Handler
111        public void onKeyValueStoreUpdate(
112                        KeyValueStoreUpdate event, PortalSession channel) 
113                                        throws InterruptedException, IOException {
114                @SuppressWarnings("unchecked")
115                Map<String,String> data = (Map<String,String>)channel
116                        .browserSession().computeIfAbsent(
117                                        PortalLocalBackedKVStore.class, k -> new HashMap<>());
118                List<String[]> actions = new ArrayList<>();
119                String keyStart = portalPrefix 
120                                + PortalLocalBackedKVStore.class.getName();
121                for (Action action: event.actions()) {
122                        String key = keyStart + (action.key().startsWith("/")
123                                        ? action.key() : ("/" + action.key()));
124                        if (action instanceof Update) {
125                                actions.add(new String [] {"u", key, 
126                                                ((Update)action).value() });
127                                data.put(action.key(), ((Update)action).value());
128                        } else if (action instanceof Deletion) {
129                                actions.add(new String [] {"d", key });
130                                data.remove(action.key());
131                        }
132                }
133                channel.respond(new JsonOutput("storeLocalData", 
134                                new Object[] { actions.toArray() }));
135        }
136
137        @Handler
138        public void onKeyValueStoreQuery(
139                        KeyValueStoreQuery event, PortalSession channel) {
140                Map<String,String> result = new HashMap<>();
141                @SuppressWarnings("unchecked")
142                Map<String,String> data = (Map<String,String>)channel
143                        .browserSession().get(PortalLocalBackedKVStore.class);
144                if (data != null) {
145                        if (!event.query().endsWith("/")) {
146                                // Single value
147                                if (data.containsKey(event.query())) {
148                                        result.put(event.query(), data.get(event.query()));
149                                }
150                        } else {
151                                for (Map.Entry<String, String> e: data.entrySet()) {
152                                        if (e.getKey().startsWith(event.query())) {
153                                                result.put(e.getKey(), e.getValue());
154                                        }
155                                }
156                        }
157                }
158                event.setResult(result);
159        }
160}
