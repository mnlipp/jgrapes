019package org.jgrapes.portal;
020
021import java.io.IOException;
022import java.util.ArrayList;
023import java.util.HashMap;
024import java.util.List;
025import java.util.Map;
026
027import org.jdrupes.json.JsonArray;
028import org.jgrapes.core.Channel;
029import org.jgrapes.core.Component;
030import org.jgrapes.core.Event;
031import org.jgrapes.core.Manager;
032import org.jgrapes.core.annotation.Handler;
033import org.jgrapes.portal.events.JsonInput;
034import org.jgrapes.portal.events.PortalReady;
035import org.jgrapes.portal.events.SimplePortalCommand;
036import org.jgrapes.util.events.KeyValueStoreQuery;
037import org.jgrapes.util.events.KeyValueStoreUpdate;
038import org.jgrapes.util.events.KeyValueStoreUpdate.Action;
039import org.jgrapes.util.events.KeyValueStoreUpdate.Deletion;
040import org.jgrapes.util.events.KeyValueStoreUpdate.Update;
041
042/**
043 * 
044 */
045public class PortalLocalBackedKVStore extends Component {
046
047    private final String portalPrefix;
048
049    /**
050     * Create a new key/value store that uses the browser's local storage
051     * for persisting the values.
052     * 
053     * @param componentChannel the channel that the component's 
054     * handlers listen on by default and that 
055     * {@link Manager#fire(Event, Channel...)} sends the event to 
056     * @param portalPrefix the portal's prefix as returned by
057     * {@link Portal#prefix()}, i.e. staring and ending with a slash
058     */
059    public PortalLocalBackedKVStore(
060            Channel componentChannel, String portalPrefix) {
061        super(componentChannel);
062        this.portalPrefix = portalPrefix;
063    }
064
065    /**
066     * Intercept {@link PortalReady} event to first get data.
067     *
068     * @param event the event
069     * @param channel the channel
070     * @throws InterruptedException the interrupted exception
071     */
072    @Handler(priority = 1000)
073    public void onPortalReady(PortalReady event, PortalSession channel)
074            throws InterruptedException {
075        // Put there by onJsonInput if retrieval has been done.
076        if (channel.browserSession().containsKey(
077            PortalLocalBackedKVStore.class)) {
078            return;
079        }
080        // Remove portal ready event from queue and save it
081        event.cancel(false);
082        channel.setAssociated(PortalLocalBackedKVStore.class, event);
083        String keyStart = portalPrefix
084            + PortalLocalBackedKVStore.class.getName() + "/";
085        channel.respond(new SimplePortalCommand("retrieveLocalData", keyStart));
086    }
087
088    @SuppressWarnings({ "unchecked", "PMD.AvoidDuplicateLiterals" })
089    private Map<String, String> getStore(PortalSession channel) {
090        return (Map<String, String>) channel
091            .browserSession().computeIfAbsent(
092                PortalLocalBackedKVStore.class, newKey -> new HashMap<>());
093    }
094
095    /**
096     * Evaluate "retrievedLocalData" response.
097     *
098     * @param event the event
099     * @param channel the channel
100     * @throws InterruptedException the interrupted exception
101     * @throws IOException Signals that an I/O exception has occurred.
102     */
103    @Handler
104    public void onJsonInput(JsonInput event, PortalSession channel)
105            throws InterruptedException, IOException {
106        if (!event.request().method().equals("retrievedLocalData")) {
107            return;
108        }
109        final String keyStart = portalPrefix
110            + PortalLocalBackedKVStore.class.getName() + "/";
111        channel.associated(PortalLocalBackedKVStore.class, PortalReady.class)
112            .ifPresent(origEvent -> {
113                // We have intercepted the portal ready, fill out store.
114                // Having a store now also shows that retrieval has been done.
115                @SuppressWarnings("PMD.DataflowAnomalyAnalysis")
116                Map<String, String> data = getStore(channel);
117                JsonArray params = (JsonArray) event.request().params();
118                params.asArray(0).arrayStream().forEach(item -> {
119                    String key = item.asString(0);
120                    if (key.startsWith(keyStart)) {
121                        data.put(key.substring(
122                            keyStart.length() - 1), item.asString(1));
123                    }
124                });
125                channel.setAssociated(PortalLocalBackedKVStore.class, null);
126                // Let others process the portal ready event
127                fire(new PortalReady(origEvent.renderSupport()), channel);
128            });
129    }
130
131    /**
132     * Handle data update events.
133     *
134     * @param event the event
135     * @param channel the channel
136     * @throws InterruptedException the interrupted exception
137     * @throws IOException Signals that an I/O exception has occurred.
138     */
139    @Handler
140    @SuppressWarnings({ "PMD.DataflowAnomalyAnalysis",
141        "PMD.AvoidInstantiatingObjectsInLoops" })
142    public void onKeyValueStoreUpdate(
143            KeyValueStoreUpdate event, PortalSession channel)
144            throws InterruptedException, IOException {
145        Map<String, String> data = getStore(channel);
146        List<String[]> actions = new ArrayList<>();
147        String keyStart = portalPrefix
148            + PortalLocalBackedKVStore.class.getName();
149        for (Action action : event.actions()) {
150            @SuppressWarnings("PMD.UselessParentheses")
151            String key = keyStart + (action.key().startsWith("/")
152                ? action.key()
153                : ("/" + action.key()));
154            if (action instanceof Update) {
155                actions.add(new String[] { "u", key,
156                    ((Update) action).value() });
157                data.put(action.key(), ((Update) action).value());
158            } else if (action instanceof Deletion) {
159                actions.add(new String[] { "d", key });
160                data.remove(action.key());
161            }
162        }
163        channel.respond(new SimplePortalCommand("storeLocalData",
164            new Object[] { actions.toArray() }));
165    }
166
167    /**
168     * Handle data query..
169     *
170     * @param event the event
171     * @param channel the channel
172     */
173    @Handler
174    @SuppressWarnings("PMD.ConfusingTernary")
175    public void onKeyValueStoreQuery(
176            KeyValueStoreQuery event, PortalSession channel) {
177        @SuppressWarnings("PMD.UseConcurrentHashMap")
178        Map<String, String> result = new HashMap<>();
179        @SuppressWarnings("unchecked")
180        Map<String, String> data = (Map<String, String>) channel
181            .browserSession().get(PortalLocalBackedKVStore.class);
182        if (data != null) {
183            if (!event.query().endsWith("/")) {
184                // Single value
185                if (data.containsKey(event.query())) {
186                    result.put(event.query(), data.get(event.query()));
187                }
188            } else {
189                for (Map.Entry<String, String> e : data.entrySet()) {
190                    if (e.getKey().startsWith(event.query())) {
191                        result.put(e.getKey(), e.getValue());
192                    }
193                }
194            }
195        }
196        event.setResult(result);
197    }
198}
