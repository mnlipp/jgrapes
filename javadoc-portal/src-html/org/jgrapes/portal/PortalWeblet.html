149        /**
150         * Instantiates a new portal weblet.
151         *
152         * @param webletChannel the weblet channel
153         * @param portal the portal
154         */
155        public PortalWeblet(Channel webletChannel, Portal portal) {
156                super(webletChannel, ChannelReplacements.create()
157                                .add(PortalChannel.class, portal.channel()));
158                this.portal = portal;
159                String portalPath = portal.prefix().getPath();
160                if (portalPath.endsWith("/")) {
161                        portalPath = portalPath.substring(0, portalPath.length() - 1);
162                }
163                portalPath = portalPath + "|**";
164                try {
165                        requestPattern = new ResourcePattern(portalPath);
166                } catch (ParseException e) {
167                        throw new IllegalArgumentException(e);
168                }
169                
170                baseTheme = new Provider();
171                
172                supportedLocales = new HashSet<>();
173                for (Locale locale: Locale.getAvailableLocales()) {
174                        if (locale.getLanguage().equals("")) {
175                                continue;
176                        }
177                        if (resourceBundleSupplier != null) {
178                                ResourceBundle rb = resourceBundleSupplier.apply(locale);
179                                if (rb.getLocale().equals(locale)) {
180                                        supportedLocales.add(locale);
181                                }
182                        }
183                        ResourceBundle rb = ResourceBundle.getBundle(getClass()
184                                        .getPackage().getName() + ".l10n", locale);
185                        if (rb.getLocale().equals(locale)) {
186                                supportedLocales.add(locale);
187                        }
188                }
189                
190                RequestHandler.Evaluator.add(this, "onGet",     portal.prefix() + "**");
191                RequestHandler.Evaluator.add(this, "onGetRedirect",
192                                portal.prefix().getPath().substring(
193                                                0, portal.prefix().getPath().length() - 1));
194                
195                portalBaseModel = createPortalBaseModel();
196        }

198        private Map<String,Object> createPortalBaseModel() {
199                // Create portal model
200                portalBaseModel = new HashMap<>();
201                portalBaseModel.put("resourceUrl", new TemplateMethodModelEx() {
202                        @Override
203                        public Object exec(@SuppressWarnings("rawtypes") List arguments)
204                                        throws TemplateModelException {
205                                @SuppressWarnings("unchecked")
206                                List<TemplateModel> args = (List<TemplateModel>)arguments;
207                                if (!(args.get(0) instanceof SimpleScalar)) {
208                                        throw new TemplateModelException("Not a string.");
209                                }
210                                return portal.prefix().resolve(
211                                                ((SimpleScalar)args.get(0)).getAsString()).getRawPath();
212                        }
213                });
214                portalBaseModel.put("useMinifiedResources", useMinifiedResources);
215                portalBaseModel.put("minifiedExtension", 
216                                useMinifiedResources ? ".min" : "");
217                portalBaseModel.put(
218                                "portalSessionRefreshInterval", portalSessionRefreshInterval);
219                portalBaseModel.put(
220                                "portalSessionInactivityTimeout", portalSessionInactivityTimeout);
221                return Collections.unmodifiableMap(portalBaseModel);
222        }

309        @RequestHandler(dynamic=true)
310        public void onGetRedirect(GetRequest event, IOSubchannel channel) 
311                        throws InterruptedException, IOException, ParseException {
312                HttpResponse response = event.httpRequest().response().get();
313                response.setStatus(HttpStatus.MOVED_PERMANENTLY)
314                        .setContentType("text", "plain", "utf-8")
315                        .setField(HttpField.LOCATION, portal.prefix());
316                channel.respond(new Response(response));
317                try {
318                        channel.respond(Output.from(portal.prefix().toString()
319                                        .getBytes("utf-8"), true));
320                } catch (UnsupportedEncodingException e) {
321                        // Supported by definition
322                }
323                event.setResult(true);
324                event.stop();
325        }

327        @RequestHandler(dynamic=true)
328        public void onGet(GetRequest event, IOSubchannel channel) 
329                        throws InterruptedException, IOException, ParseException {
330                URI requestUri = event.requestUri();
331                int prefixSegs = requestPattern.matches(requestUri);
332                // Request for portal? (Only valid with session)
333                if (prefixSegs < 0 || !event.associated(Session.class).isPresent()) {
334                        return;
335                }
336                
337                // Normalize and evaluate
338                String requestPath = ResourcePattern.removeSegments(
339                                requestUri.getPath(), prefixSegs + 1);
340                String[] requestParts = ResourcePattern.split(requestPath, 1);
341                switch (requestParts[0]) {
342                case "":
343                        renderPortal(event, channel);
344                        return;
345                case "portal-resource":
346                        ResponseCreationSupport.sendStaticContent(event, channel, 
347                                        p -> PortalWeblet.this.getClass().getResource(
348                                                        requestParts[1]), null);
349                        return;
350                case "page-resource":
351                        requestPageResource(event, channel, requestParts[1]);
352                        return;
353                case "portal-session":
354                        handleSessionRequest(event, channel, requestParts[1]);
355                        return;
356                case "theme-resource":
357                        sendThemeResource(event, channel, requestParts[1]);
358                        return;
359                case "portlet-resource":
360                        requestPortletResource(event, channel, URI.create(requestParts[1]));
361                        return;
362                }
363        }

365        private void renderPortal(GetRequest event, IOSubchannel channel)
366                throws IOException, InterruptedException {
367                event.setResult(true);
368                event.stop();
369                
370                // Because language is changed via websocket, locale cookie 
371                // may be out-dated
372                event.associated(Selection.class)
373                        .ifPresent(s -> s.prefer(s.get()[0]));
374
375                // This is a portal session now (can be connected to)
376                Session session = event.associated(Session.class).get();
377                UUID portalSessionId = UUID.randomUUID();
378                @SuppressWarnings("unchecked")
379                Map<URI,UUID> knownIds = (Map<URI,UUID>)session.computeIfAbsent(
380                                PORTAL_SESSION_IDS, k -> new HashMap<URI,UUID>());
381                knownIds.put(portal.prefix(), portalSessionId);
382                
383                // Prepare response
384                HttpResponse response = event.httpRequest().response().get();
385                MediaType mediaType = MediaType.builder().setType("text", "html")
386                                .setParameter("charset", "utf-8").build();
387                response.setField(HttpField.CONTENT_TYPE, mediaType);
388                response.setStatus(HttpStatus.OK);
389                response.setHasPayload(true);
390                channel.respond(new Response(response));
391                try (ByteBufferOutputStream bbos = new ByteBufferOutputStream(
392                                channel, channel.responsePipeline());
393                                Writer out = new OutputStreamWriter(bbos.suppressClose(), 
394                                                "utf-8")) {
395                        Map<String,Object> portalModel = new HashMap<>(portalBaseModel);
396
397                        // Portal Session UUID
398                        portalModel.put("portalSessionId", portalSessionId.toString());
399                        
400                        // Add locale
401                        final Locale locale = event.associated(Selection.class).map(
402                                        s -> s.get()[0]).orElse(Locale.getDefault());
403                        portalModel.put("locale", locale);
404
405                        // Add supported locales
406                        final Collator coll = Collator.getInstance(locale);
407                        final Comparator<LanguageInfo> comp 
408                                = new Comparator<PortalWeblet.LanguageInfo>() {
409                                @Override
410                                public int compare(LanguageInfo o1,  LanguageInfo o2) {
411                                        return coll.compare(o1.getLabel(), o2.getLabel());
412                                }
413                        };
414                        LanguageInfo[] languages = supportedLocales.stream()
415                                        .map(l -> new LanguageInfo(l))
416                                        .sorted(comp).toArray(size -> new LanguageInfo[size]);
417                        portalModel.put("supportedLanguages", languages);
418
419                        // Add localization
420                        final ResourceBundle additionalResources = resourceBundleSupplier == null
421                                        ? null : resourceBundleSupplier.apply(locale);
422                        final ResourceBundle baseResources = ResourceBundle.getBundle(
423                                        getClass().getPackage().getName() + ".l10n", locale,
424                                        ResourceBundle.Control.getNoFallbackControl(
425                                                        ResourceBundle.Control.FORMAT_DEFAULT));
426                        portalModel.put("_", new TemplateMethodModelEx() {
427                                @Override
428                                public Object exec(@SuppressWarnings("rawtypes") List arguments)
429                                                throws TemplateModelException {
430                                        @SuppressWarnings("unchecked")
431                                        List<TemplateModel> args = (List<TemplateModel>)arguments;
432                                        if (!(args.get(0) instanceof SimpleScalar)) {
433                                                throw new TemplateModelException("Not a string.");
434                                        }
435                                        String key = ((SimpleScalar)args.get(0)).getAsString();
436                                        try {
437                                                return additionalResources.getString(key);
438                                        } catch (MissingResourceException e) {
439                                                // try base resources
440                                        }
441                                        try {
442                                                return baseResources.getString(key);
443                                        } catch (MissingResourceException e) {
444                                                // no luck
445                                        }
446                                        return key;
447                                }
448                        });
449
450                        // Add themes. Doing this on every reload allows themes
451                        // to be added dynamically. Note that we must load again
452                        // (not reload) in order for this to work in an OSGi environment.
453                        themeLoader = ServiceLoader.load(ThemeProvider.class);
454                        portalModel.put("themeInfos", 
455                                        StreamSupport.stream(themeLoader().spliterator(), false)
456                                        .map(t -> new ThemeInfo(t.themeId(), t.themeName()))
457                                        .sorted().toArray(size -> new ThemeInfo[size]));
458                        
459                        Template tpl = fmConfig.getTemplate("portal.ftlh");
460                        tpl.process(portalModel, out);
461                } catch (TemplateException e) {
462                        throw new IOException(e);
463                }
464        }

466        private void sendThemeResource(GetRequest event, IOSubchannel channel,
467                        String resource) throws ParseException {
468                // Get resource
469                ThemeProvider themeProvider = event.associated(Session.class).flatMap(
470                                session -> Optional.ofNullable(session.get("themeProvider")).flatMap(
471                                                themeId -> StreamSupport
472                                                .stream(themeLoader().spliterator(), false)
473                                                .filter(t -> t.themeId().equals(themeId)).findFirst()
474                                )).orElse(baseTheme);
475                URL resourceUrl;
476                try {
477                        resourceUrl = themeProvider.getResource(resource);
478                } catch (ResourceNotFoundException e) {
479                        try {
480                                resourceUrl = baseTheme.getResource(resource);
481                        } catch (ResourceNotFoundException e1) {
482                                resourceUrl = fallbackResourceSupplier.apply(themeProvider, resource);
483                                if (resourceUrl == null) {
484                                        return;
485                                }
486                        }
487                }
488                final URL resUrl = resourceUrl;
489                ResponseCreationSupport.sendStaticContent(event, channel, 
490                                p -> resUrl, null);
491        }

493        private void requestPageResource(GetRequest event, IOSubchannel channel,
494                        String resource) throws InterruptedException {
495                // Send events to providers on portal's channel
496                PageResourceRequest pageResourceRequest = new PageResourceRequest(
497                                uriFromPath(resource),
498                                event.httpRequest().findValue(HttpField.IF_MODIFIED_SINCE, 
499                                                Converters.DATE_TIME).orElse(null),
500                                event.httpRequest(), channel, renderSupport());
501                // Make session available (associate with event, this is not
502                // a websocket request).
503                event.associated(Session.class).ifPresent(
504                                session -> pageResourceRequest.setAssociated(Session.class, session));
505                event.setResult(true);
506                event.stop();
507                fire(pageResourceRequest, portalChannel(channel));
508        }

510        private void requestPortletResource(GetRequest event, IOSubchannel channel,
511                        URI resource) throws InterruptedException {
512                String resPath = resource.getPath();
513                int sep = resPath.indexOf('/');
514                // Send events to portlets on portal's channel
515                PortletResourceRequest portletRequest = new PortletResourceRequest(
516                                resPath.substring(0, sep), 
517                                uriFromPath(resPath.substring(sep + 1)),
518                                event.httpRequest().findValue(HttpField.IF_MODIFIED_SINCE, 
519                                                Converters.DATE_TIME).orElse(null),
520                                event.httpRequest(), channel, renderSupport());
521                // Make session available (associate with event, this is not
522                // a websocket request).
523                event.associated(Session.class).ifPresent(
524                                session -> portletRequest.setAssociated(Session.class, session));
525                event.setResult(true);
526                event.stop();
527                fire(portletRequest, portalChannel(channel));
528        }

531         * The portal channel for getting resources. Resource providers
532         * respond on the same event pipeline as they receive, because
533         * handling is just a mapping to {@link ResourceRequestCompleted}.
534         *
535         * @param channel the channel
536         * @return the IO subchannel
537         */
538        private IOSubchannel portalChannel(IOSubchannel channel) {
539                @SuppressWarnings("unchecked")
540                Optional<LinkedIOSubchannel> portalChannel
541                        = (Optional<LinkedIOSubchannel>)LinkedIOSubchannel
542                                .downstreamChannel(portal, channel);
543                return portalChannel.orElseGet(
544                                () -> new PortalResourceChannel(
545                                                portal, channel, activeEventPipeline()));
546        }

548        @Handler(channels=PortalChannel.class)
549        public void onResourceRequestCompleted(
550                        ResourceRequestCompleted event, PortalResourceChannel channel) 
551                                        throws IOException, InterruptedException {
552                event.stop();
553                if (event.event().get() == null) {
554                        ResponseCreationSupport.sendResponse(event.event().httpRequest(), 
555                                        event.event().httpChannel(), HttpStatus.NOT_FOUND);
556                        return;
557                }
558                event.event().get().process();
559        }

561        private void handleSessionRequest(
562                        GetRequest event, IOSubchannel channel, String portalSessionId)
563                                        throws InterruptedException, IOException, ParseException {
564                // Must be WebSocket request.
565                if (!event.httpRequest().findField(
566                                HttpField.UPGRADE, Converters.STRING_LIST)
567                                .map(f -> f.value().containsIgnoreCase("websocket"))
568                                .orElse(false)) {
569                        return;
570                }
571                // Can only connect to sessions that have been prepared
572                // by loading the portal. (Prevents using a newly created
573                // browser session from being (re-)connected to after a 
574                // long disconnect or restart and, of course, CSF).
575                final Session browserSession = event.associated(Session.class).get();
576                @SuppressWarnings("unchecked")
577                Map<URI,UUID> knownIds = (Map<URI,UUID>)browserSession.computeIfAbsent(
578                                PORTAL_SESSION_IDS, k -> new HashMap<URI,UUID>());
579                if (!UUID.fromString(portalSessionId).equals(knownIds.get(portal.prefix()))) {
580                        channel.respond(new WebSocketAccepted(event)).get();
581                        @SuppressWarnings("resource")
582                        CharBufferWriter out = new CharBufferWriter(channel, 
583                                        channel.responsePipeline()).suppressClose();
584                        new SimplePortalCommand("reload").toJson(out);
585                        out.close();
586                        event.stop();
587                        return;
588                }
589                // Reuse old portal session if still available
590                Optional<String> oldPortalSessionId = Optional.ofNullable(
591                                event.httpRequest().queryData().get("was")).map(l -> l.get(0));
592                PortalSession portalSession = oldPortalSessionId.flatMap(
593                                opsId -> PortalSession.lookup(opsId))
594                                .map(ps -> ps.replaceId(portalSessionId))
595                                .orElse(PortalSession.lookupOrCreate(
596                                                portalSessionId, portal, portalSessionNetworkTimeout))
597                                .setUpstreamChannel(channel)
598                                .setSession(browserSession);
599                channel.setAssociated(PortalSession.class, portalSession);
600                // Channel now used as JSON input
601                channel.setAssociated(this, new WsInputReader(
602                                event.processedBy().get(), portalSession));
603                channel.respond(new WebSocketAccepted(event));
604                event.stop();
605        }

619         * 
620         * @param event the event
621         * @param wsChannel the WebSocket channel
622         * @throws IOException 
623         */
624        @Handler
625        public void onClosed(
626                        Closed event, IOSubchannel wsChannel) throws IOException {
627                Optional<WsInputReader> optWsInputReader 
628                        = wsChannel.associated(this, WsInputReader.class);
629                if (optWsInputReader.isPresent()) {
630                        wsChannel.setAssociated(this, null);
631                        optWsInputReader.get().close();
632                }
633                wsChannel.associated(PortalSession.class).ifPresent(
634                                ps -> ps.disconnected());
635        }

637        @Handler(channels=PortalChannel.class)
638        public void onPortalReady(PortalReady event, PortalSession channel) {
639                String principal =      Utils.userFromSession(channel.browserSession())
640                                .map(UserPrincipal::toString).orElse("");
641                KeyValueStoreQuery query = new KeyValueStoreQuery(
642                                "/" + principal + "/themeProvider", channel);
643                fire(query, channel);
644        }

646        @Handler(channels=PortalChannel.class)
647        public void onKeyValueStoreData(
648                        KeyValueStoreData event, PortalSession channel) 
649                                        throws JsonDecodeException {
650                Session session = channel.browserSession();
651                String principal = Utils.userFromSession(session)
652                                .map(UserPrincipal::toString).orElse("");
653                if (!event.event().query().equals("/" + principal + "/themeProvider")) {
654                        return;
655                }
656                if (!event.data().values().iterator().hasNext()) {
657                        return;
658                }
659                String requestedThemeId = event.data().values().iterator().next();
660                ThemeProvider themeProvider = Optional.ofNullable(
661                                session.get("themeProvider")).flatMap(
662                                                themeId -> StreamSupport
663                                                .stream(themeLoader().spliterator(), false)
664                                                .filter(t -> t.themeId().equals(themeId)).findFirst()
665                                ).orElse(baseTheme);
666                if (!themeProvider.themeId().equals(requestedThemeId)) {
667                        fire(new SetTheme(requestedThemeId), channel);
668                }
669        }

677        public void onSetLocale(SetLocale event, PortalSession channel)
678                        throws InterruptedException, IOException {
679                Session session = channel.browserSession();
680                if (session != null) {
681                        Selection selection = (Selection)session.get(Selection.class);
682                        if (selection != null) {
683                                supportedLocales.stream()
684                                .filter(l -> l.equals(event.locale())).findFirst()
685                                .ifPresent(l -> selection.prefer(l));
686                        }
687                }
688                channel.respond(new SimplePortalCommand("reload"));
689        }

692        public void onSetTheme(SetTheme event, PortalSession channel)
693                        throws InterruptedException, IOException {
694                ThemeProvider themeProvider = StreamSupport
695                        .stream(themeLoader().spliterator(), false)
696                        .filter(t -> t.themeId().equals(event.theme())).findFirst()
697                        .orElse(baseTheme);
698                channel.browserSession().put("themeProvider", themeProvider.themeId());
699                channel.respond(new KeyValueStoreUpdate().update(
700                                "/" + Utils.userFromSession(channel.browserSession())
701                                .map(UserPrincipal::toString).orElse("") 
702                                + "/themeProvider", themeProvider.themeId())).get();
703                channel.respond(new SimplePortalCommand("reload"));
704        }

717        private static class WsInputReader extends Thread {
718
719                private static ReferenceQueue<Object> abandoned 
720                        = new ReferenceQueue<>();
721
722                private static class RefWithThread<T> extends WeakReference<T> {
723                        public Thread watched;
724
725                        public RefWithThread(T referent, Thread thread) {
726                                super(referent, abandoned);
727                                watched = thread;
728                        }
729                }
730
731                static {
732                        Thread watchdog = new Thread(() -> {
733                                while (true) {
734                                        try {
735                                                @SuppressWarnings("unchecked")
736                                                RefWithThread<Object> ref 
737                                                        = (RefWithThread<Object>)abandoned.remove();
738                                                ref.watched.interrupt();
739                                        } catch (InterruptedException e) {
740                                                // Nothing to do
741                                        }
742                                }
743                        });
744                        watchdog.setDaemon(true);
745                        watchdog.start();
746                }
747                
748                private WeakReference<EventPipeline> pipelineRef;
749                private WeakReference<PortalSession> channelRef;
750                private PipedWriter decodeIn;
751                private Reader jsonSource;
752
753                public WsInputReader(EventPipeline wsInPipeline,
754                        PortalSession portalChannel) {
755                        this.pipelineRef = new RefWithThread<>(
756                                        wsInPipeline, WsInputReader.this);
757                        this.channelRef = new RefWithThread<>(
758                                        portalChannel, WsInputReader.this);
759                        setDaemon(true);
760                }
761
762                public void write(CharBuffer buffer) throws IOException {
763                        // Delayed initialization, allows adaption to buffer size.
764                        if (decodeIn == null) {
765                                decodeIn = new PipedWriter();
766                                jsonSource = new PipedReader(decodeIn, buffer.capacity());
767                                start();
768                        }
769                        decodeIn.append(buffer);
770                        decodeIn.flush();
771                }
772
773                public void close() throws IOException {
774                        if (decodeIn == null) {
775                                // Never started
776                                return;
777                        }
778                        decodeIn.close();
779                        try {
780                                join(1000);
781                        } catch (InterruptedException e) {
782                                // Just in case
783                                interrupt();
784                        }
785                }
786
787                public void run() {
788                        while (true) {
789                                JsonBeanDecoder jsonDecoder = JsonBeanDecoder.create(jsonSource);
790                                JsonRpc rpc;
791                                try {
792                                        rpc = jsonDecoder.readObject(DefaultJsonRpc.class);
793                                } catch (JsonDecodeException e) {
794                                        break;
795                                }
796                                if (rpc == null) {
797                                        break;
798                                }
799                                // Fully decoded JSON available.
800                                PortalSession portalSession = channelRef.get();
801                                EventPipeline eventPipeline = pipelineRef.get();
802                                if (eventPipeline == null || portalSession == null) {
803                                        break;
804                                }
805                                // Portal session established, check for special disconnect
806                                if ("disconnect".equals(rpc.method())
807                                                        && portalSession.portalSessionId().equals(
808                                                                        rpc.params().asString(0))) {
809                                        portalSession.discard();
810                                        return;
811                                }
812                                // Ordinary message from portal (view) to server.
813                                portalSession.refresh();
814                                if("keepAlive".equals(rpc.method())) {
815                                        continue;
816                                }
817                                eventPipeline.fire(new JsonInput(rpc), portalSession);
818                        }
819                }
820        }

882        /**
883         * Create a {@link URI} from a path. This is similar to calling
884         * `new URI(null, null, path, null)` with the {@link URISyntaxException}
885         * converted to a {@link IllegalArgumentException}.
886         * 
887         * @param path the path
888         * @return the uri
889         * @throws IllegalArgumentException if the string violates 
890         * RFC 2396
891         */
892        public static URI uriFromPath(String path) throws IllegalArgumentException {
893                try {
894                        return new URI(null, null, path, null);
895                } catch (URISyntaxException e) {
896                        throw new IllegalArgumentException(e);
897                }
