019package org.jgrapes.portal;
020
021import java.io.IOException;
022import java.io.InputStream;
023import java.io.OutputStream;
024import java.time.Instant;
025
026import org.jdrupes.httpcodec.protocols.http.HttpConstants.HttpStatus;
027import org.jdrupes.httpcodec.protocols.http.HttpField;
028import org.jdrupes.httpcodec.protocols.http.HttpResponse;
029import org.jdrupes.httpcodec.types.MediaType;
030import org.jgrapes.http.ResponseCreationSupport;
031import org.jgrapes.http.events.Response;
032import org.jgrapes.io.util.InputStreamPipeline;
033import org.jgrapes.portal.events.ResourceRequest;
034
035/**
036 * Returns an {@link OutputStream} as result.
037 */
038public class ResourceByInputStream extends ResourceResult {
039
040    private final InputStream stream;
041    private final MediaType mediaType;
042    private final Instant lastModifiedAt;
043    private final int maxAge;
044
045    /**
046     * Instantiates a result that is provided by an {@link OutputStream}.
047     *
048     * @param request the request
049     * @param stream the stream
050     * @param maxAge
051     */
052    public ResourceByInputStream(ResourceRequest request,
053            InputStream stream, MediaType mediaType,
054            Instant lastModifiedAt, int maxAge) {
055        super(request);
056        this.stream = stream;
057        this.mediaType = mediaType;
058        this.lastModifiedAt = lastModifiedAt;
059        this.maxAge = maxAge;
060    }
061
062    @Override
063    public void process() throws IOException, InterruptedException {
064        if (stream == null) {
065            ResponseCreationSupport.sendResponse(request().httpRequest(),
066                request().httpChannel(), HttpStatus.NOT_FOUND);
067            return;
068        }
069
070        HttpResponse response = request().httpRequest().response().get();
071        if (lastModifiedAt != null) {
072            response.setField(HttpField.LAST_MODIFIED, lastModifiedAt);
073        }
074        response.setContentType(mediaType);
075        ResponseCreationSupport.setMaxAge(response, (req, mtype) -> maxAge,
076            request().httpRequest(), mediaType);
077        response.setStatus(HttpStatus.OK);
078        request().httpChannel().respond(new Response(response));
079        // Start sending content
080        new InputStreamPipeline(stream, request().httpChannel()).run();
081    }
082
083}
