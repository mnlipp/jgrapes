019package org.jgrapes.portal.events;
020
021import java.util.HashMap;
022import java.util.Map;
023import java.util.function.BiConsumer;
024
025import org.jgrapes.portal.Portlet.RenderMode;
026import org.jgrapes.portal.RenderSupport;
027
028
029/**
030 * Sent to the portal (server) if a new portlet instance of a given 
031 * type should be added to the portal page. The portal server usually 
032 * responds with a {@link RenderPortlet} event that has as payload the
033 * HTML that displays the portlet on the portal page.
034 * 
035 * Properties may be passed with the event. The interpretation
036 * of the properties is completely dependent on the handling portlet. 
037 * 
038 * The event's result is the portlet id of the new portlet instance.
039 * 
040 * ![Event Sequence](AddPortletRequestSeq.svg)
041 * 
042 * @startuml AddPortletRequestSeq.svg
043 * hide footbox
044 * 
045 * Browser -> Portal: "addPortlet"
046 * activate Portal
047 * Portal -> Portlet: AddPortletRequest
048 * deactivate Portal
049 * activate Portlet
050 * Portlet -> Portal: RenderPortlet
051 * deactivate Portlet
052 * activate Portal
053 * Portal -> Browser: "renderPortlet"
054 * deactivate Portal
055 * 
056 * @enduml
057 * 
058 */
059public class AddPortletRequest extends RenderPortletRequestBase<String> {
060
061        private String portletType;
062        private Map<? extends Object, ? extends Object> properties = null;
063        
064        /**
065         * Creates a new event.
066         * 
067         * @param renderSupport the render support
068         * @param portletType the type of the portlet
069         * @param mode the view mode that is requested
070         */
071        public AddPortletRequest(RenderSupport renderSupport, String portletType,
072                RenderMode mode) {
073                super(renderSupport, mode);
074                this.portletType = portletType;
075        }
076
077        /**
078         * Creates a new event.
079         * 
080         * @param renderSupport the render support
081         * @param portletType the type of the portlet
082         * @param mode the view mode that is requested
083         * @param properties optional values for properties of the portlet instance
084         */
085        public AddPortletRequest(RenderSupport renderSupport, String portletType,
086                RenderMode mode, Map<?,?> properties) {
087                super(renderSupport, mode);
088                this.portletType = portletType;
089                @SuppressWarnings("unchecked")
090                Map<Object, Object> props = (Map<Object,Object>)properties;
091                this.properties = props;
092        }
093
094        /**
095         * Returns the portlet type
096         * 
097         * @return the portlet type
098         */
099        public String portletType() {
100                return portletType;
101        }
102
103        /**
104         * Returns the properties. Every event returns a mutable map,
105         * thus allowing event handlers to modify the map even if
106         * none was passed to the constructor.
107         */
108        public Map<Object,Object> properties() {
109                if (properties == null) {
110                        properties = new HashMap<>();
111                }
112                @SuppressWarnings("unchecked")
113                Map<Object, Object> props = (Map<Object, Object>) properties;
114                return props;
115        }
116        
117        /**
118         * Convenience method for adding properties one-by-one.
119         * 
120         * @param key the property key
121         * @param value the property value
122         * @return the event for easy chaining
123         */
124        public AddPortletRequest addOption(Object key, Object value) {
125                properties().put(key, value);
126                return this;
127        }
128        
129        /**
130         * Convenience method that performs the given action if a property
131         * with the given key exists.
132         * 
133         * @param key the property key
134         * @param action the action to perform
135         */
136        public AddPortletRequest ifPresent(
137                        Object key, BiConsumer<Object,Object> action) {
138                if (properties().containsKey(key)) {
139                        action.accept(key, properties().get(key));
140                }
141                return this;
142        }
143}
