019package org.jgrapes.portal.events;
020
021import java.net.URI;
022import java.net.URL;
023import java.time.Instant;
024import java.util.Optional;
025
026import org.jdrupes.httpcodec.protocols.http.HttpRequest;
027import org.jgrapes.core.Event;
028import org.jgrapes.http.events.GetRequest;
029import org.jgrapes.io.IOSubchannel;
030import org.jgrapes.portal.RenderSupport;
031import org.jgrapes.portal.ResourceByUrl;
032import org.jgrapes.portal.ResourceNotModified;
033import org.jgrapes.portal.ResourceProvided;
034import org.jgrapes.portal.ResourceResult;
035
036/**
037 * An event that signals the request for a resource by the browser.
038 * 
039 * This event is effectively a "transformed" {@link GetRequest}. It 
040 * simplifies handling of such an event by portal components, because
041 * they can simply set a result of type {@link ResourceResult} and
042 * thus need no knowledge about generating all the events required to 
043 * properly respond to a {@link GetRequest}.
044 * 
045 * The complete sequence of events is shown in the diagram.
046 * 
047 * ![Resource Response Sequence](ResourceResponseSeq.svg)
048 * 
049 * Of course, due to internal buffering, the "Response Header" data
050 * and the "Response body" data may collapse in a single message
051 * that is sent to the browser (in case of a small resource).
052 * 
053 * If a value is provided by {@link #ifModifiedSince()},
054 * and the resource has not changed since the given instant,
055 * a resource provider may set {@link ResourceNotModified} as
056 * result. This information will be forwarded to the browser.
057 * For a result of type {@link ResourceByUrl}, the check
058 * for modification will be made automatically, using information
059 * derived from the {@link URL}.
060 * 
061 * Handlers of {@link ResourceRequest} events use usually only
062 * the information provided by {@link #resourceUri()}. The other
063 * items are needed by the handler of the {@link ResourceRequestCompleted}
064 * event (the portal) to generate the response for the {@link GetRequest}.
065 * 
066 * If none of the provided {@link ResourceResult} type matches the
067 * requirements of the resource provider, it can set
068 * {@link ResourceProvided} as result. This signals that it genertes
069 * the response itself.
070 * 
071 * ![Extended Resource Response Sequence](ResourceResponseSelfSeq.svg)
072 * 
073 * @startuml ResourceResponseSeq.svg
074 * hide footbox
075 * 
076 * activate Portal
077 * actor Framework
078 * Portal -> ResourceProvider: ResourceRequest
079 * activate ResourceProvider
080 * deactivate ResourceProvider
081 * deactivate Portal
082 * Framework -> Portal: ResourceRequestCompleted
083 * activate Portal
084 * Portal -> Browser: "Response Header"
085 * loop until end of data
086 *     Portal -> Browser: Output
087 *     Portal -> Browser: "Response body"
088 *     deactivate Portal
089 * end loop
090 * deactivate ResourceProvider
091 * deactivate Browser
092 * @enduml
093 * 
094 * @startuml ResourceResponseSelfSeq.svg
095 * hide footbox
096 * 
097 * activate Portal
098 * actor Framework
099 * Portal -> ResourceProvider: ResourceRequest
100 * activate ResourceProvider
101 * Framework -> Portal: ResourceRequestCompleted
102 * deactivate Portal
103 * ResourceProvider -> Browser: "Response Header"
104 * loop until end of data
105 *     ResourceProvider -> Browser: Output
106 *     ResourceProvider -> Browser: "Response body"
107 *     deactivate Portal
108 * end loop
109 * deactivate ResourceProvider
110 * deactivate Browser
111 * 
112 * @enduml
113 */
114@SuppressWarnings("PMD.DataClass")
115public class ResourceRequest extends Event<ResourceResult> {
116
117    private final URI resourceUri;
118    private final Instant ifModifiedSince;
119    private final HttpRequest httpRequest;
120    private final IOSubchannel httpChannel;
121    private final RenderSupport renderSupport;
122
123    /**
124     * Creates a new request, including the associated 
125     * {@link ResourceRequestCompleted} event.
126     * 
127     * @param resourceUri the requested resource
128     * @param httpRequest the original HTTP request
129     * @param httpChannel the channel that the HTTP request was received on
130     * @param renderSupport the render support
131     */
132    public ResourceRequest(URI resourceUri, Instant ifModifiedSince,
133            HttpRequest httpRequest, IOSubchannel httpChannel,
134            RenderSupport renderSupport) {
135        this.resourceUri = resourceUri;
136        this.ifModifiedSince = ifModifiedSince;
137        this.httpRequest = httpRequest;
138        this.httpChannel = httpChannel;
139        this.renderSupport = renderSupport;
140        new ResourceRequestCompleted(this);
141    }
142
143    /**
144     * @return the resourceUri
145     */
146    public URI resourceUri() {
147        return resourceUri;
148    }
149
150    /**
151     * If not null, this value may be used to decide if the
152     * resource must be refreshed.
153     *
154     * @return the instant
155     */
156    public Optional<Instant> ifModifiedSince() {
157        return Optional.ofNullable(ifModifiedSince);
158    }
159
160    /**
161     * Returns the "raw" request as provided by the HTTP decoder.
162     * 
163     * @return the request
164     */
165    public HttpRequest httpRequest() {
166        return httpRequest;
167    }
168
169    /**
170     * @return the httpChannel
171     */
172    public IOSubchannel httpChannel() {
173        return httpChannel;
174    }
175
176    /**
177     * Returns the render support.
178     * 
179     * @return the render support
180     */
181    public RenderSupport renderSupport() {
182        return renderSupport;
183    }
184
185}
