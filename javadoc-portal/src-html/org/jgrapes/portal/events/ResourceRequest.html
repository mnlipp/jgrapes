019package org.jgrapes.portal.events;
020
021import java.net.MalformedURLException;
022import java.net.URI;
023import java.net.URL;
024
025import org.jdrupes.httpcodec.protocols.http.HttpRequest;
026import org.jgrapes.core.Event;
027import org.jgrapes.http.events.GetRequest;
028import org.jgrapes.io.IOSubchannel;
029import org.jgrapes.portal.RenderSupport;
030
031/**
032 * An event that signals the request for a resource by the browser.
033 * 
034 * This event is effectively a "transformed" {@link GetRequest}. It 
035 * simplifies handling of such an event by portal components, because
036 * they only have to provide a {@link URL} as result of the event and 
037 * thus need no knowledge about generating all the events required to 
038 * properly respond to a {@link GetRequest}.
039 * 
040 * The complete sequence of events is shown in the diagram.
041 * 
042 * ![Resource Response Sequence](ResourceResponseSeq.svg)
043 * 
044 * Of course, due to internal buffering, the "Response Header" data
045 * and the "Response body" data may collapse in a single message
046 * that is sent to the browser (in case of a small resource).
047 * 
048 * Handlers of {@link ResourceRequest} events use usually only
049 * the information provided by {@link #resourceUri()}. The other
050 * items are needed by the handler of the {@link ResourceRequestCompleted}
051 * event (the portal) to generate the response for the {@link GetRequest}.
052 * 
053 * If a matching resource cannot be expressed by a {@link URL},
054 * the resource provider may generate the response itself. To
055 * indicate that the request has been handled, it must set the
056 * result value of this event to {@link #RESPONSE_GENERATED}.
057 * 
058 * ![Extended Resource Response Sequence](ResourceResponseSelfSeq.svg)
059 * 
060 * @startuml ResourceResponseSeq.svg
061 * hide footbox
062 * 
063 * activate Portal
064 * actor Framework
065 * Portal -> ResourceProvider: ResourceRequest
066 * activate ResourceProvider
067 * deactivate ResourceProvider
068 * deactivate Portal
069 * Framework -> Portal: ResourceRequestCompleted
070 * activate Portal
071 * Portal -> Browser: "Response Header"
072 * loop until end of data
073 *     Portal -> Browser: Output
074 *     Portal -> Browser: "Response body"
075 *     deactivate Portal
076 * end loop
077 * deactivate ResourceProvider
078 * deactivate Browser
079 * @enduml
080 * 
081 * @startuml ResourceResponseSelfSeq.svg
082 * hide footbox
083 * 
084 * activate Portal
085 * actor Framework
086 * Portal -> ResourceProvider: ResourceRequest
087 * activate ResourceProvider
088 * Framework -> Portal: ResourceRequestCompleted
089 * deactivate Portal
090 * ResourceProvider -> Browser: "Response Header"
091 * loop until end of data
092 *     ResourceProvider -> Browser: Output
093 *     ResourceProvider -> Browser: "Response body"
094 *     deactivate Portal
095 * end loop
096 * deactivate ResourceProvider
097 * deactivate Browser
098 * 
099 * @enduml
100 */
101public class ResourceRequest extends Event<URL> {
102        
103        private static URL createResponseGeneratedUrl() {
104                try {
105                        return new URL("file:///response/generated/by/provider");
106                } catch (MalformedURLException e) {
107                        // Does not happen
108                        return null;
109                }
110        }
111
112        /**
113         * Used as result of a {@link ResourceRequest} if the response
114         * has been generated by the provider.
115         */
116        public static final URL RESPONSE_GENERATED = createResponseGeneratedUrl();
117        
118        private URI resourceUri;
119        private HttpRequest httpRequest;
120        private IOSubchannel httpChannel;
121        private RenderSupport renderSupport;
122
123        /**
124         * Creates a new request, including the associated 
125         * {@link ResourceRequestCompleted} event.
126         * 
127         * @param resourceUri the requested resource
128         * @param httpRequest the original HTTP request
129         * @param httpChannel the channel that the HTTP request was received on
130         * @param renderSupport the render support
131         */
132        public ResourceRequest(URI resourceUri, HttpRequest httpRequest,
133                        IOSubchannel httpChannel, RenderSupport renderSupport) {
134                this.resourceUri = resourceUri;
135                this.httpRequest = httpRequest;
136                this.httpChannel = httpChannel;
137                this.renderSupport = renderSupport;
138                new ResourceRequestCompleted(this);
139        }
140
141        /**
142         * Returns the "raw" request as provided by the HTTP decoder.
143         * 
144         * @return the request
145         */
146        public HttpRequest httpRequest() {
147                return httpRequest;
148        }
149
150        /**
151         * @return the httpChannel
152         */
153        public IOSubchannel httpChannel() {
154                return httpChannel;
155        }
156
157        /**
158         * @return the resourceUri
159         */
160        public URI resourceUri() {
161                return resourceUri;
162        }
163        
164        /**
165         * Returns the render support.
166         * 
167         * @return the render support
168         */
169        public RenderSupport renderSupport() {
170                return renderSupport;
171        }
172
173}
