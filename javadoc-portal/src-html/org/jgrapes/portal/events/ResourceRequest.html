019package org.jgrapes.portal.events;
020
021import java.net.MalformedURLException;
022import java.net.URI;
023import java.net.URL;
024
025import org.jdrupes.httpcodec.protocols.http.HttpRequest;
026import org.jgrapes.core.Event;
027import org.jgrapes.http.events.GetRequest;
028import org.jgrapes.io.IOSubchannel;
029import org.jgrapes.portal.RenderSupport;
030
031/**
032 * An event that signals the request for a resource by the browser.
033 * 
034 * This event is effectively a "transformed" {@link GetRequest}. It 
035 * simplifies handling of such an event by portal components, because
036 * they only have to provide a {@link URL} as result and thus need
037 * no knowledge about generating all the events required to properly
038 * respond to a {@link GetRequest}.
039 * 
040 * Handlers of {@link ResourceRequest} events use usually only
041 * the information provided by {@link #resourceUri()}. The other
042 * items are needed by the handler of the {@link ResourceRequestCompleted}
043 * event to generate the response for the {@link GetRequest}.
044 * 
045 * If a matching resource cannot be expressed by a {@link URL},
046 * the resource provider may generate the response itself. To
047 * indicate that the request has been handled it must set the
048 * result value of this event to {@link #RESPONSE_GENERATED}.
049 */
050public class ResourceRequest extends Event<URL> {
051        
052        private static URL createResponseGeneratedUrl() {
053                try {
054                        return new URL("file:///response/generated/by/provider");
055                } catch (MalformedURLException e) {
056                        // Does not happen
057                        return null;
058                }
059        }
060
061        /**
062         * Used as result of a {@link ResourceRequest} if the response
063         * has been generated by the provider.
064         */
065        public static final URL RESPONSE_GENERATED = createResponseGeneratedUrl();
066        
067        private URI resourceUri;
068        private HttpRequest httpRequest;
069        private IOSubchannel httpChannel;
070        private RenderSupport renderSupport;
071
072        /**
073         * Creates a new request, including the associated 
074         * {@link ResourceRequestCompleted} event.
075         * 
076         * @param resourceUri the requested resource
077         * @param httpRequest the original HTTP request
078         * @param httpChannel the channel that the HTTP request was received on
079         * @param renderSupport the render support
080         */
081        public ResourceRequest(URI resourceUri, HttpRequest httpRequest,
082                        IOSubchannel httpChannel, RenderSupport renderSupport) {
083                this.resourceUri = resourceUri;
084                this.httpRequest = httpRequest;
085                this.httpChannel = httpChannel;
086                this.renderSupport = renderSupport;
087                new ResourceRequestCompleted(this);
088        }
089
090        /**
091         * Returns the "raw" request as provided by the HTTP decoder.
092         * 
093         * @return the request
094         */
095        public HttpRequest httpRequest() {
096                return httpRequest;
097        }
098
099        /**
100         * @return the httpChannel
101         */
102        public IOSubchannel httpChannel() {
103                return httpChannel;
104        }
105
106        /**
107         * @return the resourceUri
108         */
109        public URI resourceUri() {
110                return resourceUri;
111        }
112        
113        /**
114         * Returns the render support.
115         * 
116         * @return the render support
117         */
118        public RenderSupport renderSupport() {
119                return renderSupport;
120        }
121
122}
