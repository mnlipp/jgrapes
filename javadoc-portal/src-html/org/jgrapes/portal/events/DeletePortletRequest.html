019package org.jgrapes.portal.events;
020
021import org.jgrapes.core.Event;
022import org.jgrapes.portal.RenderSupport;
023
024/**
025 * A request to delete a portlet instance.
026 */
027public class DeletePortletRequest extends Event<Void> {
028
029    private final RenderSupport renderSupport;
030    private final String portletId;
031
032    /**
033     * Creates a new event.
034     * 
035     * @param renderSupport the render support from the portal in case
036     * the response requires it
037     * @param portletId the portlet model that the notification is 
038     * directed at 
039     */
040    public DeletePortletRequest(RenderSupport renderSupport,
041            String portletId) {
042        this.renderSupport = renderSupport;
043        this.portletId = portletId;
044    }
045
046    /**
047     * Returns the render support.
048     * 
049     * @return the render support
050     */
051    public RenderSupport renderSupport() {
052        return renderSupport;
053    }
054
055    /**
056     * Returns the portlet id.
057     * 
058     * @return the portlet id
059     */
060    public String portletId() {
061        return portletId;
062    }
063
064}
