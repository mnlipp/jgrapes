019package org.jgrapes.portal.events;
020
021import org.jgrapes.core.CompletionEvent;
022import org.jgrapes.core.Event;
023import org.jgrapes.portal.RenderSupport;
024
025/**
026 * Signals the successful loading of the portal structure
027 * in the browser. Portlets should respond to this event by
028 * adding their types (using {@link AddPortletType} events).
029 * 
030 * After the portal ready event has been processed, a {@link PortalPrepared}
031 * event is generated as its {@link CompletionEvent}. All further action is 
032 * triggered by the {@link PortalPrepared} event.
033 */
034public class PortalReady extends Event<Void> {
035
036    private final RenderSupport renderSupport;
037
038    /**
039     * Creates a new event.
040     * 
041     * @param renderSupport the render support for generating responses
042     */
043    public PortalReady(RenderSupport renderSupport) {
044        new PortalPrepared(this);
045        this.renderSupport = renderSupport;
046    }
047
048    /**
049     * Returns the render support.
050     * 
051     * @return the render support
052     */
053    public RenderSupport renderSupport() {
054        return renderSupport;
055    }
056}
