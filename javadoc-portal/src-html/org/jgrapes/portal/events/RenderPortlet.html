019package org.jgrapes.portal.events;
020
021import java.io.BufferedReader;
022import java.io.IOException;
023import java.io.Reader;
024import java.io.StringWriter;
025import java.io.Writer;
026import java.nio.CharBuffer;
027import java.util.Arrays;
028import java.util.Collections;
029import java.util.HashSet;
030import java.util.Set;
031
032import org.jgrapes.portal.Portlet.RenderMode;
033
034/**
035 * Send to the portal page for adding or updating a complete portlet
036 * representation.
037 */
038public class RenderPortlet extends PortalCommand {
039
040        private static final Set<RenderMode> DEFAULT_SUPPORTED 
041                = Collections.unmodifiableSet(new HashSet<>(
042                                Arrays.asList(new RenderMode[] { RenderMode.Preview })));
043
044        private Class<?> portletClass;
045        private String portletId;
046        private RenderMode renderMode = RenderMode.Preview;
047        private Set<RenderMode> supportedModes = DEFAULT_SUPPORTED;
048        private boolean foreground;
049        private Reader contentReader;
050
051        /**
052         * Creates a new event.
053         * 
054         * @param portletClass the portlet class
055         * @param portletId the id of the portlet
056         */
057        public RenderPortlet(Class<?> portletClass, String portletId, 
058                        Reader contentReader) {
059                this.portletClass = portletClass;
060                this.portletId = portletId;
061                this.contentReader = contentReader;
062        }
063
064        /**
065         * Set the render mode. The default value is {@link RenderMode#Preview}.
066         * 
067         * @param renderMode the render mode to set
068         * @return the event for easy chaining
069         */
070        public RenderPortlet setRenderMode(RenderMode renderMode) {
071                this.renderMode = renderMode;
072                return this;
073        }
074        
075        /**
076         * Set the supported render modes. The default value is 
077         * {@link RenderMode#Preview}.
078         * 
079         * @param supportedModes the supported render modes to set
080         * @return the event for easy chaining
081         */
082        public RenderPortlet setSupportedModes(Set<RenderMode> supportedModes) {
083                this.supportedModes = supportedModes;
084                return this;
085        }
086        
087        /**
088         * Add the given render mode to the supported render modes.
089         * 
090         * @param supportedMode the supported render modes to add
091         * @return the event for easy chaining
092         */
093        public RenderPortlet addSupportedMode(RenderMode supportedMode) {
094                if (supportedModes == DEFAULT_SUPPORTED) {
095                        supportedModes = new HashSet<>(DEFAULT_SUPPORTED);
096                }
097                supportedModes.add(supportedMode);
098                return this;
099        }
100        
101        /**
102         * Id set, the tab with the portlet is put in the foreground
103         * when the portlet is rendered. The default value is `false`.
104         * 
105         * @param foreground if set, the portlet is put in foreground
106         * @return the event for easy chaining
107         */
108        public RenderPortlet setForeground(boolean foreground) {
109                this.foreground = foreground;
110                return this;
111        }
112        
113        public Class<?> portletClass() {
114                return portletClass;
115        }
116        
117        /**
118         * Returns the portlet id
119         * 
120         * @return the portlet id
121         */
122        public String portletId() {
123                return portletId;
124        }
125
126        /**
127         * Returns the render mode.
128         * 
129         * @return the render mode
130         */
131        public RenderMode renderMode() {
132                return renderMode;
133        }
134
135        /**
136         * Returns the supported modes.
137         * 
138         * @return the supported modes
139         */
140        public Set<RenderMode> supportedRenderModes() {
141                return supportedModes;
142        }
143
144        /**
145         * Indicates if portelt is to be put in foreground.
146         * 
147         * @return the result
148         */
149        public boolean isForeground() {
150                return foreground;
151        }
152
153        /**
154         * Returns the content reader.
155         * 
156         * @return the content reader
157         */
158        public Reader contentReader() {
159                return contentReader;
160        }
161
162        @Override
163        public void toJson(Writer writer) {
164                StringWriter content = new StringWriter();
165                CharBuffer buffer = CharBuffer.allocate(8192);
166                try (Reader in = new BufferedReader(contentReader())) {
167                        while (true) {
168                                if (in.read(buffer) < 0) {
169                                        break;
170                                }
171                                buffer.flip();
172                                content.append(buffer);
173                                buffer.clear();
174                        }
175                } catch (IOException e) {
176                        throw new IllegalStateException(e);
177                }
178                toJson(writer, "updatePortlet", portletId(), renderMode().name(),
179                                supportedRenderModes().stream().map(RenderMode::name)
180                                        .toArray(size -> new String[size]),
181                                content.toString(), isForeground());
182        }
183}
