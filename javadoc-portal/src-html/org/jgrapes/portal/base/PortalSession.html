044/**
045 * The server side representation of a window in the browser 
046 * that displays a portal page (a portal session). An instance 
047 * is created when a new portal window opens the websocket 
048 * connection to the server for the first time. If the 
049 * connection between the browser and the server is lost, 
050 * the portal code in the browser tries to establish a 
051 * new websocket connection to the same, already
052 * existing {@link PortalSession}. The {@link PortalSession} 
053 * object is thus independent of the websocket connection 
054 * that handles the actual transfer of notifications.
055 * 
056 * ![Portal Session](PortalSession.svg)
057 * 
058 * Because there is no reliable way to be notified when a
059 * window in a browser closes, {@link PortalSession}s are
060 * discarded automatically unless {@link #refresh()} is called
061 * before the timeout occurs.
062 * 
063 * {@link PortalSession} implements the {@link IOSubchannel}
064 * interface. This allows the instances to be used as channels
065 * for exchanging portal session scoped events with the 
066 * {@link Portal} component. The upstream channel
067 * (see {@link #upstreamChannel()}) is the channel of the
068 * WebSocket. It may be unavailable if the connection has
069 * been interrupted and not (yet) re-established.
070 * 
071 * The response {@link EventPipeline} must be used to send
072 * events (responses) to the portal session in the browser.
073 * No other event pipeline may be used for this purpose, else
074 * messages will interleave.
075 * 
076 * Because a portal session (channel) remains conceptually open
077 * even if the connection to the browser is lost (until the
078 * timeout occurs) {@link Closed} events from upstream are not
079 * forwarded immediately. Rather, a {@link Closed} event is
080 * fired on the channel when the timeout occurs. Method
081 * {@link #isConnected()} can be used to check the connection 
082 * state.
083 *  
084 * As a convenience, the {@link PortalSession} provides
085 * direct access to the browser session, which can 
086 * usually only be obtained from the HTTP event or WebSocket
087 * channel by looking for an association of type {@link Session}.
088 * 
089 * @startuml PortalSession.svg
090 * class PortalSession {
091 *  -{static}Map<String,PortalSession> portalSessions
092 *  +{static}findOrCreate(String portalSessionId, Manager component): PortalSession
093 *  +setTimeout(timeout: long): PortalSession
094 *  +refresh(): void
095 *  +setUpstreamChannel(IOSubchannel upstreamChannel): PortalSession
096 *  +setSession(Session browserSession): PortalSession
097 *  +upstreamChannel(): Optional<IOSubchannel>
098 *  +portalSessionId(): String
099 *  +browserSession(): Session
100 *  +locale(): Locale
101 *  +setLocale(Locale locale): void
102 * }
103 * Interface IOSubchannel {
104 * }
105 * IOSubchannel <|.. PortalSession
106 *
107 * PortalSession "1" *-- "*" PortalSession : maintains
108 * 
109 * package org.jgrapes.http {
110 *     class Session
111 * }
112 * 
113 * PortalSession "*" -up-> "1" Session: browser session
114 * @enduml
115 */
116public final class PortalSession extends DefaultIOSubchannel {
117
118    private static Map<String, WeakReference<PortalSession>> portalSessions
119        = new ConcurrentHashMap<>();
120    private static ReferenceQueue<PortalSession> unusedSessions
121        = new ReferenceQueue<>();
122
123    private String portalSessionId;
124    private final Portal portal;
125    private final Set<Locale> supportedLocales;
126    private Locale locale;
127    private long timeout;
128    private final Timer timeoutTimer;
129    private boolean active = true;
130    private boolean connected = true;
131    private Session browserSession;
132    private IOSubchannel upstreamChannel;
133
134    private static void cleanUnused() {
135        while (true) {
136            Reference<? extends PortalSession> unused = unusedSessions.poll();
137            if (unused == null) {
138                break;
139            }
140            portalSessions.remove(unused.get().portalSessionId());
141        }
142    }
143
144    /**
145     * Lookup the portal session (the channel)
146     * for the given portal session id.
147     * 
148     * @param portalSessionId the portal session id
149     * @return the channel
150     */
151    /* default */ static Optional<PortalSession>
152            lookup(String portalSessionId) {
153        cleanUnused();
154        return Optional.ofNullable(portalSessions.get(portalSessionId))
155            .flatMap(ref -> Optional.ofNullable(ref.get()));
156    }
157
158    /**
159     * Return all sessions that belong to the given portal as a new
160     * unmodifiable set.
161     *
162     * @param portal the portal
163     * @return the sets the
164     */
165    public static Set<PortalSession> byPortal(Portal portal) {
166        cleanUnused();
167        Set<PortalSession> result = new HashSet<>();
168        for (WeakReference<PortalSession> psr : portalSessions.values()) {
169            PortalSession psess = psr.get();
170            if (psess.portal.equals(portal)) {
171                result.add(psess);
172            }
173        }
174        return Collections.unmodifiableSet(result);
175    }
176
177    /**
178     * Lookup (and create if not found) the portal browserSession channel
179     * for the given portal browserSession id.
180     *
181     * @param portalSessionId the browserSession id
182     * @param portal the portal that this session belongs to
183     * class' constructor if a new channel is created, usually 
184     * the portal
185     * @param supportedLocales the locales supported by the portal
186     * @param timeout the portal session timeout in milli seconds
187     * @return the channel
188     */
189    /* default */ static PortalSession lookupOrCreate(
190            String portalSessionId, Portal portal,
191            Set<Locale> supportedLocales, long timeout) {
192        cleanUnused();
193        return portalSessions.computeIfAbsent(portalSessionId,
194            psi -> new WeakReference<>(new PortalSession(
195                portal, supportedLocales, portalSessionId, timeout),
196                unusedSessions))
197            .get();
198    }
199
200    /**
201     * Replace the id of the portal session with the new id.
202     *
203     * @param newPortalSessionId the new portal session id
204     * @return the portal session
205     */
206    /* default */ PortalSession replaceId(String newPortalSessionId) {
207        portalSessions.remove(portalSessionId);
208        portalSessionId = newPortalSessionId;
209        portalSessions.put(portalSessionId, new WeakReference<>(
210            this, unusedSessions));
211        connected = true;
212        return this;
213    }
214
215    private PortalSession(Portal portal, Set<Locale> supportedLocales,
216            String portalSessionId, long timeout) {
217        super(portal.channel(), portal.newEventPipeline());
218        this.portal = portal;
219        this.supportedLocales = supportedLocales;
220        this.portalSessionId = portalSessionId;
221        this.timeout = timeout;
222        timeoutTimer = Components.schedule(
223            tmr -> discard(), Duration.ofMillis(timeout));
224    }
225
226    /**
227     * Changes the timeout for this {@link PortalSession} to the
228     * given value.
229     * 
230     * @param timeout the timeout in milli seconds
231     * @return the portal session for easy chaining
232     */
233    public PortalSession setTimeout(long timeout) {
234        this.timeout = timeout;
235        timeoutTimer.reschedule(Duration.ofMillis(timeout));
236        return this;
237    }
238
239    /**
240     * Returns the time when this session will expire.
241     *
242     * @return the instant
243     */
244    public Instant expiresAt() {
245        return timeoutTimer.scheduledFor();
246    }
247
248    /**
249     * Resets the {@link PortalSession}'s timeout.
250     */
251    public void refresh() {
252        timeoutTimer.reschedule(Duration.ofMillis(timeout));
253    }
254
255    /**
256     * Discards this session.
257     */
258    public void discard() {
259        portalSessions.remove(portalSessionId);
260        connected = false;
261        active = false;
262        portal.newEventPipeline().fire(new Closed(), this);
263    }
264
265    /**
266     * Checks if the portal session has become stale (inactive).
267     *
268     * @return true, if is stale
269     */
270    public boolean isStale() {
271        return !active;
272    }
273
274    /* default */ void disconnected() {
275        connected = false;
276    }
277
278    /**
279     * Checks if a network connection with the browser exists.
280     *
281     * @return true, if is connected
282     */
283    public boolean isConnected() {
284        return connected;
285    }
286
287    /**
288     * Sets or updates the upstream channel. This method should only
289     * be invoked by the creator of the {@link PortalSession}, by default
290     * the {@link PortalWeblet}.
291     * 
292     * @param upstreamChannel the upstream channel (WebSocket connection)
293     * @return the portal session for easy chaining
294     */
295    public PortalSession setUpstreamChannel(IOSubchannel upstreamChannel) {
296        if (upstreamChannel == null) {
297            throw new IllegalArgumentException();
298        }
299        this.upstreamChannel = upstreamChannel;
300        return this;
301    }
302
303    /**
304     * Sets or updates associated browser session. This method should only
305     * be invoked by the creator of the {@link PortalSession}, by default
306     * the {@link PortalWeblet}.
307     * 
308     * @param browserSession the browser session
309     * @return the portal session for easy chaining
310     */
311    public PortalSession setSession(Session browserSession) {
312        this.browserSession = browserSession;
313        if (locale == null) {
314            locale = browserSession.locale();
315        }
316        return this;
317    }
318
319    /**
320     * @return the upstream channel
321     */
322    public IOSubchannel upstreamChannel() {
323        return upstreamChannel;
324    }
325
326    /**
327     * The portal session id is used in the communication between the
328     * browser and the server. It is not guaranteed to remain the same
329     * over time, even if the portal session is maintained. To prevent
330     * wrong usage, its visibility is therefore set to package. 
331     * 
332     * @return the portalSessionId
333     */
334    /* default */ String portalSessionId() {
335        return portalSessionId;
336    }
337
338    /**
339     * @return the browserSession
340     */
341    public Session browserSession() {
342        return browserSession;
343    }
344
345    /**
346     * Returns the supported locales.
347     *
348     * @return the set of locales supported by the portal
349     */
350    public Set<Locale> supportedLocales() {
351        return supportedLocales;
352    }
353
354    /**
355     * Return the portal session's locale. The locale is initialized
356     * from the browser session's locale.
357     * 
358     * @return the locale
359     */
360    public Locale locale() {
361        return locale == null ? Locale.getDefault() : locale;
362    }
363
364    /**
365     * Sets the locale for this portal session.
366     *
367     * @param locale the locale
368     * @return the portal session
369     */
370    public PortalSession setLocale(Locale locale) {
371        this.locale = locale;
372        return this;
373    }
374
375    /*
376     * (non-Javadoc)
377     * 
378     * @see java.lang.Object#toString()
379     */
380    @Override
381    public String toString() {
382        StringBuilder builder = new StringBuilder();
383        builder.append(Subchannel.toString(this));
384        Optional.ofNullable(upstreamChannel).ifPresent(upstr -> builder.append(
385            LinkedIOSubchannel.upstreamToString(upstr)));
386        return builder.toString();
387    }
