044/**
045 * The server side representation of a window in the browser 
046 * that displays a portal page (a portal session). An instance 
047 * is created when a new portal window opens the websocket 
048 * connection to the server for the first time. If the 
049 * connection between the browser and the server is lost, 
050 * the portal code in the browser tries to establish a 
051 * new websocket connection to the same, already
052 * existing {@link PortalSession}. The {@link PortalSession} 
053 * object is thus independent of the websocket connection 
054 * that handles the actual transfer of notifications.
055 * 
056 * ![Portal Session](PortalSession.svg)
057 * 
058 * Because there is no reliable way to be notified when a
059 * window in a browser closes, {@link PortalSession}s are
060 * discarded automatically unless {@link #refresh()} is called
061 * before the timeout occurs.
062 * 
063 * {@link PortalSession} implements the {@link IOSubchannel}
064 * interface. This allows the instances to be used as channels
065 * for exchanging portal session scoped events with the 
066 * {@link Portal} component. The upstream channel
067 * (see {@link #upstreamChannel()}) is the channel of the
068 * WebSocket. It may be unavailable if the connection has
069 * been interrupted and not (yet) re-established.
070 * 
071 * The response {@link EventPipeline} must be used to send
072 * events (responses) to the portal session in the browser.
073 * No other event pipeline may be used for this purpose, else
074 * messages will interleave.
075 * 
076 * Because a portal session (channel) remains conceptually open
077 * even if the connection to the browser is lost (until the
078 * timeout occurs) {@link Closed} events from upstream are not
079 * forwarded immediately. Rather, a {@link Closed} event is
080 * fired on the channel when the timeout occurs. Method
081 * {@link #isConnected()} can be used to check the connection 
082 * state.
083 *   
084 * As a convenience, the {@link PortalSession} provides
085 * direct access to the browser session, which can 
086 * usually only be obtained from the HTTP event or WebSocket
087 * channel by looking for an association of type {@link Session}.
088 * It also provides access to the {@link Locale} as maintained
089 * by the browser session.
090 * 
091 * @startuml PortalSession.svg
092 * class PortalSession {
093 *      -{static}Map<String,PortalSession> portalSessions
094 *      +{static}findOrCreate(String portalSessionId, Manager component): PortalSession
095 *  +setTimeout(timeout: long): PortalSession
096 *  +refresh(): void
097 *      +setUpstreamChannel(IOSubchannel upstreamChannel): PortalSession
098 *      +setSession(Session browserSession): PortalSession
099 *      +upstreamChannel(): Optional<IOSubchannel>
100 *      +portalSessionId(): String
101 *      +browserSession(): Session
102 *      +locale(): Locale
103 *      +setLocale(Locale locale): void
104 * }
105 * Interface IOSubchannel {
106 * }
107 * IOSubchannel <|.. PortalSession
108 *
109 * PortalSession "1" *-- "*" PortalSession : maintains
110 * 
111 * package org.jgrapes.http {
112 *     class Session
113 * }
114 * 
115 * PortalSession "*" -up-> "1" Session: browser session
116 * @enduml
117 */
118public final class PortalSession extends DefaultIOSubchannel {
119
120    private static Map<String, WeakReference<PortalSession>> portalSessions
121        = new ConcurrentHashMap<>();
122    private static ReferenceQueue<PortalSession> unusedSessions
123        = new ReferenceQueue<>();
124
125    private String portalSessionId;
126    private final Portal portal;
127    private final Set<Locale> supportedLocales;
128    private long timeout;
129    private final Timer timeoutTimer;
130    private boolean active = true;
131    private boolean connected = true;
132    private Session browserSession;
133    private IOSubchannel upstreamChannel;
134
135    private static void cleanUnused() {
136        while (true) {
137            Reference<? extends PortalSession> unused = unusedSessions.poll();
138            if (unused == null) {
139                break;
140            }
141            portalSessions.remove(unused.get().portalSessionId());
142        }
143    }
144
145    /**
146     * Lookup the portal session (the channel)
147     * for the given portal session id.
148     * 
149     * @param portalSessionId the portal session id
150     * @return the channel
151     */
152    /* default */ static Optional<PortalSession>
153            lookup(String portalSessionId) {
154        cleanUnused();
155        return Optional.ofNullable(portalSessions.get(portalSessionId))
156            .flatMap(ref -> Optional.ofNullable(ref.get()));
157    }
158
159    /**
160     * Return all sessions that belong to the given portal as a new
161     * unmodifiable set.
162     *
163     * @param portal the portal
164     * @return the sets the
165     */
166    public static Set<PortalSession> byPortal(Portal portal) {
167        cleanUnused();
168        Set<PortalSession> result = new HashSet<>();
169        for (WeakReference<PortalSession> psr : portalSessions.values()) {
170            PortalSession psess = psr.get();
171            if (psess.portal.equals(portal)) {
172                result.add(psess);
173            }
174        }
175        return Collections.unmodifiableSet(result);
176    }
177
178    /**
179     * Lookup (and create if not found) the portal browserSession channel
180     * for the given portal browserSession id.
181     *
182     * @param portalSessionId the browserSession id
183     * @param portal the portal that this session belongs to
184     * class' constructor if a new channel is created, usually 
185     * the portal
186     * @param supportedLocales the locales supported by the portal
187     * @param timeout the portal session timeout in milli seconds
188     * @return the channel
189     */
190    /* default */ static PortalSession lookupOrCreate(
191            String portalSessionId, Portal portal,
192            Set<Locale> supportedLocales, long timeout) {
193        cleanUnused();
194        return portalSessions.computeIfAbsent(portalSessionId,
195            psi -> new WeakReference<>(new PortalSession(
196                portal, supportedLocales, portalSessionId, timeout),
197                unusedSessions))
198            .get();
199    }
200
201    /**
202     * Replace the id of the portal session with the new id.
203     *
204     * @param newPortalSessionId the new portal session id
205     * @return the portal session
206     */
207    /* default */ PortalSession replaceId(String newPortalSessionId) {
208        portalSessions.remove(portalSessionId);
209        portalSessionId = newPortalSessionId;
210        portalSessions.put(portalSessionId, new WeakReference<>(
211            this, unusedSessions));
212        connected = true;
213        return this;
214    }
215
216    private PortalSession(Portal portal, Set<Locale> supportedLocales,
217            String portalSessionId, long timeout) {
218        super(portal.channel(), portal.newEventPipeline());
219        this.portal = portal;
220        this.supportedLocales = supportedLocales;
221        this.portalSessionId = portalSessionId;
222        this.timeout = timeout;
223        timeoutTimer = Components.schedule(
224            tmr -> discard(), Duration.ofMillis(timeout));
225    }
226
227    /**
228     * Changes the timeout for this {@link PortalSession} to the
229     * given value.
230     * 
231     * @param timeout the timeout in milli seconds
232     * @return the portal session for easy chaining
233     */
234    public PortalSession setTimeout(long timeout) {
235        this.timeout = timeout;
236        timeoutTimer.reschedule(Duration.ofMillis(timeout));
237        return this;
238    }
239
240    /**
241     * Returns the time when this session will expire.
242     *
243     * @return the instant
244     */
245    public Instant expiresAt() {
246        return timeoutTimer.scheduledFor();
247    }
248
249    /**
250     * Resets the {@link PortalSession}'s timeout.
251     */
252    public void refresh() {
253        timeoutTimer.reschedule(Duration.ofMillis(timeout));
254    }
255
256    /**
257     * Discards this session.
258     */
259    public void discard() {
260        portalSessions.remove(portalSessionId);
261        connected = false;
262        active = false;
263        portal.newEventPipeline().fire(new Closed(), this);
264    }
265
266    /**
267     * Checks if the portal session has become stale (inactive).
268     *
269     * @return true, if is stale
270     */
271    public boolean isStale() {
272        return !active;
273    }
274
275    /* default */ void disconnected() {
276        connected = false;
277    }
278
279    /**
280     * Checks if a network connection with the browser exists.
281     *
282     * @return true, if is connected
283     */
284    public boolean isConnected() {
285        return connected;
286    }
287
288    /**
289     * Sets or updates the upstream channel. This method should only
290     * be invoked by the creator of the {@link PortalSession}, by default
291     * the {@link PortalWeblet}.
292     * 
293     * @param upstreamChannel the upstream channel (WebSocket connection)
294     * @return the portal session for easy chaining
295     */
296    public PortalSession setUpstreamChannel(IOSubchannel upstreamChannel) {
297        if (upstreamChannel == null) {
298            throw new IllegalArgumentException();
299        }
300        this.upstreamChannel = upstreamChannel;
301        return this;
302    }
303
304    /**
305     * Sets or updates associated browser session. This method should only
306     * be invoked by the creator of the {@link PortalSession}, by default
307     * the {@link PortalWeblet}.
308     * 
309     * @param browserSession the browser session
310     * @return the portal session for easy chaining
311     */
312    public PortalSession setSession(Session browserSession) {
313        this.browserSession = browserSession;
314        return this;
315    }
316
317    /**
318     * @return the upstream channel
319     */
320    public IOSubchannel upstreamChannel() {
321        return upstreamChannel;
322    }
323
324    /**
325     * The portal session id is used in the communication between the
326     * browser and the server. It is not guaranteed to remain the same
327     * over time, even if the portal session is maintained. To prevent
328     * wrong usage, its visibility is therefore set to package. 
329     * 
330     * @return the portalSessionId
331     */
332    /* default */ String portalSessionId() {
333        return portalSessionId;
334    }
335
336    /**
337     * @return the browserSession
338     */
339    public Session browserSession() {
340        return browserSession;
341    }
342
343    /**
344     * Returns the supported locales.
345     *
346     * @return the set of locales supported by the portal
347     */
348    protected Set<Locale> supportedLocales() {
349        return supportedLocales;
350    }
351
352    /**
353     * Return the portal session's locale. The locale is obtained
354     * from the browser session.
355     * 
356     * @return the locale
357     */
358    public Locale locale() {
359        return browserSession == null ? Locale.getDefault()
360            : browserSession.locale();
361    }
362
363    /*
364     * (non-Javadoc)
365     * 
366     * @see java.lang.Object#toString()
367     */
368    @Override
369    public String toString() {
370        StringBuilder builder = new StringBuilder();
371        builder.append(Subchannel.toString(this));
372        Optional.ofNullable(upstreamChannel).ifPresent(upstr -> builder.append(
373            LinkedIOSubchannel.upstreamToString(upstr)));
374        return builder.toString();
375    }
