062 */
063public abstract class FreeMarkerPortalWeblet extends PortalWeblet {
064
065    public static final String UTF_8 = "utf-8";
066    /**
067     * Initialized with a base FreeMarker configuration.
068     */
069    protected Configuration freeMarkerConfig;
070
071    /**
072     * Instantiates a new free marker portal weblet.
073     *
074     * @param webletChannel the weblet channel
075     * @param portalChannel the portal channel
076     * @param portalPrefix the portal prefix
077     */
078    public FreeMarkerPortalWeblet(Channel webletChannel, Channel portalChannel,
079            URI portalPrefix) {
080        super(webletChannel, portalChannel, portalPrefix);
081        freeMarkerConfig = new Configuration(Configuration.VERSION_2_3_26);
082        List<TemplateLoader> loaders = new ArrayList<>();
083        Class<?> clazz = getClass();
084        while (!clazz.equals(FreeMarkerPortalWeblet.class)) {
085            loaders.add(new ClassTemplateLoader(clazz.getClassLoader(),
086                clazz.getPackage().getName().replace('.', '/')));
087            clazz = clazz.getSuperclass();
088        }
089        freeMarkerConfig.setTemplateLoader(
090            new MultiTemplateLoader(loaders.toArray(new TemplateLoader[0])));
091        freeMarkerConfig.setDefaultEncoding(UTF_8);
092        freeMarkerConfig.setTemplateExceptionHandler(
093            TemplateExceptionHandler.RETHROW_HANDLER);
094        freeMarkerConfig.setLogTemplateExceptions(false);
095    }
096
097    public FreeMarkerPortalWeblet
098            prependClassTemplateLoader(ClassLoader classloader, String path) {
099        List<TemplateLoader> loaders = new ArrayList<>();
100        loaders.add(new ClassTemplateLoader(classloader, path));
101        MultiTemplateLoader oldLoader
102            = (MultiTemplateLoader) freeMarkerConfig.getTemplateLoader();
103        for (int i = 0; i < oldLoader.getTemplateLoaderCount(); i++) {
104            loaders.add(oldLoader.getTemplateLoader(i));
105        }
106        freeMarkerConfig.setTemplateLoader(
107            new MultiTemplateLoader(loaders.toArray(new TemplateLoader[0])));
108        return this;
109    }
110
111    public FreeMarkerPortalWeblet prependClassTemplateLoader(Class<?> clazz) {
112        return prependClassTemplateLoader(clazz.getClassLoader(),
113            clazz.getPackage().getName().replace('.', '/'));
114    }
115
116    protected Map<String, Object> createPortalBaseModel() {
117        // Create portal model
118        Map<String, Object> portalModel = new HashMap<>();
119        portalModel.put("renderSupport", renderSupport());
120        portalModel.put("useMinifiedResources", useMinifiedResources());
121        portalModel.put("minifiedExtension",
122            useMinifiedResources() ? ".min" : "");
123        portalModel.put(
124            "portalSessionRefreshInterval", portalSessionRefreshInterval());
125        portalModel.put(
126            "portalSessionInactivityTimeout", portalSessionInactivityTimeout());
127        return portalModel;
128    }
129
130    protected Map<String, Object> expandPortalModel(
131            Map<String, Object> model, GetRequest event,
132            UUID portalSessionId) {
133        // Portal Session UUID
134        model.put("portalSessionId", portalSessionId.toString());
135
136        // Add locale
137        final Locale locale = event.associated(Selection.class).map(
138            sel -> sel.get()[0]).orElse(Locale.getDefault());
139        model.put("locale", locale);
140
141        // Add supported locales
142        final Collator coll = Collator.getInstance(locale);
143        final Comparator<LanguageInfo> comp
144            = new Comparator<LanguageInfo>() {
145                @Override
146                public int compare(LanguageInfo o1, LanguageInfo o2) {
147                    return coll.compare(o1.getLabel(), o2.getLabel());
148                }
149            };
150        LanguageInfo[] languages = supportedLocales().stream()
151            .map(lang -> new LanguageInfo(lang))
152            .sorted(comp).toArray(size -> new LanguageInfo[size]);
153        model.put("supportedLanguages", languages);
154
155        final ResourceBundle baseResources = portalResources(locale);
156        model.put("_", new TemplateMethodModelEx() {
157            @Override
158            public Object exec(@SuppressWarnings("rawtypes") List arguments)
159                    throws TemplateModelException {
160                @SuppressWarnings("unchecked")
161                List<TemplateModel> args = (List<TemplateModel>) arguments;
162                if (!(args.get(0) instanceof SimpleScalar)) {
163                    throw new TemplateModelException("Not a string.");
164                }
165                String key = ((SimpleScalar) args.get(0)).getAsString();
166                try {
167                    return baseResources.getString(key);
168                } catch (MissingResourceException e) { // NOPMD
169                    // no luck
170                }
171                return key;
172            }
173        });
174        return model;
175    }
176
177    @Override
178    protected void renderPortal(GetRequest event, IOSubchannel channel,
179            UUID portalSessionId) throws IOException, InterruptedException {
180        event.setResult(true);
181        event.stop();
182
183        // Prepare response
184        HttpResponse response = event.httpRequest().response().get();
185        MediaType mediaType = MediaType.builder().setType("text", "html")
186            .setParameter("charset", UTF_8).build();
187        response.setField(HttpField.CONTENT_TYPE, mediaType);
188        response.setStatus(HttpStatus.OK);
189        response.setHasPayload(true);
190        channel.respond(new Response(response));
191        try (ByteBufferOutputStream bbos = new ByteBufferOutputStream(
192            channel, channel.responsePipeline());
193                Writer out = new OutputStreamWriter(bbos.suppressClose(),
194                    UTF_8)) {
195            @SuppressWarnings("PMD.UseConcurrentHashMap")
196
197            Template tpl = freeMarkerConfig.getTemplate("portal.ftl.html");
198            Map<String, Object> portalModel = expandPortalModel(
199                createPortalBaseModel(), event, portalSessionId);
200            tpl.process(portalModel, out);
201        } catch (TemplateException e) {
202            throw new IOException(e);
203        }
204    }
205
206    /**
207    * Holds the information about the selected language.
208    */
209    public static class LanguageInfo {
210        private final Locale locale;
211
212        /**
213         * Instantiates a new language info.
214         *
215         * @param locale the locale
216         */
217        public LanguageInfo(Locale locale) {
218            this.locale = locale;
219        }
220
221        /**
222         * Gets the locale.
223         *
224         * @return the locale
225         */
226        public Locale getLocale() {
227            return locale;
228        }
229
230        /**
231         * Gets the label.
232         *
233         * @return the label
234         */
235        public String getLabel() {
236            String str = locale.getDisplayName(locale);
237            return Character.toUpperCase(str.charAt(0)) + str.substring(1);
238        }
239    }
