019package org.jgrapes.portal;
020
021import java.net.URL;
022
023/**
024 * Represents a theme provider.
025 */
026public abstract class ThemeProvider {
027
028    /**
029     * Return the id of the theme.
030     * 
031     * @return the result
032     */
033    public abstract String themeId();
034
035    /**
036     * Return the name of the theme. The default implementation 
037     * uses the theme id, replaces underscores with spaces and
038     * capitalizes the first character.
039     * 
040     * @return the result
041     */
042    @SuppressWarnings("PMD.UselessParentheses")
043    public String themeName() {
044        return (Character.toUpperCase(themeId().charAt(0))
045            + themeId().substring(1)).replace('_', ' ');
046    }
047
048    /**
049     * Find the requested resource.
050     * 
051     * @param name the resource name
052     * @return the data as input stream
053     */
054    public URL getResource(String name)
055            throws ResourceNotFoundException {
056        URL resource = getClass().getResource(name);
057        if (resource == null) {
058            throw new ResourceNotFoundException();
059        }
060        return resource;
061    }
062}
