019package org.jgrapes.portal;
020
021import java.io.IOException;
022
023import org.jgrapes.portal.events.ResourceRequest;
024
025/**
026 * Indicates that a resource provider found the resource to be
027 * unmodified.
028 */
029public class ResourceNotModified extends ResourceResult {
030
031    /**
032     * Creates a new instance.
033     *
034     * @param request the request
035     */
036    public ResourceNotModified(ResourceRequest request) {
037        super(request);
038    }
039
040    /** 
041     * Does nothing, because the resource does not have to be sent.
042     */
043    @Override
044    public void process() throws IOException, InterruptedException {
045        // Do nothing.
046    }
047
048}
