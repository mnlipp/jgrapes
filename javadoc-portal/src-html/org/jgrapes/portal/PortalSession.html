043/**
044 * The server side representation of a window in the browser 
045 * that displays a portal page (a portal session). An instance 
046 * is created when a new portal window opens the websocket 
047 * connection to the server for the first time. If the 
048 * connection between the browser and the server is lost, 
049 * the portal code in the browser tries to establish a 
050 * new websocket connection to the same, already
051 * existing {@link PortalSession}. The {@link PortalSession} 
052 * object is thus independent of the websocket connection 
053 * that handles the actual transfer of notifications.
054 * 
055 * ![Portal Session](PortalSession.svg)
056 * 
057 * Because there is no reliable way to be notified when a
058 * window in a browser closes, {@link PortalSession}s are
059 * discarded automatically unless {@link #refresh()} is called
060 * before the timeout occurs.
061 * 
062 * {@link PortalSession} implements the {@link IOSubchannel}
063 * interface. This allows the instances to be used as channels
064 * for exchanging portal session scoped events with the 
065 * {@link Portal} component. The upstream channel
066 * (see {@link #upstreamChannel()}) is the channel of the
067 * WebSocket. It may be unavailable if the connection has
068 * been interrupted and not (yet) re-established.
069 * 
070 * The response {@link EventPipeline} must be used to send
071 * events (responses) to the portal session in the browser.
072 * No other event pipeline may be used for this purpose, else
073 * messages will interleave.
074 * 
075 * Because a portal session (channel) remains conceptually open
076 * even if the connection to the browser is lost (until the
077 * timeout occurs) {@link Closed} events from upstream are not
078 * forwarded immediately. Rather, a {@link Closed} event is
079 * fired on the channel when the timeout occurs. Method
080 * {@link #isConnected()} can be used to check the connection 
081 * state.
082 *   
083 * As a convenience, the {@link PortalSession} provides
084 * direct access to the browser session, which can 
085 * usually only be obtained from the HTTP event or WebSocket
086 * channel by looking for an association of type {@link Session}.
087 * It also provides access to the {@link Locale} as maintained
088 * by the browser session.
089 * 
090 * @startuml PortalSession.svg
091 * class PortalSession {
092 *      -{static}Map<String,PortalSession> portalSessions
093 *      +{static}findOrCreate(String portalSessionId, Manager component): PortalSession
094 *  +setTimeout(timeout: long): PortalSession
095 *  +refresh(): void
096 *      +setUpstreamChannel(IOSubchannel upstreamChannel): PortalSession
097 *      +setSession(Session browserSession): PortalSession
098 *      +upstreamChannel(): Optional<IOSubchannel>
099 *      +portalSessionId(): String
100 *      +browserSession(): Session
101 *      +locale(): Locale
102 *      +setLocale(Locale locale): void
103 * }
104 * Interface IOSubchannel {
105 * }
106 * IOSubchannel <|.. PortalSession
107 *
108 * PortalSession "1" *-- "*" PortalSession : maintains
109 * 
110 * package org.jgrapes.http {
111 *     class Session
112 * }
113 * 
114 * PortalSession "*" -up-> "1" Session: browser session
115 * @enduml
116 */
117public final class PortalSession extends DefaultSubchannel {
118
119    private static Map<String, WeakReference<PortalSession>> portalSessions
120        = new ConcurrentHashMap<>();
121    private static ReferenceQueue<PortalSession> unusedSessions
122        = new ReferenceQueue<>();
123
124    private String portalSessionId;
125    private final Portal portal;
126    private long timeout;
127    private final Timer timeoutTimer;
128    private boolean active = true;
129    private boolean connected = true;
130    private Session browserSession;
131    private IOSubchannel upstreamChannel;
132
133    private static void cleanUnused() {
134        while (true) {
135            Reference<? extends PortalSession> unused = unusedSessions.poll();
136            if (unused == null) {
137                break;
138            }
139            portalSessions.remove(unused.get().portalSessionId());
140        }
141    }
142
143    /**
144     * Lookup the portal session (the channel)
145     * for the given portal session id.
146     * 
147     * @param portalSessionId the portal session id
148     * @return the channel
149     */
150    /* default */ static Optional<PortalSession>
151            lookup(String portalSessionId) {
152        cleanUnused();
153        return Optional.ofNullable(portalSessions.get(portalSessionId))
154            .flatMap(ref -> Optional.ofNullable(ref.get()));
155    }
156
157    /**
158     * Return all sessions that belong to the given portal as a new
159     * unmodifiable set.
160     *
161     * @param portal the portal
162     * @return the sets the
163     */
164    public static Set<PortalSession> byPortal(Portal portal) {
165        cleanUnused();
166        Set<PortalSession> result = new HashSet<>();
167        for (WeakReference<PortalSession> psr : portalSessions.values()) {
168            PortalSession psess = psr.get();
169            if (psess.portal.equals(portal)) {
170                result.add(psess);
171            }
172        }
173        return Collections.unmodifiableSet(result);
174    }
175
176    /**
177     * Lookup (and create if not found) the portal browserSession channel
178     * for the given portal browserSession id.
179     * 
180     * @param portalSessionId the browserSession id
181     * @param portal the portal that this session belongs to 
182     * class' constructor if a new channel is created, usually 
183     * the portal
184     * @param timeout the portal session timeout in milli seconds
185     * @return the channel
186     */
187    /* default */ static PortalSession lookupOrCreate(
188            String portalSessionId, Portal portal, long timeout) {
189        cleanUnused();
190        return portalSessions.computeIfAbsent(portalSessionId,
191            psi -> new WeakReference<>(new PortalSession(
192                portal, portalSessionId, timeout), unusedSessions))
193            .get();
194    }
195
196    /**
197     * Replace the id of the portal session with the new id.
198     *
199     * @param newPortalSessionId the new portal session id
200     * @return the portal session
201     */
202    /* default */ PortalSession replaceId(String newPortalSessionId) {
203        portalSessions.remove(portalSessionId);
204        portalSessionId = newPortalSessionId;
205        portalSessions.put(portalSessionId, new WeakReference<>(
206            this, unusedSessions));
207        connected = true;
208        return this;
209    }
210
211    private PortalSession(Portal portal, String portalSessionId,
212            long timeout) {
213        super(portal.channel(), portal.newEventPipeline());
214        this.portal = portal;
215        this.portalSessionId = portalSessionId;
216        this.timeout = timeout;
217        timeoutTimer = Components.schedule(
218            tmr -> discard(), Duration.ofMillis(timeout));
219    }
220
221    /**
222     * Changes the timeout for this {@link PortalSession} to the
223     * given value.
224     * 
225     * @param timeout the timeout in milli seconds
226     * @return the portal session for easy chaining
227     */
228    public PortalSession setTimeout(long timeout) {
229        this.timeout = timeout;
230        timeoutTimer.reschedule(Duration.ofMillis(timeout));
231        return this;
232    }
233
234    /**
235     * Returns the time when this session will expire.
236     *
237     * @return the instant
238     */
239    public Instant expiresAt() {
240        return timeoutTimer.scheduledFor();
241    }
242
243    /**
244     * Resets the {@link PortalSession}'s timeout.
245     */
246    public void refresh() {
247        timeoutTimer.reschedule(Duration.ofMillis(timeout));
248    }
249
250    /**
251     * Discards this session.
252     */
253    public void discard() {
254        portalSessions.remove(portalSessionId);
255        connected = false;
256        active = false;
257        portal.newEventPipeline().fire(new Closed(), this);
258    }
259
260    /**
261     * Checks if the portal session has become stale (inactive).
262     *
263     * @return true, if is stale
264     */
265    public boolean isStale() {
266        return !active;
267    }
268
269    /* default */ void disconnected() {
270        connected = false;
271    }
272
273    /**
274     * Checks if a network connection with the browser exists.
275     *
276     * @return true, if is connected
277     */
278    public boolean isConnected() {
279        return connected;
280    }
281
282    /**
283     * Sets or updates the upstream channel. This method should only
284     * be invoked by the creator of the {@link PortalSession}, by default
285     * the {@link PortalWeblet}.
286     * 
287     * @param upstreamChannel the upstream channel (WebSocket connection)
288     * @return the portal session for easy chaining
289     */
290    public PortalSession setUpstreamChannel(IOSubchannel upstreamChannel) {
291        if (upstreamChannel == null) {
292            throw new IllegalArgumentException();
293        }
294        this.upstreamChannel = upstreamChannel;
295        return this;
296    }
297
298    /**
299     * Sets or updates associated browser session. This method should only
300     * be invoked by the creator of the {@link PortalSession}, by default
301     * the {@link PortalWeblet}.
302     * 
303     * @param browserSession the browser session
304     * @return the portal session for easy chaining
305     */
306    public PortalSession setSession(Session browserSession) {
307        this.browserSession = browserSession;
308        return this;
309    }
310
311    /**
312     * @return the upstream channel
313     */
314    public IOSubchannel upstreamChannel() {
315        return upstreamChannel;
316    }
317
318    /**
319     * The portal session id is used in the communication between the
320     * browser and the server. It is not guaranteed to remain the same
321     * over time, even if the portal session is maintained. To prevent
322     * wrong usage, its visibility is therefore set to package. 
323     * 
324     * @return the portalSessionId
325     */
326    /* default */ String portalSessionId() {
327        return portalSessionId;
328    }
329
330    /**
331     * @return the browserSession
332     */
333    public Session browserSession() {
334        return browserSession;
335    }
336
337    /**
338     * Return the portal session's locale. The locale is obtained
339     * from the browser session.
340     * 
341     * @return the locale
342     */
343    public Locale locale() {
344        return browserSession == null ? Locale.getDefault()
345            : browserSession.locale();
346    }
347
348    /*
349     * (non-Javadoc)
350     * 
351     * @see java.lang.Object#toString()
352     */
353    @Override
354    public String toString() {
355        StringBuilder builder = new StringBuilder();
356        builder.append(IOSubchannel.toString(this));
357        Optional.ofNullable(upstreamChannel).ifPresent(upstr -> builder.append(
358            LinkedIOSubchannel.upstreamToString(upstr)));
359        return builder.toString();
360    }
