019package org.jgrapes.portal;
020
021import java.lang.ref.WeakReference;
022import java.time.Duration;
023import java.util.Locale;
024import java.util.Map;
025import java.util.Optional;
026import java.util.concurrent.ConcurrentHashMap;
027
028import org.jgrapes.core.Components;
029import org.jgrapes.core.Components.Timer;
030import org.jgrapes.core.EventPipeline;
031import org.jgrapes.core.Manager;
032import org.jgrapes.http.LanguageSelector.Selection;
033import org.jgrapes.http.Session;
034import org.jgrapes.io.IOSubchannel;
035import org.jgrapes.io.IOSubchannel.DefaultSubchannel;
036import org.jgrapes.io.events.Close;
037import org.jgrapes.io.events.Closed;
038
039/**
040 * The server side representation of a window in the browser 
041 * that displays a portal page (a portal session). An instance 
042 * is created when a new portal window opens the websocket 
043 * connection to the server for the first time. If the 
044 * connection between the browser and the server is lost, 
045 * the portal code in the browser tries to establish a 
046 * new websocket connection to the same, already
047 * existing {@link PortalSession}. The {@link PortalSession} 
048 * object is thus independent of the websocket connection 
049 * that handles the actual transfer of notifications.
050 * 
051 * ![Portal Session](PortalSession.svg)
052 * 
053 * Because there is no reliable way to be notified when a
054 * window in a browser closes, {@link PortalSession}s are
055 * discarded automatically unless {@link #refresh()} is called
056 * before the timeout occurs.
057 * 
058 * {@link PortalSession} implements the {@link IOSubchannel}
059 * interface. This allows the instances to be used as channels
060 * for exchanging portal session scoped events with the 
061 * {@link Portal} component. The upstream channel
062 * (see {@link #upstreamChannel()}) is the channel of the
063 * WebSocket. It may be unavailable if the connection has
064 * been interrupted and not (yet) re-established.
065 * 
066 * As a convenience, the {@link PortalSession} provides
067 * direct access to the browser session, which can 
068 * usually only be obtained from the HTTP event or WebSocket
069 * channel by looking for an association of type {@link Session}.
070 * It also provides access to the {@link Locale} as maintained
071 * by the upstream web socket.
072 * 
073 * @startuml PortalSession.svg
074 * class PortalSession {
075 *      -{static}Map<String,PortalSession> portalSessions
076 *      +{static}findOrCreate(String portalSessionId, Manager component): PortalSession
077 *  +setTimeout(timeout: long): PortalSession
078 *  +refresh(): void
079 *      +setUpstreamChannel(IOSubchannel upstreamChannel): PortalSession
080 *      +setSession(Session browserSession): PortalSession
081 *      +upstreamChannel(): Optional<IOSubchannel>
082 *      +portalSessionId(): String
083 *      +browserSession(): Session
084 *      +locale(): Locale
085 *      +setLocale(Locale locale): void
086 * }
087 * Interface IOSubchannel {
088 * }
089 * IOSubchannel <|.. PortalSession
090 *
091 * PortalSession "1" *-- "*" PortalSession : maintains
092 * 
093 * package org.jgrapes.http {
094 *     class Session
095 * }
096 * 
097 * PortalSession "*" -up-> "1" Session: browser session
098 * @enduml
099 */
100public class PortalSession extends DefaultSubchannel {
101
102        private static Map<String,PortalSession> portalSessions
103                = new ConcurrentHashMap<>();
104        
105        private String portalSessionId;
106        private boolean closed = false;
107        private long timeout;
108        private Timer timeoutTimer;
109        private Session browserSession = null;
110        private Locale locale = Locale.getDefault();
111        private EventPipeline eventPipeline = null;
112        // Must be weak, else there will always be a reference to the 
113        // upstream channel and, through the reverseMap, to this object.
114        private WeakReference<IOSubchannel> upstreamChannel = null;
115        
116        /**
117         * Lookup (and create if not found) the portal browserSession channel
118         * for the given portal browserSession id.
119         * 
120         * @param component the component to pass to the super 
121         * class' constructor if a new channel is created, usually 
122         * the portal
123         * @param portalSessionId the browserSession id
124         * @param timeout the portal session timeout in milli seconds
125         * @return the channel
126         */
127        public static PortalSession findOrCreate(
128                        String portalSessionId, Manager component, long timeout) {
129                return portalSessions.computeIfAbsent(portalSessionId, 
130                                psi -> new PortalSession(component, portalSessionId, timeout));
131        }
132        
133        private PortalSession(Manager component, String portalSessionId,
134                        long timeout) {
135                super(component);
136                this.portalSessionId = portalSessionId;
137                this.timeout = timeout;
138                timeoutTimer = Components.schedule(
139                                t -> close(), Duration.ofMillis(timeout));
140        }
141
142        /**
143         * Changes the timeout for this {@link PortalSession} to the
144         * given value.
145         * 
146         * @param timeout the timeout in milli seconds
147         * @return the portal session for easy chaining
148         */
149        public PortalSession setTimeout(long timeout) {
150                this.timeout = timeout;
151                timeoutTimer.reschedule(Duration.ofMillis(timeout));
152                return this;
153        }
154        
155        /**
156         * Resets the {@link PortalSession}'s timeout.
157         */
158        public void refresh() {
159                timeoutTimer.reschedule(Duration.ofMillis(timeout));
160        }
161
162        /**
163         * Close the portal session.
164         */
165        public void close() {
166                if (!closed) {
167                        closed = true;
168                        upstreamChannel().ifPresent(up -> up.respond(new Close()));
169                        if (eventPipeline != null) {
170                                eventPipeline.fire(new Closed(), PortalSession.this);
171                        }
172                        portalSessions.remove(portalSessionId);
173                }
174        }
175        
176        /**
177         * Sets the event pipeline used by upstream to send events 
178         * over this channel. This event pipeline is used to send
179         * the {@link Closed} event when the portal session times out.
180         * 
181         * @param eventPipeline the event pipeline
182         * @return the portal session for easy chaining
183         */
184        public PortalSession setEventPipeline(EventPipeline eventPipeline) {
185                this.eventPipeline = eventPipeline;
186                return this;
187        }
188        
189        /**
190         * Sets or updates the upstream channel. This method should only
191         * be invoked by the creator of the {@link PortalSession}, by default
192         * the {@link PortalView}.
193         * 
194         * @param upstreamChannel the upstream channel (WebSocket connection)
195         * @return the portal session for easy chaining
196         */
197        public PortalSession setUpstreamChannel(IOSubchannel upstreamChannel) {
198                if (upstreamChannel == null) {
199                        this.upstreamChannel = null;
200                } else {
201                        this.upstreamChannel = new WeakReference<IOSubchannel>(upstreamChannel);
202                        upstreamChannel.associated(Selection.class).ifPresent(
203                                        s -> locale = s.get()[0]);
204                }
205                return this;
206        }
207
208        /**
209         * Sets or updates associated browser session. This method should only
210         * be invoked by the creator of the {@link PortalSession}, by default
211         * the {@link PortalView}.
212         * 
213         * @param browserSession the browser session
214         * @return the portal session for easy chaining
215         */
216        public PortalSession setSession(Session browserSession) {
217                this.browserSession = browserSession;
218                return this;
219        }
220
221        /**
222         * @return the upstream channel
223         */
224        public Optional<IOSubchannel> upstreamChannel() {
225                return Optional.ofNullable(upstreamChannel)
226                                .flatMap(weakRef -> Optional.ofNullable(weakRef.get()));
227        }
228        
229        /**
230         * @return the portalSessionId
231         */
232        public String portalSessionId() {
233                return portalSessionId;
234        }
235
236        /**
237         * @return the browserSession
238         */
239        public Session browserSession() {
240                return browserSession;
241        }
242
243        /**
244         * Return the portal session's locale. The locale is obtained
245         * from the upstream channel set with 
246         * {@link #setUpstreamChannel(IOSubchannel)}.
247         * 
248         * @return the locale
249         */
250        public Locale locale() {
251                return locale;
252        }
253
254        /**
255         * Set the locale on the upstream channel.
256         * 
257         * @param locale the locale to set
258         */
259        public void setLocale(Locale locale) {
260                upstreamChannel().ifPresent(u -> 
261                        u.associated(Selection.class).ifPresent(s -> s.prefer(locale)));
262        }
263}
