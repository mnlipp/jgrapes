019package org.jgrapes.portal;
020
021import java.lang.ref.WeakReference;
022import java.time.Duration;
023import java.time.Instant;
024import java.util.HashSet;
025import java.util.Locale;
026import java.util.Map;
027import java.util.Optional;
028import java.util.Set;
029import java.util.concurrent.ConcurrentHashMap;
030
031import org.jgrapes.core.Components;
032import org.jgrapes.core.Components.Timer;
033import org.jgrapes.core.EventPipeline;
034import org.jgrapes.http.Session;
035import org.jgrapes.io.IOSubchannel;
036import org.jgrapes.io.IOSubchannel.DefaultSubchannel;
037import org.jgrapes.io.events.Close;
038import org.jgrapes.io.events.Closed;
039
040/**
041 * The server side representation of a window in the browser 
042 * that displays a portal page (a portal session). An instance 
043 * is created when a new portal window opens the websocket 
044 * connection to the server for the first time. If the 
045 * connection between the browser and the server is lost, 
046 * the portal code in the browser tries to establish a 
047 * new websocket connection to the same, already
048 * existing {@link PortalSession}. The {@link PortalSession} 
049 * object is thus independent of the websocket connection 
050 * that handles the actual transfer of notifications.
051 * 
052 * ![Portal Session](PortalSession.svg)
053 * 
054 * Because there is no reliable way to be notified when a
055 * window in a browser closes, {@link PortalSession}s are
056 * discarded automatically unless {@link #refresh()} is called
057 * before the timeout occurs.
058 * 
059 * {@link PortalSession} implements the {@link IOSubchannel}
060 * interface. This allows the instances to be used as channels
061 * for exchanging portal session scoped events with the 
062 * {@link Portal} component. The upstream channel
063 * (see {@link #upstreamChannel()}) is the channel of the
064 * WebSocket. It may be unavailable if the connection has
065 * been interrupted and not (yet) re-established.
066 * 
067 * As a convenience, the {@link PortalSession} provides
068 * direct access to the browser session, which can 
069 * usually only be obtained from the HTTP event or WebSocket
070 * channel by looking for an association of type {@link Session}.
071 * It also provides access to the {@link Locale} as maintained
072 * by the browser session.
073 * 
074 * @startuml PortalSession.svg
075 * class PortalSession {
076 *      -{static}Map<String,PortalSession> portalSessions
077 *      +{static}findOrCreate(String portalSessionId, Manager component): PortalSession
078 *  +setTimeout(timeout: long): PortalSession
079 *  +refresh(): void
080 *      +setUpstreamChannel(IOSubchannel upstreamChannel): PortalSession
081 *      +setSession(Session browserSession): PortalSession
082 *      +upstreamChannel(): Optional<IOSubchannel>
083 *      +portalSessionId(): String
084 *      +browserSession(): Session
085 *      +locale(): Locale
086 *      +setLocale(Locale locale): void
087 * }
088 * Interface IOSubchannel {
089 * }
090 * IOSubchannel <|.. PortalSession
091 *
092 * PortalSession "1" *-- "*" PortalSession : maintains
093 * 
094 * package org.jgrapes.http {
095 *     class Session
096 * }
097 * 
098 * PortalSession "*" -up-> "1" Session: browser session
099 * @enduml
100 */
101public class PortalSession extends DefaultSubchannel {
102
103        private static Map<String,PortalSession> portalSessions
104                = new ConcurrentHashMap<>();
105        
106        private String portalSessionId;
107        private Portal portal;
108        private boolean closed = false;
109        private long timeout;
110        private Timer timeoutTimer;
111        private Session browserSession = null;
112        private EventPipeline eventPipeline = null;
113        // Must be weak, else there will always be a reference to the 
114        // upstream channel and, through the reverseMap, to this object.
115        private WeakReference<IOSubchannel> upstreamChannel = null;
116        
117        /**
118         * Lookup the portal browserSession channel
119         * for the given portal browserSession id.
120         * 
121         * @param portalSessionId the browserSession id
122         * @return the channel
123         */
124        public static Optional<PortalSession> lookup(String portalSessionId) {
125                return Optional.ofNullable(portalSessions.get(portalSessionId));
126        }
127        
128        /**
129         * Return all sessions that belong to the given portal.
130         *
131         * @param portal the portal
132         * @return the sets the
133         */
134        public static Set<PortalSession> byPortal(Portal portal) {
135                Set<PortalSession> result  = new HashSet<>();
136                for (PortalSession ps: portalSessions.values()) {
137                        if (ps.portal.equals(portal)) {
138                                result.add(ps);
139                        }
140                }
141                return result;
142        }
143        
144        /**
145         * Lookup (and create if not found) the portal browserSession channel
146         * for the given portal browserSession id.
147         * 
148         * @param portalSessionId the browserSession id
149         * @param portal the portal that this session belongs to 
150         * class' constructor if a new channel is created, usually 
151         * the portal
152         * @param timeout the portal session timeout in milli seconds
153         * @return the channel
154         */
155        public static PortalSession lookupOrCreate(
156                        String portalSessionId, Portal portal, long timeout) {
157                return portalSessions.computeIfAbsent(portalSessionId, 
158                                psi -> new PortalSession(portal, portalSessionId, timeout));
159        }
160        
161        
162        /**
163         * Replace the id of the portal session with the new id.
164         *
165         * @param newPortalSessionId the new portal session id
166         * @return the portal session
167         */
168        public PortalSession replaceId(String newPortalSessionId) {
169                portalSessions.remove(portalSessionId);
170                portalSessionId = newPortalSessionId;
171                portalSessions.put(portalSessionId, this);
172                return this;
173        }
174        
175        private PortalSession(Portal portal, String portalSessionId,
176                        long timeout) {
177                super(portal);
178                this.portal = portal;
179                this.portalSessionId = portalSessionId;
180                this.timeout = timeout;
181                timeoutTimer = Components.schedule(
182                                t -> close(), Duration.ofMillis(timeout));
183        }
184
185        /**
186         * Changes the timeout for this {@link PortalSession} to the
187         * given value.
188         * 
189         * @param timeout the timeout in milli seconds
190         * @return the portal session for easy chaining
191         */
192        public PortalSession setTimeout(long timeout) {
193                this.timeout = timeout;
194                timeoutTimer.reschedule(Duration.ofMillis(timeout));
195                return this;
196        }
197        
198        /**
199         * Returns the time when this session will expire.
200         *
201         * @return the instant
202         */
203        public Instant expiresAt() {
204                return timeoutTimer.scheduledFor();
205        }
206        
207        /**
208         * Resets the {@link PortalSession}'s timeout.
209         */
210        public void refresh() {
211                timeoutTimer.reschedule(Duration.ofMillis(timeout));
212        }
213
214        /**
215         * Close the portal session.
216         */
217        public void close() {
218                if (!closed) {
219                        closed = true;
220                        upstreamChannel().ifPresent(up -> up.respond(new Close()));
221                        if (eventPipeline != null) {
222                                eventPipeline.fire(new Closed(), PortalSession.this);
223                        }
224                        portalSessions.remove(portalSessionId);
225                }
226        }
227        
228        /**
229         * Sets the event pipeline used by upstream to send events 
230         * over this channel. This event pipeline is used to send
231         * the {@link Closed} event when the portal session times out.
232         * 
233         * @param eventPipeline the event pipeline
234         * @return the portal session for easy chaining
235         */
236        public PortalSession setEventPipeline(EventPipeline eventPipeline) {
237                this.eventPipeline = eventPipeline;
238                return this;
239        }
240        
241        /**
242         * Sets or updates the upstream channel. This method should only
243         * be invoked by the creator of the {@link PortalSession}, by default
244         * the {@link PortalView}.
245         * 
246         * @param upstreamChannel the upstream channel (WebSocket connection)
247         * @return the portal session for easy chaining
248         */
249        public PortalSession setUpstreamChannel(IOSubchannel upstreamChannel) {
250                if (upstreamChannel == null) {
251                        this.upstreamChannel = null;
252                } else {
253                        this.upstreamChannel = new WeakReference<IOSubchannel>(upstreamChannel);
254                }
255                return this;
256        }
257
258        /**
259         * Sets or updates associated browser session. This method should only
260         * be invoked by the creator of the {@link PortalSession}, by default
261         * the {@link PortalView}.
262         * 
263         * @param browserSession the browser session
264         * @return the portal session for easy chaining
265         */
266        public PortalSession setSession(Session browserSession) {
267                this.browserSession = browserSession;
268                return this;
269        }
270
271        /**
272         * @return the upstream channel
273         */
274        public Optional<IOSubchannel> upstreamChannel() {
275                return Optional.ofNullable(upstreamChannel)
276                                .flatMap(weakRef -> Optional.ofNullable(weakRef.get()));
277        }
278        
279        /**
280         * @return the portalSessionId
281         */
282        public String portalSessionId() {
283                return portalSessionId;
284        }
285
286        /**
287         * @return the browserSession
288         */
289        public Session browserSession() {
290                return browserSession;
291        }
292
293        /**
294         * Return the portal session's locale. The locale is obtained
295         * from the browser session.
296         * 
297         * @return the locale
298         */
299        public Locale locale() {
300                return browserSession == null ? Locale.getDefault() 
301                                : browserSession.locale();
302        }
303}
