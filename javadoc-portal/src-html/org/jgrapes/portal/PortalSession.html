019package org.jgrapes.portal;
020
021import java.lang.ref.Reference;
022import java.lang.ref.ReferenceQueue;
023import java.lang.ref.WeakReference;
024import java.time.Duration;
025import java.time.Instant;
026import java.util.HashSet;
027import java.util.Locale;
028import java.util.Map;
029import java.util.Optional;
030import java.util.Set;
031import java.util.concurrent.ConcurrentHashMap;
032
033import org.jgrapes.core.Components;
034import org.jgrapes.core.Components.Timer;
035import org.jgrapes.core.EventPipeline;
036import org.jgrapes.http.Session;
037import org.jgrapes.io.IOSubchannel;
038import org.jgrapes.io.IOSubchannel.DefaultSubchannel;
039import org.jgrapes.io.events.Closed;
040import org.jgrapes.io.util.LinkedIOSubchannel;
041
042/**
043 * The server side representation of a window in the browser 
044 * that displays a portal page (a portal session). An instance 
045 * is created when a new portal window opens the websocket 
046 * connection to the server for the first time. If the 
047 * connection between the browser and the server is lost, 
048 * the portal code in the browser tries to establish a 
049 * new websocket connection to the same, already
050 * existing {@link PortalSession}. The {@link PortalSession} 
051 * object is thus independent of the websocket connection 
052 * that handles the actual transfer of notifications.
053 * 
054 * ![Portal Session](PortalSession.svg)
055 * 
056 * Because there is no reliable way to be notified when a
057 * window in a browser closes, {@link PortalSession}s are
058 * discarded automatically unless {@link #refresh()} is called
059 * before the timeout occurs.
060 * 
061 * {@link PortalSession} implements the {@link IOSubchannel}
062 * interface. This allows the instances to be used as channels
063 * for exchanging portal session scoped events with the 
064 * {@link Portal} component. The upstream channel
065 * (see {@link #upstreamChannel()}) is the channel of the
066 * WebSocket. It may be unavailable if the connection has
067 * been interrupted and not (yet) re-established.
068 * 
069 * The response {@link EventPipeline} must be used to send
070 * events (responses) to the portal session in the browser.
071 * No other event pipeline may be used for this purpose, else
072 * messages will interleave.
073 * 
074 * As a convenience, the {@link PortalSession} provides
075 * direct access to the browser session, which can 
076 * usually only be obtained from the HTTP event or WebSocket
077 * channel by looking for an association of type {@link Session}.
078 * It also provides access to the {@link Locale} as maintained
079 * by the browser session.
080 * 
081 * @startuml PortalSession.svg
082 * class PortalSession {
083 *      -{static}Map<String,PortalSession> portalSessions
084 *      +{static}findOrCreate(String portalSessionId, Manager component): PortalSession
085 *  +setTimeout(timeout: long): PortalSession
086 *  +refresh(): void
087 *      +setUpstreamChannel(IOSubchannel upstreamChannel): PortalSession
088 *      +setSession(Session browserSession): PortalSession
089 *      +upstreamChannel(): Optional<IOSubchannel>
090 *      +portalSessionId(): String
091 *      +browserSession(): Session
092 *      +locale(): Locale
093 *      +setLocale(Locale locale): void
094 * }
095 * Interface IOSubchannel {
096 * }
097 * IOSubchannel <|.. PortalSession
098 *
099 * PortalSession "1" *-- "*" PortalSession : maintains
100 * 
101 * package org.jgrapes.http {
102 *     class Session
103 * }
104 * 
105 * PortalSession "*" -up-> "1" Session: browser session
106 * @enduml
107 */
108public class PortalSession extends DefaultSubchannel {
109
110        private static Map<String,WeakReference<PortalSession>> portalSessions
111                = new ConcurrentHashMap<>();
112        private static ReferenceQueue<PortalSession> unusedSessions
113                = new ReferenceQueue<>();
114        
115        private String portalSessionId;
116        private Portal portal;
117        private long timeout;
118        private Timer timeoutTimer;
119        private Session browserSession = null;
120        private IOSubchannel upstreamChannel = null;
121        
122        private static void cleanUnused() {
123                while(true) {
124                        Reference<? extends PortalSession> unused = unusedSessions.poll();
125                        if (unused == null) {
126                                break;
127                        }
128                        portalSessions.remove(unused.get().portalSessionId());
129                }
130        }
131        
132        /**
133         * Lookup the portal browserSession channel
134         * for the given portal browserSession id.
135         * 
136         * @param portalSessionId the browserSession id
137         * @return the channel
138         */
139        public static Optional<PortalSession> lookup(String portalSessionId) {
140                cleanUnused();
141                return Optional.ofNullable(portalSessions.get(portalSessionId))
142                                .flatMap(wr -> Optional.ofNullable(wr.get()));
143        }
144        
145        /**
146         * Return all sessions that belong to the given portal.
147         *
148         * @param portal the portal
149         * @return the sets the
150         */
151        public static Set<PortalSession> byPortal(Portal portal) {
152                cleanUnused();
153                Set<PortalSession> result  = new HashSet<>();
154                for (WeakReference<PortalSession> psr: portalSessions.values()) {
155                        PortalSession ps = psr.get();
156                        if (ps.portal.equals(portal)) {
157                                result.add(ps);
158                        }
159                }
160                return result;
161        }
162        
163        /**
164         * Lookup (and create if not found) the portal browserSession channel
165         * for the given portal browserSession id.
166         * 
167         * @param portalSessionId the browserSession id
168         * @param portal the portal that this session belongs to 
169         * class' constructor if a new channel is created, usually 
170         * the portal
171         * @param timeout the portal session timeout in milli seconds
172         * @return the channel
173         */
174        public static PortalSession lookupOrCreate(
175                        String portalSessionId, Portal portal, long timeout) {
176                cleanUnused();
177                return portalSessions.computeIfAbsent(portalSessionId, 
178                                psi -> new WeakReference<>(new PortalSession(
179                                                portal, portalSessionId, timeout), unusedSessions))
180                                .get();
181        }
182        
183        
184        /**
185         * Replace the id of the portal session with the new id.
186         *
187         * @param newPortalSessionId the new portal session id
188         * @return the portal session
189         */
190        public PortalSession replaceId(String newPortalSessionId) {
191                portalSessions.remove(portalSessionId);
192                portalSessionId = newPortalSessionId;
193                portalSessions.put(portalSessionId, new WeakReference<>(
194                                this, unusedSessions));
195                return this;
196        }
197        
198        private PortalSession(Portal portal, String portalSessionId,
199                        long timeout) {
200                super(portal.channel(), portal.newEventPipeline());
201                this.portal = portal;
202                this.portalSessionId = portalSessionId;
203                this.timeout = timeout;
204                timeoutTimer = Components.schedule(
205                                t -> discard(), Duration.ofMillis(timeout));
206        }
207
208        /**
209         * Changes the timeout for this {@link PortalSession} to the
210         * given value.
211         * 
212         * @param timeout the timeout in milli seconds
213         * @return the portal session for easy chaining
214         */
215        public PortalSession setTimeout(long timeout) {
216                this.timeout = timeout;
217                timeoutTimer.reschedule(Duration.ofMillis(timeout));
218                return this;
219        }
220        
221        /**
222         * Returns the time when this session will expire.
223         *
224         * @return the instant
225         */
226        public Instant expiresAt() {
227                return timeoutTimer.scheduledFor();
228        }
229        
230        /**
231         * Resets the {@link PortalSession}'s timeout.
232         */
233        public void refresh() {
234                timeoutTimer.reschedule(Duration.ofMillis(timeout));
235        }
236
237        /**
238         * Discards this session.
239         */
240        public void discard() {
241                portalSessions.remove(portalSessionId);
242                portal.newEventPipeline().fire(new Closed(), this);
243        }
244
245        /**
246         * Sets or updates the upstream channel. This method should only
247         * be invoked by the creator of the {@link PortalSession}, by default
248         * the {@link PortalWeblet}.
249         * 
250         * @param upstreamChannel the upstream channel (WebSocket connection)
251         * @return the portal session for easy chaining
252         */
253        public PortalSession setUpstreamChannel(IOSubchannel upstreamChannel) {
254                if (upstreamChannel == null) {
255                        throw new IllegalArgumentException();
256                }
257                this.upstreamChannel = upstreamChannel;
258                return this;
259        }
260
261        /**
262         * Sets or updates associated browser session. This method should only
263         * be invoked by the creator of the {@link PortalSession}, by default
264         * the {@link PortalWeblet}.
265         * 
266         * @param browserSession the browser session
267         * @return the portal session for easy chaining
268         */
269        public PortalSession setSession(Session browserSession) {
270                this.browserSession = browserSession;
271                return this;
272        }
273
274        /**
275         * @return the upstream channel
276         */
277        public IOSubchannel upstreamChannel() {
278                return upstreamChannel;
279        }
280        
281        /**
282         * @return the portalSessionId
283         */
284        public String portalSessionId() {
285                return portalSessionId;
286        }
287
288        /**
289         * @return the browserSession
290         */
291        public Session browserSession() {
292                return browserSession;
293        }
294
295        /**
296         * Return the portal session's locale. The locale is obtained
297         * from the browser session.
298         * 
299         * @return the locale
300         */
301        public Locale locale() {
302                return browserSession == null ? Locale.getDefault() 
303                                : browserSession.locale();
304        }
305        
306        /* (non-Javadoc)
307         * @see java.lang.Object#toString()
308         */
309        public String toString() {
310                StringBuilder builder = new StringBuilder();
311                builder.append(IOSubchannel.toString(this));
312                Optional.ofNullable(upstreamChannel).ifPresent(up -> builder.append(
313                                LinkedIOSubchannel.upstreamToString(up)));
314                return builder.toString();
315        }
316        
317}
