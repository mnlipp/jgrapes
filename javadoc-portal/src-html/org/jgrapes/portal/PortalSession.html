019package org.jgrapes.portal;
020
021import java.lang.ref.WeakReference;
022import java.time.Duration;
023import java.time.Instant;
024import java.util.HashSet;
025import java.util.Locale;
026import java.util.Map;
027import java.util.Optional;
028import java.util.Set;
029import java.util.concurrent.ConcurrentHashMap;
030
031import org.jgrapes.core.Components;
032import org.jgrapes.core.Components.Timer;
033import org.jgrapes.core.EventPipeline;
034import org.jgrapes.http.Session;
035import org.jgrapes.io.IOSubchannel;
036import org.jgrapes.io.IOSubchannel.DefaultSubchannel;
037import org.jgrapes.io.events.Close;
038import org.jgrapes.io.events.Closed;
039import org.jgrapes.io.util.LinkedIOSubchannel;
040
041/**
042 * The server side representation of a window in the browser 
043 * that displays a portal page (a portal session). An instance 
044 * is created when a new portal window opens the websocket 
045 * connection to the server for the first time. If the 
046 * connection between the browser and the server is lost, 
047 * the portal code in the browser tries to establish a 
048 * new websocket connection to the same, already
049 * existing {@link PortalSession}. The {@link PortalSession} 
050 * object is thus independent of the websocket connection 
051 * that handles the actual transfer of notifications.
052 * 
053 * ![Portal Session](PortalSession.svg)
054 * 
055 * Because there is no reliable way to be notified when a
056 * window in a browser closes, {@link PortalSession}s are
057 * discarded automatically unless {@link #refresh()} is called
058 * before the timeout occurs.
059 * 
060 * {@link PortalSession} implements the {@link IOSubchannel}
061 * interface. This allows the instances to be used as channels
062 * for exchanging portal session scoped events with the 
063 * {@link Portal} component. The upstream channel
064 * (see {@link #upstreamChannel()}) is the channel of the
065 * WebSocket. It may be unavailable if the connection has
066 * been interrupted and not (yet) re-established.
067 * 
068 * The response {@link EventPipeline} must be used to send
069 * events (responses) to the portal session in the browser.
070 * No other event pipeline may be used for this purpose, else
071 * messages will interleave.
072 * 
073 * As a convenience, the {@link PortalSession} provides
074 * direct access to the browser session, which can 
075 * usually only be obtained from the HTTP event or WebSocket
076 * channel by looking for an association of type {@link Session}.
077 * It also provides access to the {@link Locale} as maintained
078 * by the browser session.
079 * 
080 * @startuml PortalSession.svg
081 * class PortalSession {
082 *      -{static}Map<String,PortalSession> portalSessions
083 *      +{static}findOrCreate(String portalSessionId, Manager component): PortalSession
084 *  +setTimeout(timeout: long): PortalSession
085 *  +refresh(): void
086 *      +setUpstreamChannel(IOSubchannel upstreamChannel): PortalSession
087 *      +setSession(Session browserSession): PortalSession
088 *      +upstreamChannel(): Optional<IOSubchannel>
089 *      +portalSessionId(): String
090 *      +browserSession(): Session
091 *      +locale(): Locale
092 *      +setLocale(Locale locale): void
093 * }
094 * Interface IOSubchannel {
095 * }
096 * IOSubchannel <|.. PortalSession
097 *
098 * PortalSession "1" *-- "*" PortalSession : maintains
099 * 
100 * package org.jgrapes.http {
101 *     class Session
102 * }
103 * 
104 * PortalSession "*" -up-> "1" Session: browser session
105 * @enduml
106 */
107public class PortalSession extends DefaultSubchannel {
108
109        private static Map<String,PortalSession> portalSessions
110                = new ConcurrentHashMap<>();
111        
112        private String portalSessionId;
113        private Portal portal;
114        private boolean closed = false;
115        private long timeout;
116        private Timer timeoutTimer;
117        private Session browserSession = null;
118        private EventPipeline eventPipeline = null;
119        // Must be weak, else there will always be a reference to the 
120        // upstream channel and, through the reverseMap, to this object.
121        private WeakReference<IOSubchannel> upstreamChannel = null;
122        
123        /**
124         * Lookup the portal browserSession channel
125         * for the given portal browserSession id.
126         * 
127         * @param portalSessionId the browserSession id
128         * @return the channel
129         */
130        public static Optional<PortalSession> lookup(String portalSessionId) {
131                return Optional.ofNullable(portalSessions.get(portalSessionId));
132        }
133        
134        /**
135         * Return all sessions that belong to the given portal.
136         *
137         * @param portal the portal
138         * @return the sets the
139         */
140        public static Set<PortalSession> byPortal(Portal portal) {
141                Set<PortalSession> result  = new HashSet<>();
142                for (PortalSession ps: portalSessions.values()) {
143                        if (ps.portal.equals(portal)) {
144                                result.add(ps);
145                        }
146                }
147                return result;
148        }
149        
150        /**
151         * Lookup (and create if not found) the portal browserSession channel
152         * for the given portal browserSession id.
153         * 
154         * @param portalSessionId the browserSession id
155         * @param portal the portal that this session belongs to 
156         * class' constructor if a new channel is created, usually 
157         * the portal
158         * @param timeout the portal session timeout in milli seconds
159         * @return the channel
160         */
161        public static PortalSession lookupOrCreate(
162                        String portalSessionId, Portal portal, long timeout) {
163                return portalSessions.computeIfAbsent(portalSessionId, 
164                                psi -> new PortalSession(portal, portalSessionId, timeout));
165        }
166        
167        
168        /**
169         * Replace the id of the portal session with the new id.
170         *
171         * @param newPortalSessionId the new portal session id
172         * @return the portal session
173         */
174        public PortalSession replaceId(String newPortalSessionId) {
175                portalSessions.remove(portalSessionId);
176                portalSessionId = newPortalSessionId;
177                portalSessions.put(portalSessionId, this);
178                return this;
179        }
180        
181        private PortalSession(Portal portal, String portalSessionId,
182                        long timeout) {
183                super(portal.channel(), portal.newEventPipeline());
184                this.portal = portal;
185                this.portalSessionId = portalSessionId;
186                this.timeout = timeout;
187                timeoutTimer = Components.schedule(
188                                t -> close(), Duration.ofMillis(timeout));
189        }
190
191        /**
192         * Changes the timeout for this {@link PortalSession} to the
193         * given value.
194         * 
195         * @param timeout the timeout in milli seconds
196         * @return the portal session for easy chaining
197         */
198        public PortalSession setTimeout(long timeout) {
199                this.timeout = timeout;
200                timeoutTimer.reschedule(Duration.ofMillis(timeout));
201                return this;
202        }
203        
204        /**
205         * Returns the time when this session will expire.
206         *
207         * @return the instant
208         */
209        public Instant expiresAt() {
210                return timeoutTimer.scheduledFor();
211        }
212        
213        /**
214         * Resets the {@link PortalSession}'s timeout.
215         */
216        public void refresh() {
217                timeoutTimer.reschedule(Duration.ofMillis(timeout));
218        }
219
220        /**
221         * Close the portal session.
222         */
223        public void close() {
224                if (!closed) {
225                        closed = true;
226                        upstreamChannel().ifPresent(up -> up.respond(new Close()));
227                        if (eventPipeline != null) {
228                                eventPipeline.fire(new Closed(), PortalSession.this);
229                        }
230                        portalSessions.remove(portalSessionId);
231                }
232        }
233        
234        /**
235         * Sets the event pipeline used by upstream to send events 
236         * over this channel. This event pipeline is used to send
237         * the {@link Closed} event when the portal session times out.
238         * 
239         * @param eventPipeline the event pipeline
240         * @return the portal session for easy chaining
241         */
242        public PortalSession setEventPipeline(EventPipeline eventPipeline) {
243                this.eventPipeline = eventPipeline;
244                return this;
245        }
246        
247        /**
248         * Sets or updates the upstream channel. This method should only
249         * be invoked by the creator of the {@link PortalSession}, by default
250         * the {@link PortalWeblet}.
251         * 
252         * @param upstreamChannel the upstream channel (WebSocket connection)
253         * @return the portal session for easy chaining
254         */
255        public PortalSession setUpstreamChannel(IOSubchannel upstreamChannel) {
256                if (upstreamChannel == null) {
257                        this.upstreamChannel = null;
258                } else {
259                        this.upstreamChannel = new WeakReference<IOSubchannel>(upstreamChannel);
260                }
261                return this;
262        }
263
264        /**
265         * Sets or updates associated browser session. This method should only
266         * be invoked by the creator of the {@link PortalSession}, by default
267         * the {@link PortalWeblet}.
268         * 
269         * @param browserSession the browser session
270         * @return the portal session for easy chaining
271         */
272        public PortalSession setSession(Session browserSession) {
273                this.browserSession = browserSession;
274                return this;
275        }
276
277        /**
278         * @return the upstream channel
279         */
280        public Optional<IOSubchannel> upstreamChannel() {
281                return Optional.ofNullable(upstreamChannel)
282                                .flatMap(weakRef -> Optional.ofNullable(weakRef.get()));
283        }
284        
285        /**
286         * @return the portalSessionId
287         */
288        public String portalSessionId() {
289                return portalSessionId;
290        }
291
292        /**
293         * @return the browserSession
294         */
295        public Session browserSession() {
296                return browserSession;
297        }
298
299        /**
300         * Return the portal session's locale. The locale is obtained
301         * from the browser session.
302         * 
303         * @return the locale
304         */
305        public Locale locale() {
306                return browserSession == null ? Locale.getDefault() 
307                                : browserSession.locale();
308        }
309        
310        /* (non-Javadoc)
311         * @see java.lang.Object#toString()
312         */
313        public String toString() {
314                StringBuilder builder = new StringBuilder();
315                builder.append(IOSubchannel.toString(this));
316                upstreamChannel().ifPresent(up -> builder.append(
317                                LinkedIOSubchannel.upstreamToString(up)));
318                builder.append(")");
319                return builder.toString();
320        }
321        
322}
