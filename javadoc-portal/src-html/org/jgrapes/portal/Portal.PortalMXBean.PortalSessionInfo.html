098    public Portal(Channel componentChannel, Channel webletChannel, URI prefix) {
099        super(componentChannel);
100        this.prefix = URI.create(prefix.getPath().endsWith("/")
101            ? prefix.getPath()
102            : (prefix.getPath() + "/"));
103        view = attach(new PortalWeblet(webletChannel, this));
104        MBeanView.addPortal(this);
105    }

176    /**
177     * Handle JSON input.
178     *
179     * @param event the event
180     * @param channel the channel
181     * @throws InterruptedException the interrupted exception
182     * @throws IOException Signals that an I/O exception has occurred.
183     */
184    @Handler
185    @SuppressWarnings("PMD.DataflowAnomalyAnalysis")
186    public void onJsonInput(JsonInput event, PortalSession channel)
187            throws InterruptedException, IOException {
188        // Send events to portlets on portal's channel
189        JsonArray params = event.request().params();
190        switch (event.request().method()) {
191        case "portalReady": {
192            fire(new PortalReady(view.renderSupport()), channel);
193            break;
194        }
195        case "addPortlet": {
196            fire(new AddPortletRequest(view.renderSupport(),
197                params.asString(0), RenderMode.valueOf(
198                    params.asString(1))),
199                channel);
200            break;
201        }
202        case "deletePortlet": {
203            fire(new DeletePortletRequest(
204                view.renderSupport(), params.asString(0)), channel);
205            break;
206        }
207        case "portalLayout": {
208            List<String> previewLayout = params.asArray(0).stream().map(
209                value -> (String) value).collect(Collectors.toList());
210            List<String> tabsLayout = params.asArray(1).stream().map(
211                value -> (String) value).collect(Collectors.toList());
212            JsonObject xtraInfo = (JsonObject) params.get(2);
213            fire(new PortalLayoutChanged(
214                previewLayout, tabsLayout, xtraInfo), channel);
215            break;
216        }
217        case "renderPortlet": {
218            fire(new RenderPortletRequest(view.renderSupport(),
219                params.asString(0),
220                RenderMode.valueOf(params.asString(1)),
221                (Boolean) params.asBoolean(2)), channel);
222            break;
223        }
224        case "setLocale": {
225            fire(new SetLocale(Locale.forLanguageTag(params.asString(0))),
226                channel);
227            break;
228        }
229        case "setTheme": {
230            fire(new SetTheme(params.asString(0)), channel);
231            break;
232        }
233        case "notifyPortletModel": {
234            fire(new NotifyPortletModel(view.renderSupport(),
235                params.asString(0), params.asString(1),
236                params.size() <= 2
237                    ? JsonArray.EMPTY_ARRAY
238                    : params.asArray(2)),
239                channel);
240            break;
241        }
242        default:
243            // Ignore unknown
244            break;
245        }
246    }

248    /**
249     * Handle network configured condition.
250     *
251     * @param event the event
252     * @param channel the channel
253     * @throws InterruptedException the interrupted exception
254     * @throws IOException Signals that an I/O exception has occurred.
255     */
256    @Handler
257    public void onPortalConfigured(
258            PortalConfigured event, PortalSession channel)
259            throws InterruptedException, IOException {
260        channel.respond(new SimplePortalCommand("portalConfigured"));
261    }
262
263    /**
264     * Fallback handler that sends a {@link DeletePortlet} event 
265     * if the {@link RenderPortletRequest} event has not been handled
266     * successfully.
267     *
268     * @param event the event
269     * @param channel the channel
270     */
271    @Handler(priority = -1000000)
272    public void onRenderPortlet(
273            RenderPortletRequest event, PortalSession channel) {
274        if (!event.hasBeenRendered()) {
275            channel.respond(new DeletePortlet(event.portletId()));
276        }
277    }
278
279    /**
280     * Discard all portal sessions on stop.
281     *
282     * @param event the event
283     */
284    @Handler
285    public void onStop(Stop event) {
286        for (PortalSession ps : PortalSession.byPortal(this)) {
287            ps.discard();
288        }
289    }

291    /**
292     * The MBeans view of a portal.
293     */
294    @SuppressWarnings({ "PMD.CommentRequired", "PMD.AvoidDuplicateLiterals" })
295    public interface PortalMXBean {
296
297        @SuppressWarnings("PMD.CommentRequired")
298        class PortalSessionInfo {
299
300            private final PortalSession session;
301
302            public PortalSessionInfo(PortalSession session) {
303                super();
304                this.session = session;
305            }
306
307            public String getChannel() {
308                return session.upstreamChannel().toString();
309            }
310
311            public String getExpiresAt() {
312                return session.expiresAt().atZone(ZoneId.systemDefault())
313                    .toString();
314            }
315        }
316
317        String getComponentPath();
318
319        String getPrefix();
320
321        boolean isUseMinifiedResources();
322
323        void setUseMinifiedResources(boolean useMinifiedResources);
324
325        SortedMap<String, PortalSessionInfo> getPortalSessions();
326    }
327
328    @SuppressWarnings("PMD.CommentRequired")
329    public static class PortalInfo implements PortalMXBean {
330
331        private static MBeanServer mbs
332            = ManagementFactory.getPlatformMBeanServer();
333
334        private ObjectName mbeanName;
335        private final WeakReference<Portal> portalRef;
336
337        public PortalInfo(Portal portal) {
338            try {
339                mbeanName = new ObjectName("org.jgrapes.portal:type="
340                    + Portal.class.getSimpleName() + ",name="
341                    + ObjectName.quote(Components.simpleObjectName(portal)
342                        + " (" + portal.prefix.toString() + ")"));
343            } catch (MalformedObjectNameException e) {
344                // Should not happen
345                logger.log(Level.WARNING, e.getMessage(), e);
346            }
347            portalRef = new WeakReference<>(portal);
348            try {
349                mbs.unregisterMBean(mbeanName);
350            } catch (Exception e) { // NOPMD
351                // Just in case, should not work
352            }
353            try {
354                mbs.registerMBean(this, mbeanName);
355            } catch (InstanceAlreadyExistsException | MBeanRegistrationException
356                    | NotCompliantMBeanException e) {
357                // Should not happen
358                logger.log(Level.WARNING, e.getMessage(), e);
359            }
360        }
361
362        public Optional<Portal> portal() {
363            Portal portal = portalRef.get();
364            if (portal == null) {
365                try {
366                    mbs.unregisterMBean(mbeanName);
367                } catch (Exception e) { // NOPMD
368                    // Should work.
369                }
370            }
371            return Optional.ofNullable(portal);
372        }
373
374        @Override
375        public String getComponentPath() {
376            return portal().map(mgr -> mgr.componentPath()).orElse("<removed>");
377        }
378
379        @Override
380        public String getPrefix() {
381            return portal().map(
382                portal -> portal.prefix().toString()).orElse("<unknown>");
383        }
384
385        @Override
386        public boolean isUseMinifiedResources() {
387            return portal().map(
388                portal -> portal.view.useMinifiedResources())
389                .orElse(false);
390        }
391
392        @Override
393        public void setUseMinifiedResources(boolean useMinifiedResources) {
394            portal().ifPresent(portal -> portal.view.setUseMinifiedResources(
395                useMinifiedResources));
396        }
397
398        @Override
399        @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
400        public SortedMap<String, PortalSessionInfo> getPortalSessions() {
401            SortedMap<String, PortalSessionInfo> result = new TreeMap<>();
402            portal().ifPresent(portal -> {
403                for (PortalSession ps : PortalSession.byPortal(portal)) {
404                    result.put(Components.simpleObjectName(ps),
405                        new PortalSessionInfo(ps));
406                }
407            });
408            return result;
409        }
410    }

412    /**
413     * An MBean interface for getting information about all portals.
414     * 
415     * There is currently no summary information. However, the (periodic)
416     * invocation of {@link PortalSummaryMXBean#getPortals()} ensures
417     * that entries for removed {@link Portal}s are unregistered.
418     */
419    @SuppressWarnings("PMD.CommentRequired")
420    public interface PortalSummaryMXBean {
421
422        Set<PortalMXBean> getPortals();
423
424    }
425
426    /**
427     * Provides an MBean view of the portal.
428     */
429    @SuppressWarnings("PMD.CommentRequired")
430    private static class MBeanView implements PortalSummaryMXBean {
431
432        private static Set<PortalInfo> portalInfos = new HashSet<>();
433
434        public static void addPortal(Portal portal) {
435            synchronized (portalInfos) {
436                portalInfos.add(new PortalInfo(portal));
437            }
438        }
439
440        @Override
441        public Set<PortalMXBean> getPortals() {
442            Set<PortalInfo> expired = new HashSet<>();
443            synchronized (portalInfos) {
444                for (PortalInfo portalInfo : portalInfos) {
445                    if (!portalInfo.portal().isPresent()) {
446                        expired.add(portalInfo);
447                    }
448                }
449                portalInfos.removeAll(expired);
450            }
451            @SuppressWarnings("unchecked")
452            Set<PortalMXBean> result = (Set<PortalMXBean>) (Object) portalInfos;
453            return result;
454        }
455    }
456
457    static {
458        try {
459            MBeanServer mbs = ManagementFactory.getPlatformMBeanServer();
460            ObjectName mxbeanName = new ObjectName("org.jgrapes.portal:type="
461                + Portal.class.getSimpleName() + "s");
462            mbs.registerMBean(new MBeanView(), mxbeanName);
463        } catch (MalformedObjectNameException | InstanceAlreadyExistsException
464                | MBeanRegistrationException | NotCompliantMBeanException e) {
465            // Should not happen
466            logger.log(Level.WARNING, e.getMessage(), e);
467        }
468    }
