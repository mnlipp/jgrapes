019package org.jgrapes.portal;
020
021import java.io.IOException;
022import java.util.Collections;
023import java.util.HashMap;
024import java.util.List;
025import java.util.Map;
026import java.util.Optional;
027
028import org.jdrupes.json.JsonBeanDecoder;
029import org.jdrupes.json.JsonBeanEncoder;
030import org.jdrupes.json.JsonDecodeException;
031import org.jgrapes.core.Channel;
032import org.jgrapes.core.Component;
033import org.jgrapes.core.annotation.Handler;
034import org.jgrapes.http.Session;
035import org.jgrapes.io.IOSubchannel;
036import org.jgrapes.portal.events.LastPortalLayout;
037import org.jgrapes.portal.events.PortalLayoutChanged;
038import org.jgrapes.portal.events.PortalPrepared;
039import org.jgrapes.portal.events.PortalReady;
040import org.jgrapes.portal.events.RenderPortletRequest;
041import org.jgrapes.util.events.KeyValueStoreData;
042import org.jgrapes.util.events.KeyValueStoreQuery;
043import org.jgrapes.util.events.KeyValueStoreUpdate;

045/**
046 * A component that restores the portal layout,
047 * using key/value events for persisting the data between sessions.
048 * 
049 * ![Boot Event Sequence](KVPPBootSeq.svg)
050 * 
051 * This component requires another component that handles the key/value
052 * store events ({@link KeyValueStoreUpdate}, {@link KeyValueStoreQuery})
053 * used by this component for implementing persistence. When the portal becomes
054 * ready, this policy sends a query for the persisted data.
055 * 
056 * When the portal has been prepared, the policy sends the last layout
057 * as retrieved from persistent storage to the portal and then generates
058 * render events for all portlets contained in this layout.
059 * 
060 * Each time the layout is changed in the portal, the portal sends the
061 * new layout data and this component updates the persistent storage
062 * accordingly.
063 * 
064 * @startuml KVPPBootSeq.svg
065 * hide footbox
066 * 
067 * Browser -> Portal: "portalReady"
068 * activate Portal
069 * Portal -> KVStoreBasedPortalPolicy: PortalReady
070 * deactivate Portal
071 * activate KVStoreBasedPortalPolicy
072 * KVStoreBasedPortalPolicy -> "KV Store": KeyValueStoreQuery
073 * activate "KV Store"
074 * "KV Store" -> KVStoreBasedPortalPolicy: KeyValueStoreData
075 * deactivate "KV Store"
076 * deactivate KVStoreBasedPortalPolicy
077 * 
078 * actor System
079 * System -> KVStoreBasedPortalPolicy: PortalPrepared
080 * activate KVStoreBasedPortalPolicy
081 * KVStoreBasedPortalPolicy -> Portal: LastPortalLayout
082 * activate Portal
083 * Portal -> Browser: "lastPortalLayout"
084 * deactivate Portal
085 * loop for all portlets to be displayed
086 *     KVStoreBasedPortalPolicy -> PortletX: RenderPortletRequest
087 *     activate PortletX
088 *     PortletX -> Portal: RenderPortlet
089 *     deactivate PortletX
090 *     activate Portal
091 *     Portal -> Browser: "renderPortlet"
092 *     deactivate Portal
093 * end
094 * deactivate KVStoreBasedPortalPolicy
095 * 
096 * Browser -> Portal: "portalLayout"
097 * activate Portal
098 * Portal -> KVStoreBasedPortalPolicy: PortalLayoutChanged
099 * deactivate Portal
100 * activate KVStoreBasedPortalPolicy
101 * KVStoreBasedPortalPolicy -> "KV Store": KeyValueStoreUpdate
102 * deactivate KVStoreBasedPortalPolicy
103 * 
104 * @enduml
105 */
106public class KVStoreBasedPortalPolicy extends Component {
107
108        /**
109         * Creates a new component with its channel set to
110         * itself.
111         */
112        public KVStoreBasedPortalPolicy() {
113        }
114
115        /**
116         * Creates a new component with its channel set to the given channel.
117         * 
118         * @param componentChannel
119         */
120        public KVStoreBasedPortalPolicy(Channel componentChannel) {
121                super(componentChannel);
122        }
123
124        /**
125         * Intercept the {@link PortalReady} event. Request the 
126         * session data from the key/value store and resume.
127         * 
128         * @param event
129         * @param channel
130         * @throws InterruptedException
131         */
132        @Handler
133        public void onPortalReady(PortalReady event, PortalSession channel) 
134                        throws InterruptedException {
135                PortalSessionDataStore sessionDs = channel.associated(
136                                PortalSessionDataStore.class, 
137                                () -> new PortalSessionDataStore(channel.browserSession()));
138                sessionDs.onPortalReady(event, channel);
139        }
140
141        @Handler
142        public void onKeyValueStoreData(
143                        KeyValueStoreData event, PortalSession channel) 
144                                        throws JsonDecodeException {
145                Optional<PortalSessionDataStore> optSessionDs 
146                        = channel.associated(PortalSessionDataStore.class);
147                if (optSessionDs.isPresent()) {
148                        optSessionDs.get().onKeyValueStoreData(event, channel);
149                }
150        }
151
152        @Handler
153        public void onPortalPrepared(
154                        PortalPrepared event, PortalSession channel) {
155                channel.associated(PortalSessionDataStore.class).ifPresent(
156                                ps -> ps.onPortalPrepared(event, channel));
157        }
158
159        @Handler
160        public void onPortalLayoutChanged(PortalLayoutChanged event, 
161                        PortalSession channel) throws IOException {
162                Optional<PortalSessionDataStore>optDs = channel.associated(
163                                PortalSessionDataStore.class);
164                if (optDs.isPresent()) {
165                        optDs.get().onPortalLayoutChanged(event, channel);
166                }
167        }
168        
169        private class PortalSessionDataStore {
170
171                private String storagePath;
172                private Map<String,Object> persisted = null;
173                
174                public PortalSessionDataStore(Session session) {
175                        storagePath = "/" 
176                                        + Utils.userFromSession(session)
177                                        .map(UserPrincipal::toString).orElse("")
178                                        + "/" + KVStoreBasedPortalPolicy.class.getName();
179                }
180                
181                public void onPortalReady(PortalReady event, IOSubchannel channel) 
182                                throws InterruptedException {
183                        if (persisted != null) {
184                                return;
185                        }
186                        KeyValueStoreQuery query = new KeyValueStoreQuery(
187                                        storagePath, channel);
188                        fire(query, channel);
189                }
190
191                public void onKeyValueStoreData(
192                                KeyValueStoreData event, IOSubchannel channel) 
193                                                throws JsonDecodeException {
194                        if (!event.event().query().equals(storagePath)) {
195                                return;
196                        }
197                        String data = event.data().get(storagePath);
198                        if (data != null) {
199                                JsonBeanDecoder decoder = JsonBeanDecoder.create(data);
200                                @SuppressWarnings("unchecked")
201                                Class<Map<String,Object>> cls 
202                                        = (Class<Map<String,Object>>)(Class<?>)HashMap.class;
203                                persisted = decoder.readObject(cls);
204                        }
205                }
206                
207                public void onPortalPrepared(
208                                PortalPrepared event, IOSubchannel channel) {
209                        if (persisted == null) {
210                                // Retrieval was not successful
211                                persisted = new HashMap<>();
212                        }
213                        // Make sure data is consistent
214                        @SuppressWarnings("unchecked")
215                        List<List<String>> previewLayout = (List<List<String>>)persisted
216                                .computeIfAbsent(
217                                        "previewLayout", k -> { return Collections.emptyList(); });
218                        @SuppressWarnings("unchecked")
219                        List<String> tabsLayout = (List<String>)persisted.computeIfAbsent(
220                                        "tabsLayout", k -> { return Collections.emptyList(); });
221
222                        // Update layout
223                        channel.respond(new LastPortalLayout(previewLayout, tabsLayout));
224                        
225                        // Restore portlets
226                        for (String portletId: tabsLayout) {
227                                fire(new RenderPortletRequest(
228                                                event.event().renderSupport(), portletId,
229                                                Portlet.RenderMode.View, false), channel);
230                        }
231                        for (List<String> column: previewLayout) {
232                                for (String portletId: column) {
233                                        fire(new RenderPortletRequest(
234                                                        event.event().renderSupport(), portletId,
235                                                        Portlet.RenderMode.Preview, true), channel);
236                                }
237                        }
238                }
239                
240                public void onPortalLayoutChanged(PortalLayoutChanged event,
241                                IOSubchannel channel) throws IOException {
242                        persisted.put("previewLayout", event.previewLayout());
243                        persisted.put("tabsLayout", event.tabsLayout());
244                        
245                        // Now store.
246                        JsonBeanEncoder encoder = JsonBeanEncoder.create();
247                        encoder.writeObject(persisted);
248                        fire(new KeyValueStoreUpdate()
249                                        .update(storagePath, encoder.toJson()), channel);
250                }
251
252        }
253}
