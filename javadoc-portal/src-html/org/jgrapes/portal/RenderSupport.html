019package org.jgrapes.portal;
020
021import java.net.URI;
022
023import org.jgrapes.portal.events.PageResourceRequest;
024import org.jgrapes.portal.events.PortletResourceRequest;
025
026/**
027 * Provides support for creating URIs in the portal scope that
028 * are forwarded to components listening on the portal channel.  
029 */
030public interface RenderSupport {
031
032    /**
033     * Create a reference to a resource provided by a page resource
034     * provider. Requesting the resulting URI results in a 
035     * {@link PageResourceRequest}.
036     * 
037     * @param uri the URI made available as
038     * {@link PageResourceRequest#resourceUri()}  
039     * @return the resulting URI
040     */
041    URI pageResource(URI uri);
042
043    /**
044     * Convenience method that converts the path to an URI
045     * before calling {@link #pageResource(URI)}.
046     * 
047     * @param path the path 
048     * @return the resulting URI
049     */
050    default URI pageResource(String path) {
051        return pageResource(PortalWeblet.uriFromPath(path));
052    }
053
054    /**
055     * Create a reference to a resource provided by a portlet
056     * of the given type. Requesting the resulting URI results
057     * in a {@link PortletResourceRequest}.
058     * 
059     * @param portletType the portlet type
060     * @param uri the URI made available as 
061     * {@link PortletResourceRequest#resourceUri()}
062     * @return the resulting URI
063     */
064    URI portletResource(String portletType, URI uri);
065
066    /**
067     * Convenience method that converts the path to an URI
068     * before calling {@link #portletResource(String, URI)}.
069     * 
070     * @param portletType the portlet type
071     * @param path the path 
072     * @return the resulting URI
073     */
074    default URI portletResource(String portletType, String path) {
075        return portletResource(portletType, PortalWeblet.uriFromPath(path));
076    }
077
078    /**
079     * Indicates if minified resources should be used.
080     * 
081     * @return the setting
082     */
083    boolean useMinifiedResources();
084}
