070/**
071 * A portlet used to display information to the user. Instances
072 * may be used as a kind of note, i.e. created and configured by
073 * a user himself. A typical use case, however, is to create
074 * an instance during startup by a portal policy.
075 */
076public class MarkdownDisplayPortlet extends FreeMarkerPortlet {
077
078        /**
079         * The supported preferences.
080         */
081        public enum Preferences { PORTLET_ID, TITLE, PREVIEW_SOURCE, 
082                VIEW_SOURCE, DELETABLE, EDITABLE_BY }
083        
084        /**
085         * Creates a new component with its channel set to the given 
086         * channel.

088         * @param componentChannel the channel that the component's 
089         * handlers listen on by default and that 
090         * {@link Manager#fire(Event, Channel...)} sends the event to 
091         */
092        public MarkdownDisplayPortlet(Channel componentChannel) {
093                super(componentChannel, false);
094        }

096        private String storagePath(Session session) {
097                return "/" + Utils.userFromSession(session)
098                        .map(UserPrincipal::toString).orElse("")
099                        + "/portlets/" + MarkdownDisplayPortlet.class.getName() + "/";
100        }

102        @Handler
103        public void onPortalReady(PortalReady event, PortalSession portalSession) 
104                        throws TemplateNotFoundException, MalformedTemplateNameException, 
105                        ParseException, IOException {
106                ResourceBundle resourceBundle = resourceBundle(portalSession.locale());
107                // Add MarkdownDisplayPortlet resources to page
108                portalSession.respond(new AddPortletType(type())
109                                .setDisplayName(resourceBundle.getString("portletName"))
110                                .addScript(new ScriptResource()
111                                                .setRequires(new String[] {"markdown-it.github.io",
112                                                                "github.com/markdown-it/markdown-it-abbr",
113                                                                "github.com/markdown-it/markdown-it-container",
114                                                                "github.com/markdown-it/markdown-it-deflist",
115                                                                "github.com/markdown-it/markdown-it-emoji",
116                                                                "github.com/markdown-it/markdown-it-footnote",
117                                                                "github.com/markdown-it/markdown-it-ins",
118                                                                "github.com/markdown-it/markdown-it-mark",
119                                                                "github.com/markdown-it/markdown-it-sub",
120                                                                "github.com/markdown-it/markdown-it-sup"})
121                                                .setScriptUri(event.renderSupport().portletResource(
122                                                                type(), "MarkdownDisplay-functions.ftl.js")))
123                                .addCss(PortalView.uriFromPath("MarkdownDisplay-style.css"))
124                                .setInstantiable());
125                KeyValueStoreQuery query = new KeyValueStoreQuery(
126                                storagePath(portalSession.browserSession()), portalSession);
127                fire(query, portalSession);
128        }

130        @Handler
131        public void onKeyValueStoreData(
132                        KeyValueStoreData event, PortalSession channel) 
133                                        throws JsonDecodeException {
134                if (!event.event().query().equals(storagePath(channel.browserSession()))) {
135                        return;
136                }
137                for (String json: event.data().values()) {
138                        MarkdownDisplayModel model = JsonBeanDecoder.create(json)
139                                        .readObject(MarkdownDisplayModel.class);
140                        putInSession(channel.browserSession(), model);
141                }
142        }

144        /**
145         * Adds the portlet to the portal. The portlet supports the 
146         * following options (see {@link AddPortletRequest#properties()}
147         * and {@link Preferences}):
148         * 
149         * * `PORTLET_ID` (String): The portlet id.
150         * 
151         * * `TITLE` (String): The portlet title.
152         * 
153         * * `PREVIEW_SOURCE` (String): The markdown source that is rendered 
154         * in the portlet preview.
155         * 
156         * * `VIEW_SOURCE` (String): The markdown source that is rendered 
157         * in the portlet view.
158         * 
159         * * `DELETABLE` (Boolean): Indicates that the portlet may be 
160         * deleted from the overview page.
161         * 
162         * * `EDITABLE_BY` (Set&lt;Principal&gt;): The principals that may edit 
163         * the portlet instance.
164         */
165        @Override
166        public String doAddPortlet(AddPortletRequest event,
167                        PortalSession portalSession) throws Exception {
168                ResourceBundle resourceBundle = resourceBundle(portalSession.locale());
169                
170                // Create new model
171                String portletId = (String)event.properties().get(Preferences.PORTLET_ID);
172                if (portletId == null) {
173                        portletId = generatePortletId();
174                }
175                MarkdownDisplayModel model = putInSession(
176                                portalSession.browserSession(), 
177                                new MarkdownDisplayModel(portletId));
178                model.setTitle((String)event.properties().getOrDefault(Preferences.TITLE, 
179                                resourceBundle.getString("portletName")));
180                model.setPreviewContent((String)event.properties().getOrDefault(
181                                Preferences.PREVIEW_SOURCE, ""));
182                model.setViewContent((String)event.properties().getOrDefault(
183                                Preferences.VIEW_SOURCE, ""));
184                model.setDeletable((Boolean)event.properties().getOrDefault(
185                                Preferences.DELETABLE,  Boolean.TRUE));
186                @SuppressWarnings("unchecked")
187                Set<Principal> editableBy = (Set<Principal>)event.properties().get(
188                                Preferences.EDITABLE_BY);
189                model.setEditableBy(editableBy);
190                
191                // Save model
192                String jsonState = JsonBeanEncoder.create()
193                                .writeObject(model).toJson();
194                portalSession.respond(new KeyValueStoreUpdate().update(
195                                storagePath(portalSession.browserSession()) + model.getPortletId(),
196                                jsonState));
197                
198                // Send HTML
199                Set<RenderMode> modes = renderModes(model);
200                Template tpl = freemarkerConfig().getTemplate(
201                                "MarkdownDisplay-preview.ftl.html");
202                portalSession.respond(new RenderPortlet(
203                                MarkdownDisplayPortlet.class, model.getPortletId(),
204                                templateProcessor(tpl, fmModel(event, portalSession, model)))
205                                .setRenderMode(DeleteablePreview).setSupportedModes(modes)
206                                .setForeground(true));
207                
208                // Fill in data
209                updateView(portalSession, model, portalSession.locale());
210                return portletId;
211        }

213        /* (non-Javadoc)
214         * @see org.jgrapes.portal.AbstractPortlet#doRenderPortlet
215         */
216        @Override
217        protected void doRenderPortlet(RenderPortletRequest event,
218                PortalSession portalSession, String portletId, 
219                Serializable retrievedState) throws Exception {
220                MarkdownDisplayModel model = (MarkdownDisplayModel)retrievedState;
221                Set<RenderMode> modes = renderModes(model);
222                if (model.getViewContent() != null && !model.getViewContent().isEmpty()) {
223                        modes.add(View);
224                }
225                switch (event.renderMode()) {
226                case Preview:
227                case DeleteablePreview: {
228                        Template tpl = freemarkerConfig().getTemplate("MarkdownDisplay-preview.ftl.html");
229                        portalSession.respond(new RenderPortlet(
230                                        MarkdownDisplayPortlet.class, model.getPortletId(), 
231                                        templateProcessor(tpl, fmModel(event, portalSession, model)))
232                                        .setRenderMode(event.renderMode()).setSupportedModes(modes)
233                                        .setForeground(event.isForeground()));
234                        updateView(portalSession, model, portalSession.locale());
235                        break;
236                }
237                case View: {
238                        Template tpl = freemarkerConfig().getTemplate("MarkdownDisplay-view.ftl.html");
239                        portalSession.respond(new RenderPortlet(
240                                        MarkdownDisplayPortlet.class, model.getPortletId(), 
241                                        templateProcessor(tpl, fmModel(event, portalSession, model)))
242                                        .setRenderMode(View).setSupportedModes(modes)
243                                        .setForeground(event.isForeground()));
244                        updateView(portalSession, model, portalSession.locale());
245                        break;
246                }
247                case Edit: {
248                        Template tpl = freemarkerConfig().getTemplate("MarkdownDisplay-edit.ftl.html");
249                        portalSession.respond(new RenderPortlet(
250                                        MarkdownDisplayPortlet.class, model.getPortletId(), 
251                                        templateProcessor(tpl, fmModel(event, portalSession, model)))
252                                        .setRenderMode(Edit).setSupportedModes(modes));
253                        break;
254                }
255                default:
256                        break;
257                }
258        }
259
260        private Set<RenderMode> renderModes(MarkdownDisplayModel model) {
261                Set<RenderMode> modes = new HashSet<>();
262                modes.add(model.isDeletable() ? DeleteablePreview : Preview);
263                if (model.getViewContent() != null && !model.getViewContent().isEmpty()) {
264                        modes.add(View);
265                }
266                if (model.getEditableBy() == null) {
267                        modes.add(Edit);
268                }
269                return modes;
270        }
271        
272        private void updateView(IOSubchannel channel, MarkdownDisplayModel model,
273                Locale locale) {
274                channel.respond(new NotifyPortletView(type(),
275                                model.getPortletId(), "updateAll", model.getTitle(), 
276                                model.getPreviewContent(), model.getViewContent(),
277                                renderModes(model)));
278        }

282         */
283        @Override
284        protected void doDeletePortlet(DeletePortletRequest event,
285                PortalSession channel, String portletId, 
286                Serializable retrievedState) throws Exception {
287                channel.respond(new KeyValueStoreUpdate().delete(
288                                storagePath(channel.browserSession()) + portletId));
289                channel.respond(new DeletePortlet(portletId));
290        }

292        /* (non-Javadoc)
293         * @see org.jgrapes.portal.AbstractPortlet#doNotifyPortletModel
294         */
295        @Override
296        protected void doNotifyPortletModel(NotifyPortletModel event,
297                PortalSession portalSession, Serializable portletState)
298                throws Exception {
299                event.stop();
300                Map<Preferences, String> properties = new HashMap<>();
301                if (!event.params().isNull(0)) {
302                        properties.put(Preferences.TITLE, event.params().getString(0));
303                }
304                if (!event.params().isNull(1)) {
305                        properties.put(Preferences.PREVIEW_SOURCE, event.params().getString(1));
306                }
307                if (!event.params().isNull(2)) {
308                        properties.put(Preferences.VIEW_SOURCE, event.params().getString(2));
309                }
310                fire(new UpdatePortletModel(event.portletId(), properties), portalSession);
311        }

313        @SuppressWarnings("unchecked")
314        @Handler
315        public void onUpdatePortletModel(UpdatePortletModel event, 
316                        PortalSession portalSession) {
317                stateFromSession(portalSession.browserSession(), event.portletId(), 
318                                MarkdownDisplayModel.class).ifPresent(model -> {
319                                        event.ifPresent(Preferences.TITLE, 
320                                                        (key, value) -> model.setTitle((String)value))
321                                        .ifPresent(Preferences.PREVIEW_SOURCE, 
322                                                        (key, value) -> model.setPreviewContent((String)value))
323                                        .ifPresent(Preferences.VIEW_SOURCE, 
324                                                        (key, value) -> model.setViewContent((String)value))
325                                        .ifPresent(Preferences.DELETABLE, 
326                                                        (key, value) -> model.setDeletable((Boolean)value))
327                                        .ifPresent(Preferences.EDITABLE_BY, 
328                                                        (key, value) -> {
329                                                                model.setEditableBy((Set<Principal>)value);
330                                                        });
331                                        String jsonState = JsonBeanEncoder.create()
332                                                        .writeObject(model).toJson();
333                                        portalSession.respond(new KeyValueStoreUpdate().update(
334                                                        storagePath(portalSession.browserSession()) 
335                                                        + model.getPortletId(), jsonState));
336                                        updateView(portalSession, model, portalSession.locale());
337                                });
338        }
