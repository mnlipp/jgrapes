068/**
069 * A portlet used to display information to the user. Instances
070 * may be used as a kind of note, i.e. created and configured by
071 * a user himself. A typical use case, however, is to create
072 * an instance during startup by a portal policy.
073 */
074@SuppressWarnings("PMD.DataClass")
075public class MarkdownDisplayPortlet extends FreeMarkerPortlet {
076
077    /** Property for forcing a portlet id (used for singleton instaces). */
078    public static final String PORTLET_ID = "PortletId";
079    /** Property for setting a title. */
080    public static final String TITLE = "Title";
081    /** Property for setting the preview source. */
082    public static final String PREVIEW_SOURCE = "PreviewSource";
083    /** Property for setting the view source. */
084    public static final String VIEW_SOURCE = "ViewSource";
085    /** Boolean property that controls if the preview is deletable. */
086    public static final String DELETABLE = "Deletable";
087    /** Property of type `Set<Principal>` for restricting who 
088     * can edit the content. */
089    public static final String EDITABLE_BY = "EditableBy";

103    private String storagePath(Session session) {
104        return "/" + Utils.userFromSession(session)
105            .map(UserPrincipal::toString).orElse("")
106            + "/portlets/" + MarkdownDisplayPortlet.class.getName() + "/";
107    }

109    /**
110     * On {@link PortalReady}, fire the {@link AddPortletType}.
111     *
112     * @param event the event
113     * @param portalSession the portal session
114     * @throws TemplateNotFoundException the template not found exception
115     * @throws MalformedTemplateNameException the malformed template name exception
116     * @throws ParseException the parse exception
117     * @throws IOException Signals that an I/O exception has occurred.
118     */
119    @Handler
120    public void onPortalReady(PortalReady event, PortalSession portalSession)
121            throws TemplateNotFoundException, MalformedTemplateNameException,
122            ParseException, IOException {
123        ResourceBundle resourceBundle = resourceBundle(portalSession.locale());
124        // Add MarkdownDisplayPortlet resources to page
125        portalSession.respond(new AddPortletType(type())
126            .setDisplayName(resourceBundle.getString("portletName"))
127            .addScript(new ScriptResource()
128                .setRequires(new String[] { "markdown-it.github.io",
129                    "github.com/markdown-it/markdown-it-abbr",
130                    "github.com/markdown-it/markdown-it-container",
131                    "github.com/markdown-it/markdown-it-deflist",
132                    "github.com/markdown-it/markdown-it-emoji",
133                    "github.com/markdown-it/markdown-it-footnote",
134                    "github.com/markdown-it/markdown-it-ins",
135                    "github.com/markdown-it/markdown-it-mark",
136                    "github.com/markdown-it/markdown-it-sub",
137                    "github.com/markdown-it/markdown-it-sup" })
138                .setScriptUri(event.renderSupport().portletResource(
139                    type(), "MarkdownDisplay-functions.ftl.js")))
140            .addCss(event.renderSupport(), PortalWeblet.uriFromPath(
141                "MarkdownDisplay-style.css"))
142            .setInstantiable());
143        KeyValueStoreQuery query = new KeyValueStoreQuery(
144            storagePath(portalSession.browserSession()), portalSession);
145        fire(query, portalSession);
146    }

148    /**
149     * Restore portlet information, if contained in the event.
150     *
151     * @param event the event
152     * @param channel the channel
153     * @throws JsonDecodeException the json decode exception
154     */
155    @Handler
156    public void onKeyValueStoreData(
157            KeyValueStoreData event, PortalSession channel)
158            throws JsonDecodeException {
159        if (!event.event().query()
160            .equals(storagePath(channel.browserSession()))) {
161            return;
162        }
163        for (String json : event.data().values()) {
164            MarkdownDisplayModel model = JsonBeanDecoder.create(json)
165                .readObject(MarkdownDisplayModel.class);
166            putInSession(channel.browserSession(), model);
167        }
168    }

170    /**
171     * Adds the portlet to the portal. The portlet supports the 
172     * following options (see {@link AddPortletRequest#properties()}:
173     * 
174     * * `PORTLET_ID` (String): The portlet id.
175     * 
176     * * `TITLE` (String): The portlet title.
177     * 
178     * * `PREVIEW_SOURCE` (String): The markdown source that is rendered 
179     *   in the portlet preview.
180     * 
181     * * `VIEW_SOURCE` (String): The markdown source that is rendered 
182     *   in the portlet view.
183     * 
184     * * `DELETABLE` (Boolean): Indicates that the portlet may be 
185     *   deleted from the overview page.
186     * 
187     * * `EDITABLE_BY` (Set&lt;Principal&gt;): The principals that may edit 
188     *   the portlet instance.
189     */
190    @Override
191    public String doAddPortlet(AddPortletRequest event,
192            PortalSession portalSession) throws Exception {
193        ResourceBundle resourceBundle = resourceBundle(portalSession.locale());
194
195        // Create new model
196        String portletId = (String) event.properties().get(PORTLET_ID);
197        if (portletId == null) {
198            portletId = generatePortletId();
199        }
200        MarkdownDisplayModel model = putInSession(
201            portalSession.browserSession(),
202            new MarkdownDisplayModel(portletId));
203        model.setTitle((String) event.properties().getOrDefault(TITLE,
204            resourceBundle.getString("portletName")));
205        model.setPreviewContent((String) event.properties().getOrDefault(
206            PREVIEW_SOURCE, ""));
207        model.setViewContent((String) event.properties().getOrDefault(
208            VIEW_SOURCE, ""));
209        model.setDeletable((Boolean) event.properties().getOrDefault(
210            DELETABLE, Boolean.TRUE));
211        @SuppressWarnings("unchecked")
212        Set<Principal> editableBy = (Set<Principal>) event.properties().get(
213            EDITABLE_BY);
214        model.setEditableBy(editableBy);
215
216        // Save model
217        String jsonState = JsonBeanEncoder.create()
218            .writeObject(model).toJson();
219        portalSession.respond(new KeyValueStoreUpdate().update(
220            storagePath(portalSession.browserSession()) + model.getPortletId(),
221            jsonState));
222
223        // Send HTML
224        renderPortlet(event, portalSession, model);
225        return portletId;
226    }

234    protected void doRenderPortlet(RenderPortletRequest event,

241    @SuppressWarnings("PMD.DataflowAnomalyAnalysis")
242    private void renderPortlet(RenderPortletRequestBase<?> event,
243            PortalSession portalSession, MarkdownDisplayModel model)
244            throws TemplateNotFoundException, MalformedTemplateNameException,
245            ParseException, IOException {
246        Set<RenderMode> modes = renderModes(model);
247        if (model.getViewContent() != null
248            && !model.getViewContent().isEmpty()) {
249            modes.add(RenderMode.View);
250        }
251        switch (event.renderMode()) {
252        case Preview:
253        case DeleteablePreview: {
254            Template tpl = freemarkerConfig()
255                .getTemplate("MarkdownDisplay-preview.ftl.html");
256            portalSession.respond(new RenderPortletFromTemplate(event,
257                MarkdownDisplayPortlet.class, model.getPortletId(),
258                tpl, fmModel(event, portalSession, model))
259                    .setRenderMode(event.renderMode()).setSupportedModes(modes)
260                    .setForeground(event.isForeground()));
261            updateView(portalSession, model);
262            break;
263        }
264        case View: {
265            Template tpl = freemarkerConfig()
266                .getTemplate("MarkdownDisplay-view.ftl.html");
267            portalSession.respond(new RenderPortletFromTemplate(event,
268                MarkdownDisplayPortlet.class, model.getPortletId(),
269                tpl, fmModel(event, portalSession, model))
270                    .setRenderMode(RenderMode.View).setSupportedModes(modes)
271                    .setForeground(event.isForeground()));
272            updateView(portalSession, model);
273            break;
274        }
275        case Edit: {
276            Template tpl = freemarkerConfig()
277                .getTemplate("MarkdownDisplay-edit.ftl.html");
278            portalSession.respond(new RenderPortletFromTemplate(event,
279                MarkdownDisplayPortlet.class, model.getPortletId(),
280                tpl, fmModel(event, portalSession, model))
281                    .setRenderMode(RenderMode.Edit).setSupportedModes(modes));
282            break;
283        }
284        default:
285            break;
286        }
287    }
288
289    private Set<RenderMode> renderModes(MarkdownDisplayModel model) {
290        Set<RenderMode> modes = new HashSet<>();
291        modes.add(model.isDeletable() ? RenderMode.DeleteablePreview
292            : RenderMode.Preview);
293        if (model.getViewContent() != null
294            && !model.getViewContent().isEmpty()) {
295            modes.add(RenderMode.View);
296        }
297        if (model.getEditableBy() == null) {
298            modes.add(RenderMode.Edit);
299        }
300        return modes;
301    }
302
303    private void updateView(IOSubchannel channel, MarkdownDisplayModel model) {
304        channel.respond(new NotifyPortletView(type(),
305            model.getPortletId(), "updateAll", model.getTitle(),
306            model.getPreviewContent(), model.getViewContent(),
307            renderModes(model)));
308    }

318            Serializable retrievedState) throws Exception {

328     */
329    @Override
330    protected void doNotifyPortletModel(NotifyPortletModel event,
331            PortalSession portalSession, Serializable portletState)
332            throws Exception {
333        event.stop();
334        @SuppressWarnings("PMD.UseConcurrentHashMap")
335        Map<String, String> properties = new HashMap<>();
336        if (event.params().get(0) != null) {
337            properties.put(TITLE, event.params().asString(0));
338        }
339        if (event.params().get(1) != null) {
340            properties.put(PREVIEW_SOURCE, event.params().asString(1));
341        }
342        if (event.params().get(2) != null) {
343            properties.put(VIEW_SOURCE, event.params().asString(2));
344        }
345        fire(new UpdatePortletModel(event.portletId(), properties),
346            portalSession);
347    }

349    /**
350     * Stores the modified properties using a {@link KeyValueStoreUpdate}
351     * event and updates the view with a {@link NotifyPortletView}. 
352     *
353     * @param event the event
354     * @param portalSession the portal session
355     */
356    @SuppressWarnings("unchecked")
357    @Handler
358    public void onUpdatePortletModel(UpdatePortletModel event,
359            PortalSession portalSession) {
360        stateFromSession(portalSession.browserSession(), event.portletId(),
361            MarkdownDisplayModel.class).ifPresent(model -> {
362                event.ifPresent(TITLE,
363                    (key, value) -> model.setTitle((String) value))
364                    .ifPresent(PREVIEW_SOURCE,
365                        (key, value) -> model.setPreviewContent((String) value))
366                    .ifPresent(VIEW_SOURCE,
367                        (key, value) -> model.setViewContent((String) value))
368                    .ifPresent(DELETABLE,
369                        (key, value) -> model.setDeletable((Boolean) value))
370                    .ifPresent(EDITABLE_BY,
371                        (key, value) -> {
372                            model.setEditableBy((Set<Principal>) value);
373                        });
374                try {
375                    String jsonState = JsonBeanEncoder.create()
376                        .writeObject(model).toJson();
377                    portalSession.respond(new KeyValueStoreUpdate().update(
378                        storagePath(portalSession.browserSession())
379                            + model.getPortletId(),
380                        jsonState));
381                    updateView(portalSession, model);
382                } catch (IOException e) { // NOPMD
383                    // Won't happen, uses internal writer
384                }
385            });
386    }
