056/**
057 * 
058 */
059public class SysInfoPortlet extends FreeMarkerPortlet {
060
061    private static final Set<RenderMode> MODES = RenderMode.asSet(
062        RenderMode.DeleteablePreview, RenderMode.View);
063
064    /**
065     * The periodically generated update event.
066     */
067    public static class Update extends Event<Void> {
068    }
069
070    /**
071     * Creates a new component with its channel set to the given channel.
072     * 
073     * @param componentChannel the channel that the component's handlers listen
074     *            on by default and that {@link Manager#fire(Event, Channel...)}
075     *            sends the event to
076     */
077    public SysInfoPortlet(Channel componentChannel) {
078        super(componentChannel, true);
079        setPeriodicRefresh(Duration.ofSeconds(1), () -> new Update());
080    }
081
082    /**
083     * On {@link PortalReady}, fire the {@link AddPortletType}.
084     *
085     * @param event the event
086     * @param portalSession the portal session
087     * @throws TemplateNotFoundException the template not found exception
088     * @throws MalformedTemplateNameException the malformed template name
089     *             exception
090     * @throws ParseException the parse exception
091     * @throws IOException Signals that an I/O exception has occurred.
092     */
093    @Handler
094    public void onPortalReady(PortalReady event, PortalSession portalSession)
095            throws TemplateNotFoundException, MalformedTemplateNameException,
096            ParseException, IOException {
097        ResourceBundle resourceBundle = resourceBundle(portalSession.locale());
098        // Add SysInfoPortlet resources to page
099        portalSession.respond(new AddPortletType(type())
100            .setDisplayName(resourceBundle.getString("portletName"))
101            .addScript(new ScriptResource()
102                .setRequires(new String[] { "chartjs.org" })
103                .setScriptUri(event.renderSupport().portletResource(
104                    type(), "SysInfo-functions.ftl.js")))
105            .addCss(event.renderSupport(), PortalWeblet.uriFromPath(
106                "SysInfo-style.css"))
107            .setInstantiable());
108    }
109
110    /*
111     * (non-Javadoc)
112     * 
113     * @see org.jgrapes.portal.AbstractPortlet#generatePortletId()
114     */
115    @Override
116    protected String generatePortletId() {
117        return type() + "-" + super.generatePortletId();
118    }
119
120    /*
121     * (non-Javadoc)
122     * 
123     * @see org.jgrapes.portal.AbstractPortlet#modelFromSession
124     */
125    @SuppressWarnings("unchecked")
126    @Override
127    protected <T extends Serializable> Optional<T> stateFromSession(
128            Session session, String portletId, Class<T> type) {
129        if (portletId.startsWith(type() + "-")) {
130            return Optional.of((T) new SysInfoModel(portletId));
131        }
132        return Optional.empty();
133    }
134
135    @Override
136    public String doAddPortlet(AddPortletRequest event,
137            PortalSession portalSession) throws Exception {
138        String portletId = generatePortletId();
139        SysInfoModel portletModel = putInSession(
140            portalSession.browserSession(), new SysInfoModel(portletId));
141        renderPortlet(event, portalSession, portletModel);
142        return portletId;
143    }
144
145    /*
146     * (non-Javadoc)
147     * 
148     * @see org.jgrapes.portal.AbstractPortlet#doRenderPortlet
149     */
150    @Override
151    protected void doRenderPortlet(RenderPortletRequest event,
152            PortalSession portalSession, String portletId,
153            Serializable retrievedState) throws Exception {
154        SysInfoModel portletModel = (SysInfoModel) retrievedState;
155        renderPortlet(event, portalSession, portletModel);
156    }
157
158    @SuppressWarnings("PMD.DataflowAnomalyAnalysis")
159    private void renderPortlet(RenderPortletRequestBase<?> event,
160            PortalSession portalSession, SysInfoModel portletModel)
161            throws TemplateNotFoundException, MalformedTemplateNameException,
162            ParseException, IOException {
163        switch (event.renderMode()) {
164        case Preview:
165        case DeleteablePreview: {
166            Template tpl
167                = freemarkerConfig().getTemplate("SysInfo-preview.ftl.html");
168            portalSession.respond(new RenderPortletFromTemplate(event,
169                SysInfoPortlet.class, portletModel.getPortletId(),
170                tpl, fmModel(event, portalSession, portletModel))
171                    .setRenderMode(RenderMode.DeleteablePreview)
172                    .setSupportedModes(MODES)
173                    .setForeground(event.isForeground()));
174            updateView(portalSession, portletModel.getPortletId());
175            break;
176        }
177        case View: {
178            Template tpl
179                = freemarkerConfig().getTemplate("SysInfo-view.ftl.html");
180            portalSession.respond(new RenderPortletFromTemplate(event,
181                SysInfoPortlet.class, portletModel.getPortletId(),
182                tpl, fmModel(event, portalSession, portletModel))
183                    .setSupportedModes(MODES)
184                    .setForeground(event.isForeground()));
185            break;
186        }
187        default:
188            break;
189        }
190    }
191
192    private void updateView(PortalSession portalSession, String portletId) {
193        if (!portalSession.isConnected()) {
194            return;
195        }
196        Runtime runtime = Runtime.getRuntime();
197        portalSession.respond(new NotifyPortletView(type(),
198            portletId, "updateMemorySizes",
199            System.currentTimeMillis(), runtime.maxMemory(),
200            runtime.totalMemory(),
201            runtime.totalMemory() - runtime.freeMemory()));
202    }
203
204    /*
205     * (non-Javadoc)
206     * 
207     * @see org.jgrapes.portal.AbstractPortlet#doDeletePortlet
208     */
209    @Override
210    protected void doDeletePortlet(DeletePortletRequest event,
211            PortalSession portalSession, String portletId,
212            Serializable retrievedState) throws Exception {
213        portalSession.respond(new DeletePortlet(portletId));
214    }
215
216    /**
217     * Handle the periodic update event by sending {@link NotifyPortletView}
218     * events.
219     *
220     * @param event the event
221     * @param portalSession the portal session
222     */
223    @Handler
224    public void onUpdate(Update event, PortalSession portalSession) {
225        for (String portletId : portletIds(portalSession)) {
226            updateView(portalSession, portletId);
227        }
228    }
229
230    @Override
231    @SuppressWarnings("PMD.DoNotCallGarbageCollectionExplicitly")
232    protected void doNotifyPortletModel(NotifyPortletModel event,
233            PortalSession portalSession, Serializable portletState)
234            throws Exception {
235        event.stop();
236        System.gc();
237        for (String portletId : portletIds(portalSession)) {
238            updateView(portalSession, portletId);
239        }
240    }
241
242    /**
243     * The portlet's model.
244     */
245    @SuppressWarnings("serial")
246    public static class SysInfoModel extends PortletBaseModel {
247
248        /**
249         * Creates a new model with the given type and id.
250         * 
251         * @param portletId the portlet id
252         */
253        @ConstructorProperties({ "portletId" })
254        public SysInfoModel(String portletId) {
255            super(portletId);
256        }
257
258        /**
259         * Return the system properties.
260         *
261         * @return the properties
262         */
263        public Properties systemProperties() {
264            return System.getProperties();
265        }
266
267        /**
268         * Return the {@link Runtime}.
269         *
270         * @return the runtime
271         */
272        public Runtime runtime() {
273            return Runtime.getRuntime();
274        }
275    }
