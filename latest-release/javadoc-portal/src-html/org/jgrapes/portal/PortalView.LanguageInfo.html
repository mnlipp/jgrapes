133        public PortalView(Portal portal, Channel componentChannel) {
134                super(componentChannel);
135                this.portal = portal;
136                if (fmConfig == null) {
137                        fmConfig = new Configuration(Configuration.VERSION_2_3_26);
138                        fmConfig.setClassLoaderForTemplateLoading(
139                                        getClass().getClassLoader(), "org/jgrapes/portal");
140                        fmConfig.setDefaultEncoding("utf-8");
141                        fmConfig.setTemplateExceptionHandler(
142                                        TemplateExceptionHandler.RETHROW_HANDLER);
143                fmConfig.setLogTemplateExceptions(false);
144                }
145                baseTheme = new Provider();
146                
147                supportedLocales = new HashSet<>();
148                for (Locale locale: Locale.getAvailableLocales()) {
149                        if (locale.getLanguage().equals("")) {
150                                continue;
151                        }
152                        if (resourceBundleSupplier != null) {
153                                ResourceBundle rb = resourceBundleSupplier.apply(locale);
154                                if (rb.getLocale().equals(locale)) {
155                                        supportedLocales.add(locale);
156                                }
157                        }
158                        ResourceBundle rb = ResourceBundle.getBundle(getClass()
159                                        .getPackage().getName() + ".l10n", locale);
160                        if (rb.getLocale().equals(locale)) {
161                                supportedLocales.add(locale);
162                        }
163                }
164                
165                RequestHandler.Evaluator.add(this, "onGet",     portal.prefix() + "**");
166                RequestHandler.Evaluator.add(this, "onGetRedirect",
167                                portal.prefix().getPath().substring(
168                                                0, portal.prefix().getPath().length() - 1));
169                
170                portalBaseModel = createPortalBaseModel();
171
172                // Handlers attached to the portal side channel
173                Handler.Evaluator.add(this, "onPortalReady", portal.channel());
174                Handler.Evaluator.add(this, "onKeyValueStoreData", portal.channel());
175                Handler.Evaluator.add(
176                                this, "onPortletResourceResponse", portal.channel());
177                Handler.Evaluator.add(this, "onOutput", portal.channel());
178                Handler.Evaluator.add(this, "onJsonOutput", portal.channel());
179                Handler.Evaluator.add(this, "onSetLocale", portal.channel());
180                Handler.Evaluator.add(this, "onSetTheme", portal.channel());
181        }

183        private Map<String,Object> createPortalBaseModel() {
184                // Create portal model
185                portalBaseModel = new HashMap<>();
186                portalBaseModel.put("resourceUrl", new TemplateMethodModelEx() {
187                        @Override
188                        public Object exec(@SuppressWarnings("rawtypes") List arguments)
189                                        throws TemplateModelException {
190                                @SuppressWarnings("unchecked")
191                                List<TemplateModel> args = (List<TemplateModel>)arguments;
192                                if (!(args.get(0) instanceof SimpleScalar)) {
193                                        throw new TemplateModelException("Not a string.");
194                                }
195                                return portal.prefix().resolve(
196                                                ((SimpleScalar)args.get(0)).getAsString()).getRawPath();
197                        }
198                });
199                portalBaseModel.put("useMinifiedResources", useMinifiedResources);
200                portalBaseModel.put("minifiedExtension", 
201                                useMinifiedResources ? ".min" : "");
202                portalBaseModel.put(
203                                "portalSessionRefreshInterval", portalSessionRefreshInterval);
204                portalBaseModel.put(
205                                "portalSessionInactivityTimeout", portalSessionInactivityTimeout);
206                return Collections.unmodifiableMap(portalBaseModel);
207        }

303        @RequestHandler(dynamic=true)
304        public void onGetRedirect(GetRequest event, IOSubchannel channel) 
305                        throws InterruptedException, IOException, ParseException {
306                HttpResponse response = event.httpRequest().response().get();
307                response.setStatus(HttpStatus.MOVED_PERMANENTLY)
308                        .setContentType("text", "plain", "utf-8")
309                        .setField(HttpField.LOCATION, portal.prefix());
310                fire(new Response(response), channel);
311                try {
312                        fire(Output.wrap(portal.prefix().toString()
313                                        .getBytes("utf-8"), true), channel);
314                } catch (UnsupportedEncodingException e) {
315                        // Supported by definition
316                }
317                event.stop();
318        }

320        @RequestHandler(dynamic=true)
321        public void onGet(GetRequest event, IOSubchannel channel) 
322                        throws InterruptedException, IOException {
323                URI requestUri = event.requestUri();
324                // Append trailing slash, if missing
325                if ((requestUri.getRawPath() + "/").equals(
326                                portal.prefix().getRawPath())) {
327                        requestUri = portal.prefix();
328                }
329                
330                // Request for portal?
331                if (!requestUri.getRawPath().startsWith(portal.prefix().getRawPath())) {
332                        return;
333                }
334                
335                // Normalize and evaluate
336                requestUri = portal.prefix().relativize(
337                                URI.create(requestUri.getRawPath()));
338                if (requestUri.getRawPath().isEmpty()) {
339                        if (event.httpRequest().findField(
340                                        HttpField.UPGRADE, Converters.STRING_LIST)
341                                        .map(f -> f.value().containsIgnoreCase("websocket"))
342                                        .orElse(false)) {
343                                channel.setAssociated(this, new WsInputReader());
344                                channel.respond(new WebSocketAccepted(event));
345                                event.stop();
346                                return;
347                        }
348                        renderPortal(event, channel);
349                        return;
350                }
351                URI subUri = uriFromPath("portal-resource/").relativize(requestUri);
352                if (!subUri.equals(requestUri)) {
353                        sendPortalResource(event, channel, subUri.getPath());
354                        return;
355                }
356                subUri = uriFromPath("page-resource/").relativize(requestUri);
357                if (!subUri.equals(requestUri)) {
358                        requestPageResource(event, channel, subUri);
359                        return;
360                }
361                subUri = uriFromPath("portal-session/").relativize(requestUri);
362                if (!subUri.equals(requestUri)) {
363                        if (event.httpRequest().findField(
364                                        HttpField.UPGRADE, Converters.STRING_LIST)
365                                        .map(f -> f.value().containsIgnoreCase("websocket"))
366                                        .orElse(false)) {
367                                // Establish portlet session
368                                Optional<Session> optSession = event.associated(Session.class);
369                                if (!optSession.isPresent()) {
370                                        return;
371                                }
372                                String portalSessionId = subUri.getPath();
373                                PortalSession portalSession = PortalSession
374                                                .findOrCreate(portalSessionId, portal, portalSessionNetworkTimeout)
375                                                .setUpstreamChannel(channel)
376                                                .setEventPipeline(activeEventPipeline())
377                                                .setSession(optSession.get());
378                                channel.setAssociated(PortalSession.class, portalSession);
379                                // Channel now used as JSON input
380                                channel.setAssociated(this, new WsInputReader());
381                                channel.respond(new WebSocketAccepted(event));
382                                event.stop();
383                        }
384                        return;
385                }
386                subUri = uriFromPath("theme-resource/").relativize(requestUri);
387                if (!subUri.equals(requestUri)) {
388                        sendThemeResource(event, channel, subUri.getPath());
389                        return;
390                }
391                subUri = uriFromPath("portlet-resource/").relativize(requestUri);
392                if (!subUri.equals(requestUri)) {
393                        requestPortletResource(event, channel, subUri);
394                        return;
395                }
396        }

398        private void renderPortal(GetRequest event, IOSubchannel channel)
399                throws IOException, InterruptedException {
400                event.stop();
401                
402                // Because language is changed via websocket, locale cookie 
403                // may be out-dated
404                event.associated(Selection.class)
405                        .ifPresent(s -> s.prefer(s.get()[0]));
406                
407                // Prepare response
408                HttpResponse response = event.httpRequest().response().get();
409                MediaType mediaType = MediaType.builder().setType("text", "html")
410                                .setParameter("charset", "utf-8").build();
411                response.setField(HttpField.CONTENT_TYPE, mediaType);
412                response.setStatus(HttpStatus.OK);
413                response.setHasPayload(true);
414                channel.respond(new Response(response));
415                try (Writer out = new OutputStreamWriter(new ByteBufferOutputStream(
416                                channel, channel.responsePipeline()), "utf-8")) {
417                        Map<String,Object> portalModel = new HashMap<>(portalBaseModel);
418
419                        // Add locale
420                        final Locale locale = event.associated(Selection.class).map(
421                                        s -> s.get()[0]).orElse(Locale.getDefault());
422                        portalModel.put("locale", locale);
423
424                        // Add supported locales
425                        final Collator coll = Collator.getInstance(locale);
426                        final Comparator<LanguageInfo> comp 
427                                = new Comparator<PortalView.LanguageInfo>() {
428                                @Override
429                                public int compare(LanguageInfo o1,  LanguageInfo o2) {
430                                        return coll.compare(o1.getLabel(), o2.getLabel());
431                                }
432                        };
433                        LanguageInfo[] languages = supportedLocales.stream()
434                                        .map(l -> new LanguageInfo(l))
435                                        .sorted(comp).toArray(size -> new LanguageInfo[size]);
436                        portalModel.put("supportedLanguages", languages);
437
438                        // Add localization
439                        final ResourceBundle additionalResources = resourceBundleSupplier == null
440                                        ? null : resourceBundleSupplier.apply(locale);
441                        final ResourceBundle baseResources = ResourceBundle.getBundle(
442                                        getClass().getPackage().getName() + ".l10n", locale,
443                                        ResourceBundle.Control.getNoFallbackControl(
444                                                        ResourceBundle.Control.FORMAT_DEFAULT));
445                        portalModel.put("_", new TemplateMethodModelEx() {
446                                @Override
447                                public Object exec(@SuppressWarnings("rawtypes") List arguments)
448                                                throws TemplateModelException {
449                                        @SuppressWarnings("unchecked")
450                                        List<TemplateModel> args = (List<TemplateModel>)arguments;
451                                        if (!(args.get(0) instanceof SimpleScalar)) {
452                                                throw new TemplateModelException("Not a string.");
453                                        }
454                                        String key = ((SimpleScalar)args.get(0)).getAsString();
455                                        try {
456                                                return additionalResources.getString(key);
457                                        } catch (MissingResourceException e) {
458                                                // try base resources
459                                        }
460                                        try {
461                                                return baseResources.getString(key);
462                                        } catch (MissingResourceException e) {
463                                                // no luck
464                                        }
465                                        return key;
466                                }
467                        });
468
469                        // Add themes. Doing this on every reload allows themes
470                        // to be added dynamically. Note that we must load again
471                        // (not reload) in order for this to work in an OSGi environment.
472                        themeLoader = ServiceLoader.load(ThemeProvider.class);
473                        portalModel.put("themeInfos", 
474                                        StreamSupport.stream(themeLoader().spliterator(), false)
475                                        .map(t -> new ThemeInfo(t.themeId(), t.themeName()))
476                                        .sorted().toArray(size -> new ThemeInfo[size]));
477                        
478                        Template tpl = fmConfig.getTemplate("portal.ftlh");
479                        tpl.process(portalModel, out);
480                } catch (TemplateException e) {
481                        throw new IOException(e);
482                }
483        }

485        private void sendPortalResource(GetRequest event, IOSubchannel channel,
486                        String resource) {
487                // Look for content
488                InputStream in = this.getClass().getResourceAsStream(resource);
489                if (in == null) {
490                        return;
491                }
492                
493                // Send header
494                HttpResponse response = event.httpRequest().response().get();
495                prepareResourceResponse(response, event.requestUri(), false);
496                channel.respond(new Response(response));
497                
498                // Send content
499                channel.responsePipeline().executorService()
500                        .submit(new InputStreamPipeline(in, channel));
501                
502                // Done
503                event.stop();
504        }

506        private void sendThemeResource(GetRequest event, IOSubchannel channel,
507                        String resource) {
508                // Get resource
509                ThemeProvider themeProvider = event.associated(Session.class).flatMap(
510                                session -> Optional.ofNullable(session.get("themeProvider")).flatMap(
511                                                themeId -> StreamSupport
512                                                .stream(themeLoader().spliterator(), false)
513                                                .filter(t -> t.themeId().equals(themeId)).findFirst()
514                                )).orElse(baseTheme);
515                InputStream resIn;
516                try {
517                        resIn = themeProvider.getResourceAsStream(resource);
518                } catch (ResourceNotFoundException e) {
519                        try {
520                                resIn = baseTheme.getResourceAsStream(resource);
521                        } catch (ResourceNotFoundException e1) {
522                                resIn = fallbackResourceSupplier.apply(themeProvider, resource);
523                                if (resIn == null) {
524                                        return;
525                                }
526                        }
527                }
528
529                // Send header
530                HttpResponse response = event.httpRequest().response().get();
531                prepareResourceResponse(response, event.requestUri(), false);
532                channel.respond(new Response(response));
533
534                // Send content
535                channel.responsePipeline().executorService()
536                        .submit(new InputStreamPipeline(resIn, channel));
537
538                // Done
539                event.stop();
540        }

542        public static void prepareResourceResponse(
543                        HttpResponse response, URI request, boolean noCache) {
544                response.setContentType(request);
545                // Set max age in cache-control header
546                List<Directive> directives = new ArrayList<>();
547                if (noCache) {
548                        directives.add(new Directive("no-cache"));
549                } else {
550                        directives.add(new Directive("max-age", 600));
551                }
552                response.setField(HttpField.CACHE_CONTROL, directives);
553                response.setField(HttpField.LAST_MODIFIED, Instant.now());
554
555                response.setStatus(HttpStatus.OK);
556        }

558        private void requestPageResource(GetRequest event, IOSubchannel channel,
559                        URI resource) throws InterruptedException {
560                // Send events to providers on portal's channel
561                PageResourceRequest pageResourceRequest = new PageResourceRequest(
562                                uriFromPath(resource.getPath()),
563                                event.httpRequest(), channel, renderSupport());
564                // Make session available (associate with event, this is not
565                // a websocket request).
566                event.associated(Session.class).ifPresent(
567                                session -> pageResourceRequest.setAssociated(Session.class, session));
568                if (Boolean.TRUE.equals(newEventPipeline().fire(
569                                pageResourceRequest, portalChannel(channel)).get())) {
570                        event.stop();
571                }
572        }

574        private void requestPortletResource(GetRequest event, IOSubchannel channel,
575                        URI resource) throws InterruptedException {
576                String resPath = resource.getPath();
577                int sep = resPath.indexOf('/');
578                // Send events to portlets on portal's channel
579                PortletResourceRequest portletRequest = new PortletResourceRequest(
580                                resPath.substring(0, sep), 
581                                uriFromPath(resPath.substring(sep + 1)),
582                                event.httpRequest(), channel, renderSupport());
583                // Make session available (associate with event, this is not
584                // a websocket request).
585                event.associated(Session.class).ifPresent(
586                                session -> portletRequest.setAssociated(Session.class, session));
587                if (Boolean.TRUE.equals(newEventPipeline().fire(
588                                portletRequest, portalChannel(channel)).get())) {
589                        event.stop();
590                }
591        }

593        @Handler
594        public void onInput(Input<ManagedCharBuffer> event, IOSubchannel wsChannel)
595                        throws IOException {
596                Optional<WsInputReader> optWsInputReader 
597                        = wsChannel.associated(this, WsInputReader.class);
598                if (!optWsInputReader.isPresent()) {
599                        return;
600                }
601                JsonObject json = optWsInputReader.get().toJsonObject(event);
602                if (json != null) {
603                        // Fully decoded JSON available.
604                        Optional<PortalSession> psc = wsChannel.associated(
605                                        PortalSession.class);
606                        if (psc.isPresent()) {
607                                // Portal session established, check for special disconnect
608                                if ("disconnect".equals(json.getString("method"))
609                                                && psc.get().portalSessionId().equals(
610                                                                json.getJsonArray("params").getString(0))) {
611                                        psc.get().close();
612                                        return;
613                                }
614                                // Ordinary message from portal (view) to server.
615                                psc.get().refresh();
616                                if("keepAlive".equals(json.getString("method"))) {
617                                        return;
618                                }
619                                fire(new JsonInput(json), psc.get());
620                                return;
621                        }
622                }
623        }

643        public void onPortalReady(PortalReady event, PortalSession channel) {
644                String principal =      Utils.userFromSession(channel.browserSession())
645                                .map(UserPrincipal::toString).orElse("");
646                KeyValueStoreQuery query = new KeyValueStoreQuery(
647                                "/" + principal + "/themeProvider", true);
648                channel.setAssociated(this, new CompletionLock(event, 3000));
649                fire(query, channel);

652        @Handler(dynamic=true)
653        public void onKeyValueStoreData(
654                        KeyValueStoreData event, PortalSession channel) 
655                                        throws JsonDecodeException {
656                Session session = channel.browserSession();
657                String principal = Utils.userFromSession(session)
658                                .map(UserPrincipal::toString).orElse("");
659                if (!event.event().query().equals("/" + principal + "/themeProvider")) {
660                        return;
661                }
662                channel.associated(this, CompletionLock.class)
663                        .ifPresent(lock -> lock.remove());
664                if (!event.data().values().iterator().hasNext()) {
665                        return;
666                }
667                String requestedThemeId = event.data().values().iterator().next();
668                ThemeProvider themeProvider = Optional.ofNullable(
669                                session.get("themeProvider")).flatMap(
670                                                themeId -> StreamSupport
671                                                .stream(themeLoader().spliterator(), false)
672                                                .filter(t -> t.themeId().equals(themeId)).findFirst()
673                                ).orElse(baseTheme);
674                if (!themeProvider.themeId().equals(requestedThemeId)) {
675                        fire(new SetTheme(requestedThemeId), channel);
676                }
677        }

711                        throws InterruptedException, IOException {
712                ThemeProvider themeProvider = StreamSupport

714                        .filter(t -> t.themeId().equals(event.theme())).findFirst()
715                        .orElse(baseTheme);

717                channel.respond(new KeyValueStoreUpdate().update(
718                                "/" + Utils.userFromSession(channel.browserSession())
719                                .map(UserPrincipal::toString).orElse("") 
720                                + "/themeProvider", themeProvider.themeId())).get();
721                fire(new JsonOutput("reload"), channel);
722        }

738        private class WsInputReader {
739
740                private PipedWriter decodeWriter;
741                private Future<JsonObject> decodeResult;
742                
743                public JsonObject toJsonObject(Input<ManagedCharBuffer> event)
744                                                throws IOException {
745                        CharBuffer buffer = event.buffer().backingBuffer();
746                        if (decodeWriter == null) {
747                                decodeWriter = new PipedWriter();
748                                PipedReader reader = new PipedReader(
749                                                decodeWriter, buffer.capacity());
750                                decodeResult = activeEventPipeline().executorService()
751                                        .submit(new DecodeTask(reader));
752                        }
753                        decodeWriter.append(buffer);
754                        if (event.isEndOfRecord()) {
755                                decodeWriter.close();
756                                decodeWriter = null;
757                                try {
758                                        return decodeResult.get();
759                                } catch (ExecutionException e) {
760                                        if (e.getCause() instanceof IOException) {
761                                                throw (IOException)e.getCause();
762                                        }
763                                        throw new IOException(e);
764                                } catch (InterruptedException e) {
765                                        throw new IOException(e);
766                                }
767                        }
768                        return null;
769                }
770                
771                private class DecodeTask implements Callable<JsonObject> {
772
773                        private Reader reader;
774                        
775                        public DecodeTask(Reader reader) {
776                                this.reader = reader;
777                        }
778
779                        @Override
780                        public JsonObject call() throws IOException {
781                                try (Reader in = reader) {
782                                        JsonReader reader = Json.createReader(in);
783                                        return reader.readObject();
784                                }
785                        }
786                }
787        }

851         * `new URI(null, null, path, null)` with the {@link URISyntaxException}
852         * converted to a {@link IllegalArgumentException}.
853         * 
854         * @param path the path
855         * @return the uri
856         * @throws IllegalArgumentException if the string violates
