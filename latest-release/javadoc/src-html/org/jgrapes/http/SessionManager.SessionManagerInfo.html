055/**
056 * A base class for session managers. A session manager associates 
057 * {@link Request} events with a {@link Session} object using 
058 * `Session.class` as association identifier. The {@link Request}
059 * handler has a default priority of 1000.
060 * 
061 * Managers track requests using a cookie with a given name and path. The 
062 * path is a prefix that has to be matched by the request, often "/".
063 * If no cookie with the given name (see {@link #idName()}) is found,
064 * a new cookie with that name and the specified path is created.
065 * The cookie's value is the unique session id that is used to lookup
066 * the session object.
067 * 
068 * Session managers provide additional support for web sockets. If a
069 * web socket is accepted, the session associated with the request
070 * is automatically associated with the {@link IOSubchannel} that
071 * is subsequently used for the web socket events. This allows
072 * handlers for web socket messages to access the session like
073 * {@link Request} handlers.
074 * 
075 * @see EventBase#setAssociated(Object, Object)
076 * @see "[OWASP Session Management Cheat Sheet](https://www.owasp.org/index.php/Session_Management_Cheat_Sheet)"
077 */
078@SuppressWarnings({ "PMD.DataClass", "PMD.AvoidPrintStackTrace" })
079public abstract class SessionManager extends Component {
080
081    private static SecureRandom secureRandom = new SecureRandom();
082
083    private String idName = "id";
084    private String path = "/";
085    private long absoluteTimeout = 9 * 60 * 60 * 1000;
086    private long idleTimeout = 30 * 60 * 1000;
087    private int maxSessions = 1000;

122    /**
123     * Creates a new session manager with the given channel and path.
124     * The manager handles all requests that match the given path, using
125     * the same rules as browsers do for selecting the cookies that
126     * are to be sent.
127     *  
128     * @param componentChannel the component channel
129     * @param path the path

131    public SessionManager(Channel componentChannel, String path) {
132        this(componentChannel, derivePattern(path), 1000, path);
133    }

135    /**
136     * Derives the resource pattern from the path.
137     *
138     * @param path the path
139     * @return the pattern
140     */
141    @SuppressWarnings("PMD.DataflowAnomalyAnalysis")
142    protected static String derivePattern(String path) {
143        String pattern;
144        if ("/".equals(path)) {
145            pattern = "/**";
146        } else {
147            String patternBase = path;
148            if (patternBase.endsWith("/")) {
149                patternBase = path.substring(0, path.length() - 1);
150            }
151            pattern = path + "," + path + "/**";
152        }
153        return pattern;
154    }

156    /**
157     * Creates a new session manager using the given channel and path.
158     * The manager handles only requests that match the given pattern.
159     * The handler is registered with the given priority.
160     * 
161     * This constructor can be used if special handling of top level
162     * requests is needed.
163     *
164     * @param componentChannel the component channel
165     * @param pattern the path part of a {@link ResourcePattern}
166     * @param priority the priority
167     * @param path the path
168     */
169    public SessionManager(Channel componentChannel, String pattern,
170            int priority, String path) {
171        super(componentChannel);
172        this.path = path;
173        RequestHandler.Evaluator.add(this, "onRequest", pattern, priority);
174        MBeanView.addManager(this);
175    }

261    /**
262     * Associates the event with a {@link Session} object
263     * using `Session.class` as association identifier.
264     * 
265     * @param event the event
266     */
267    @RequestHandler(dynamic = true)
268    public void onRequest(Request.In event) {
269        if (event.associated(Session.class).isPresent()) {
270            return;
271        }
272        final HttpRequest request = event.httpRequest();
273        Optional<String> requestedSessionId = request.findValue(
274            HttpField.COOKIE, Converters.COOKIE_LIST)
275            .flatMap(cookies -> cookies.stream().filter(
276                cookie -> cookie.getName().equals(idName()))
277                .findFirst().map(HttpCookie::getValue));
278        if (requestedSessionId.isPresent()) {
279            String sessionId = requestedSessionId.get();
280            synchronized (this) {
281                Optional<Session> session = lookupSession(sessionId);
282                if (session.isPresent()) {
283                    Instant now = Instant.now();
284                    if ((absoluteTimeout <= 0
285                        || Duration.between(session.get().createdAt(),
286                            now).toMillis() < absoluteTimeout)
287                        && (idleTimeout <= 0
288                            || Duration.between(session.get().lastUsedAt(),
289                                now).toMillis() < idleTimeout)) {
290                        event.setAssociated(Session.class, session.get());
291                        session.get().updateLastUsedAt();
292                        return;
293                    }
294                    // Invalidate, too old
295                    removeSession(sessionId);
296                }
297            }
298        }
299        String sessionId = createSessionId(request.response().get());
300        Session session = createSession(sessionId);
301        event.setAssociated(Session.class, session);
302    }

304    /**

334    /**
335     * Creates a session id and adds the corresponding cookie to the
336     * response.
337     * 
338     * @param response the response
339     * @return the session id
340     */
341    protected String createSessionId(HttpResponse response) {
342        StringBuilder sessionIdBuilder = new StringBuilder();
343        byte[] bytes = new byte[16];
344        secureRandom.nextBytes(bytes);
345        for (byte b : bytes) {
346            sessionIdBuilder.append(Integer.toHexString(b & 0xff));
347        }
348        String sessionId = sessionIdBuilder.toString();
349        HttpCookie sessionCookie = new HttpCookie(idName(), sessionId);
350        sessionCookie.setPath(path);
351        sessionCookie.setHttpOnly(true);
352        response.computeIfAbsent(HttpField.SET_COOKIE, CookieList::new)
353            .value().add(sessionCookie);
354        response.computeIfAbsent(
355            HttpField.CACHE_CONTROL, CacheControlDirectives::new)
356            .value().add(new Directive("no-cache", "SetCookie, Set-Cookie2"));
357        return sessionId;
358    }

360    /**
361     * Discards the given session.
362     * 
363     * @param event the event
364     */
365    @Handler(channels = Channel.class)
366    public void discard(DiscardSession event) {
367        removeSession(event.session().id());
368    }

370    /**
371     * Associates the channel with the session from the upgrade request.
372     * 
373     * @param event the event
374     * @param channel the channel
375     */
376    @Handler(priority = 1000)
377    public void onProtocolSwitchAccepted(
378            ProtocolSwitchAccepted event, IOSubchannel channel) {
379        event.requestEvent().associated(Session.class)
380            .ifPresent(session -> {
381                channel.setAssociated(Session.class, session);
382            });
383    }

405    /**
406     * The session manager information.
407     */
408    public static class SessionManagerInfo implements SessionManagerMXBean {
409
410        private static MBeanServer mbs
411            = ManagementFactory.getPlatformMBeanServer();
412
413        private ObjectName mbeanName;
414        private final WeakReference<SessionManager> sessionManagerRef;
415
416        /**
417         * Instantiates a new session manager info.
418         *
419         * @param sessionManager the session manager
420         */
421        @SuppressWarnings({ "PMD.AvoidCatchingGenericException",
422            "PMD.EmptyCatchBlock" })
423        public SessionManagerInfo(SessionManager sessionManager) {
424            try {
425                mbeanName = new ObjectName("org.jgrapes.http:type="
426                    + SessionManager.class.getSimpleName() + ",name="
427                    + ObjectName.quote(Components.simpleObjectName(
428                        sessionManager)));
429            } catch (MalformedObjectNameException e) {
430                // Won't happen
431            }
432            sessionManagerRef = new WeakReference<>(sessionManager);
433            try {
434                mbs.unregisterMBean(mbeanName);
435            } catch (Exception e) {
436                // Just in case, should not work
437            }
438            try {
439                mbs.registerMBean(this, mbeanName);
440            } catch (InstanceAlreadyExistsException | MBeanRegistrationException
441                    | NotCompliantMBeanException e) {
442                // Have to live with that
443            }
444        }
445
446        /**
447         * Returns the session manager.
448         *
449         * @return the optional session manager
450         */
451        @SuppressWarnings({ "PMD.AvoidCatchingGenericException",
452            "PMD.EmptyCatchBlock" })
453        public Optional<SessionManager> manager() {
454            SessionManager manager = sessionManagerRef.get();
455            if (manager == null) {
456                try {
457                    mbs.unregisterMBean(mbeanName);
458                } catch (MBeanRegistrationException
459                        | InstanceNotFoundException e) {
460                    // Should work.
461                }
462            }
463            return Optional.ofNullable(manager);
464        }
465
466        @Override
467        public String getComponentPath() {
468            return manager().map(mgr -> mgr.componentPath())
469                .orElse("<removed>");
470        }
471
472        @Override
473        public String getPath() {
474            return manager().map(mgr -> mgr.path).orElse("<unknown>");
475        }
476
477        @Override
478        public int getMaxSessions() {
479            return manager().map(mgr -> mgr.maxSessions()).orElse(0);
480        }
481
482        @Override
483        public int getAbsoluteTimeout() {
484            return manager().map(mgr -> mgr.absoluteTimeout()).orElse(0);
485        }
486
487        @Override
488        public int getIdleTimeout() {
489            return manager().map(mgr -> mgr.idleTimeout()).orElse(0);
490        }
491
492        @Override
493        public int getSessionCount() {
494            return manager().map(mgr -> mgr.sessionCount()).orElse(0);
495        }
496    }

498    /**
499     * An MBean interface for getting information about all session
500     * managers.
501     * 
502     * There is currently no summary information. However, the (periodic)
503     * invocation of {@link SessionManagerSummaryMXBean#getManagers()} ensures
504     * that entries for removed {@link SessionManager}s are unregistered.
505     */
506    public interface SessionManagerSummaryMXBean {
507
508        /**
509         * Gets the managers.
510         *
511         * @return the managers
512         */
513        Set<SessionManagerMXBean> getManagers();
514    }
515
516    /**
517     * The MBean view.
518     */
519    private static class MBeanView implements SessionManagerSummaryMXBean {
520        private static Set<SessionManagerInfo> managerInfos = new HashSet<>();
521
522        /**
523         * Adds a manager.
524         *
525         * @param manager the manager
526         */
527        public static void addManager(SessionManager manager) {
528            synchronized (managerInfos) {
529                managerInfos.add(new SessionManagerInfo(manager));
530            }
531        }
532
533        @Override
534        public Set<SessionManagerMXBean> getManagers() {
535            Set<SessionManagerInfo> expired = new HashSet<>();
536            synchronized (managerInfos) {
537                for (SessionManagerInfo managerInfo : managerInfos) {
538                    if (!managerInfo.manager().isPresent()) {
539                        expired.add(managerInfo);
540                    }
541                }
542                managerInfos.removeAll(expired);
543            }
544            @SuppressWarnings("unchecked")
545            Set<SessionManagerMXBean> result
546                = (Set<SessionManagerMXBean>) (Object) managerInfos;
547            return result;
548        }
549    }
550
551    static {
552        try {
553            MBeanServer mbs = ManagementFactory.getPlatformMBeanServer();
554            ObjectName mxbeanName = new ObjectName("org.jgrapes.http:type="
555                + SessionManager.class.getSimpleName() + "s");
556            mbs.registerMBean(new MBeanView(), mxbeanName);
557        } catch (MalformedObjectNameException | InstanceAlreadyExistsException
558                | MBeanRegistrationException | NotCompliantMBeanException e) {
559            // Does not happen
560            e.printStackTrace();
561        }
562    }
