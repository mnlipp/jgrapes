060/**
061 * A base class for components that generate responses to
062 * HTTP requests which are based on a FreeMarker template.
063 */
064public class FreeMarkerRequestHandler extends Component {
065        static final Pattern templatePattern 
066                = Pattern.compile(".*\\.ftl\\.[a-z]+$");
067
068        private ClassLoader contentLoader;
069        private String contentPath;
070        private String prefix;
071        private Configuration fmConfig = null;
072        private MaxAgeCalculator maxAgeCalculator = null;
073
074        /**
075         * Instantiates a new free marker request handler.
076         * 
077         * The prefix is removed from the request paths before resolving
078         * them against the content root. A prefix must start with a
079         * slash and must not end with a slash.
080         *
081         * @param componentChannel the component channel
082         * @param contentLoader the content loader
083         * @param contentPath the content path
084         * @param prefix the prefix used in requests
085         */
086        public FreeMarkerRequestHandler(Channel componentChannel, 
087                        ClassLoader contentLoader, String contentPath, String prefix) {
088                super(componentChannel);
089                this.contentLoader = contentLoader;
090                if (contentPath.startsWith("/")) {
091                        contentPath = contentPath.substring(1);
092                }
093                if (contentPath.endsWith("/")) {
094                        contentPath = contentPath.substring(0, contentPath.length() - 1);
095                }
096                this.contentPath = contentPath;
097                this.prefix = prefix;
098        }
099
100        /**
101         * @return the maxAgeCalculator
102         */
103        public MaxAgeCalculator maxAgeCalculator() {
104                return maxAgeCalculator;
105        }
106
107        /**
108         * Sets the {@link MaxAgeCalculator} for generating the `Cache-Control` 
109         * (`max-age`) header of the response. The default is `null`. This
110         * causes 0 to be provided for responses generated from templates and the 
111         * {@link ResponseCreationSupport#DEFAULT_MAX_AGE_CALCULATOR} to be
112         * used for static content.
113         * 
114         * @param maxAgeCalculator the maxAgeCalculator to set
115         */
116        public void setMaxAgeCalculator(MaxAgeCalculator maxAgeCalculator) {
117                this.maxAgeCalculator = maxAgeCalculator;
118        }
119
120        /**
121         * Removes the prefix specified in the constructor from the
122         * path in the request. Checks if the resulting path  
123         * ends with `*.ftl.*`. If so, processes the template with the
124         * {@link #sendProcessedTemplate(Request, IOSubchannel, String)} (which 
125         * uses {@link #fmSessionModel(Optional)}) and sends the result. 
126         * Else, tries to serve static content with the optionally 
127         * shortened path.
128         * 
129         * @param event the event
130         * @param channel the channel
131         * @throws ParseException the parse exception
132         */
133        protected void doGet(Request event, IOSubchannel channel)
134                        throws ParseException {
135                final HttpRequest request = event.httpRequest();
136                URI subUri = ResponseCreationSupport.uriFromPath(prefix)
137                                .relativize(request.requestUri());
138                if (subUri.equals(request.requestUri())) {
139                        return;
140                }
141                if (!templatePattern.matcher(event.requestUri().getPath()).matches()) {
142                        ResponseCreationSupport.sendStaticContent(
143                                        event, channel, path -> contentLoader.getResource(
144                                                        FreeMarkerRequestHandler.this.contentPath 
145                                                        + "/" + subUri.getPath()), maxAgeCalculator);
146                        return;
147                }
148                sendProcessedTemplate(event, channel, subUri.getPath());
149        }
150
151        /**
152         * Render a response using the given template.
153         *
154         * @param event
155         *            the event
156         * @param channel
157         *            the channel
158         * @param tpl
159         *            the template
160         */
161        protected void sendProcessedTemplate(
162                        Request event, IOSubchannel channel, Template tpl) {
163                // Prepare response
164                HttpResponse response = event.httpRequest().response().get();
165                MediaType mediaType = contentType(event.requestUri());
166                response.setContentType(mediaType);
167
168                // Send response 
169                response.setStatus(HttpStatus.OK);
170                response.setField(HttpField.LAST_MODIFIED, Instant.now());
171                if (maxAgeCalculator == null) {
172                        ResponseCreationSupport.setMaxAge(response, 
173                                        (req, mt) -> 0, event.httpRequest(), mediaType);
174                } else {
175                        ResponseCreationSupport.setMaxAge(
176                                        response, maxAgeCalculator, event.httpRequest(), mediaType);
177                }
178                event.stop();
179                channel.respond(new Response(response));
180
181                // Send content
182                try (Writer out = new OutputStreamWriter(
183                        new ByteBufferOutputStream(channel, channel.responsePipeline()),
184                        "utf-8")) {
185                        Map<String, Object> model = fmSessionModel(
186                                event.associated(Session.class));
187                        tpl.setLocale((Locale)model.get("locale"));
188                        tpl.process(model, out);
189                } catch (IOException | TemplateException e) {
190                        // Too late to do anything about this (header was sent).
191                }
192        }
193
194        /**
195         * Render a response using the template obtained from the config with
196         * {@link Configuration#getTemplate(String)} and the given path.
197         *
198         * @param event
199         *            the event
200         * @param channel
201         *            the channel
202         * @param path
203         *            the path
204         */
205        protected void sendProcessedTemplate(
206                        Request event, IOSubchannel channel, String path) {
207                try {
208                        // Get template (no need to continue if this fails).
209                        Template tpl = freemarkerConfig().getTemplate(path);
210                        sendProcessedTemplate(event, channel, tpl);
211                } catch (Exception e) {
212                        return;
213                }
214        }
215
216        /**
217         * Creates the configuration for freemarker template processing.
218         * 
219         * @return the configuration
220         */
221        protected Configuration freemarkerConfig() {
222                if (fmConfig == null) {
223                        fmConfig = new Configuration(Configuration.VERSION_2_3_26);
224                        fmConfig.setClassLoaderForTemplateLoading(
225                                contentLoader, contentPath);
226                        fmConfig.setDefaultEncoding("utf-8");
227                        fmConfig.setTemplateExceptionHandler(
228                                TemplateExceptionHandler.RETHROW_HANDLER);
229                        fmConfig.setLogTemplateExceptions(false);
230                }
231                return fmConfig;
232        }
233
234        /**
235         * Build a freemarker model holding the information associated with the
236         * session.
237         * 
238         * This model provides: 
239         * 
240         * * The `locale` (of type {@link Locale}). 
241         * 
242         * * The `resourceBundle` (of type {@link ResourceBundle}). 
243         * 
244         * * A function "`_`" that looks up the given key in the 
245         *   resource bundle.
246         * 
247         * @param session
248         *            the session
249         * @return the model
250         */
251        protected Map<String, Object> fmSessionModel(Optional<Session> session) {
252                final Map<String, Object> model = new HashMap<>();
253                Locale locale = session.map(s -> s.locale()).orElse(Locale.getDefault());
254                model.put("locale", locale);
255                final ResourceBundle resourceBundle = resourceBundle(locale);
256                model.put("resourceBundle", resourceBundle);
257                model.put("_", new TemplateMethodModelEx() {
258                        @Override
259                        public Object exec(@SuppressWarnings("rawtypes") List arguments)
260                                throws TemplateModelException {
261                                @SuppressWarnings("unchecked")
262                                List<TemplateModel> args = (List<TemplateModel>) arguments;
263                                if (!(args.get(0) instanceof SimpleScalar)) {
264                                        throw new TemplateModelException("Not a string.");
265                                }
266                                String key = ((SimpleScalar) args.get(0)).getAsString();
267                                try {
268                                        return resourceBundle.getString(key);
269                                } catch (MissingResourceException e) {
270                                        // no luck
271                                }
272                                return key;
273                        }
274                });
275                return model;
276        }
277
278        /**
279         * Used to get the content type when generating a response with
280         * {@link #sendProcessedTemplate(Request, IOSubchannel, Template)}. May be
281         * overridden by derived classes. This implementation simply invokes
282         * {@link HttpResponse#contentType(URI)}.
283         *
284         * @param request the request
285         * @return the content type
286         */
287        protected MediaType contentType(URI request) {
288                return HttpResponse.contentType(request);
289        }
290
291        /**
292         * Provides a resource bundle for localization.
293         * The default implementation looks up a bundle using the
294         * package name plus "l10n" as base name.
295         * 
296         * @return the resource bundle
297         */
298        protected ResourceBundle resourceBundle(Locale locale) {
299                return ResourceBundle.getBundle(
300                        contentPath.replace('/', '.') + ".l10n", locale, 
301                        contentLoader, ResourceBundle.Control.getNoFallbackControl(
302                                        ResourceBundle.Control.FORMAT_DEFAULT));
303        }
