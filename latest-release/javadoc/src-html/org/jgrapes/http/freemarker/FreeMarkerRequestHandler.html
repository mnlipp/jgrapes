066/**
067 * A base class for components that generate responses to
068 * HTTP requests which are based on a FreeMarker template.
069 */
070@SuppressWarnings("PMD.DataClass")
071public class FreeMarkerRequestHandler extends Component {
072    public static final Pattern TEMPLATE_PATTERN
073        = Pattern.compile(".*\\.ftl\\.[a-z]+$");
074
075    private ClassLoader contentLoader;
076    private String contentPath;
077    private URI prefix;
078    private ResourcePattern prefixPattern;
079    private Configuration fmConfig;
080    private MaxAgeCalculator maxAgeCalculator;
081
082    /**
083     * Instantiates a new free marker request handler.
084     * 
085     * The prefix path is removed from the request paths before resolving
086     * them against the content root. A prefix path must start with a
087     * slash and must end with a slash. If the request handler
088     * should respond to top-level requests, the prefix must be
089     * a single slash.
090     *
091     * @param componentChannel the component channel
092     * @param channelReplacements the channel replacements to apply
093     * to the `channels` elements of the {@link Handler} annotations
094     * @param contentLoader the content loader
095     * @param contentPath the content path
096     * @param prefix the prefix used in requests
097     */
098    public FreeMarkerRequestHandler(Channel componentChannel,
099            ChannelReplacements channelReplacements,
100            ClassLoader contentLoader, String contentPath, URI prefix) {
101        super(componentChannel, channelReplacements);
102        String prefixPath = prefix.getPath();
103        if (!prefixPath.startsWith("/") || !prefixPath.endsWith("/")) {
104            throw new IllegalArgumentException("Illegal prefix: " + prefix);
105        }
106        this.prefix = prefix;
107        this.contentLoader = contentLoader;
108        if (contentPath.startsWith("/")) {
109            contentPath = contentPath.substring(1);
110        }
111        if (contentPath.endsWith("/")) {
112            contentPath = contentPath.substring(0, contentPath.length() - 1);
113        }
114        this.contentPath = contentPath;
115        try {
116            this.prefixPattern = new ResourcePattern(
117                prefixPath.substring(0, prefixPath.length() - 1) + "|**");
118        } catch (ParseException e) {
119            throw new IllegalArgumentException(e);
120        }
121    }

178    /**
179     * Sets the {@link MaxAgeCalculator} for generating the `Cache-Control` 
180     * (`max-age`) header of the response. The default is `null`. This
181     * causes 0 to be provided for responses generated from templates and the 
182     * {@link ResponseCreationSupport#DEFAULT_MAX_AGE_CALCULATOR} to be
183     * used for static content.
184     * 
185     * @param maxAgeCalculator the maxAgeCalculator to set
186     */
187    public void setMaxAgeCalculator(MaxAgeCalculator maxAgeCalculator) {
188        this.maxAgeCalculator = maxAgeCalculator;
189    }
190
191    /**
192     * Removes the prefix specified in the constructor from the
193     * path in the request. Checks if the resulting path  
194     * ends with `*.ftl.*`. If so, processes the template with the
195     * {@link #sendProcessedTemplate(Request, IOSubchannel, String)} (which 
196     * uses {@link #fmSessionModel(Optional)}) and sends the result. 
197     * Else, tries to serve static content with the optionally 
198     * shortened path.
199     * 
200     * @param event the event
201     * @param channel the channel
202     * @throws ParseException the parse exception
203     */
204    @SuppressWarnings("PMD.DataflowAnomalyAnalysis")
205    protected void doRespond(Request event, IOSubchannel channel)
206            throws ParseException {
207        final HttpRequest request = event.httpRequest();
208        prefixPattern.pathRemainder(request.requestUri()).ifPresent(path -> {
209            boolean success = false;
210            if (TEMPLATE_PATTERN.matcher(path).matches()) {
211                success = sendProcessedTemplate(event, channel, path);
212            } else {
213                success = ResponseCreationSupport.sendStaticContent(
214                    request, channel, requestPath -> contentLoader
215                        .getResource(contentPath + "/" + path),
216                    maxAgeCalculator);
217            }
218            event.setResult(true);
219            event.stop();
220            if (!success) {
221                channel.respond(new Close());
222            }
223        });
224    }
225
226    /**
227     * Render a response using the given template. Send the 
228     * {@link Response} and at least one {@link Output} event.
229     *
230     * If the method does not return `true`, the invoker should close
231     * the connection because it is most likely in an undefined state.
232     *
233     * @param event the request event
234     * @param channel the channel
235     * @param tpl the template
236     * @return false if an error occurred
237     */
238    protected boolean sendProcessedTemplate(
239            Request event, IOSubchannel channel, Template tpl) {
240        // Prepare response
241        HttpResponse response = event.httpRequest().response().get();
242        MediaType mediaType = contentType(
243            ResponseCreationSupport.uriFromPath(tpl.getSourceName()));
244        response.setContentType(mediaType);
245
246        // Send response
247        response.setStatus(HttpStatus.OK);
248        response.setField(HttpField.LAST_MODIFIED, Instant.now());
249        if (maxAgeCalculator == null) {
250            ResponseCreationSupport.setMaxAge(response,
251                (req, mtype) -> 0, event.httpRequest(), mediaType);
252        } else {
253            ResponseCreationSupport.setMaxAge(
254                response, maxAgeCalculator, event.httpRequest(), mediaType);
255        }
256        channel.respond(new Response(response));
257
258        // Send content
259        try (ByteBufferOutputStream bbos = new ByteBufferOutputStream(
260            channel, channel.responsePipeline());
261                Writer out = new OutputStreamWriter(
262                    bbos.suppressClose(), "utf-8")) {
263            Map<String, Object> model = fmSessionModel(
264                event.associated(Session.class));
265            tpl.setLocale((Locale) model.get("locale"));
266            tpl.process(model, out);
267            return true;
268        } catch (IOException | TemplateException e) {
269            // Too late to do anything about this (header was sent).
270            fire(new Error(event, e), channel);
271        }
272        return false;
273    }
274
275    /**
276     * Render a response using the template obtained from the config with
277     * {@link Configuration#getTemplate(String)} and the given path.
278     *
279     * @param event
280     *            the event
281     * @param channel
282     *            the channel
283     * @param path
284     *            the path
285     */
286    protected boolean sendProcessedTemplate(
287            Request event, IOSubchannel channel, String path) {
288        try {
289            // Get template (no need to continue if this fails).
290            Template tpl = freemarkerConfig().getTemplate(path);
291            return sendProcessedTemplate(event, channel, tpl);
292        } catch (IOException e) {
293            fire(new Error(event, e), channel);
294            return false;
295        }
296    }
297
298    /**
299     * Creates the configuration for freemarker template processing.
300     * 
301     * @return the configuration
302     */
303    protected Configuration freemarkerConfig() {
304        if (fmConfig == null) {
305            fmConfig = new Configuration(Configuration.VERSION_2_3_26);
306            fmConfig.setClassLoaderForTemplateLoading(
307                contentLoader, contentPath);
308            fmConfig.setDefaultEncoding("utf-8");
309            fmConfig.setTemplateExceptionHandler(
310                TemplateExceptionHandler.RETHROW_HANDLER);
311            fmConfig.setLogTemplateExceptions(false);
312        }
313        return fmConfig;
314    }
315
316    /**
317     * Build a freemarker model holding the information associated with the
318     * session.
319     * 
320     * This model provides: 
321     * 
322     * * The `locale` (of type {@link Locale}). 
323     * 
324     * * The `resourceBundle` (of type {@link ResourceBundle}). 
325     * 
326     * * A function "`_`" that looks up the given key in the 
327     *   resource bundle.
328     * 
329     * @param session
330     *            the session
331     * @return the model
332     */
333    protected Map<String, Object> fmSessionModel(Optional<Session> session) {
334        @SuppressWarnings("PMD.UseConcurrentHashMap")
335        final Map<String, Object> model = new HashMap<>();
336        Locale locale = session.map(
337            sess -> sess.locale()).orElse(Locale.getDefault());
338        model.put("locale", locale);
339        final ResourceBundle resourceBundle = resourceBundle(locale);
340        model.put("resourceBundle", resourceBundle);
341        model.put("_", new TemplateMethodModelEx() {
342            @Override
343            @SuppressWarnings("PMD.EmptyCatchBlock")
344            public Object exec(@SuppressWarnings("rawtypes") List arguments)
345                    throws TemplateModelException {
346                @SuppressWarnings("unchecked")
347                List<TemplateModel> args = (List<TemplateModel>) arguments;
348                if (!(args.get(0) instanceof SimpleScalar)) {
349                    throw new TemplateModelException("Not a string.");
350                }
351                String key = ((SimpleScalar) args.get(0)).getAsString();
352                try {
353                    return resourceBundle.getString(key);
354                } catch (MissingResourceException e) {
355                    // no luck
356                }
357                return key;
358            }
359        });
360        return model;
361    }

363    /**
364     * Used to get the content type when generating a response with
365     * {@link #sendProcessedTemplate(Request, IOSubchannel, Template)}. May be
366     * overridden by derived classes. This implementation simply invokes
367     * {@link HttpResponse#contentType(URI)}.
368     *
369     * @param request the request
370     * @return the content type
371     */
372    protected MediaType contentType(URI request) {
373        return HttpResponse.contentType(request);
374    }
375
376    /**
377     * Provides a resource bundle for localization.
378     * The default implementation looks up a bundle using the
379     * package name plus "l10n" as base name.
380     * 
381     * @return the resource bundle
382     */
383    protected ResourceBundle resourceBundle(Locale locale) {
384        return ResourceBundle.getBundle(
385            contentPath.replace('/', '.') + ".l10n", locale,
386            contentLoader, ResourceBundle.Control.getNoFallbackControl(
387                ResourceBundle.Control.FORMAT_DEFAULT));
388    }
