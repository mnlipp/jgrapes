036/**
037 * The base class for all HTTP requests such as {@link Request.In.Get},
038 * {@link Request.In.Post} etc.

040 * @param <R> the generic type

044    /**
045     * @param channels
046     */

049    }

051    /**
052     * The base class for all incoming HTTP requests. Incoming
053     * request flow downstream and are served by the framework. 
054     * 
055     * A result of `true` indicates that the request has been processed, 
056     * i.e. a response has been sent or will sent.
057     */
058    @SuppressWarnings("PMD.ShortClassName")
059    public static class In extends Request<Boolean> {
060
061        @SuppressWarnings("PMD.AvoidFieldNameMatchingTypeName")
062        private final HttpRequest request;
063        private int matchLevels;
064        private MatchValue matchValue;
065        private URI matchUri;
066        private URI uri;
067
068        /**
069         * Creates a new request event with the associated {@link Completed}
070         * event.
071         * 
072         * @param protocol the protocol as reported by {@link #requestUri()}
073         * @param request the request data
074         * @param matchLevels the number of elements from the request path
075         * to use in the match value (see {@link #matchValue})
076         * @param channels the channels associated with this event
077         */
078        @SuppressWarnings("PMD.UselessParentheses")
079        public In(String protocol, HttpRequest request,
080                int matchLevels, Channel... channels) {
081            super(channels);
082            new Completed(this);
083            this.request = request;
084            this.matchLevels = matchLevels;
085            try {
086                URI headerInfo = new URI(protocol, null,
087                    request.host(), request.port(), null, null, null);
088                setRequestUri(headerInfo.resolve(request.requestUri()));
089                Iterator<String> segs = PathSpliterator.stream(
090                    requestUri().getPath()).skip(1).iterator();
091                StringBuilder pattern = new StringBuilder(20);
092                for (int i = 0; i < matchLevels && segs.hasNext(); i++) {
093                    pattern.append('/').append(segs.next());
094                }
095                if (segs.hasNext()) {
096                    pattern.append("/â€¦");
097                }
098                String matchPath = pattern.toString();
099                URI uri = requestUri();
100                matchUri = new URI(uri.getScheme(), null, uri.getHost(),
101                    uri.getPort(), uri.getPath(), null, null);
102                matchValue = new MatchValue(getClass(),
103                    (uri.getScheme() == null ? ""
104                        : (uri.getScheme() + "://"))
105                        + (uri.getHost() == null ? ""
106                            : (uri.getHost() + (uri.getPort() == -1 ? ""
107                                : (":" + uri.getPort()))))
108                        + matchPath);
109            } catch (URISyntaxException e) {
110                throw new IllegalArgumentException(e);
111            }
112        }
113
114        /**
115         * Creates the appropriate derived request event type from
116         * a given {@link HttpRequest}.
117         * 
118         * @param request the HTTP request
119         * @param secure whether the request was received over a secure channel
120         * @param matchLevels the match levels
121         * @return the request event
122         */
123        @SuppressWarnings("PMD.AvoidDuplicateLiterals")
124        public static In fromHttpRequest(
125                HttpRequest request, boolean secure, int matchLevels) {
126            switch (request.method()) {
127            case "OPTIONS":
128                return new Options(request, secure, matchLevels);
129            case "GET":
130                return new Get(request, secure, matchLevels);
131            case "HEAD":
132                return new Head(request, secure, matchLevels);
133            case "POST":
134                return new Post(request, secure, matchLevels);
135            case "PUT":
136                return new Put(request, secure, matchLevels);
137            case "DELETE":
138                return new Delete(request, secure, matchLevels);
139            case "TRACE":
140                return new Trace(request, secure, matchLevels);
141            case "CONNECT":
142                return new Connect(request, secure, matchLevels);
143            default:
144                return new In(secure ? "https" : "http", request, matchLevels);
145            }
146        }

176        /**
177         * The match value consists of the event class and a URI.
178         * The URI is similar to the request URI but its path elements
179         * are shortened as specified in the constructor.
180         * 
181         * The match value is used as key in a map that speeds up
182         * the lookup of handlers. Having the complete URI in the match
183         * value would inflate this map.
184         *
185         * @return the object
186         * @see org.jgrapes.core.Event#defaultCriterion()
187         */
188        @Override
189        public Object defaultCriterion() {
190            return matchValue;
191        }
192
193        /*
194         * (non-Javadoc)
195         * 
196         * @see org.jgrapes.core.Event#isMatchedBy(java.lang.Object)
197         */
198        @Override
199        public boolean isEligibleFor(Object value) {
200            if (!(value instanceof MatchValue)) {
201                return super.isEligibleFor(value);
202            }
203            MatchValue mval = (MatchValue) value;
204            if (!mval.type.isAssignableFrom(matchValue.type)) {
205                return false;
206            }
207            if (mval.resource instanceof ResourcePattern) {
208                return ((ResourcePattern) mval.resource).matches(matchUri,
209                    matchLevels) >= 0;
210            }
211            return mval.resource.equals(matchValue.resource);
212        }
213
214        /*
215         * (non-Javadoc)
216         * 
217         * @see java.lang.Object#toString()
218         */
219        @Override
220        @SuppressWarnings("PMD.AvoidLiteralsInIfCondition")
221        public String toString() {
222            StringBuilder builder = new StringBuilder();
223            builder.append(Components.objectName(this))
224                .append(" [\"");
225            String path = request.requestUri().getPath();
226            if (path.length() > 15) {
227                builder.append("...")
228                    .append(path.substring(path.length() - 12));
229            } else {
230                builder.append(path);
231            }
232            builder.append('\"');
233            if (channels().length > 0) {
234                builder.append(", channels=");
235                builder.append(Channel.toString(channels()));
236            }
237            builder.append(']');
238            return builder.toString();
239        }
240
241        /**
242         * Creates the match value.
243         *
244         * @param type the type
245         * @param resource the resource
246         * @return the object
247         */
248        public static Object createMatchValue(
249                Class<?> type, ResourcePattern resource) {
250            return new MatchValue(type, resource);
251        }
252
253        /**
254         * Represents a match value.
255         */
256        private static class MatchValue {
257            private final Class<?> type;
258            private final Object resource;
259
260            /**
261             * @param type
262             * @param resource
263             */
264            public MatchValue(Class<?> type, Object resource) {
265                super();
266                this.type = type;
267                this.resource = resource;
268            }
269
270            /*
271             * (non-Javadoc)
272             * 
273             * @see java.lang.Object#hashCode()
274             */
275            @Override
276            @SuppressWarnings("PMD.DataflowAnomalyAnalysis")
277            public int hashCode() {
278                @SuppressWarnings("PMD.AvoidFinalLocalVariable")
279                final int prime = 31;
280                int result = 1;
281                result = prime * result
282                    + ((resource == null) ? 0 : resource.hashCode());
283                result
284                    = prime * result + ((type == null) ? 0 : type.hashCode());
285                return result;
286            }
287
288            /*
289             * (non-Javadoc)
290             * 
291             * @see java.lang.Object#equals(java.lang.Object)
292             */
293            @Override
294            public boolean equals(Object obj) {
295                if (this == obj) {
296                    return true;
297                }
298                if (obj == null) {
299                    return false;
300                }
301                if (getClass() != obj.getClass()) {
302                    return false;
303                }
304                MatchValue other = (MatchValue) obj;
305                if (resource == null) {
306                    if (other.resource != null) {
307                        return false;
308                    }
309                } else if (!resource.equals(other.resource)) {
310                    return false;
311                }
312                if (type == null) {
313                    if (other.type != null) {
314                        return false;
315                    }
316                } else if (!type.equals(other.type)) {
317                    return false;
318                }
319                return true;
320            }

322        }

528    /**
529     * The base class for all outgoing HTTP requests. Outgoing
530     * request flow upstream and are served externally. 
531     * 
532     * A result of `true` indicates that the request has been processed, 
533     * i.e. a response has been sent or will sent.
534     */
535    @SuppressWarnings("PMD.ShortClassName")
536    public static class Out extends Request<Void> {
537
538        private HttpRequest request;
539        private BiConsumer<Request.Out, TcpChannel> connectedCallback;
540
541        /**
542         * Instantiates a new request.
543         *
544         * @param method the method
545         * @param url the url
546         */
547        public Out(String method, URL url) {
548            try {
549                request = new HttpRequest(method, url.toURI(),
550                    HttpProtocol.HTTP_1_1, false);
551            } catch (URISyntaxException e) {
552                // This should not happen because every valid URL can be
553                // converted to a URI.
554                throw new IllegalArgumentException(e);
555            }
556        }
557
558        /**
559         * Sets a "connected callback". When the {@link Out} event is
560         * created, the network connection is not yet known. Some
561         * header fields' values, however, need e.g. the port information
562         * from the connection. Therefore a callback may be set which is
563         * invoked when the connection has been obtained that will be used
564         * to send the request.
565         *
566         * @param connectedCallback the connected callback
567         * @return the out
568         */
569        public Out setConnectedCallback(
570                BiConsumer<Request.Out, TcpChannel> connectedCallback) {
571            this.connectedCallback = connectedCallback;
572            return this;
573        }
574
575        /**
576         * Returns the connected callback.
577         *
578         * @return the connected callback, if set
579         */
580        public Optional<BiConsumer<Request.Out, TcpChannel>>
581                connectedCallback() {
582            return Optional.ofNullable(connectedCallback);
583        }
584
585        /**
586         * The HTTP request that will be sent by the event.
587         *
588         * @return the http request
589         */
590        public HttpRequest httpRequest() {
591            return request;
592        }
593
594        /**
595         * Returns an absolute URI of the request.
596         * 
597         * @return the URI
598         */
599        public URI requestUri() {
600            return request.requestUri();
601        }
602
603        /*
604         * (non-Javadoc)
605         * 
606         * @see java.lang.Object#toString()
607         */
608        @Override
609        public String toString() {
610            StringBuilder builder = new StringBuilder();
611            builder.append(Components.objectName(this))
612                .append(" [").append(request.toString());
613            if (channels().length > 0) {
614                builder.append(", channels=");
615                builder.append(Channel.toString(channels()));
616            }
617            builder.append(']');
618            return builder.toString();
619        }
