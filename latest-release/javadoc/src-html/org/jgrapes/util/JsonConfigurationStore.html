019package org.jgrapes.util;
020
021import java.io.File;
022import java.io.IOException;
023import java.io.InputStreamReader;
024import java.io.OutputStreamWriter;
025import java.io.Reader;
026import java.io.Writer;
027import java.nio.file.Files;
028import java.util.HashMap;
029import java.util.Map;
030import java.util.Optional;
031import java.util.StringTokenizer;
032import java.util.prefs.BackingStoreException;
033import org.jdrupes.json.JsonBeanDecoder;
034import org.jdrupes.json.JsonBeanEncoder;
035import org.jdrupes.json.JsonDecodeException;
036import org.jgrapes.core.Channel;
037import org.jgrapes.core.Component;
038import org.jgrapes.core.EventPipeline;
039import org.jgrapes.core.annotation.Handler;
040import org.jgrapes.core.events.Start;
041import org.jgrapes.util.events.ConfigurationUpdate;
042import org.jgrapes.util.events.InitialPreferences;
043
044/**
045 * This component provides a store for an application's configuration
046 * backed by a JSON file. The JSON object described by this file 
047 * represents the root directory. If an entry does not start with a
048 * slash, it represents the key of a key value pair. If it does
049 * starts with a slash, the value is another JSON object that
050 * describes the respective subdirectory.
051 * 
052 * The component reads the initial values from {@link File} passed
053 * to the constructor. During application bootstrap, it 
054 * intercepts the {@link Start} event using a handler with  priority 
055 * 999999. When receiving this event, it fires all known preferences 
056 * values on the channels of the start event as a 
057 * {@link InitialPreferences} event, using a new {@link EventPipeline}
058 * and waiting for its completion. Then, allows the intercepted 
059 * {@link Start} event to continue. 
060 * 
061 * Components that depend on configuration values define handlers
062 * for {@link ConfigurationUpdate} events and adapt themselves to the values 
063 * received. Note that due to the intercepted {@link Start} event, the initial
064 * preferences values are received before the {@link Start} event, so
065 * components' configurations can be rearranged before they actually
066 * start doing something.
067 *
068 * Besides initially publishing the stored preferences values,
069 * the component also listens for {@link ConfigurationUpdate} events
070 * on its channel and updates the JSON file (may be suppressed).
071 */
072public class JsonConfigurationStore extends Component {
073
074    private File file;
075    private Map<String, Object> cache;
076
077    /**
078     * Creates a new component with its channel set to the given 
079     * channel and the given file.
080     * 
081     * @param componentChannel the channel 
082     * @param file the file used to store the JSON
083     * @throws JsonDecodeException
084     */
085    public JsonConfigurationStore(Channel componentChannel, File file)
086            throws IOException {
087        this(componentChannel, file, true);
088    }
089
090    /**
091     * Creates a new component with its channel set to the given 
092     * channel and the given file.
093     * 
094     * @param componentChannel the channel 
095     * @param file the file used to store the JSON
096     * @throws JsonDecodeException
097     */
098    @SuppressWarnings("PMD.ShortVariable")
099    public JsonConfigurationStore(Channel componentChannel, File file,
100            boolean update) throws IOException {
101        super(componentChannel);
102        if (update) {
103            Handler.Evaluator.add(this, "onConfigurationUpdate",
104                channel().defaultCriterion());
105        }
106        this.file = file;
107        if (!file.exists()) {
108            cache = new HashMap<>();
109        }
110        try (Reader in = new InputStreamReader(
111            Files.newInputStream(file.toPath()), "utf-8")) {
112            @SuppressWarnings("unchecked")
113            Map<String, Object> confCache
114                = (Map<String, Object>) JsonBeanDecoder.create(in).readObject();
115            cache = confCache;
116        } catch (JsonDecodeException e) {
117            throw new IOException(e);
118        }
119    }
120
121    /**
122     * Intercepts the {@link Start} event and fires a
123     * {@link ConfigurationUpdate} event.
124     *
125     * @param event the event
126     * @throws BackingStoreException the backing store exception
127     * @throws InterruptedException the interrupted exception
128     */
129    @Handler(priority = 999999, channels = Channel.class)
130    public void onStart(Start event)
131            throws BackingStoreException, InterruptedException {
132        ConfigurationUpdate updEvt = new ConfigurationUpdate();
133        addPrefs(updEvt, "/", cache);
134        newEventPipeline().fire(updEvt, event.channels()).get();
135    }
136
137    private void addPrefs(
138            ConfigurationUpdate updEvt, String path, Map<String, ?> map) {
139        for (Map.Entry<String, ?> e : map.entrySet()) {
140            if (e.getValue() instanceof Map) {
141                @SuppressWarnings("unchecked")
142                Map<String, ?> value = (Map<String, ?>) e.getValue();
143                addPrefs(updEvt, ("/".equals(path) ? "" : path)
144                    + e.getKey(), value);
145                continue;
146            }
147            updEvt.add(path, e.getKey(), e.getValue().toString());
148        }
149    }
150
151    /**
152     * Merges and saves configuration updates.
153     *
154     * @param event the event
155     * @throws IOException Signals that an I/O exception has occurred.
156     */
157    @Handler(dynamic = true)
158    @SuppressWarnings({ "PMD.DataflowAnomalyAnalysis",
159        "PMD.AvoidLiteralsInIfCondition",
160        "PMD.AvoidInstantiatingObjectsInLoops" })
161    public void onConfigurationUpdate(ConfigurationUpdate event)
162            throws IOException {
163        boolean changed = false;
164        for (String path : event.paths()) {
165            if ("/".equals(path) && !event.values(path).isPresent()) {
166                // Special case, "remove root", i.e. all configuration data
167                cache.clear();
168                changed = true;
169            }
170            changed = changed || handleSegment(cache,
171                new StringTokenizer(path, "/"), event.values(path));
172        }
173        if (changed) {
174            try (Writer out = new OutputStreamWriter(
175                Files.newOutputStream(file.toPath()), "utf-8");
176                    JsonBeanEncoder enc = JsonBeanEncoder.create(out)) {
177                enc.writeObject(cache);
178            }
179        }
180    }
181
182    @SuppressWarnings("PMD.DataflowAnomalyAnalysis")
183    private boolean handleSegment(Map<String, Object> currentMap,
184            StringTokenizer tokenizer, Optional<Map<String, String>> values) {
185        if (!tokenizer.hasMoreTokens()) {
186            // "Leave" map
187            return mergeValues(currentMap, values.get());
188        }
189        boolean changed = false;
190        String nextSegment = "/" + tokenizer.nextToken();
191        if (!tokenizer.hasMoreTokens() && !values.isPresent()) {
192            // Next segment is last segment from path and we must remove
193            if (currentMap.containsKey(nextSegment)) {
194                // Delete sub-map.
195                currentMap.remove(nextSegment);
196                changed = true;
197            }
198            return changed;
199        }
200        // Check if next map exists
201        @SuppressWarnings("unchecked")
202        Map<String, Object> nextMap
203            = (Map<String, Object>) currentMap.get(nextSegment);
204        if (nextMap == null) {
205            // Doesn't exist, new sub-map
206            changed = true;
207            nextMap = new HashMap<>();
208            currentMap.put(nextSegment, nextMap);
209        }
210        // Continue with sub-map
211        return changed || handleSegment(nextMap, tokenizer, values);
212    }
213
214    @SuppressWarnings("PMD.DataflowAnomalyAnalysis")
215    private boolean mergeValues(Map<String, Object> currentMap,
216            Map<String, String> values) {
217        boolean changed = false;
218        for (Map.Entry<String, String> e : values.entrySet()) {
219            if (e.getValue() == null) {
220                // Delete from map
221                if (currentMap.containsKey(e.getKey())) {
222                    currentMap.remove(e.getKey());
223                    changed = true;
224                }
225                continue;
226            }
227            String oldValue = Optional.ofNullable(currentMap.get(e.getKey()))
228                .map(val -> val.toString()).orElse(null);
229            if (oldValue == null || !e.getValue().equals(oldValue)) {
230                currentMap.put(e.getKey(), e.getValue());
231                changed = true;
232            }
233        }
234        return changed;
235    }
236}
