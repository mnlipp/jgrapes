019package org.jgrapes.util.events;
020
021import java.util.Map;
022import org.jgrapes.core.Channel;
023import org.jgrapes.core.CompletionEvent;
024
025/**
026 * Defines an optional completion event for a {@link KeyValueStoreQuery}.
027 * 
028 * Because {@link KeyValueStoreQuery} events have a result, this event
029 * is not automatically generated for every {@link KeyValueStoreQuery}.
030 * Rather, it must be created explicitly.
031 */
032public class KeyValueStoreData extends CompletionEvent<KeyValueStoreQuery> {
033
034    /**
035     * @param monitoredEvent
036     * @param channels
037     */
038    public KeyValueStoreData(KeyValueStoreQuery monitoredEvent,
039            Channel... channels) {
040        super(monitoredEvent, channels);
041    }
042
043    /**
044     * A shortcut to get the result of the completed query event.
045     * 
046     * @return the data
047     */
048    public Map<String, String> data() {
049        try {
050            return event().get();
051        } catch (InterruptedException e) {
052            // Can only happen if invoked before completion
053            throw new IllegalStateException(e);
054        }
055    }
056}
