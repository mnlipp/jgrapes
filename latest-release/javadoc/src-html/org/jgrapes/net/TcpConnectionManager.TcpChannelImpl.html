049/**
050 * Provides a base class for the {@link TcpServer} and the {@link TcpConnector}.
051 */
052@SuppressWarnings({ "PMD.ExcessiveImports", "PMD.ExcessivePublicCount",
053    "PMD.NcssCount", "PMD.EmptyCatchBlock", "PMD.AvoidDuplicateLiterals",
054    "PMD.ExcessiveClassLength" })
055public abstract class TcpConnectionManager extends Component {
056
057    private int bufferSize;
058    protected final Set<TcpChannelImpl> channels = new HashSet<>();
059    private ExecutorService executorService;
060
061    /**
062     * Creates a new server using the given channel.
063     * 
064     * @param componentChannel the component's channel
065     */
066    public TcpConnectionManager(Channel componentChannel) {
067        super(componentChannel);
068    }

070    /**
071     * Sets the buffer size for the send an receive buffers.
072     * If no size is set, the system defaults will be used.
073     * 
074     * @param bufferSize the size to use for the send and receive buffers
075     * @return the TCP connection manager for easy chaining
076     */
077    public TcpConnectionManager setBufferSize(int bufferSize) {
078        this.bufferSize = bufferSize;
079        return this;
080    }

082    /**
083     * Return the configured buffer size.
084     *
085     * @return the bufferSize
086     */
087    public int bufferSize() {
088        return bufferSize;
089    }

091    /**
092     * Sets an executor service to be used by the event pipelines
093     * that process the data from the network. Setting this
094     * to an executor service with a limited number of threads
095     * allows to control the maximum load from the network.
096     * 
097     * @param executorService the executorService to set
098     * @return the TCP connection manager for easy chaining
099     * @see Manager#newEventPipeline(ExecutorService)
100     */
101    public TcpConnectionManager
102            setExecutorService(ExecutorService executorService) {
103        this.executorService = executorService;
104        return this;
105    }

107    /**
108     * Returns the executor service.
109     *
110     * @return the executorService
111     */
112    public ExecutorService executorService() {
113        return executorService;
114    }

116    /**
117     * Writes the data passed in the event. 
118     * 
119     * The end of record flag is used to determine if a channel is 
120     * eligible for purging. If the flag is set and all output has 
121     * been processed, the channel is purgeable until input is 
122     * received or another output event causes the state to be 
123     * reevaluated. 
124     *
125     * @param event the event
126     * @param channel the channel
127     * @throws InterruptedException the interrupted exception
128     */
129    @Handler
130    public void onOutput(Output<ByteBuffer> event,
131            TcpChannelImpl channel) throws InterruptedException {
132        if (channels.contains(channel)) {
133            channel.write(event);
134        }
135    }
136
137    /**
138     * Removes the channel from the set of registered channels.
139     *
140     * @param channel the channel
141     * @return true, if channel was registered
142     */
143    protected boolean removeChannel(TcpChannelImpl channel) {
144        synchronized (channels) {
145            return channels.remove(channel);
146        }
147    }

159    /**
160     * The close state.
161     */
162    private enum ConnectionState {
163        OPEN, DELAYED_EVENT, DELAYED_REQUEST, HALF_CLOSED, CLOSED
164    }
165
166    /**
167     * The purgeable state.
168     */
169    private enum PurgeableState {
170        NO, PENDING, YES
171    }

173    /**
174     * The internal representation of a connection. 
175     */
176    protected class TcpChannelImpl
177            extends DefaultIOSubchannel implements NioHandler, TcpChannel {
178
179        private final SocketChannel nioChannel;
180        private final SocketAddress localAddress;
181        private final SocketAddress remoteAddress;
182        private EventPipeline downPipeline;
183        private final ManagedBufferPool<ManagedBuffer<ByteBuffer>,
184                ByteBuffer> readBuffers;
185        private Registration registration;
186        private int selectionKeys;
187        private final Queue<
188                ManagedBuffer<ByteBuffer>.ByteBufferView> pendingWrites
189                    = new ArrayDeque<>();
190        private ConnectionState connState = ConnectionState.OPEN;
191        private PurgeableState purgeable = PurgeableState.NO;
192        private long becamePurgeableAt;
193
194        /**
195         * @param nioChannel the channel
196         * @throws IOException if an I/O error occured
197         */
198        public TcpChannelImpl(SocketChannel nioChannel) throws IOException {
199            super(channel(), newEventPipeline());
200            this.nioChannel = nioChannel;
201            // Copy, because they are only available while channel is open.
202            localAddress = nioChannel.getLocalAddress();
203            remoteAddress = nioChannel.getRemoteAddress();
204            if (executorService == null) {
205                downPipeline = newEventPipeline();
206            } else {
207                downPipeline = newEventPipeline(executorService);
208            }
209            String channelName
210                = Components.objectName(TcpConnectionManager.this)
211                    + "." + Components.objectName(this);
212
213            // Prepare write buffers
214            int writeBufferSize = bufferSize == 0
215                ? nioChannel.socket().getSendBufferSize()
216                : bufferSize;
217            setByteBufferPool(new ManagedBufferPool<>(ManagedBuffer::new,
218                () -> {
219                    return ByteBuffer.allocate(writeBufferSize);
220                }, 2)
221                    .setName(channelName + ".upstream.buffers"));
222
223            // Prepare read buffers
224            int readBufferSize = bufferSize == 0
225                ? nioChannel.socket().getReceiveBufferSize()
226                : bufferSize;
227            readBuffers = new ManagedBufferPool<>(ManagedBuffer::new,
228                () -> {
229                    return ByteBuffer.allocate(readBufferSize);
230                }, 2)
231                    .setName(channelName + ".downstream.buffers");
232
233            // Register with dispatcher
234            nioChannel.configureBlocking(false);
235            TcpConnectionManager.this.fire(
236                new NioRegistration(this, nioChannel, 0,
237                    TcpConnectionManager.this),
238                Channel.BROADCAST);
239        }

279        /**

281         * 
282         * @param event the completed event
283         * @throws InterruptedException if the execution was interrupted
284         * @throws IOException if an I/O error occurred
285         */
286        public void registrationComplete(NioRegistration event)
287                throws InterruptedException, IOException {
288            registration = event.get();
289            selectionKeys |= SelectionKey.OP_READ;
290            registration.updateInterested(selectionKeys);
291        }

293        /**
294         * Checks if is purgeable.
295         *
296         * @return true, if is purgeable
297         */
298        public boolean isPurgeable() {
299            return purgeable == PurgeableState.YES;
300        }
301
302        /**
303         * Gets the the time when the connection became purgeable.
304         *
305         * @return the time
306         */
307        public long purgeableSince() {
308            return becamePurgeableAt;
309        }

311        /**
312         * Write the data on this channel.
313         * 
314         * @param event the event
315         */
316        public void write(Output<ByteBuffer> event)
317                throws InterruptedException {
318            synchronized (pendingWrites) {
319                if (!nioChannel.isOpen()) {
320                    return;
321                }
322                ManagedBuffer<ByteBuffer>.ByteBufferView reader
323                    = event.buffer().newByteBufferView();
324                if (!pendingWrites.isEmpty()) {
325                    reader.managedBuffer().lockBuffer();
326                    purgeable = event.isEndOfRecord() ? PurgeableState.PENDING
327                        : PurgeableState.NO;
328                    pendingWrites.add(reader);
329                    return;
330                }
331                try {
332                    nioChannel.write(reader.get());
333                } catch (IOException e) {
334                    forceClose(e);
335                    return;
336                }
337                if (!reader.get().hasRemaining()) {
338                    if (event.isEndOfRecord()) {
339                        becamePurgeableAt = System.currentTimeMillis();
340                        purgeable = PurgeableState.YES;
341                    } else {
342                        purgeable = PurgeableState.NO;
343                    }
344                    return;
345                }
346                reader.managedBuffer().lockBuffer();
347                purgeable = event.isEndOfRecord() ? PurgeableState.PENDING
348                    : PurgeableState.NO;
349                pendingWrites.add(reader);
350                selectionKeys |= SelectionKey.OP_WRITE;
351                registration.updateInterested(selectionKeys);
352            }
353        }
354
355        @Override
356        public void handleOps(int ops) throws InterruptedException {
357            if ((ops & SelectionKey.OP_READ) != 0) {
358                handleReadOp();
359            }
360            if ((ops & SelectionKey.OP_WRITE) != 0) {
361                handleWriteOp();
362            }
363        }
364
365        /**
366         * Gets a buffer from the pool and reads available data into it.
367         * Sends the result as event. 
368         * 
369         * @throws InterruptedException
370         * @throws IOException
371         */
372        @SuppressWarnings("PMD.EmptyCatchBlock")
373        private void handleReadOp() throws InterruptedException {
374            ManagedBuffer<ByteBuffer> buffer;
375            buffer = readBuffers.acquire();
376            try {
377                int bytes = buffer.fillFromChannel(nioChannel);
378                if (bytes == 0) {
379                    buffer.unlockBuffer();
380                    return;
381                }
382                if (bytes > 0) {
383                    purgeable = PurgeableState.NO;
384                    downPipeline.fire(Input.fromSink(buffer, false), this);
385                    return;
386                }
387            } catch (IOException e) {
388                // Buffer already unlocked by fillFromChannel
389                forceClose(e);
390                return;
391            }
392            // EOF (-1) from other end
393            buffer.unlockBuffer();
394            synchronized (nioChannel) {
395                if (connState == ConnectionState.HALF_CLOSED) {
396                    // Other end confirms our close, complete close
397                    try {
398                        nioChannel.close();
399                    } catch (IOException e) {
400                        // Ignored for close
401                    }
402                    connState = ConnectionState.CLOSED;
403                    downPipeline.fire(new Closed(), this);
404                    return;
405                }
406            }
407            // Other end initiates close
408            selectionKeys &= ~SelectionKey.OP_READ;
409            registration.updateInterested(selectionKeys);
410            downPipeline.submit("SendHalfClosed", () -> {
411                try {
412                    // Inform downstream and wait until everything has settled.
413                    newEventPipeline().fire(new HalfClosed(), this).get();
414                    // All settled.
415                    removeChannel(this);
416                    downPipeline.fire(new Closed(), this);
417                    // Close our end if everything has been written.
418                    synchronized (pendingWrites) {
419                        synchronized (nioChannel) {
420                            try {
421                                if (!pendingWrites.isEmpty()) {
422                                    // Pending writes, delay close
423                                    connState = ConnectionState.DELAYED_REQUEST;
424                                    return;
425                                }
426                                // Nothing left to do, close
427                                nioChannel.close();
428                                connState = ConnectionState.CLOSED;
429                            } catch (IOException e) {
430                                // Ignored for close
431                            }
432                        }
433                    }
434                } catch (InterruptedException e) {
435                    // Nothing to do about this
436                }
437            });
438        }
439
440        /**
441         * Checks if there is still data to be written. This may be
442         * a left over in an incompletely written buffer or a complete
443         * pending buffer. 
444         * 
445         * @throws IOException
446         * @throws InterruptedException 
447         */
448        @SuppressWarnings({ "PMD.DataflowAnomalyAnalysis",
449            "PMD.EmptyCatchBlock",
450            "PMD.AvoidBranchingStatementAsLastInLoop" })
451        private void handleWriteOp() throws InterruptedException {
452            while (true) {
453                ManagedBuffer<ByteBuffer>.ByteBufferView head = null;
454                synchronized (pendingWrites) {
455                    if (pendingWrites.isEmpty()) {
456                        // Nothing left to write, stop getting ops
457                        selectionKeys &= ~SelectionKey.OP_WRITE;
458                        registration.updateInterested(selectionKeys);
459                        // Was the connection closed while we were writing?
460                        if (connState == ConnectionState.DELAYED_REQUEST
461                            || connState == ConnectionState.DELAYED_EVENT) {
462                            synchronized (nioChannel) {
463                                try {
464                                    if (connState == ConnectionState.DELAYED_REQUEST) {
465                                        // Delayed close request from other end,
466                                        // complete
467                                        nioChannel.close();
468                                        connState = ConnectionState.CLOSED;
469                                    }
470                                    if (connState == ConnectionState.DELAYED_EVENT) {
471                                        // Delayed close from this end, initiate
472                                        nioChannel.shutdownOutput();
473                                        connState = ConnectionState.HALF_CLOSED;
474                                    }
475                                } catch (IOException e) {
476                                    // Ignored for close
477                                }
478                            }
479                        } else {
480                            if (purgeable == PurgeableState.PENDING) {
481                                purgeable = PurgeableState.YES;
482                            }
483                        }
484                        break; // Nothing left to do
485                    }
486                    head = pendingWrites.peek();
487                    if (!head.get().hasRemaining()) {
488                        // Nothing left in head buffer, try next
489                        head.managedBuffer().unlockBuffer();
490                        pendingWrites.remove();
491                        continue;
492                    }
493                }
494                try {
495                    nioChannel.write(head.get()); // write...
496                } catch (IOException e) {
497                    forceClose(e);
498                    return;
499                }
500                break; // ... and wait for next op
501            }
502        }
503
504        /**
505         * Closes this channel.
506         * 
507         * @throws IOException if an error occurs
508         * @throws InterruptedException if the execution was interrupted 
509         */
510        public void close() throws IOException, InterruptedException {
511            if (!removeChannel(this)) {
512                return;
513            }
514            synchronized (pendingWrites) {
515                if (!pendingWrites.isEmpty()) {
516                    // Pending writes, delay close until done
517                    connState = ConnectionState.DELAYED_EVENT;
518                    return;
519                }
520                // Nothing left to do, proceed
521                synchronized (nioChannel) {
522                    if (nioChannel.isOpen()) {
523                        // Initiate close, must be confirmed by other end
524                        nioChannel.shutdownOutput();
525                        connState = ConnectionState.HALF_CLOSED;
526                    }
527                }
528            }
529        }
530
531        @SuppressWarnings("PMD.EmptyCatchBlock")
532        private void forceClose(Throwable error) throws InterruptedException {
533            try {
534                nioChannel.close();
535                connState = ConnectionState.CLOSED;
536            } catch (IOException e) {
537                // Closed only to make sure, any failure can be ignored.
538            }
539            if (removeChannel(this)) {
540                Closed evt = new Closed(error);
541                downPipeline.fire(evt, this);
542            }
543        }
