081/**
082 * Provides a TCP server. The server binds to the given address. If the
083 * address is {@code null}, address and port are automatically assigned.
084 * The port may be overwritten by a configuration event
085 * (see {@link #onConfigurationUpdate(ConfigurationUpdate)}).
086 * 
087 * For each established connection, the server creates a new
088 * {@link LinkedIOSubchannel}. The servers basic operation is to
089 * fire {@link Input} (and {@link Closed}) events on the
090 * appropriate subchannel in response to data received from the
091 * network and to handle {@link Output} (and {@link Close}) events 
092 * on the subchannel and forward the information to the network
093 * connection.
094 * 
095 * The server supports limiting the number of concurrent connections
096 * with a {@link PermitsPool}. If such a pool is set as connection
097 * limiter (see {@link #setConnectionLimiter(PermitsPool)}), a
098 * permit is acquired for each new connection attempt. If no more
099 * permits are available, the server sends a {@link Purge} event on
100 * each channel that is purgeable for at least the time span
101 * set with {@link #setMinimalPurgeableTime(long)}. Purgeability 
102 * is derived from the end of record flag of {@link Output} events
103 * (see {@link #onOutput(Output, TcpChannel)}. When using this feature, 
104 * make sure that connections are either short lived or the application
105 * level components support the {@link Purge} event. Else, it may become
106 * impossible to establish new connections.
107 */
108@SuppressWarnings({ "PMD.ExcessiveImports", "PMD.ExcessivePublicCount",

110    "PMD.ExcessiveClassLength" })

125    /**
126     * The purger thread.
127     */
128    private class Purger extends Thread implements AvailabilityListener {
129
130        private boolean permitsAvailable = true;
131
132        /**
133         * Instantiates a new purger.
134         */
135        public Purger() {
136            setName(Components.simpleObjectName(this));
137            setDaemon(true);
138        }
139
140        @Override
141        public void availabilityChanged(PermitsPool pool, boolean available) {
142            if (registration == null) {
143                return;
144            }
145            synchronized (this) {
146                permitsAvailable = available;
147                registration.updateInterested(
148                    permitsAvailable ? SelectionKey.OP_ACCEPT : 0);
149                if (!permitsAvailable) {
150                    this.notifyAll();
151                }
152            }
153        }
154
155        @Override
156        @SuppressWarnings({ "PMD.AvoidInstantiatingObjectsInLoops",
157            "PMD.DataflowAnomalyAnalysis" })
158        public void run() {
159            if (connLimiter == null) {
160                return;
161            }
162            try {
163                connLimiter.addListener(this);
164                while (serverSocketChannel.isOpen()) {
165                    synchronized (this) {
166                        while (permitsAvailable) {
167                            wait();
168                        }
169                    }
170                    // Copy to avoid ConcurrentModificationException
171                    List<TcpChannel> candidates;
172                    synchronized (channels) {
173                        candidates = new ArrayList<>(channels);
174                    }
175                    long purgeableSince
176                        = System.currentTimeMillis() - minimumPurgeableTime;
177                    candidates = candidates.stream()
178                        .filter(channel -> channel.isPurgeable()
179                            && channel.becamePurgeableAt < purgeableSince)
180                        .sorted(new Comparator<TcpChannel>() {
181                            @Override
182                            public int compare(TcpChannel c1, TcpChannel c2) {
183                                if (c1.becamePurgeableAt < c2.becamePurgeableAt) {
184                                    return 1;
185                                }
186                                if (c1.becamePurgeableAt > c2.becamePurgeableAt) {
187                                    return -1;
188                                }
189                                return 0;
190                            }
191                        })
192                        .collect(Collectors.toList());
193                    for (TcpChannel channel : candidates) {
194                        // Sorting may have taken time...
195                        if (!channel.isPurgeable()) {
196                            continue;
197                        }
198                        channel.downPipeline.fire(new Purge(), channel);
199                        // Continue only as long as necessary
200                        if (permitsAvailable) {
201                            break;
202                        }
203                    }
204                    sleep(1000);
205                }
206            } catch (InterruptedException e) {
207                // Fall through
208            } finally {
209                connLimiter.removeListener(this);
210            }
211        }
212
213    }

429     * Handles the successful channel registration.

431     * @param event the event
432     * @throws InterruptedException the interrupted exception
433     * @throws IOException Signals that an I/O exception has occurred.
434     */
435    @Handler(channels = Self.class)
436    public void onRegistered(NioRegistration.Completed event)
437            throws InterruptedException, IOException {
438        NioHandler handler = event.event().handler();
439        if (handler == this) {
440            if (event.event().get() == null) {
441                fire(new Error(event,
442                    "Registration failed, no NioDispatcher?"));
443                return;
444            }
445            registration = event.event().get();
446            purger = new Purger();
447            purger.start();
448            fire(new Ready(serverSocketChannel.getLocalAddress()));
449            return;
450        }
451        if (handler instanceof TcpChannel) {
452            ((TcpChannel) handler)
453                .registrationComplete(event.event());
454        }
455    }

457    /*
458     * (non-Javadoc)
459     * 
460     * @see org.jgrapes.io.NioSelectable#handleOps(int)
461     */
462    @Override
463    public void handleOps(int ops) {
464        if ((ops & SelectionKey.OP_ACCEPT) == 0 || closing) {
465            return;
466        }
467        synchronized (channels) {
468            if (connLimiter != null && !connLimiter.tryAcquire()) {
469                return;
470            }
471            try {
472                SocketChannel socketChannel = serverSocketChannel.accept();
473                channels.add(new TcpChannel(socketChannel));
474            } catch (IOException e) {
475                fire(new IOError(null, e));
476            }
477        }
478    }

480    private boolean removeChannel(TcpChannel channel) {
481        synchronized (channels) {
482            if (!channels.remove(channel)) {
483                // Closed already
484                return false;
485            }
486            // In case the server is shutting down
487            channels.notifyAll();
488        }
489        if (connLimiter != null) {
490            connLimiter.release();
491        }
492        return true;
493    }

516    /**
517     * Shuts down the server or one of the connections to the server.
518     *
519     * @param event the event
520     * @throws IOException if an I/O exception occurred
521     * @throws InterruptedException if the execution was interrupted
522     */
523    @Handler
524    @SuppressWarnings("PMD.DataflowAnomalyAnalysis")
525    public void onClose(Close event) throws IOException, InterruptedException {
526        boolean subOnly = true;
527        for (Channel channel : event.channels()) {
528            if (channel instanceof TcpChannel) {
529                if (channels.contains(channel)) {
530                    ((TcpChannel) channel).close();
531                }
532            } else {
533                subOnly = false;
534            }
535        }
536        if (subOnly || !serverSocketChannel.isOpen()) {
537            // Closed already
538            fire(new Closed());
539            return;
540        }
541        synchronized (channels) {
542            closing = true;
543            // Copy to avoid concurrent modification exception
544            Set<TcpChannel> conns = new HashSet<>(channels);
545            for (TcpChannel conn : conns) {
546                conn.close();
547            }
548            while (!channels.isEmpty()) {
549                channels.wait();
550            }
551        }
552        serverSocketChannel.close();
553        purger.interrupt();
554        closing = false;
555        fire(new Closed());
556    }

561     * in closing all open connections by the runtime system
562     * and thus in {@link Closed} events on all subchannels.

591    /**
592     * The internal representation of a connected client. 
593     */
594    private class TcpChannel
595            extends DefaultSubchannel implements NioHandler {
596
597        private final SocketChannel nioChannel;
598        private EventPipeline downPipeline;
599        private final ManagedBufferPool<ManagedBuffer<ByteBuffer>,
600                ByteBuffer> readBuffers;
601        private Registration registration;
602        private final Queue<
603                ManagedBuffer<ByteBuffer>.ByteBufferView> pendingWrites
604                    = new ArrayDeque<>();
605        private boolean pendingClose;
606        private PurgeableState purgeable = PurgeableState.NO;
607        private long becamePurgeableAt;
608
609        /**
610         * @param nioChannel the channel
611         * @throws IOException if an I/O error occured
612         */
613        public TcpChannel(SocketChannel nioChannel) throws IOException {
614            super(channel(), newEventPipeline());
615            this.nioChannel = nioChannel;
616            if (executorService == null) {
617                downPipeline = newEventPipeline();
618            } else {
619                downPipeline = newEventPipeline(executorService);
620            }
621
622            String channelName = Components.objectName(TcpServer.this)
623                + "." + Components.objectName(this);
624            int writeBufferSize = bufferSize == 0
625                ? nioChannel.socket().getSendBufferSize()
626                : bufferSize;
627            setByteBufferPool(new ManagedBufferPool<>(ManagedBuffer::new,
628                () -> {
629                    return ByteBuffer.allocate(writeBufferSize);
630                }, 2)
631                    .setName(channelName + ".upstream.buffers"));
632
633            int readBufferSize = bufferSize == 0
634                ? nioChannel.socket().getReceiveBufferSize()
635                : bufferSize;
636            readBuffers = new ManagedBufferPool<>(ManagedBuffer::new,
637                () -> {
638                    return ByteBuffer.allocate(readBufferSize);
639                }, 2)
640                    .setName(channelName + ".downstream.buffers");
641
642            // Register with dispatcher
643            nioChannel.configureBlocking(false);
644            TcpServer.this.fire(
645                new NioRegistration(this, nioChannel, 0, TcpServer.this),
646                Channel.BROADCAST);
647        }
648
649        /**
650         * Invoked when registration has completed.
651         * 
652         * @param event the completed event
653         * @throws InterruptedException if the execution was interrupted
654         * @throws IOException if an I/O error occurred
655         */
656        public void registrationComplete(NioRegistration event)
657                throws InterruptedException, IOException {
658            registration = event.get();
659            downPipeline.fire(new Accepted(nioChannel.getLocalAddress(),
660                nioChannel.getRemoteAddress(), false,
661                Collections.emptyList()), this);
662            registration.updateInterested(SelectionKey.OP_READ);
663
664        }
665
666        public boolean isPurgeable() {
667            return purgeable == PurgeableState.YES;
668        }
669
670        /**
671         * Write the data on this channel.
672         * 
673         * @param event the event
674         */
675        public void write(Output<ByteBuffer> event)
676                throws InterruptedException {
677            synchronized (pendingWrites) {
678                if (!nioChannel.isOpen()) {
679                    return;
680                }
681                ManagedBuffer<ByteBuffer>.ByteBufferView reader
682                    = event.buffer().newByteBufferView();
683                if (!pendingWrites.isEmpty()) {
684                    reader.managedBuffer().lockBuffer();
685                    purgeable = event.isEndOfRecord() ? PurgeableState.PENDING
686                        : PurgeableState.NO;
687                    pendingWrites.add(reader);
688                    return;
689                }
690                try {
691                    nioChannel.write(reader.get());
692                } catch (IOException e) {
693                    removeChannel(e);
694                    return;
695                }
696                if (!reader.get().hasRemaining()) {
697                    if (event.isEndOfRecord()) {
698                        becamePurgeableAt = System.currentTimeMillis();
699                        purgeable = PurgeableState.YES;
700                    } else {
701                        purgeable = PurgeableState.NO;
702                    }
703                    return;
704                }
705                reader.managedBuffer().lockBuffer();
706                purgeable = event.isEndOfRecord() ? PurgeableState.PENDING
707                    : PurgeableState.NO;
708                pendingWrites.add(reader);
709                registration.updateInterested(
710                    SelectionKey.OP_READ | SelectionKey.OP_WRITE);
711            }
712        }
713
714        @Override
715        public void handleOps(int ops) throws InterruptedException {
716            if ((ops & SelectionKey.OP_READ) != 0) {
717                handleReadOp();
718            }
719            if ((ops & SelectionKey.OP_WRITE) != 0) {
720                handleWriteOp();
721            }
722        }
723
724        /**
725         * Gets a buffer from the pool and reads available data into it.
726         * Sends the result as event. 
727         * 
728         * @throws InterruptedException
729         * @throws IOException
730         */
731        @SuppressWarnings("PMD.EmptyCatchBlock")
732        private void handleReadOp() throws InterruptedException {
733            ManagedBuffer<ByteBuffer> buffer;
734            buffer = readBuffers.acquire();
735            try {
736                int bytes = buffer.fillFromChannel(nioChannel);
737                if (bytes == 0) {
738                    buffer.unlockBuffer();
739                    return;
740                }
741                if (bytes > 0) {
742                    purgeable = PurgeableState.NO;
743                    downPipeline.fire(Input.fromSink(buffer, false), this);
744                    return;
745                }
746            } catch (IOException e) {
747                // Buffer already unlocked by fillFromChannel
748                removeChannel(e);
749                return;
750            }
751            // EOF (-1) from client
752            buffer.unlockBuffer();
753            synchronized (nioChannel) {
754                if (nioChannel.socket().isOutputShutdown()) {
755                    // Client confirms our close, complete close
756                    try {
757                        nioChannel.close();
758                    } catch (IOException e) {
759                        // Ignored for close
760                    }
761                    return;
762                }
763
764            }
765            // Client initiates close
766            removeChannel(null);
767            synchronized (pendingWrites) {
768                synchronized (nioChannel) {
769                    try {
770                        if (!pendingWrites.isEmpty()) {
771                            // Pending writes, delay close
772                            pendingClose = true;
773                            // Mark as client initiated close
774                            nioChannel.shutdownInput();
775                            return;
776                        }
777                        // Nothing left to do, close
778                        nioChannel.close();
779                    } catch (IOException e) {
780                        // Ignored for close
781                    }
782                }
783            }
784        }
785
786        /**
787         * Checks if there is still data to be written. This may be
788         * a left over in an incompletely written buffer or a complete
789         * pending buffer. 
790         * 
791         * @throws IOException
792         * @throws InterruptedException 
793         */
794        @SuppressWarnings({ "PMD.DataflowAnomalyAnalysis",
795            "PMD.EmptyCatchBlock",
796            "PMD.AvoidBranchingStatementAsLastInLoop" })
797        private void handleWriteOp() throws InterruptedException {
798            while (true) {
799                ManagedBuffer<ByteBuffer>.ByteBufferView head = null;
800                synchronized (pendingWrites) {
801                    if (pendingWrites.isEmpty()) {
802                        // Nothing left to write, stop getting ops
803                        registration.updateInterested(SelectionKey.OP_READ);
804                        // Was the connection closed while we were writing?
805                        if (pendingClose) {
806                            synchronized (nioChannel) {
807                                try {
808                                    if (nioChannel.socket().isInputShutdown()) {
809                                        // Delayed close from client, complete
810                                        nioChannel.close();
811                                    } else {
812                                        // Delayed close from server, initiate
813                                        nioChannel.shutdownOutput();
814                                    }
815                                } catch (IOException e) {
816                                    // Ignored for close
817                                }
818                            }
819                            pendingClose = false;
820                        } else {
821                            if (purgeable == PurgeableState.PENDING) {
822                                purgeable = PurgeableState.YES;
823                            }
824                        }
825                        break; // Nothing left to do
826                    }
827                    head = pendingWrites.peek();
828                    if (!head.get().hasRemaining()) {
829                        // Nothing left in head buffer, try next
830                        head.managedBuffer().unlockBuffer();
831                        pendingWrites.remove();
832                        continue;
833                    }
834                }
835                try {
836                    nioChannel.write(head.get()); // write...
837                } catch (IOException e) {
838                    removeChannel(e);
839                    return;
840                }
841                break; // ... and wait for next op
842            }
843        }
844
845        /**
846         * Closes this channel.
847         * 
848         * @throws IOException if an error occurs
849         * @throws InterruptedException if the execution was interrupted 
850         */
851        public void close() throws IOException, InterruptedException {
852            removeChannel(null);
853            synchronized (pendingWrites) {
854                if (!pendingWrites.isEmpty()) {
855                    // Pending writes, delay close until done
856                    pendingClose = true;
857                    return;
858                }
859                // Nothing left to do, proceed
860                synchronized (nioChannel) {
861                    if (nioChannel.isOpen()) {
862                        // Initiate close, must be confirmed by client
863                        nioChannel.shutdownOutput();
864                    }
865                }
866            }
867        }
868
869        @SuppressWarnings("PMD.EmptyCatchBlock")
870        private void removeChannel(Throwable error)
871                throws InterruptedException {
872            if (error != null) {
873                try {
874                    nioChannel.close();
875                } catch (IOException e) {
876                    // Closed only to make sure, any failure can be ignored.
877                }
878            }
879            if (TcpServer.this.removeChannel(this)) {
880                Closed evt = new Closed(error);
881                downPipeline.fire(evt, this);
882                evt.get();
883            }
884        }
885
886        /*
887         * (non-Javadoc)
888         * 
889         * @see org.jgrapes.io.IOSubchannel.DefaultSubchannel#toString()
890         */
891        @SuppressWarnings("PMD.CommentRequired")
892        public String toString() {
893            return IOSubchannel.toString(this);
894        }
895    }

1000                    + ObjectName.quote(Components.objectName(server)

1008                // Just in case, should not work
1009            }
1010            try {
1011                mbs.registerMBean(this, mbeanName);
1012            } catch (InstanceAlreadyExistsException | MBeanRegistrationException
1013                    | NotCompliantMBeanException e) {
1014                // Have to live with that
1015            }

1026            TcpServer server = serverRef.get();
1027            if (server == null) {

1029                    mbs.unregisterMBean(mbeanName);

1111     */
1112    private static class MBeanView implements TcpServerSummaryMXBean {
1113        private static Set<TcpServerInfo> serverInfos = new HashSet<>();
1114
1115        /**
1116         * Adds the server to the reported servers.
1117         *
1118         * @param server the server
1119         */

1121            synchronized (serverInfos) {
1122                serverInfos.add(new TcpServerInfo(server));
1123            }
1124        }
1125
1126        /**
1127         * Returns the infos.
1128         *
1129         * @return the sets the
1130         */
1131        private Set<TcpServerInfo> infos() {
1132            Set<TcpServerInfo> expired = new HashSet<>();
1133            synchronized (serverInfos) {
1134                for (TcpServerInfo serverInfo : serverInfos) {
1135                    if (!serverInfo.server().isPresent()) {
1136                        expired.add(serverInfo);
1137                    }
1138                }
1139                serverInfos.removeAll(expired);
1140            }
1141            return serverInfos;
1142        }

1144        @SuppressWarnings("unchecked")
1145        @Override
1146        public Set<TcpServerMXBean> getServers() {
1147            return (Set<TcpServerMXBean>) (Object) infos();
1148        }

1150        @Override
1151        public IntSummaryStatistics getConnectionsPerServerStatistics() {
1152            return infos().stream().map(info -> info.server().get())
1153                .filter(ref -> ref != null).collect(
1154                    Collectors.summarizingInt(srv -> srv.channels.size()));
1155        }
