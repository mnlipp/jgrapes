070/**
071 * Provides a TCP server. The server binds to the given address. If the
072 * address is {@code null}, address and port are automatically assigned.
073 * The port may be overwritten by a configuration event
074 * (see {@link #onConfigurationUpdate(ConfigurationUpdate)}).
075 * 
076 * For each established connection, the server creates a new
077 * {@link LinkedIOSubchannel}. The servers basic operation is to
078 * fire {@link Input} (and {@link Closed}) events on the
079 * appropriate subchannel in response to data received from the
080 * network and to handle {@link Output} (and {@link Close}) events 
081 * on the subchannel and forward the information to the network
082 * connection.
083 * 
084 * The server supports limiting the number of concurrent connections
085 * with a {@link PermitsPool}. If such a pool is set as connection
086 * limiter (see {@link #setConnectionLimiter(PermitsPool)}), a
087 * permit is acquired for each new connection attempt. If no more
088 * permits are available, the server sends a {@link Purge} event on
089 * each channel that is purgeable for at least the time span
090 * set with {@link #setMinimalPurgeableTime(long)}. Purgeability 
091 * is derived from the end of record flag of {@link Output} events
092 * (see {@link #onOutput(Output, TcpChannelImpl)}. When using this feature, 
093 * make sure that connections are either short lived or the application
094 * level components support the {@link Purge} event. Else, it may become
095 * impossible to establish new connections.
096 */
097@SuppressWarnings({ "PMD.ExcessiveImports", "PMD.ExcessivePublicCount",
098    "PMD.NcssCount", "PMD.EmptyCatchBlock", "PMD.AvoidDuplicateLiterals",
099    "PMD.ExcessiveClassLength" })

111    /**
112     * The purger thread.
113     */
114    private class Purger extends Thread implements AvailabilityListener {
115
116        private boolean permitsAvailable = true;
117
118        /**
119         * Instantiates a new purger.
120         */
121        public Purger() {
122            setName(Components.simpleObjectName(this));
123            setDaemon(true);
124        }
125
126        @Override
127        public void availabilityChanged(PermitsPool pool, boolean available) {
128            if (registration == null) {
129                return;
130            }
131            synchronized (this) {
132                permitsAvailable = available;
133                registration.updateInterested(
134                    permitsAvailable ? SelectionKey.OP_ACCEPT : 0);
135                if (!permitsAvailable) {
136                    this.notifyAll();
137                }
138            }
139        }
140
141        @Override
142        @SuppressWarnings({ "PMD.AvoidInstantiatingObjectsInLoops",
143            "PMD.DataflowAnomalyAnalysis" })
144        public void run() {
145            if (connLimiter == null) {
146                return;
147            }
148            try {
149                connLimiter.addListener(this);
150                while (serverSocketChannel.isOpen()) {
151                    synchronized (this) {
152                        while (permitsAvailable) {
153                            wait();
154                        }
155                    }
156                    // Copy to avoid ConcurrentModificationException
157                    List<TcpChannelImpl> candidates;
158                    synchronized (channels) {
159                        candidates = new ArrayList<>(channels);
160                    }
161                    long purgeableSince
162                        = System.currentTimeMillis() - minimumPurgeableTime;
163                    candidates = candidates.stream()
164                        .filter(channel -> channel.isPurgeable()
165                            && channel.purgeableSince() < purgeableSince)
166                        .sorted(new Comparator<TcpChannelImpl>() {
167                            @Override
168                            @SuppressWarnings("PMD.ShortVariable")
169                            public int compare(TcpChannelImpl c1,
170                                    TcpChannelImpl c2) {
171                                if (c1.purgeableSince() < c2
172                                    .purgeableSince()) {
173                                    return 1;
174                                }
175                                if (c1.purgeableSince() > c2
176                                    .purgeableSince()) {
177                                    return -1;
178                                }
179                                return 0;
180                            }
181                        })
182                        .collect(Collectors.toList());
183                    for (TcpChannelImpl channel : candidates) {
184                        // Sorting may have taken time...
185                        if (!channel.isPurgeable()) {
186                            continue;
187                        }
188                        channel.downPipeline().fire(new Purge(), channel);
189                        // Continue only as long as necessary
190                        if (permitsAvailable) {
191                            break;
192                        }
193                    }
194                    sleep(1000);
195                }
196            } catch (InterruptedException e) {
197                // Fall through
198            } finally {
199                connLimiter.removeListener(this);
200            }
201        }
202
203    }

233    /**
234     * The component can be configured with events that include
235     * a path (see @link {@link ConfigurationUpdate#paths()})
236     * that matches this components path (see {@link Manager#componentPath()}).
237     * 
238     * The following properties are recognized:
239     * 
240     * `hostname`
241     * : If given, is used as first parameter for 
242     *   {@link InetSocketAddress#InetSocketAddress(String, int)}.
243     * 
244     * `port`
245     * : If given, is used as parameter for 
246     *   {@link InetSocketAddress#InetSocketAddress(String, int)} 
247     *   or {@link InetSocketAddress#InetSocketAddress(int)}, 
248     *   depending on whether a host name is specified. Defaults to "0".
249     *   
250     * `backlog`
251     * : See {@link #setBacklog(int)}.
252     * 
253     * `bufferSize`
254     * : See {@link #setBufferSize(int)}.
255     * 
256     * @param event the event
257     */
258    @Handler
259    @SuppressWarnings("PMD.ConfusingTernary")
260    public void onConfigurationUpdate(ConfigurationUpdate event) {
261        event.values(componentPath()).ifPresent(values -> {
262            String hostname = values.get("hostname");
263            if (hostname != null) {
264                setServerAddress(new InetSocketAddress(hostname,
265                    Integer.parseInt(values.getOrDefault("port", "0"))));
266            } else if (values.containsKey("port")) {
267                setServerAddress(new InetSocketAddress(
268                    Integer.parseInt(values.get("port"))));
269            }
270            Optional.ofNullable(values.get("backlog")).ifPresent(
271                value -> setBacklog(Integer.parseInt(value)));
272            Optional.ofNullable(values.get("bufferSize")).ifPresent(
273                value -> setBufferSize(Integer.parseInt(value)));
274        });
275    }

357    /**
358     * Starts the server.
359     * 
360     * @param event the start event
361     * @throws IOException if an I/O exception occurred
362     */
363    @Handler
364    public void onStart(Start event) throws IOException {
365        closing = false;
366        serverSocketChannel = ServerSocketChannel.open();
367        serverSocketChannel.bind(serverAddress, backlog);
368        MBeanView.addServer(this);
369        fire(new NioRegistration(this, serverSocketChannel,
370            SelectionKey.OP_ACCEPT, this), Channel.BROADCAST);
371    }

373    /**
374     * Handles the successful channel registration.
375     *
376     * @param event the event
377     * @throws InterruptedException the interrupted exception
378     * @throws IOException Signals that an I/O exception has occurred.
379     */
380    @Handler(channels = Self.class)
381    public void onRegistered(NioRegistration.Completed event)
382            throws InterruptedException, IOException {
383        NioHandler handler = event.event().handler();
384        if (handler == this) {
385            if (event.event().get() == null) {
386                fire(new Error(event,
387                    "Registration failed, no NioDispatcher?"));
388                return;
389            }
390            registration = event.event().get();
391            purger = new Purger();
392            purger.start();
393            fire(new Ready(serverSocketChannel.getLocalAddress()));
394            return;
395        }
396        if (handler instanceof TcpChannelImpl) {
397            TcpChannelImpl channel = (TcpChannelImpl) handler;
398            channel.downPipeline()
399                .fire(new Accepted(channel.nioChannel().getLocalAddress(),
400                    channel.nioChannel().getRemoteAddress(), false,
401                    Collections.emptyList()), channel);
402            channel.registrationComplete(event.event());
403        }
404    }

406    /*
407     * (non-Javadoc)
408     * 
409     * @see org.jgrapes.io.NioSelectable#handleOps(int)
410     */
411    @Override
412    public void handleOps(int ops) {
413        if ((ops & SelectionKey.OP_ACCEPT) == 0 || closing) {
414            return;
415        }
416        synchronized (channels) {
417            if (connLimiter != null && !connLimiter.tryAcquire()) {
418                return;
419            }
420            try {
421                SocketChannel socketChannel = serverSocketChannel.accept();
422                if (socketChannel == null) {
423                    // "False alarm"
424                    if (connLimiter != null) {
425                        connLimiter.release();
426                    }
427                    return;
428                }
429                channels.add(new TcpChannelImpl(socketChannel));
430            } catch (IOException e) {
431                fire(new IOError(null, e));
432            }
433        }
434    }
435
436    @Override
437    protected boolean removeChannel(TcpChannelImpl channel) {
438        synchronized (channels) {
439            if (!channels.remove(channel)) {
440                // Closed already
441                return false;
442            }
443            // In case the server is shutting down
444            channels.notifyAll();
445        }
446        if (connLimiter != null) {
447            connLimiter.release();
448        }
449        return true;
450    }

452    /**
453     * Shuts down the server or one of the connections to the server.
454     *
455     * @param event the event
456     * @throws IOException if an I/O exception occurred
457     * @throws InterruptedException if the execution was interrupted
458     */
459    @Handler
460    @SuppressWarnings("PMD.DataflowAnomalyAnalysis")
461    public void onClose(Close event) throws IOException, InterruptedException {
462        boolean closeServer = false;
463        for (Channel channel : event.channels()) {
464            if (channels.contains(channel)) {
465                ((TcpChannelImpl) channel).close();
466                continue;
467            }
468            if (channel instanceof Subchannel) {
469                // Some subchannel that we're not interested in.
470                continue;
471            }
472            // Close event on "main" channel
473            closeServer = true;
474        }
475        if (!closeServer) {
476            // Only connection(s) were to be closed.
477            return;
478        }
479        if (!serverSocketChannel.isOpen()) {
480            // Closed already
481            fire(new Closed());
482            return;
483        }
484        synchronized (channels) {
485            closing = true;
486            // Copy to avoid concurrent modification exception
487            Set<TcpChannelImpl> conns = new HashSet<>(channels);
488            for (TcpChannelImpl conn : conns) {
489                conn.close();
490            }
491            while (!channels.isEmpty()) {
492                channels.wait();
493            }
494        }
495        serverSocketChannel.close();
496        purger.interrupt();
497        closing = false;
498        fire(new Closed());
499    }

501    /**
502     * Shuts down the server by firing a {@link Close} using the
503     * server as channel. Note that this automatically results
504     * in closing all open connections by the runtime system
505     * and thus in {@link Closed} events on all subchannels.
506     * 
507     * @param event the event
508     */
509    @Handler(priority = -1000)
510    public void onStop(Stop event) {
511        if (closing || !serverSocketChannel.isOpen()) {
512            return;
513        }
514        newSyncEventPipeline().fire(new Close(), this);
515    }

599    public static class TcpServerInfo implements TcpServerMXBean {
600
601        private static MBeanServer mbs
602            = ManagementFactory.getPlatformMBeanServer();
603
604        private ObjectName mbeanName;
605        private final WeakReference<TcpServer> serverRef;
606
607        /**
608         * Instantiates a new tcp server info.
609         *
610         * @param server the server
611         */
612        @SuppressWarnings({ "PMD.EmptyCatchBlock",
613            "PMD.AvoidCatchingGenericException" })
614        public TcpServerInfo(TcpServer server) {
615            serverRef = new WeakReference<>(server);
616            try {
617                int port = server.serverAddress().getPort();
618                mbeanName = new ObjectName("org.jgrapes.io:type="
619                    + TcpServer.class.getSimpleName() + ",name="
620                    + ObjectName.quote(Components.objectName(server)
621                        + " (:" + port + ")"));
622            } catch (MalformedObjectNameException e) {
623                // Should not happen
624            }
625            try {
626                mbs.unregisterMBean(mbeanName);
627            } catch (Exception e) {
628                // Just in case, should not work
629            }
630            try {
631                mbs.registerMBean(this, mbeanName);
632            } catch (InstanceAlreadyExistsException | MBeanRegistrationException
633                    | NotCompliantMBeanException e) {
634                // Have to live with that
635            }
636        }
637
638        /**
639         * Server.
640         *
641         * @return the optional
642         */
643        @SuppressWarnings({ "PMD.AvoidCatchingGenericException",
644            "PMD.EmptyCatchBlock" })
645        public Optional<TcpServer> server() {
646            TcpServer server = serverRef.get();
647            if (server == null) {
648                try {
649                    mbs.unregisterMBean(mbeanName);
650                } catch (Exception e) {
651                    // Should work.
652                }
653            }
654            return Optional.ofNullable(server);
655        }
656
657        /*

682         */
683        @Override
684        public int getChannelCount() {
685            return server().map(server -> server.channels.size()).orElse(0);
686        }
687
688        /*
689         * (non-Javadoc)
690         * 
691         * @see org.jgrapes.net.TcpServer.TcpServerMXBean#getChannels()
692         */
693        @Override
694        @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
695        public SortedMap<String, ChannelInfo> getChannels() {
696            return server().map(server -> {
697                SortedMap<String, ChannelInfo> result = new TreeMap<>();
698                for (TcpChannelImpl channel : server.channels) {
699                    result.put(channel.nioChannel().socket()
700                        .getRemoteSocketAddress().toString(),
701                        new ChannelInfo(channel));
702                }
703                return result;
704            }).orElse(Collections.emptySortedMap());
705        }

729    /**
730     * The MBeanView.
731     */
732    private static class MBeanView implements TcpServerSummaryMXBean {
733        private static Set<TcpServerInfo> serverInfos = new HashSet<>();
734
735        /**
736         * Adds the server to the reported servers.
737         *
738         * @param server the server
739         */
740        public static void addServer(TcpServer server) {
741            synchronized (serverInfos) {
742                serverInfos.add(new TcpServerInfo(server));
743            }
744        }
745
746        /**
747         * Returns the infos.
748         *
749         * @return the sets the
750         */
751        private Set<TcpServerInfo> infos() {
752            Set<TcpServerInfo> expired = new HashSet<>();
753            synchronized (serverInfos) {
754                for (TcpServerInfo serverInfo : serverInfos) {
755                    if (!serverInfo.server().isPresent()) {
756                        expired.add(serverInfo);
757                    }
758                }
759                serverInfos.removeAll(expired);
760            }
761            return serverInfos;
762        }
763
764        @SuppressWarnings("unchecked")
765        @Override
766        public Set<TcpServerMXBean> getServers() {
767            return (Set<TcpServerMXBean>) (Object) infos();
768        }
769
770        @Override
771        public IntSummaryStatistics getConnectionsPerServerStatistics() {
772            return infos().stream().map(info -> info.server().get())
773                .filter(ref -> ref != null).collect(
774                    Collectors.summarizingInt(srv -> srv.channels.size()));
775        }
776    }
