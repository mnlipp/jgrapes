059/**
060 * A component that receives and send byte buffers on an
061 * encrypted channel and sends and receives the corresponding
062 * decrypted data on its own channel.
063 */
064@SuppressWarnings("PMD.ExcessiveImports")
065public class SslServer extends Component {
066
067    @SuppressWarnings("PMD.VariableNamingConventions")
068    private static final Logger logger
069        = Logger.getLogger(SslServer.class.getName());
070
071    private final SSLContext sslContext;
072
073    /**
074     * Represents the encrypted channel in annotations.
075     */
076    private class EncryptedChannel extends ClassChannel {
077    }
078
079    /**
080     * @param plainChannel the component's channel
081     * @param encryptedChannel the channel with the encrypted data
082     * @param sslContext the SSL context to use
083     */
084    public SslServer(Channel plainChannel, Channel encryptedChannel,
085            SSLContext sslContext) {
086        super(plainChannel, ChannelReplacements.create()
087            .add(EncryptedChannel.class, encryptedChannel));
088        this.sslContext = sslContext;
089    }
090
091    /**
092     * Creates a new downstream connection as {@link LinkedIOSubchannel} 
093     * of the network connection together with an {@link SSLEngine}.
094     * 
095     * @param event
096     *            the accepted event
097     */
098    @Handler(channels = EncryptedChannel.class)
099    public void onAccepted(Accepted event, IOSubchannel encryptedChannel) {
100        new PlainChannel(event, encryptedChannel);
101    }
102
103    /**
104     * Handles encrypted data from upstream (the network). The data is 
105     * send through the {@link SSLEngine} and events are sent downstream 
106     * (and in the initial phases upstream) according to the conversion 
107     * results.
108     * 
109     * @param event the event
110     * @param encryptedChannel the channel for exchanging the encrypted data
111     * @throws InterruptedException 
112     * @throws SSLException 
113     * @throws ExecutionException 
114     */
115    @Handler(channels = EncryptedChannel.class)
116    public void onInput(
117            Input<ByteBuffer> event, IOSubchannel encryptedChannel)
118            throws InterruptedException, SSLException, ExecutionException {
119        @SuppressWarnings({ "unchecked", "PMD.AvoidDuplicateLiterals" })
120        final Optional<PlainChannel> plainChannel
121            = (Optional<PlainChannel>) LinkedIOSubchannel
122                .downstreamChannel(this, encryptedChannel);
123        if (plainChannel.isPresent()) {
124            plainChannel.get().sendDownstream(event);
125        }
126    }
127
128    /**
129     * Handles a close event from the encrypted channel (client).
130     * 
131     * @param event the event
132     * @param encryptedChannel the channel for exchanging the encrypted data
133     * @throws InterruptedException 
134     * @throws SSLException 
135     */
136    @Handler(channels = EncryptedChannel.class)
137    public void onClosed(Closed event, IOSubchannel encryptedChannel)
138            throws SSLException, InterruptedException {
139        @SuppressWarnings("unchecked")
140        final Optional<PlainChannel> plainChannel
141            = (Optional<PlainChannel>) LinkedIOSubchannel
142                .downstreamChannel(this, encryptedChannel);
143        if (plainChannel.isPresent()) {
144            plainChannel.get().upstreamClosed();
145        }
146    }
147
148    /**
149     * Forwards a {@link Purge} event downstream.
150     *
151     * @param event the event
152     * @param encryptedChannel the encrypted channel
153     */
154    @Handler(channels = EncryptedChannel.class)
155    public void onPurge(Purge event, IOSubchannel encryptedChannel) {
156        @SuppressWarnings("unchecked")
157        final Optional<PlainChannel> plainChannel
158            = (Optional<PlainChannel>) LinkedIOSubchannel
159                .downstreamChannel(this, encryptedChannel);
160        if (plainChannel.isPresent()) {
161            plainChannel.get().purge();
162        }
163    }
164
165    /**
166     * Handles an {@link IOError} event from the encrypted channel (client)
167     * by sending it downstream.
168     * 
169     * @param event the event
170     * @param encryptedChannel the channel for exchanging the encrypted data
171     * @throws InterruptedException 
172     * @throws SSLException 
173     */
174    @Handler(channels = EncryptedChannel.class)
175    public void onIOError(IOError event, IOSubchannel encryptedChannel)
176            throws SSLException, InterruptedException {
177        @SuppressWarnings("unchecked")
178        final Optional<PlainChannel> plainChannel
179            = (Optional<PlainChannel>) LinkedIOSubchannel
180                .downstreamChannel(this, encryptedChannel);
181        plainChannel.ifPresent(channel -> fire(new IOError(event), channel));
182    }
183
184    /**
185     * Sends decrypted data through the engine and then upstream.
186     * 
187     * @param event
188     *            the event with the data
189     * @throws InterruptedException if the execution was interrupted
190     * @throws SSLException if some SSL related problem occurs
191     */
192    @Handler
193    public void onOutput(Output<ByteBuffer> event,
194            PlainChannel plainChannel)
195            throws InterruptedException, SSLException {
196        if (plainChannel.hub() != this) {
197            return;
198        }
199        plainChannel.sendUpstream(event);
200    }
201
202    /**
203     * Forwards a close event upstream.
204     * 
205     * @param event
206     *            the close event
207     * @throws SSLException if an SSL related problem occurs
208     * @throws InterruptedException if the execution was interrupted
209     */
210    @Handler
211    public void onClose(Close event, PlainChannel plainChannel)
212            throws InterruptedException, SSLException {
213        if (plainChannel.hub() != this) {
214            return;
215        }
216        plainChannel.close(event);
217    }
218
219    /**
220     * Represents the plain channel.
221     */
222    private class PlainChannel extends LinkedIOSubchannel {
223        public SocketAddress localAddress;
224        public SocketAddress remoteAddress;
225        public SSLEngine sslEngine;
226        private final ManagedBufferPool<ManagedBuffer<ByteBuffer>,
227                ByteBuffer> downstreamPool;
228        private boolean isInputClosed;
229        private ByteBuffer carryOver;
230
231        /**
232         * Instantiates a new plain channel.
233         *
234         * @param event the event
235         * @param upstreamChannel the upstream channel
236         */
237        public PlainChannel(Accepted event, IOSubchannel upstreamChannel) {
238            super(SslServer.this, channel(), upstreamChannel,
239                newEventPipeline());
240            localAddress = event.localAddress();
241            remoteAddress = event.remoteAddress();
242            if (remoteAddress instanceof InetSocketAddress) {
243                sslEngine = sslContext.createSSLEngine(
244                    ((InetSocketAddress) remoteAddress).getAddress()
245                        .getHostAddress(),
246                    ((InetSocketAddress) remoteAddress).getPort());
247            } else {
248                sslEngine = sslContext.createSSLEngine();
249            }
250            sslEngine.setUseClientMode(false);
251
252            String channelName = Components.objectName(SslServer.this)
253                + "." + Components.objectName(this);
254            // Create buffer pools, adding 50 to application buffer size, see
255            // https://docs.oracle.com/javase/8/docs/technotes/guides/security/jsse/samples/sslengine/SSLEngineSimpleDemo.java
256            int decBufSize = sslEngine.getSession()
257                .getApplicationBufferSize() + 50;
258            downstreamPool = new ManagedBufferPool<>(ManagedBuffer::new,
259                () -> {
260                    return ByteBuffer.allocate(decBufSize);
261                }, 2)
262                    .setName(channelName + ".downstream.buffers");
263            int encBufSize = sslEngine.getSession().getPacketBufferSize();
264            setByteBufferPool(new ManagedBufferPool<>(ManagedBuffer::new,
265                () -> {
266                    return ByteBuffer.allocate(encBufSize);
267                }, 2)
268                    .setName(channelName + ".upstream.buffers"));
269        }
270
271        /**
272         * Sends input downstream.
273         *
274         * @param event the event
275         * @throws SSLException the SSL exception
276         * @throws InterruptedException the interrupted exception
277         * @throws ExecutionException the execution exception
278         */
279        @SuppressWarnings({ "PMD.CyclomaticComplexity",
280            "PMD.ExcessiveMethodLength", "PMD.NcssCount",
281            "PMD.NPathComplexity" })
282        public void sendDownstream(Input<ByteBuffer> event)
283                throws SSLException, InterruptedException, ExecutionException {
284            ManagedBuffer<ByteBuffer> unwrapped = downstreamPool.acquire();
285            ByteBuffer input = event.buffer().duplicate();
286            if (carryOver != null) {
287                if (carryOver.remaining() < input.remaining()) {
288                    // Shouldn't happen with carryOver having packet size
289                    // bytes left, have seen it happen nevertheless.
290                    carryOver.flip();
291                    ByteBuffer extCarryOver = ByteBuffer.allocate(
292                        carryOver.remaining() + input.remaining());
293                    extCarryOver.put(carryOver);
294                    carryOver = extCarryOver;
295                }
296                carryOver.put(input);
297                carryOver.flip();
298                input = carryOver;
299                carryOver = null;
300            }
301            SSLEngineResult unwrapResult;
302            while (true) {
303                unwrapResult = sslEngine.unwrap(
304                    input, unwrapped.backingBuffer());
305                // Handle any handshaking procedures
306                switch (unwrapResult.getHandshakeStatus()) {
307                case NEED_TASK:
308                    while (true) {
309                        Runnable runnable = sslEngine.getDelegatedTask();
310                        if (runnable == null) {
311                            break;
312                        }
313                        // Having this handled by the response thread is
314                        // probably not really necessary, but as the delegated
315                        // task usually includes sending upstream...
316                        @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
317                        FutureTask<Boolean> task
318                            = new FutureTask<>(runnable, true);
319                        upstreamChannel().responsePipeline()
320                            .executorService().submit(task);
321                        task.get();
322                    }
323                    continue;
324
325                case NEED_WRAP:
326                    ManagedBuffer<ByteBuffer> feedback
327                        = upstreamChannel().byteBufferPool().acquire();
328                    SSLEngineResult wrapResult = sslEngine.wrap(
329                        ManagedBuffer.EMPTY_BYTE_BUFFER
330                            .backingBuffer(),
331                        feedback.backingBuffer());
332                    upstreamChannel().respond(Output.fromSink(feedback, false));
333                    if (wrapResult
334                        .getHandshakeStatus() == HandshakeStatus.FINISHED) {
335                        fireAccepted();
336                    }
337                    continue;
338
339                case FINISHED:
340                    fireAccepted();
341                    // fall through
342                case NEED_UNWRAP:
343                    // sslEngine.unwrap sometimes return NEED_UNWRAP in
344                    // combination with CLOSED, though this doesn't really
345                    // make sense.
346                    if (unwrapResult.getStatus() != Status.BUFFER_UNDERFLOW
347                        && unwrapResult.getStatus() != Status.CLOSED) {
348                        continue;
349                    }
350                    break;
351
352                default:
353                    break;
354                }
355
356                // Just to make sure...
357                if (unwrapResult.getStatus() == Status.BUFFER_OVERFLOW) {
358                    unwrapped.replaceBackingBuffer(ByteBuffer.allocate(
359                        sslEngine.getSession()
360                            .getApplicationBufferSize() + 50));
361                }
362
363                // If we get here, handshake has completed or no input is left
364                if (unwrapResult.getStatus() != Status.OK) {
365                    // Underflow, overflow or closed
366                    break;
367                }
368            }
369
370            if (unwrapped.position() == 0) {
371                // Was only handshake
372                unwrapped.unlockBuffer();
373            } else {
374                // forward data received
375                fire(Input.fromSink(unwrapped, sslEngine.isInboundDone()),
376                    this);
377            }
378
379            // final message?
380            if (unwrapResult.getStatus() == Status.CLOSED
381                && !isInputClosed) {
382                Closed evt = new Closed();
383                newEventPipeline().fire(evt, this);
384                evt.get();
385                isInputClosed = true;
386                return;
387            }
388
389            // Check if data from incomplete packet remains in input buffer
390            if (input.hasRemaining()) {
391                // Actually, packet buffer size should be sufficient,
392                // but since this is hard to test and doesn't really matter...
393                carryOver = ByteBuffer.allocate(input.remaining()
394                    + sslEngine.getSession().getPacketBufferSize() + 50);
395                carryOver.put(input);
396            }
397        }

399        /**
400         * Send output upstream.
401         *
402         * @param event the event
403         * @throws SSLException the SSL exception
404         * @throws InterruptedException the interrupted exception
405         */
406        public void sendUpstream(Output<ByteBuffer> event)
407                throws SSLException, InterruptedException {
408            ByteBuffer output = event.buffer().backingBuffer().duplicate();
409            while (output.hasRemaining() && !sslEngine.isInboundDone()) {
410                ManagedBuffer<ByteBuffer> out
411                    = upstreamChannel().byteBufferPool().acquire();
412                sslEngine.wrap(output, out.backingBuffer());
413                upstreamChannel().respond(
414                    Output.fromSink(out, event.isEndOfRecord()
415                        && !output.hasRemaining()));
416            }
417        }
418
419        /**
420         * Close the connection.
421         *
422         * @param event the event
423         * @throws InterruptedException the interrupted exception
424         * @throws SSLException the SSL exception
425         */
426        public void close(Close event)
427                throws InterruptedException, SSLException {
428            sslEngine.closeOutbound();
429            while (!sslEngine.isOutboundDone()) {
430                ManagedBuffer<ByteBuffer> feedback
431                    = upstreamChannel().byteBufferPool().acquire();
432                sslEngine.wrap(ManagedBuffer.EMPTY_BYTE_BUFFER
433                    .backingBuffer(), feedback.backingBuffer());
434                upstreamChannel().respond(Output.fromSink(feedback, false));
435            }
436            upstreamChannel().respond(new Close());
437        }
438
439        @SuppressWarnings("PMD.DataflowAnomalyAnalysis")
440        private void fireAccepted() {
441            List<SNIServerName> snis = Collections.emptyList();
442            if (sslEngine.getSession() instanceof ExtendedSSLSession) {
443                snis = ((ExtendedSSLSession) sslEngine.getSession())
444                    .getRequestedServerNames();
445            }
446            fire(new Accepted(
447                localAddress, remoteAddress, true, snis), this);
448        }
449
450        /**
451         * Forwards the {@link Closed} event downstream.
452         *
453         * @throws SSLException the SSL exception
454         * @throws InterruptedException the interrupted exception
455         */
456        public void upstreamClosed()
457                throws SSLException, InterruptedException {
458            if (!isInputClosed) {
459                // was not properly closed on SSL layer
460                Closed evt = new Closed();
461                newEventPipeline().fire(evt, this);
462                evt.get();
463            }
464            try {
465                sslEngine.closeInbound();
466                while (!sslEngine.isOutboundDone()) {
467                    ManagedBuffer<ByteBuffer> feedback
468                        = upstreamChannel().byteBufferPool().acquire();
469                    sslEngine.wrap(ManagedBuffer.EMPTY_BYTE_BUFFER
470                        .backingBuffer(), feedback.backingBuffer());
471                    upstreamChannel().respond(Output.fromSink(feedback, false));
472                }
473            } catch (SSLException e) {
474                // Several clients (notably chromium, see
475                // https://bugs.chromium.org/p/chromium/issues/detail?id=118366
476                // don't close the connection properly. So nobody is really
477                // interested in this message
478                logger.log(Level.FINEST, e.getMessage(), e);
479            }
480        }
481
482        /**
483         * Fire a {@link Purge} event downstream.
484         */
485        public void purge() {
486            fire(new Purge(), this);
487        }
488    }
