066/**
067 * A component that receives and sends byte buffers on an
068 * encrypted channel and sends and receives the corresponding
069 * decrypted data on a plain channel.
070 * 
071 * The encrypted channel is assumed to be the network side
072 * ("upstream") and therefore {@link Input} events represent
073 * encrypted data and are decoded to {@link Output} events on
074 * the plain channel ("downstream") and vice versa.
075 */
076@SuppressWarnings({ "PMD.ExcessiveImports" })
077public class SslCodec extends Component {
078

081        = Logger.getLogger(SslCodec.class.getName());

084    private final SSLContext sslContext;

099    public SslCodec(Channel plainChannel, Channel encryptedChannel,
100            SSLContext sslContext) {
101        super(plainChannel, ChannelReplacements.create()
102            .add(EncryptedChannel.class, encryptedChannel));
103        this.encryptedChannel = encryptedChannel;
104        this.sslContext = sslContext;
105    }

107    /**
108     * Creates a new codec to be used as client.
109     * 
110     * @param plainChannel the component's channel
111     * @param encryptedChannel the channel with the encrypted data
112     * @param dontValidate if `true` accept all kinds of certificates
113     */
114    @SuppressWarnings({ "PMD.DataflowAnomalyAnalysis", "PMD.CommentRequired",
115        "PMD.ReturnEmptyArrayRatherThanNull", "PMD.UncommentedEmptyMethodBody",
116        "PMD.AvoidDuplicateLiterals" })
117    public SslCodec(Channel plainChannel, Channel encryptedChannel,
118            boolean dontValidate) {
119        super(plainChannel, ChannelReplacements.create()
120            .add(EncryptedChannel.class, encryptedChannel));
121        this.encryptedChannel = encryptedChannel;
122        try {
123            final SSLContext sslContext = SSLContext.getInstance("SSL");
124            if (dontValidate) {
125                // Create a trust manager that does not validate certificate
126                // chains
127                final TrustManager[] trustAllCerts = {
128                    new X509TrustManager() {
129                        public X509Certificate[] getAcceptedIssuers() {
130                            return null;
131                        }
132
133                        public void checkClientTrusted(
134                                X509Certificate[] certs, String authType) {
135                        }
136
137                        public void checkServerTrusted(
138                                X509Certificate[] certs, String authType) {
139                        }
140                    }
141                };
142                sslContext.init(null, trustAllCerts, null);
143            } else {
144                sslContext.init(null, null, null);
145            }
146            this.sslContext = sslContext;
147        } catch (NoSuchAlgorithmException | KeyManagementException e) {
148            throw new IllegalArgumentException(e);
149        }
150    }

186    /**
187     * Handles encrypted data from upstream (the network). The data is 
188     * send through the {@link SSLEngine} and events are sent downstream 
189     * (and in the initial phases upstream) according to the conversion 
190     * results.
191     * 
192     * @param event the event
193     * @param encryptedChannel the channel for exchanging the encrypted data
194     * @throws InterruptedException 
195     * @throws SSLException 
196     * @throws ExecutionException 
197     */
198    @Handler(channels = EncryptedChannel.class)
199    public void onInput(
200            Input<ByteBuffer> event, IOSubchannel encryptedChannel)
201            throws InterruptedException, SSLException, ExecutionException {
202        @SuppressWarnings({ "unchecked", "PMD.AvoidDuplicateLiterals" })
203        final Optional<PlainChannel> plainChannel
204            = (Optional<PlainChannel>) LinkedIOSubchannel
205                .downstreamChannel(this, encryptedChannel);
206        if (plainChannel.isPresent()) {
207            plainChannel.get().sendDownstream(event);
208        }
209    }

323    /**
324     * Represents the plain channel.
325     */
326    @SuppressWarnings("PMD.DataflowAnomalyAnalysis")
327    private class PlainChannel extends LinkedIOSubchannel {
328        public SocketAddress localAddress;
329        public SocketAddress remoteAddress;
330        public SSLEngine sslEngine;
331        private EventPipeline downPipeline;
332        private ManagedBufferPool<ManagedBuffer<ByteBuffer>,
333                ByteBuffer> downstreamPool;
334        private ByteBuffer carryOver;
335        private boolean[] inputProcessed = { false };
336
337        /**
338         * Instantiates a new plain channel from an accepted connection.
339         *
340         * @param event the event
341         * @param upstreamChannel the upstream channel
342         */
343        public PlainChannel(Accepted event, IOSubchannel upstreamChannel) {
344            super(SslCodec.this, channel(), upstreamChannel,
345                newEventPipeline());
346            localAddress = event.localAddress();
347            remoteAddress = event.remoteAddress();
348            init();
349            sslEngine.setUseClientMode(false);
350        }
351
352        /**
353         * Instantiates a new plain channel from an initiated connection.
354         *
355         * @param event the event
356         * @param upstreamChannel the upstream channel
357         */
358        public PlainChannel(Connected event, IOSubchannel upstreamChannel) {
359            super(SslCodec.this, channel(), upstreamChannel,
360                newEventPipeline());
361            localAddress = event.localAddress();
362            remoteAddress = event.remoteAddress();
363            init();
364            sslEngine.setUseClientMode(true);
365
366            // Forward downstream
367            downPipeline.fire(
368                new Connected(event.localAddress(), event.remoteAddress()),
369                this);
370        }
371
372        private void init() {
373            if (remoteAddress instanceof InetSocketAddress) {
374                sslEngine = sslContext.createSSLEngine(
375                    ((InetSocketAddress) remoteAddress).getAddress()
376                        .getHostAddress(),
377                    ((InetSocketAddress) remoteAddress).getPort());
378            } else {
379                sslEngine = sslContext.createSSLEngine();
380            }
381            String channelName = Components.objectName(SslCodec.this)
382                + "." + Components.objectName(this);
383            // Create buffer pools, adding 50 to decoded application buffer
384            // size, see
385            // https://docs.oracle.com/javase/8/docs/technotes/guides/security/jsse/samples/sslengine/SSLEngineSimpleDemo.java
386            final int appBufSize
387                = sslEngine.getSession().getApplicationBufferSize();
388            downstreamPool = new ManagedBufferPool<>(ManagedBuffer::new,
389                () -> ByteBuffer.allocate(appBufSize + 50), 2)
390                    .setName(channelName + ".downstream.buffers");
391            // Provide buffers with application buffer size
392            // for use by downstream components.
393            setByteBufferPool(new ManagedBufferPool<>(ManagedBuffer::new,
394                () -> ByteBuffer.allocate(appBufSize), 2)
395                    .setName(channelName + ".upstream.buffers"));
396            downPipeline = newEventPipeline();
397            // Buffers for sending encrypted data upstream will be
398            // obtained from upstream() and resized if required.
399        }
400
401        /**
402         * Sends input downstream.
403         *
404         * @param event the event
405         * @throws SSLException the SSL exception
406         * @throws InterruptedException the interrupted exception
407         * @throws ExecutionException the execution exception
408         */
409        public void sendDownstream(Input<ByteBuffer> event)
410                throws SSLException, InterruptedException, ExecutionException {
411            ByteBuffer input = event.buffer().duplicate();
412            if (carryOver != null) {
413                if (carryOver.remaining() < input.remaining()) {
414                    // Shouldn't happen with carryOver having packet size
415                    // bytes left, have seen it happen nevertheless.
416                    carryOver.flip();
417                    ByteBuffer extCarryOver = ByteBuffer.allocate(
418                        carryOver.remaining() + input.remaining());
419                    extCarryOver.put(carryOver);
420                    carryOver = extCarryOver;
421                }
422                carryOver.put(input);
423                carryOver.flip();
424                input = carryOver;
425                carryOver = null;
426            }
427
428            // Main processing
429            processInput(input);
430
431            // Check if data from incomplete packet remains in input buffer
432            if (input.hasRemaining()) {
433                // Actually, packet buffer size should be sufficient,
434                // but since this is hard to test and doesn't really matter...
435                carryOver = ByteBuffer.allocate(input.remaining()
436                    + sslEngine.getSession().getPacketBufferSize() + 50);
437                carryOver.put(input);
438            }
439        }
440
441        @SuppressWarnings({ "PMD.CyclomaticComplexity", "PMD.NcssCount",
442            "PMD.AvoidInstantiatingObjectsInLoops", "PMD.ExcessiveMethodLength",
443            "PMD.NPathComplexity" })
444        private SSLEngineResult processInput(ByteBuffer input)
445                throws SSLException, InterruptedException, ExecutionException {
446            SSLEngineResult unwrapResult;
447            ManagedBuffer<ByteBuffer> unwrapped = downstreamPool.acquire();
448            while (true) {
449                unwrapResult
450                    = sslEngine.unwrap(input, unwrapped.backingBuffer());
451                synchronized (inputProcessed) {
452                    inputProcessed[0] = true;
453                    inputProcessed.notifyAll();
454                }
455                // Handle any handshaking procedures
456                switch (unwrapResult.getHandshakeStatus()) {
457                case NEED_TASK:
458                    while (true) {
459                        Runnable runnable = sslEngine.getDelegatedTask();
460                        if (runnable == null) {
461                            break;
462                        }
463                        // Having this handled by the response thread is
464                        // probably not really necessary, but as the delegated
465                        // task usually includes sending upstream...
466                        @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
467                        FutureTask<Boolean> task
468                            = new FutureTask<>(runnable, true);
469                        upstreamChannel().responsePipeline()
470                            .executorService().submit(task);
471                        task.get();
472                    }
473                    continue;
474
475                case NEED_WRAP:
476                    ManagedBuffer<ByteBuffer> feedback
477                        = acquireUpstreamBuffer();
478                    synchronized (sslEngine) {
479                        SSLEngineResult wrapResult = sslEngine.wrap(
480                            ManagedBuffer.EMPTY_BYTE_BUFFER.backingBuffer(),
481                            feedback.backingBuffer());
482                        // JDK11 sometimes returns NEED_WRAP (together
483                        // with status is CLOSED) but does not produce
484                        // anything when wrapping (all of which does
485                        // not make sense).
486                        if (feedback.position() == 0
487                            && unwrapResult.getStatus() == Status.CLOSED) {
488                            feedback.unlockBuffer();
489                            break;
490                        }
491                        upstreamChannel()
492                            .respond(Output.fromSink(feedback, false));
493                        if (wrapResult
494                            .getHandshakeStatus() == HandshakeStatus.FINISHED) {
495                            fireAccepted();
496                        }
497                    }
498                    continue;
499
500                case FINISHED:
501                    fireAccepted();
502                    break;
503
504                default:
505                    break;
506                }
507
508                // Anything to forward downstream?
509                if (unwrapped.position() > 0) {
510                    // forward unwrapped data
511                    downPipeline.fire(Input.fromSink(unwrapped,
512                        sslEngine.isInboundDone()), this);
513                    unwrapped = null;
514                }
515
516                // If we have a buffer overflow or everything was okay
517                // and there's data left, we try again, else we quit.
518                if (unwrapResult.getStatus() != Status.BUFFER_OVERFLOW
519                    && (unwrapResult.getStatus() != Status.OK
520                        || !input.hasRemaining())) {
521                    // Underflow or closed
522                    if (unwrapped != null) {
523                        unwrapped.unlockBuffer();
524                    }
525                    break;
526                }
527
528                // Make sure we have an output buffer.
529                if (unwrapped == null) {
530                    unwrapped = downstreamPool.acquire();
531                }
532            }
533            return unwrapResult;
534        }
535
536        @SuppressWarnings("PMD.DataflowAnomalyAnalysis")
537        private void fireAccepted() {
538            List<SNIServerName> snis = Collections.emptyList();
539            if (sslEngine.getSession() instanceof ExtendedSSLSession) {
540                snis = ((ExtendedSSLSession) sslEngine.getSession())
541                    .getRequestedServerNames();
542            }
543            downPipeline.fire(new Accepted(
544                localAddress, remoteAddress, true, snis), this);
545        }

547        /**
548         * Send output upstream.
549         *
550         * @param event the event
551         * @throws SSLException the SSL exception
552         * @throws InterruptedException the interrupted exception
553         * @throws ExecutionException 
554         */
555        public void sendUpstream(Output<ByteBuffer> event)
556                throws SSLException, InterruptedException, ExecutionException {
557            ByteBuffer output = event.buffer().backingBuffer().duplicate();
558            processOutput(output, event.isEndOfRecord());
559        }
560
561        @SuppressWarnings({ "PMD.DataflowAnomalyAnalysis",
562            "PMD.CyclomaticComplexity", "PMD.NcssCount",
563            "PMD.NPathComplexity" })
564        private void processOutput(ByteBuffer output, boolean eor)
565                throws InterruptedException, SSLException, ExecutionException {
566            ManagedBuffer<ByteBuffer> wrapped = acquireUpstreamBuffer();
567            while (true) {
568                SSLEngineResult wrapResult;
569                // Cheap synchronization: no (relevant) input
570                inputProcessed[0] = false;
571                synchronized (sslEngine) {
572                    wrapResult
573                        = sslEngine.wrap(output, wrapped.backingBuffer());
574                    // Anything to be sent upstream?
575                    if (wrapped.position() > 0) {
576                        upstreamChannel().respond(Output.fromSink(wrapped,
577                            sslEngine.isInboundDone()
578                                || eor && !output.hasRemaining()));
579                        wrapped = null;
580                    }
581                }
582                switch (wrapResult.getHandshakeStatus()) {
583                case NEED_TASK:
584                    while (true) {
585                        Runnable runnable = sslEngine.getDelegatedTask();
586                        if (runnable == null) {
587                            break;
588                        }
589                        runnable.run();
590                    }
591                    continue;
592
593                case NEED_UNWRAP:
594                    // Input required. Wait until
595                    // input becomes available and retry.
596                    synchronized (inputProcessed) {
597                        while (!inputProcessed[0]) {
598                            inputProcessed.wait();
599                        }
600                    }
601                    break;
602
603                default:
604                    break;
605                }
606
607                // If we have a buffer overflow or everything was okay
608                // and there's data left, we try again, else we quit.
609                if (wrapResult.getStatus() != Status.BUFFER_OVERFLOW
610                    && (wrapResult.getStatus() != Status.OK
611                        || !output.hasRemaining())) {
612                    // Underflow or closed
613                    if (wrapped != null) {
614                        wrapped.unlockBuffer();
615                    }
616                    // Warn if data is discarded
617                    if (output.hasRemaining()) {
618                        logger.warning(() -> toString()
619                            + ": Upstream data discarded, SSLEngine status: "
620                            + wrapResult.getStatus());
621                    }
622                    break;
623                }
624
625                // Was handshake (or partial content), get new buffer and try
626                // again
627                if (wrapped == null) {
628                    wrapped = acquireUpstreamBuffer();
629                }
630            }
631        }

633        /**
634         * Close the connection.
635         *
636         * @param event the event
637         * @throws InterruptedException the interrupted exception
638         * @throws SSLException the SSL exception
639         */
640        public void close(Close event)
641                throws InterruptedException, SSLException {
642            sslEngine.closeOutbound();
643            while (!sslEngine.isOutboundDone()) {
644                ManagedBuffer<ByteBuffer> feedback = acquireUpstreamBuffer();
645                sslEngine.wrap(ManagedBuffer.EMPTY_BYTE_BUFFER
646                    .backingBuffer(), feedback.backingBuffer());
647                upstreamChannel().respond(Output.fromSink(feedback, false));
648            }
649            upstreamChannel().respond(new Close());
650        }

652        /**
653         * Handles the {@link HalfClosed} event.
654         *
655         * @throws SSLException the SSL exception
656         * @throws InterruptedException the interrupted exception
657         */
658        public void upstreamHalfClosed()
659                throws SSLException, InterruptedException {
660            if (sslEngine.isInboundDone()) {
661                // Was properly closed on SSL level
662                return;
663            }
664            try {
665                sslEngine.closeInbound();
666                sslEngine.closeOutbound();
667                while (!sslEngine.isOutboundDone()) {
668                    ManagedBuffer<ByteBuffer> feedback
669                        = acquireUpstreamBuffer();
670                    SSLEngineResult result = sslEngine.wrap(
671                        ManagedBuffer.EMPTY_BYTE_BUFFER.backingBuffer(),
672                        feedback.backingBuffer());
673                    // This is required for/since JDK 11. It claims that
674                    // outbound is not done, but doesn't produce any
675                    // additional
676                    // data.
677                    if (result.getStatus() == Status.CLOSED
678                        || feedback.position() == 0) {
679                        feedback.unlockBuffer();
680                        break;
681                    }
682                    upstreamChannel()
683                        .respond(Output.fromSink(feedback, false));
684                }
685            } catch (SSLException e) {
686                // Several clients (notably chromium, see
687                // https://bugs.chromium.org/p/chromium/issues/detail?id=118366
688                // don't close the connection properly. So nobody is really
689                // interested in this message
690                logger.log(Level.FINEST, e.getMessage(), e);
691            } catch (InterruptedException e) {
692                logger.log(Level.WARNING, e.getMessage(), e);
693            }
694        }

707        private ManagedBuffer<ByteBuffer> acquireUpstreamBuffer()
708                throws InterruptedException {
709            ManagedBuffer<ByteBuffer> feedback
710                = upstreamChannel().byteBufferPool().acquire();
711            int encSize
712                = sslEngine.getSession().getPacketBufferSize() + 50;
713            if (feedback.capacity() < encSize) {
714                feedback.replaceBackingBuffer(ByteBuffer.allocate(
715                    encSize));
716            }
717            return feedback;
718        }
