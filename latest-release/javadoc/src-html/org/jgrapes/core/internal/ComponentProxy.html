019package org.jgrapes.core.internal;
020
021import java.lang.reflect.Field;
022
023import org.jgrapes.core.Channel;
024import org.jgrapes.core.ComponentType;
025import org.jgrapes.core.Manager;
026import org.jgrapes.core.NamedChannel;
027import org.jgrapes.core.annotation.ComponentManager;
028import org.jgrapes.core.annotation.Handler;
029
030/**
031 * The ComponentProxy is a special ComponentVertex that references the
032 * object implementing the Component interface (instead of being
033 * its base class).
034 */
035public class ComponentProxy extends ComponentVertex {
036
037        /** The reference to the actual component. */
038        private ComponentType component = null;
039        /** The referenced component's channel. */
040        private Channel componentChannel = Channel.BROADCAST;
041        
042        private static Field getManagerField(Class<?> clazz) {
043                try {
044                        while (true) {
045                                for (Field field: clazz.getDeclaredFields()) {
046                                        if (Manager.class.isAssignableFrom(field.getType())
047                                                        && field.getAnnotation(ComponentManager.class) != null) {
048                                                return field;
049                                        }
050                                }
051                                clazz = clazz.getSuperclass();
052                                if (clazz == null) {
053                                        throw new IllegalArgumentException(
054                                                        "Components must have a manager attribute");
055                                }
056                        }
057                } catch (SecurityException e) {
058                        throw (RuntimeException)(new IllegalArgumentException(
059                                        "Cannot access component's manager attribute")).initCause(e);
060                }
061        }
062
063        private static Channel getComponentChannel(Field field) {
064                ComponentManager cma = field.getAnnotation(ComponentManager.class);
065                if (cma.channel() != Handler.NoChannel.class) {
066                        if (cma.channel() != Channel.BROADCAST.defaultCriterion()) {
067                                try {
068                                        return cma.channel().newInstance();
069                                } catch (InstantiationException | IllegalAccessException e) {
070                                        // Ignored
071                                }
072                        }
073                        return Channel.BROADCAST;
074                }
075                if (!cma.namedChannel().equals("")) {
076                        return new NamedChannel(cma.namedChannel());
077                }
078                return Channel.SELF;
079        }
080        
081        /**
082         * Create a new component proxy for the component and assign it to 
083         * the specified field which must be of type {@link Manager}.
084         * 
085         * @param field the field that gets the proxy assigned
086         * @param componentChannel the componen't channel
087         * @param component the component
088         */
089        private ComponentProxy(
090                        Field field, ComponentType component, Channel componentChannel) {
091                this.component = component;
092                try {
093                        field.set(component, this);
094                        if (componentChannel == null) {
095                                componentChannel = getComponentChannel(field);
096                        }
097                        if (componentChannel.equals(Channel.SELF)) {
098                                componentChannel = this;
099                        }
100                        this.componentChannel = componentChannel;
101                        initComponentsHandlers();
102                } catch (SecurityException | IllegalAccessException e) {
103                        throw (RuntimeException)(new IllegalArgumentException(
104                                        "Cannot access component's manager attribute")).initCause(e);
105                }
106        }
107
108        /**
109         * Return the component node for a component that is represented
110         * by a proxy in the tree.
111         * 
112         * @param component the component
113         * @param componentChannel the component's channel
114         * @return the node representing the component in the tree
115         */
116        static ComponentVertex getComponentProxy(
117                        ComponentType component, Channel componentChannel) {
118                ComponentProxy componentProxy = null;
119                try {
120                        Field field = getManagerField(component.getClass());
121                        synchronized (component) {
122                                if (!field.isAccessible()) {
123                                        field.setAccessible(true);
124                                        componentProxy = (ComponentProxy)field.get(component);
125                                        if (componentProxy == null) {
126                                                componentProxy = new ComponentProxy(
127                                                                field, component, componentChannel);
128                                        }
129                                        field.setAccessible(false);
130                                } else {
131                                        componentProxy = (ComponentProxy)field.get(component);
132                                        if (componentProxy == null) {
133                                                componentProxy = new ComponentProxy(
134                                                                field, component, componentChannel);
135                                        }
136                                }
137                        }
138                } catch (SecurityException | IllegalAccessException e) {
139                        throw (RuntimeException)(new IllegalArgumentException(
140                                        "Cannot access component's manager attribute")).initCause(e);
141                }
142                return componentProxy;
143        }
144
145        public ComponentType component() {
146                return component;
147        }
148
149        /* (non-Javadoc)
150         * @see org.jgrapes.core.Manager#getChannel()
151         */
152        @Override
153        public Channel channel() {
154                return componentChannel;
155        }
156
157        /**
158         * Return the object itself as value.
159         */
160        @Override
161        public Object defaultCriterion() {
162                return this;
163        }
164
165        /**
166         * Matches the object itself (using identity comparison) or the
167         * {@link Channel} class.
168         * 
169         * @see Channel#isEligibleFor(Object)
170         */
171        @Override
172        public boolean isEligibleFor(Object value) {
173                return value.equals(Channel.class) 
174                                || value == defaultCriterion();
175        }
176}
