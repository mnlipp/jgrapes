019package org.jgrapes.core.internal;
020
021import java.util.concurrent.Callable;
022
023import org.jgrapes.core.Event;
024import org.jgrapes.core.EventPipeline;
025
026/**
027 * A base class for events that perform an action instead of being
028 * handled. Use to synchronize some action with th eother events on
029 * an {@link EventPipeline}.
030 */
031public abstract class ActionEvent<T> extends Event<T> {
032
033    private static Object defaultCriterion = new Object();
034
035    /* default */ static <V> ActionEvent<V> create(Callable<V> action) {
036        return new CallableActionEvent<>(action);
037    }
038
039    /* default */ static ActionEvent<Void> create(Runnable action) {
040        return new RunnableActionEvent(action);
041    }
042
043    @SuppressWarnings("PMD.SignatureDeclareThrowsException")
044    /* default */ abstract void execute() throws Exception;
045
046    @Override
047    public boolean isEligibleFor(Object criterion) {
048        return criterion == defaultCriterion; // NOPMD, comparing references
049    }
050
051    @Override
052    public Object defaultCriterion() {
053        return defaultCriterion;
054    }
055
056    /**
057     * An {@link ActionEvent} that executes a {@link Callable}.
058     *
059     * @param <V> the value type
060     */
061    private static class CallableActionEvent<V> extends ActionEvent<V> {
062        private final Callable<V> action;
063
064        /**
065         * Instantiates a new callable action event.
066         *
067         * @param callable the callable
068         */
069        public CallableActionEvent(Callable<V> callable) {
070            this.action = callable;
071        }
072
073        @SuppressWarnings("PMD.SignatureDeclareThrowsException")
074        /* default */ void execute() throws Exception {
075            setResult(action.call());
076        }
077    }
078
079    /**
080     * An {@link ActionEvent} that executes a {@link Runnable}.
081     */
082    private static class RunnableActionEvent extends ActionEvent<Void> {
083        private final Runnable action;
084
085        /**
086         * Instantiates a new runnable action event.
087         *
088         * @param action the action
089         */
090        public RunnableActionEvent(Runnable action) {
091            this.action = action;
092        }
093
094        @SuppressWarnings("PMD.SignatureDeclareThrowsException")
095        /* default */ void execute() throws Exception {
096            action.run();
097        }
098
099    }
100}
