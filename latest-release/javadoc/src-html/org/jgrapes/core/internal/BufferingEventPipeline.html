019package org.jgrapes.core.internal;
020
021import java.util.concurrent.ExecutorService;
022import org.jgrapes.core.Channel;
023import org.jgrapes.core.Components;
024import org.jgrapes.core.Event;
025import org.jgrapes.core.events.Start;
026
027/**
028 * The buffering event pipeline is used before a tree has been started. 
029 * It simply buffers all events until a {@link Start} event is added.
030 */
031public class BufferingEventPipeline implements InternalEventPipeline {
032
033    private final ComponentTree componentTree;
034    /** Buffered events. */
035    private EventQueue buffered = new EventQueue();
036    /** The event pipeline that we delegate to after the start
037     * event has been detected. */
038    private InternalEventPipeline activePipeline;
039
040    /**
041     * Instantiates a new buffering event pipeline.
042     *
043     * @param componentTree the component tree
044     */
045    /* default */ BufferingEventPipeline(ComponentTree componentTree) {
046        super();
047        this.componentTree = componentTree;
048    }
049
050    @Override
051    public void merge(InternalEventPipeline other) {
052        synchronized (this) {
053            if (!(other instanceof BufferingEventPipeline)) {
054                throw new IllegalArgumentException(
055                    "Can only merge events from an BufferingEventPipeline.");
056            }
057            buffered.addAll(((BufferingEventPipeline) other).retrieveEvents());
058        }
059    }
060
061    @Override
062    public <T extends Event<?>> T add(T event, Channel... channels) {
063        synchronized (this) {
064            // If thread1 adds the start event and thread2 gets here before we
065            // have changed the event processor for the tree, forward the
066            // event to the event processor that should already have been used.
067            if (activePipeline != null) {
068                activePipeline.add(event, channels);
069                return event;
070            }
071            // Invoke although argument is null!
072            ((EventBase<?>) event).generatedBy(null);
073            buffered.add(event, channels);
074            if (event instanceof Start) {
075                // Merge all events into a "standard" event processor
076                // and set it as default processor for the tree (with
077                // any thread specific pipelines taking precedence).
078                EventProcessor processor = new EventProcessor(componentTree);
079                activePipeline
080                    = new FeedBackPipelineFilter(componentTree, processor);
081                componentTree.setEventPipeline(activePipeline);
082                processor.add(buffered);
083            }
084            return event;
085        }
086    }
087
088    /* default */ EventQueue retrieveEvents() {
089        synchronized (this) {
090            EventQueue old = buffered;
091            buffered = new EventQueue();
092            return old;
093        }
094    }
095
096    /*
097     * (non-Javadoc)
098     * 
099     * @see org.jgrapes.core.internal.InternalEventPipeline#executorService()
100     */
101    @Override
102    public ExecutorService executorService() {
103        return Components.defaultExecutorService();
104    }
105
106    /*
107     * (non-Javadoc)
108     * 
109     * @see java.lang.Object#toString()
110     */
111    @Override
112    public String toString() {
113        StringBuilder builder = new StringBuilder(50);
114        builder.append("BufferingEventPipeline [");
115        if (buffered != null) {
116            builder.append("buffered=");
117            builder.append(buffered);
118        }
119        builder.append(']');
120        return builder.toString();
121    }
122}
