238    private static Map<Object, String> objectIds // NOPMD
239        = new WeakHashMap<>();
240    private static Map<Class<?>, AtomicLong> idCounters // NOPMD
241        = new WeakHashMap<>();

243    private static String getId(Class<?> scope, Object object) {
244        if (object == null) {
245            return "?";
246        }
247        synchronized (objectIds) {
248            return objectIds.computeIfAbsent(object,
249                key -> Long.toString(idCounters
250                    .computeIfAbsent(scope, newKey -> new AtomicLong())
251                    .incrementAndGet()));
252
253        }
254    }

263    public static String className(Class<?> clazz) {

271    /**
272     * Returns the simple name of a class. Contrary to 
273     * {@link Class#getSimpleName()}, this method returns
274     * the last segement of the full name for anonymous
275     * classes (instead of an empty string).
276     * 
277     * @param clazz the class
278     * @return the name
279     */
280    public static String simpleClassName(Class<?> clazz) {
281        if (!clazz.isAnonymousClass()) {
282            return clazz.getSimpleName();
283        }
284        // Simple name of anonymous class is empty
285        String name = clazz.getName();
286        int lastDot = name.lastIndexOf('.');
287        if (lastDot <= 0) {
288            return name;
289        }
290        return name.substring(lastDot + 1);
291    }

293    /**
294     * Returns an id of the object that is unique within a specific scope. Ids
295     * are generated and looked up in the scope of the object's class unless the
296     * class implements {@link IdInfoProvider}.
297     * 
298     * @param object
299     *            the object
300     * @return the object's name
301     */
302    public static String objectId(Object object) {
303        if (object == null) {
304            return "?";
305        }
306        if (object instanceof IdInfoProvider) {
307            return getId(((IdInfoProvider) object).idScope(),
308                ((IdInfoProvider) object).idObject());
309        } else {
310            return getId(object.getClass(), object);
311        }
312    }

355     * {@link Components#schedule(TimeoutHandler, Instant)}.
356     */
357    public static class Timer {
358        private final Scheduler scheduler;
359        private final TimeoutHandler timeoutHandler;
360        private Instant scheduledFor;
361
362        private Timer(Scheduler scheduler,
363                TimeoutHandler timeoutHandler, Instant scheduledFor) {
364            this.scheduler = scheduler;
365            this.timeoutHandler = timeoutHandler;
366            this.scheduledFor = scheduledFor;
367        }
368
369        /**
370         * Reschedules the timer for the given instant.
371         * 
372         * @param scheduledFor the instant
373         */
374        public void reschedule(Instant scheduledFor) {
375            scheduler.reschedule(this, scheduledFor);
376        }
377
378        /**
379         * Reschedules the timer for the given duration after now.
380         * 
381         * @param scheduledFor the timeout
382         */
383        public void reschedule(Duration scheduledFor) {
384            reschedule(Instant.now().plus(scheduledFor));
385        }
386
387        /**
388         * Returns the timeout handler of this timer.
389         * 
390         * @return the handler
391         */
392        public TimeoutHandler timeoutHandler() {
393            return timeoutHandler;
394        }
395
396        /**
397         * Returns the instant that this handler is scheduled for.
398         * 
399         * @return the instant or `null` if the timer has been cancelled.
400         */
401        public Instant scheduledFor() {
402            return scheduledFor;
403        }
404
405        /**
406         * Cancels this timer.
407         */
408        public void cancel() {
409            scheduler.cancel(this);
410        }
411    }

433    /**
434     * A general purpose scheduler.
435     */
436    private static class Scheduler extends Thread {
437
438        private final PriorityQueue<Timer> timers
439            = new PriorityQueue<>(10,
440                Comparator.comparing(Timer::scheduledFor));
441
442        /**
443         * Instantiates a new scheduler.
444         */
445        public Scheduler() {
446            setName("Components.Scheduler");
447            setDaemon(true);
448            start();
449        }
450
451        /**
452         * Schedule the handler and return the resulting timer.
453         *
454         * @param timeoutHandler the timeout handler
455         * @param scheduledFor the scheduled for
456         * @return the timer
457         */
458        public Timer schedule(
459                TimeoutHandler timeoutHandler, Instant scheduledFor) {
460            @SuppressWarnings("PMD.AccessorClassGeneration")
461            Timer timer = new Timer(this, timeoutHandler, scheduledFor);
462            synchronized (timers) {
463                timers.add(timer);
464                timers.notifyAll();
465            }
466            return timer;
467        }
468
469        private void reschedule(Timer timer, Instant scheduledFor) {
470            synchronized (timers) {
471                timers.remove(timer);
472                timer.scheduledFor = scheduledFor;
473                timers.add(timer);
474                timers.notifyAll();
475            }
476        }
477
478        private void cancel(Timer timer) {
479            synchronized (timers) {
480                timers.remove(timer);
481                timers.notifyAll();
482                timer.scheduledFor = null;
483            }
484        }
485
486        @Override
487        public void run() {
488            while (true) {
489                while (true) {
490                    @SuppressWarnings("PMD.AvoidFinalLocalVariable")
491                    final Timer first;
492                    synchronized (timers) {
493                        first = timers.peek();
494                        if (first == null
495                            || first.scheduledFor().isAfter(Instant.now())) {
496                            break;
497                        }
498                        timers.poll();
499                    }
500                    timerExecutorService.submit(
501                        () -> first.timeoutHandler().timeout(first));
502                }
503                try {
504                    synchronized (timers) {
505                        if (timers.size() == 0) {
506                            timers.wait();
507                        } else {
508                            timers
509                                .wait(Math.max(1,
510                                    Duration.between(Instant.now(),
511                                        timers.peek().scheduledFor())
512                                        .toMillis()));
513                        }
514                    }
515                } catch (Exception e) { // NOPMD
516                    // Keep running.
517                }
518            }
519        }
520    }
