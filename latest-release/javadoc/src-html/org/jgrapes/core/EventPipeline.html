019package org.jgrapes.core;
020
021import java.util.concurrent.ExecutorService;
022
023import org.jgrapes.core.Components.IdInfoProvider;
024
025/**
026 * An interface that describes a queue of events that are sent to the components
027 * of the associated tree. Any events fired by the components while handling
028 * an event from the pipeline are added at the end of the pipeline.
029 * 
030 * An event pipeline is run by a single thread from aan executor service. 
031 * Adding several events to the same pipeline therefore ensures that they 
032 * are executed in sequence.
033 */
034public interface EventPipeline extends IdInfoProvider {
035
036        /**
037         * Add an event to be sent to components listening for such events on
038         * the given channels to the end of the queue. If no channels are
039         * specified as parameters, the event is fired on the event's 
040         * channel (see {@link Event#channels()}). If the event doesn't
041         * specify channels either, the channel depends on how the
042         * event pipeline was obtained. Event pipelines obtained from a
043         * component's manager use the component's channel as fall back.
044         * 
045         * @param <T> the event's type
046         * @param event the event to process
047         * @param channels the channels that the event was fired on
048         * @return the event (for easy chaining)
049         */
050        <T extends Event<?>> T fire(T event, Channel... channels);
051
052        /**
053         * All pipelines use the same id scope to make them uniquely identifiable
054         * by their number.
055         */
056        @Override
057        default Class<?> idScope() {
058                return EventPipeline.class;
059        }
060        
061        /**
062         * Returns the executor service used by this event pipeline to
063         * obtain its thread. If a component needs to create a thread for its
064         * own use, it should obtain it from the component's event pipeline's
065         * executor service in order to conform with standard resource
066         * management.
067         * 
068         * @return the executor service
069         */
070        ExecutorService executorService();
071}
