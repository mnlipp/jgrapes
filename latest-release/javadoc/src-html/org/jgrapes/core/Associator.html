019package org.jgrapes.core;
020
021import java.util.Optional;
022import java.util.function.Supplier;
023
024/**
025 * Implemented by classes that allow arbitrary objects to be associated
026 * with instances.
027 */
028public interface Associator {
029        
030        /**
031         * Establishes a "named" association to an associated object. Note that 
032         * anything that represents an id can be used as value for 
033         * parameter `name`, it does not necessarily have to be a string.
034         * 
035         * Passing `null` as parameter `with` clears the association.
036         * 
037         * @param by the "name"
038         * @param with the object to be associated
039         * @return the sub channel for easy chaining
040         */
041        Associator setAssociated(Object by, Object with);
042
043        /**
044         * Retrieves the associated object following the association 
045         * with the given "name". This general version of the method
046         * supports the retrieval of values of arbitrary types
047         * associated by any "name" types. 
048         * 
049         * @param by the "name"
050         * @param type the type of the value to be retrieved
051         * @param <V> the type of the value to be retrieved
052         * @return the associate, if any
053         */
054        <V> Optional<V> associated(Object by, Class<V> type);
055        
056        /**
057         * Retrieves the associated object following the association 
058         * with the given "name". If no association exists, the
059         * object is created and the association is established.  
060         * 
061         * @param by the "name"
062         * @param supplier the supplier
063         * @param <V> the type of the value to be retrieved
064         * @return the associate, if any
065         */
066        @SuppressWarnings("unchecked")
067        default <V> V associated(Object by, Supplier<V> supplier) {
068                return (V)associated(by, Object.class).orElseGet(() -> {
069                        V associated = supplier.get();
070                        setAssociated(by, associated);
071                        return associated;
072                });
073        }
074        
075        /**
076         * Retrieves the associated object following the association 
077         * with the given name. This convenience methods simplifies the
078         * retrieval of String values associated by a (real) name.
079         * 
080         * @param by the name
081         * @return the associate, if any
082         */
083        default Optional<String> associated(String by) {
084                return associated(by, String.class);
085        }
086        
087        /**
088         * Retrieves the associated object following the association 
089         * with the given class. The associated object must be an instance
090         * of the given class.
091         * 
092         * @param <V> the type of the value
093         * @param by the name
094         * @return the associate, if any
095         */
096        default <V> Optional<V> associated(Class<V> by) {
097                return associated(by, by);
098        }
099
100}
