040/**
041 * This class is the base class for all events.
042 *  
043 * By default (i.e. as implemented by this class), the event's kind is
044 * represented by its Java class and the eligibility is based on
045 * the "is a" relationship between classes. An event is eligible if its class
046 * is equal to or a super class of the class used as criterion. 
047 * This default behavior can be changed by overriding the
048 * methods from {@link Eligible}. See {@link NamedEvent} as an example.
049 * 
050 * @param <T>
051 *            the result type of the event. Use {@link Void} if handling the
052 *            event does not produce a result
053 */
054public class Event<T> extends EventBase<T> {
055
056        /** The channels that this event is to be fired on if no
057         * channels are specified explicitly when firing. */
058        private Channel[] channels = null;
059        /** Indicates that the event should not processed further. */
060        private boolean stopped = false;
061        /** The results of handling the event (if any). */
062        private List<T> results;
063        /** Context data. */
064        private Map<Object,Object> contextData = null;
065        private boolean cancelled = false;
066        
067        /**
068         * Creates a new event. Passing channels is equivalent to first
069         * creating the event and then calling {@link #setChannels(Channel...)}
070         * with the given channels.
071         * 
072         * @param channels the channels to set
073         */
074        public Event(Channel... channels) {
075                super();
076                if (channels.length > 0) {
077                        setChannels(channels);
078                }
079        }
080
081        /**
082         * Returns the class of this event as representation of its kind.
083         * 
084         * @return the class of this event
085         * 
086         * @see org.jgrapes.core.Eligible#defaultCriterion()
087         */
088        @Override
089        public Object defaultCriterion() {
090                return getClass();
091        }
092
093        /**
094         * Returns <code>true</code> if the `criterion` 
095         * is of the same class or a base class of this event's class.
096         * 
097         * @see org.jgrapes.core.Eligible#isEligibleFor(java.lang.Object)
098         */
099        @Override
100        public boolean isEligibleFor(Object criterion) {
101                return Class.class.isInstance(criterion)
102                                && ((Class<?>)criterion).isAssignableFrom(getClass());
103        }
104
105        /**
106         * Return the event pipeline that currently processes the event
107         * (if any).
108         * 
109         * @return the event pipeline if the event is being processed
110         */
111        public Optional<EventPipeline> processedBy() {
112                return super.processedBy();
113        }
114
115        /**
116         * Implements the default behavior for handling events thrown
117         * by a handler. Fires a {@link HandlingError handling error} event
118         * for this event and the given throwable.
119         * 
120         * @see HandlingError
121         */
122        @Override
123        protected void handlingError(
124                        EventPipeline eventProcessor, Throwable throwable) {
125                eventProcessor.fire(
126                                new HandlingError(this, throwable), channels());
127        }
128
129        /**
130         * Sets the channels that the event is fired on if no channels
131         * are specified explicitly when firing the event
132         * (see {@link org.jgrapes.core.Manager#fire(Event, Channel...)}).
133         * 
134         * @param channels the channels to set
135         * @return the object for easy chaining
136         * 
137         * @throws IllegalStateException if the method is called after
138         * this event has been fired
139         */
140        public Event<T> setChannels(Channel... channels) {
141                if (enqueued()) {
142                        throw new IllegalStateException(
143                                        "Channels cannot be changed after fire");
144                }
145                this.channels = channels;
146                return (Event<T>)this;
147        }
148
149        /**
150         * Returns the channels associated with the event. Before an
151         * event has been fired, this returns the channels set with
152         * {@link #setChannels(Channel[])}. After an event has been
153         * fired, this returns the channels that the event has
154         * effectively been fired on 
155         * (see {@link Manager#fire(Event, Channel...)}).
156         * 
157         * @return the channels
158         */
159        @Override
160        public Channel[] channels() {
161                return channels;
162        }
163
164        /**
165         * Returns the subset of channels that are assignable to the given type.
166         * 
167         * @param <C> the given type's class
168         * @param type the class to look for
169         * @return the filtered channels
170         * @see #channels()
171         */
172        @SuppressWarnings("unchecked")
173        public <C> C[] channels(Class<C> type) {
174                return Arrays.stream(channels)
175                        .filter(c -> type.isAssignableFrom(c.getClass())).toArray(
176                                        size -> (C[])Array.newInstance(type, size));
177        }
178
179        /**
180         * Execute the given handler for all channels of the given type.
181         * 
182         * @param <E> the type of the event
183         * @param <C> the type of the channel
184         * @param type the channel type
185         * @param handler the handler
186         */
187        @SuppressWarnings("unchecked")
188        public <E extends EventBase<?>, C extends Channel> void forChannels(
189                        Class<C> type, BiConsumer<E, C> handler) {
190                Arrays.stream(channels)
191                        .filter(c -> type.isAssignableFrom(c.getClass()))
192                        .forEach(c -> handler.accept((E)this, (C)c));
193        }
194        
195        /**
196         * Returns the events to be thrown when this event has completed
197         * (see {@link #isDone()}).
198         * 
199         * @return the completed events
200         */
201        public Set<Event<?>> completionEvents() {
202                return completionEvents == null ? Collections.emptySet()
203                                : Collections.unmodifiableSet(completionEvents);
204        }
205
206        /**
207         * Adds the given event to the events to be thrown when this event 
208         * has completed (see {@link #isDone()}). Such an event is called 
209         * a "completion event".
210         * 
211         * Completion events are considered to be caused by the event that 
212         * caused the completed event. If an event *e1* caused an event
213         * *e2* which has a completion event *e2c*, *e1* is only put in 
214         * state completed when *e2c* has been handled.
215         * 
216         * Completion events are handled by the same {@link EventProcessor}
217         * as the event that has been completed.
218         * 
219         * @param completionEvent the completion event to add
220         * @return the object for easy chaining
221         */
222        public Event<T> addCompletionEvent(Event<?> completionEvent) {
223                if (completionEvents == null) {
224                        completionEvents = new HashSet<>();
225                }
226                completionEvents.add(completionEvent);
227                return (Event<T>)this;
228        }
229
230        /* (non-Javadoc)
231         * @see org.jgrapes.core.internal.EventBase#setRequiresResult(boolean)
232         */
233        @Override
234        public Event<T> setRequiresResult(boolean value) {
235                return (Event<T>)super.setRequiresResult(value);
236        }
237
238        /**
239         * Check if this event has completed. An event is completed
240         * if 
241         *  * all its handlers have been invoked (or the event has
242         *    been stopped or cancelled), 
243         *  * all events caused by it have completed,
244         *  * no {@link CompletionLock}s remain, and  
245         *  * a result has been set (only required if 
246         *    {@link #setRequiresResult(boolean)} has been called with `true`).
247         * 
248         * @return the completed state
249         */
250        @Override
251        public boolean isDone() {
252                return completed;
253        }
254
255        /**
256         * Invoked after all handlers for the event have been executed. 
257         * May be overridden by derived classes to cause some immediate effect
258         * (instead of e.g. waiting for the completion event). The default 
259         * implementation does nothing. This method is invoked by the event 
260         * handler thread and must not block.
261         */
262        protected void handled() {
263        }
264        
265        /**
266         * Can be called during the execution of an event handler to indicate
267         * that the event should not be processed further. All remaining 
268         * handlers for this event will be skipped.
269         * 
270         * @return the object for easy chaining
271         */
272        public Event<T> stop() {
273                stopped = true;
274                return this;
275        }
276
277        /**
278         * Returns <code>true</code> if {@link #stop} has been called.
279         * 
280         * @return the stopped state
281         */
282        public boolean isStopped() {
283                return stopped;
284        }
285
286        /**
287         * Prevents the invocation of further handlers (like {@link #stop()} 
288         * and (in addition) the invocation of any added completed events.
289         * 
290         * @param mayInterruptIfRunning ignored
291         * @return `false` if the event has already been completed
292         * @see java.util.concurrent.Future#cancel(boolean)
293         */
294        @Override
295        public boolean cancel(boolean mayInterruptIfRunning) {
296                if (!completed && !cancelled) {
297                        stop();
298                        cancelled = true;
299                        return true;
300                }
301                return false;
302        }
303
304        @Override
305        public boolean isCancelled() {
306                return cancelled;
307        }
308
309        /**
310         * Sets the result of handling this event. If this method is invoked 
311         * more then once, the various results are collected in a list. This
312         * can happen if the event is handled by several components. 
313         * 
314         * @param result the result to set
315         * @return the object for easy chaining
316         */
317        public synchronized Event<T> setResult(T result) {
318                if (results == null) {
319                        // Make sure that we have a valid result before 
320                        // calling decrementOpen
321                        results = new ArrayList<T>();
322                        results.add(result);
323                        firstResultAssigned();
324                        return (Event<T>)this;
325                }
326                results.add(result);
327                return (Event<T>)this;
328        }
329
330        /**
331         * Allows access to the intermediate result before the 
332         * completion of the event. 
333         * 
334         * @return the intermediate results or `null`
335         */
336        protected List<T> currentResults() {
337                return Collections.unmodifiableList(results);
338        }
339        
340        /**
341         * Tie the result of this event to the result of the other event.
342         * Changes of either event's results will subsequently be applied
343         * to both events.
344         * <P>
345         * This is useful when an event is replaced by another event during
346         * handling like:
347         * {@code fire((new Event()).tieTo(oldEvent.stop()))}  
348         * 
349         * @param other the event to tie to
350         * @return the object for easy chaining
351         */
352        public synchronized Event<T> tieTo(Event<T> other) {
353                if (other.results == null) {
354                        other.results = new ArrayList<T>();
355                }
356                results = other.results;
357                return (Event<T>)this;
358        }
359        
360        /**
361         * Waits for the event to be completed (see {@link #isDone()})
362         * and returns the first (or only) result.
363         * 
364         * @see Future#get()
365         */
366        @Override
367        public T get() throws InterruptedException {
368                while (true) {
369                        synchronized(this) {
370                                if (completed) {
371                                        return ((results == null || results.isEmpty()))
372                                                        ? null : results.get(0);
373                                }
374                                wait();
375                        }
376                }
377        }
378
379        /**
380         * Causes the invoking thread to wait until the processing of the 
381         * event has been completed (see {@link #isDone()}) or the given 
382         * timeout has expired and returns the first (or only) result. 
383         * 
384         * @return the result
385         * @see java.util.concurrent.Future#get(long, java.util.concurrent.TimeUnit)
386         */
387        @Override
388        public T get(long timeout, TimeUnit unit)
389                throws InterruptedException, TimeoutException {
390                synchronized(this) {
391                        if (completed) {
392                                return (results == null || results.isEmpty())
393                                                ? null : results.get(0);
394                        }
395                        wait(unit.toMillis(timeout));
396                }
397                if (completed) {
398                        return (results == null || results.isEmpty())
399                                        ? null : results.get(0);
400                }
401                throw new TimeoutException();
402        }
403
404        /**
405         * Waits for the event to be completed (see {@link #isDone()})
406         * and returns the list of results (which may be empty if the
407         * event's result type is {@link Void}).
408         * 
409         * @return the results
410         * @see Future#get()
411         */
412        public List<T> results() throws InterruptedException {
413                while (true) {
414                        synchronized(this) {
415                                if (completed) {
416                                        return (results == null ? Collections.emptyList() 
417                                                        : Collections.unmodifiableList(results));
418                                }
419                                wait();
420                        }
421                }
422        }
423
424        /**
425         * Causes the invoking thread to wait until the processing of the 
426         * event has been completed (see {@link #isDone()}) or given timeout 
427         * has expired and returns the list of results (which may be empty
428         * if the event's result type is {@link Void}). 
429         * 
430         * @return the results
431         * @see java.util.concurrent.Future#get(long, java.util.concurrent.TimeUnit)
432         */
433        public List<T> results(long timeout, TimeUnit unit)
434                throws InterruptedException, TimeoutException {
435                synchronized(this) {
436                        if (completed) {
437                                return (results == null ? Collections.emptyList() 
438                                                : Collections.unmodifiableList(results));
439                        }
440                        wait(unit.toMillis(timeout));
441                }
442                if (completed) {
443                        return (results == null ? Collections.emptyList() 
444                                        : Collections.unmodifiableList(results));
445                }
446                throw new TimeoutException();
447        }
448
449        @Override
450        public Event<T> setAssociated(Object by, Object with) {
451                if (contextData == null) {
452                        contextData = new ConcurrentHashMap<>();
453                }
454                if (with != null) {
455                        contextData.put(by, with);
456                } else {
457                        contextData.remove(by);
458                }
459                return (Event<T>)this;
460        }
461
462        @Override
463        public <V> Optional<V> associated(Object by, Class<V> type) {
464                if (contextData == null) {
465                        return Optional.empty();
466                }
467                return Optional.ofNullable(contextData.get(by))
468                        .filter(found -> type.isAssignableFrom(found.getClass()))
469                        .map(match -> type.cast(match));
470        }
471        
472        /* (non-Javadoc)
473         * @see java.lang.Object#toString()
474         */
475        @Override
476        public String toString() {
477                StringBuilder builder = new StringBuilder();
478                builder.append(Components.objectName(this));
479                builder.append(" [");
480                if (channels != null) {
481                        builder.append("channels=");
482                        builder.append(Channel.toString(channels));
483                }
484                builder.append("]");
485                return builder.toString();
486        }
