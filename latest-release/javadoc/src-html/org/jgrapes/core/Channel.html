019package org.jgrapes.core;
020
021/**
022 * Classes that implements this interface can be used as a communication bus 
023 * for sending events between components.
024 * <P>
025 * For ordinary usage, the implementing classes {@link ClassChannel}
026 * and {@link NamedChannel} should be sufficient. If another type of
027 * <code>Channel</code> is needed, its implementation of this interface 
028 * must make sure that {@link Eligible#isEligibleFor(Object)} returns
029 * <code>true</code> if called with <code>Channel.class</code>
030 * as parameter, else channels of the new type will not participate
031 * in broadcasts.
032 * <P>
033 * Objects of type <code>Channel</code> must be immutable.
034 * 
035 * @see Channel#BROADCAST
036 */
037public interface Channel extends Eligible {
038
039        /**
040         * A special channel object that can be passed to the constructor
041         * of {@link Component#Component(Channel)}.
042         * 
043         * @see Component#Component(Channel)
044         */
045        public static final Channel SELF = new ClassChannel();
046        
047        /**
048         * A special channel that can be used to send events to
049         * all components.
050         */
051        public static final Channel BROADCAST = new ClassChannel() {
052
053                /**
054                 * Returns <code>Channel.class</code>, the value that must
055                 * by definition be matched by any channel.
056                 * 
057                 * @return <code>Channel.class</code>
058                 */
059                @Override
060                public Object defaultCriterion() {
061                        return Channel.class;
062                }
063
064                /**
065                 * Always returns {@code true} because the broadcast channel
066                 * is matched by every channel.
067                 * 
068                 * @return {@code true}
069                 */
070                @Override
071                public boolean isEligibleFor(Object criterion) {
072                        return true;
073                }
074
075                /* (non-Javadoc)
076                 * @see org.jgrapes.core.ClassChannel#toString()
077                 */
078                @Override
079                public String toString() {
080                        return "BROADCAST";
081                }
082        };
083
084        /**
085         * Returns a textual representation of a channel's criterion.
086         * 
087         * @param criterion the criterion
088         * @return the representation
089         */
090        static String criterionToString(Object criterion) {
091                StringBuilder builder = new StringBuilder();
092                if (criterion instanceof Class) {
093                        if (criterion == Channel.class) {
094                                builder.append("BROADCAST");
095                        } else {
096                                builder.append(Components.classToString((Class<?>) criterion));
097                        }
098                } else {
099                        builder.append(criterion);
100                }
101                return builder.toString();
102        }
103
104        /**
105         * Returns a textual representation of a channel.
106         * 
107         * @param channel the channel
108         * @return the representation
109         */
110        static String toString(Channel channel) {
111                StringBuilder builder = new StringBuilder();
112                if ((channel instanceof ClassChannel)
113                        || (channel instanceof NamedChannel)) {
114                        builder.append(criterionToString(channel.defaultCriterion()));
115                } else if (channel == channel.defaultCriterion()) {
116                        builder.append(Components.objectName(channel));
117                } else {
118                        builder.append(channel.toString());
119                }
120                return builder.toString();
121        }
122
123        /**
124         * Returns a textual representation of an array of channels.
125         * 
126         * @param channels the channels
127         * @return the representation
128         */
129        static String toString(Channel[] channels) {
130                StringBuilder builder = new StringBuilder();
131                builder.append("[");
132                boolean first = true;
133                for (Channel c: channels) {
134                        if (!first) {
135                                builder.append(", ");
136                        }
137                        builder.append(Channel.toString(c));
138                        first = false;
139                }
140                builder.append("]");
141                return builder.toString();
142        }
143
144}
