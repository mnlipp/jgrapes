019package org.jgrapes.core;
020
021import org.jgrapes.core.annotation.Handler;
022
023/**
024 * Instances of this this interface can be used as a communication 
025 * bus for sending events between components. The instances work
026 * as identifiers of channels. Their only functionality is defined
027 * by the {@link Eligible} interface, which allows a channel
028 * (used as attribute of an {@link Event}) to be matched against 
029 * a criterion specified in a {@link Handler}.
030 * 
031 * The need to use the {@link Eligible} interface for comparison
032 * arises from the fact that we cannot use objects as values in
033 * annotations. It must therefore be possible to match channels
034 * (objects) against criteria that can be expressed as constant 
035 * values.
036 * 
037 * Some values have been defined to represent special criteria.
038 * 
039 * * If the value `Channel.class` is specified as criterion in
040 *   a handler, all channel instances match. It is the "catch-all"
041 *   criterion.
042 * 
043 * * If the value `{@link Default}.class` is specified as criterion
044 *   in a handler, the channels from an {@link Event} are
045 *   matched agains the criterion from the component's channel
046 *   (returned by the {@link Manager#channel() channel()} method).  
047 * 
048 * The predefined {@link #BROADCAST} channel is a channel instance
049 * that implements the {@link Eligible} interface in such a way that
050 * all criteria match. Events fired on the {@link #BROADCAST} channel
051 * will therefore be accepted by all handlers (as its name suggests).
052 * 
053 * For ordinary usage, the implementing classes {@link ClassChannel}
054 * and {@link NamedChannel} should be sufficient. If another type of
055 * `Channel` is needed, its implementation must make sure that 
056 * {@link Eligible#isEligibleFor(Object)} returns
057 * `true` if called with `Channel.class` as parameter, else channels 
058 * of the new type will not be delivered to "catch-all" handlers.
059 * 
060 * Objects of type <code>Channel</code> must be immutable.
061 * 
062 * @see Channel#BROADCAST
063 */
064public interface Channel extends Eligible {
065
066        /**
067         * This interface's class can be used to specify the component's 
068         * channel (see {@link Component#channel()}) as criterion in 
069         * handler annotations.
070         * 
071         * Using the component's channel for comparison is the default 
072         * if no channels are specified in the annotation, so specifying 
073         * only this class in the handler annotation is equivalent
074         * to specifying no channel at all. This special channel type is required
075         * if you want to specify a handler that handles events fired on the 
076         * component's channel or on additional channels.
077         */
078        public interface Default extends Channel {
079        }
080        
081        /**
082         * A special channel object that can be passed as argument to 
083         * the constructor of {@link Component#Component(Channel)}. 
084         * Doing this sets the component's channel to the component 
085         * (which is not available as argument when calling the 
086         * constructor).
087         * 
088         * @see Component#Component(Channel)
089         */
090        public static final Channel SELF = new ClassChannel();
091        
092        /**
093         * A special channel instance that can be used to send events to
094         * all components.
095         */
096        public static final Channel BROADCAST = new ClassChannel() {
097
098                /**
099                 * Returns <code>Channel.class</code>, the value that must
100                 * by definition be matched by any channel.
101                 * 
102                 * @return <code>Channel.class</code>
103                 */
104                @Override
105                public Object defaultCriterion() {
106                        return Channel.class;
107                }
108
109                /**
110                 * Always returns {@code true} because the broadcast channel
111                 * is matched by every channel.
112                 * 
113                 * @return {@code true}
114                 */
115                @Override
116                public boolean isEligibleFor(Object criterion) {
117                        return true;
118                }
119
120                /* (non-Javadoc)
121                 * @see org.jgrapes.core.ClassChannel#toString()
122                 */
123                @Override
124                public String toString() {
125                        return "BROADCAST";
126                }
127        };
128
129        /**
130         * Returns a textual representation of a channel's criterion.
131         * 
132         * @param criterion the criterion
133         * @return the representation
134         */
135        static String criterionToString(Object criterion) {
136                StringBuilder builder = new StringBuilder();
137                if (criterion instanceof Class) {
138                        if (criterion == Channel.class) {
139                                builder.append("BROADCAST");
140                        } else {
141                                builder.append(Components.className((Class<?>) criterion));
142                        }
143                } else {
144                        builder.append(criterion);
145                }
146                return builder.toString();
147        }
148
149        /**
150         * Returns a textual representation of a channel.
151         * 
152         * @param channel the channel
153         * @return the representation
154         */
155        static String toString(Channel channel) {
156                StringBuilder builder = new StringBuilder();
157                if ((channel instanceof ClassChannel)
158                        || (channel instanceof NamedChannel)) {
159                        builder.append(criterionToString(channel.defaultCriterion()));
160                } else if (channel == channel.defaultCriterion()) {
161                        builder.append(Components.objectName(channel));
162                } else {
163                        builder.append(channel.toString());
164                }
165                return builder.toString();
166        }
167
168        /**
169         * Returns a textual representation of an array of channels.
170         * 
171         * @param channels the channels
172         * @return the representation
173         */
174        static String toString(Channel[] channels) {
175                StringBuilder builder = new StringBuilder();
176                builder.append("[");
177                boolean first = true;
178                for (Channel c: channels) {
179                        if (!first) {
180                                builder.append(", ");
181                        }
182                        builder.append(Channel.toString(c));
183                        first = false;
184                }
185                builder.append("]");
186                return builder.toString();
187        }
188
189}
