057/**
058 * A queue based buffer pool. Using buffers from a pool is an important
059 * feature for limiting the computational resources for an {@link IOSubchannel}.
060 * A producer of {@link Output} events that simply creates its own buffers
061 * may produce and enqueue a large number of events that are not consumed
062 * as fast as they are produced. 
063 * 
064 * Using a buffer pool with a typical size of two synchronizes the 
065 * producer and the consumers of events nicely. The producer
066 * (thread) holds one buffer and fills it, the consumer (thread) holds 
067 * the other buffer and works with its content. If the producer finishes
068 * before the consumer, it has to stop until the consumer has processed 
069 * previous event and releases the buffer. The consumer can continue
070 * without delay, because the data has already been prepared and enqueued
071 * as the next event.
072 * 
073 * One of the biggest problems when using a pool can be to identify 
074 * leaking buffers, i.e. buffers that are not properly returned to the pool.
075 * This implementation therefore tracks all created buffers 
076 * (with a small overhead) and logs a warning if a buffer is no longer
077 * used (referenced) but has not been returned to the pool. If the
078 * log level for {@link ManagedBufferPool} is set to {@link Level#FINE},
079 * the warning also includes a stack trace of the call to {@link #acquire()}
080 * that handed out the buffer. Providing this information in addition 
081 * obviously requires a larger overhead and is therefore limited to the
082 * finer log levels.
083 *
084 * @param <W> the type of the wrapped (managed) buffer
085 * @param <T> the type of the content buffer that is wrapped
086 */
087@SuppressWarnings({ "PMD.ExcessiveImports", "PMD.NcssCount",
088    "PMD.EmptyCatchBlock" })
089public class ManagedBufferPool<W extends ManagedBuffer<T>, T extends Buffer>
090        implements BufferCollector<W> {

131    /**
132     * Create a pool that contains a varying number of (wrapped) buffers.
133     * The pool is initially empty. When buffers are requested and none 
134     * are left in the pool, new buffers are created up to the given 
135     * upper limit. Recollected buffers are put in the pool until it holds
136     * the number specified by the lower threshold. Any additional 
137     * recollected buffers are discarded. 
138     * 
139     * @param wrapper the function that converts buffers to managed buffers
140     * @param bufferFactory a function that creates a new buffer
141     * @param lowerThreshold the number of buffers kept in the pool
142     * @param upperLimit the maximum number of buffers
143     */
144    public ManagedBufferPool(BiFunction<T, BufferCollector<W>, W> wrapper,
145            Supplier<T> bufferFactory, int lowerThreshold, int upperLimit) {
146        this.wrapper = wrapper;
147        this.bufferFactory = bufferFactory;
148        preservedBufs = lowerThreshold;
149        maximumBufs = upperLimit;
150        createdBufs = new AtomicInteger();
151        queue = new ArrayBlockingQueue<>(lowerThreshold);
152        bufferMonitor = new BufferMonitor(upperLimit);
153        MBeanView.addPool(this);
154    }

164    public ManagedBufferPool(BiFunction<T, BufferCollector<W>, W> wrapper,
165            Supplier<T> bufferFactory, int buffers) {

201    private W createBuffer() {
202        createdBufs.incrementAndGet();
203        W buffer = wrapper.apply(this.bufferFactory.get(), this);
204        bufferMonitor.put(buffer, new BufferProperties());
205        bufferSize = buffer.capacity();
206        return buffer;
207    }
208
209    /**
210     * Removes the buffer from the pool.
211     * 
212     * @param buffer the buffer to remove
213     */
214    private void removeBuffer(W buffer) {
215        createdBufs.decrementAndGet();
216        if (bufferMonitor.remove(buffer) == null) {
217            if (logger.isLoggable(Level.FINE)) {
218                logger.log(Level.WARNING,
219                    "Attempt to remove unknown buffer from pool.",
220                    new Throwable());
221            } else {
222                logger.warning("Attempt to remove unknown buffer from pool.");
223            }
224        }
225    }

227    /**
228     * Returns the size of the buffers managed by this pool.
229     * 
230     * @return the buffer size
231     */
232    public int bufferSize() {
233        if (bufferSize < 0) {
234            createBuffer().unlockBuffer();
235        }
236        return bufferSize;
237    }

239    /**
240     * Acquires a managed buffer from the pool. If the pool is empty,
241     * waits for a buffer to become available. The acquired buffer has 
242     * a lock count of one.
243     * 
244     * @return the acquired buffer
245     * @throws InterruptedException if the current thread is interrupted
246     */
247    @SuppressWarnings("PMD.GuardLogStatement")
248    public W acquire() throws InterruptedException {
249        // Stop draining, because we obviously need this kind of buffers
250        Optional.ofNullable(idleTimer.getAndSet(null)).ifPresent(
251            timer -> timer.cancel());
252        if (createdBufs.get() < maximumBufs) {
253            // Haven't reached maximum, so if no buffer is queued, create one.
254            W buffer = queue.poll();
255            if (buffer != null) {
256                buffer.lockBuffer();
257                return buffer;
258            }
259            return createBuffer();
260        }
261        // Wait for buffer to become available.
262        if (logger.isLoggable(Level.FINE)) {
263            // If configured, log message after waiting some time.
264            W buffer = queue.poll(acquireWarningLimit, TimeUnit.MILLISECONDS);
265            if (buffer != null) {
266                buffer.lockBuffer();
267                return buffer;
268            }
269            logger.log(Level.FINE,
270                Thread.currentThread().getName() + " waiting > "
271                    + acquireWarningLimit + "ms for buffer, while executing:",
272                new Throwable());
273        }
274        W buffer = queue.take();
275        buffer.lockBuffer();
276        return buffer;
277    }

279    /**
280     * Re-adds the buffer to the pool. The buffer is cleared.
281     *
282     * @param buffer the buffer
283     * @see org.jgrapes.io.util.BufferCollector#recollect(org.jgrapes.io.util.ManagedBuffer)
284     */
285    @Override
286    public void recollect(W buffer) {
287        if (queue.size() < preservedBufs) {
288            long effectiveDrainDelay
289                = drainDelay > 0 ? drainDelay : defaultDrainDelay;
290            if (effectiveDrainDelay > 0) {
291                // Enqueue
292                buffer.clear();
293                queue.add(buffer);
294                Timer old = idleTimer.getAndSet(Components.schedule(this::drain,
295                    Duration.ofMillis(effectiveDrainDelay)));
296                if (old != null) {
297                    old.cancel();
298                }
299                return;
300            }
301        }
302        // Discard
303        removeBuffer(buffer);
304    }
305
306    @SuppressWarnings("PMD.UnusedFormalParameter")
307    private void drain(Timer timer) {
308        idleTimer.set(null);
309        while (true) {
310            W buffer = queue.poll();
311            if (buffer == null) {
312                break;
313            }
314            removeBuffer(buffer);
315        }
316    }

362    /**
363     * This is basically a WeakHashMap. We cannot use WeakHashMap
364     * because there is no "hook" into the collection of orphaned
365     * references, which is what we want here.
366     */
367    @SuppressWarnings("PMD.DataflowAnomalyAnalysis")
368    private class BufferMonitor {
369
370        private Entry<W>[] data;
371        private int indexMask;
372        private final ReferenceQueue<W> orphanedEntries
373            = new ReferenceQueue<>();
374
375        /**
376         * An Entry.
377         *
378         * @param <B> the generic type
379         */
380        private class Entry<B extends ManagedBuffer<?>> extends WeakReference<B>
381                implements Map.Entry<B, BufferProperties> {
382            /* default */ final int index;
383            /* default */ BufferProperties props;
384            /* default */ Entry<B> next;
385
386            /**
387             * Instantiates a new entry.
388             *
389             * @param buffer the buffer
390             * @param props the props
391             * @param queue the queue
392             * @param index the index
393             * @param next the next
394             */
395            /* default */ Entry(B buffer, BufferProperties props,
396                    ReferenceQueue<B> queue, int index, Entry<B> next) {
397                super(buffer, queue);
398                this.index = index;
399                this.props = props;
400                this.next = next;
401            }
402
403            @Override
404            public B getKey() {
405                return get();
406            }
407
408            @Override
409            public BufferProperties getValue() {
410                return props;
411            }
412
413            @Override
414            public BufferProperties setValue(BufferProperties props) {
415                return this.props = props;
416            }
417        }
418
419        /**
420         * @param data
421         */
422        @SuppressWarnings("unchecked")
423        public BufferMonitor(int maxBuffers) {
424            int lists = 1;
425            while (lists < maxBuffers) {
426                lists <<= 1;
427                indexMask = (indexMask << 1) + 1;
428            }
429            data = new Entry[lists];
430        }
431
432        /**
433         * Put an entry in the map.
434         *
435         * @param buffer the buffer
436         * @param properties the properties
437         * @return the buffer properties
438         */
439        @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
440        public BufferProperties put(W buffer, BufferProperties properties) {
441            check();
442            int index = buffer.hashCode() & indexMask;
443            synchronized (data) {
444                Entry<W> entry = data[index];
445                Entry<W> prev = null;
446                while (true) {
447                    if (entry == null) {
448                        // Not found, create new.
449                        entry = new Entry<>(buffer, properties,
450                            orphanedEntries, index, null);
451                        if (prev == null) {
452                            data[index] = entry; // Is first.
453                        } else {
454                            prev.next = entry; // Is next (last).
455                        }
456                        return properties;
457                    }
458                    if (entry.getKey() == buffer) {
459                        // Found, update.
460                        BufferProperties old = entry.getValue();
461                        entry.setValue(properties);
462                        return old;
463                    }
464                    prev = entry;
465                    entry = entry.next;
466                }
467            }
468        }
469
470        /**
471         * Returns the properties for the given buffer.
472         *
473         * @param buffer the buffer
474         * @return the buffer properties
475         */
476        @SuppressWarnings("unused")
477        public BufferProperties get(ManagedBuffer<?> buffer) {
478            check();
479            int index = buffer.hashCode() & indexMask;
480            synchronized (data) {
481                Entry<W> entry = data[index];
482                while (entry != null) {
483                    if (entry.getKey() == buffer) {
484                        return entry.getValue();
485                    }
486                    entry = entry.next;
487                }
488                return null;
489            }
490        }
491
492        /**
493         * Removes the given buffer.
494         *
495         * @param buffer the buffer
496         * @return the buffer properties
497         */
498        public BufferProperties remove(ManagedBuffer<?> buffer) {
499            check();
500            int index = buffer.hashCode() & indexMask;
501            synchronized (data) {
502                Entry<W> entry = data[index];
503                Entry<W> prev = null;
504                while (entry != null) {
505                    if (entry.getKey() == buffer) {
506                        if (prev == null) {
507                            data[index] = entry.next; // Was first.
508                        } else {
509                            prev.next = entry.next;
510                        }
511                        return entry.getValue();
512                    }
513                    prev = entry;
514                    entry = entry.next;
515                }
516                return null;
517            }
518        }
519
520        @SuppressWarnings("PMD.CompareObjectsWithEquals")
521        private BufferProperties remove(Entry<W> toBeRemoved) {
522            synchronized (data) {
523                Entry<W> entry = data[toBeRemoved.index];
524                Entry<W> prev = null;
525                while (entry != null) {
526                    if (entry == toBeRemoved) {
527                        if (prev == null) {
528                            data[toBeRemoved.index] = entry.next; // Was first.
529                        } else {
530                            prev.next = entry.next;
531                        }
532                        return entry.getValue();
533                    }
534                    prev = entry;
535                    entry = entry.next;
536                }
537                return null;
538            }
539        }
540
541        private void check() {
542            while (true) {
543                @SuppressWarnings("unchecked")
544                Entry<W> entry = (Entry<W>) orphanedEntries.poll();
545                if (entry == null) {
546                    return;
547                }
548                // Managed buffer has not been properly recollected, fix.
549                BufferProperties props = remove(entry);
550                if (props == null) {
551                    return;
552                }
553                createdBufs.decrementAndGet();
554                // Create warning
555                if (logger.isLoggable(Level.WARNING)) {
556                    @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
557                    final StringBuilder msg = new StringBuilder(
558                        "Orphaned buffer from pool ");
559                    msg.append(name());
560                    StackTraceElement[] trace = props.createdBy();
561                    if (trace != null) {
562                        msg.append(", created");
563                        for (StackTraceElement e : trace) {
564                            msg.append(System.lineSeparator());
565                            msg.append("\tat ");
566                            msg.append(e.toString());
567                        }
568                    }
569                    logger.warning(msg.toString());
570                }
571            }
572        }
573    }

660        /**
661         * Three views on the existing pool.
662         */
663        class PoolInfos {
664            private final SortedMap<String, PoolInfo> allPools;
665            private final SortedMap<String, PoolInfo> nonEmptyPools;
666            private final SortedMap<String, PoolInfo> usedPools;
667
668            /**
669             * Instantiates a new pool infos.
670             *
671             * @param pools the pools
672             */
673            public PoolInfos(Set<ManagedBufferPool<?, ?>> pools) {
674                allPools = new TreeMap<>();
675                nonEmptyPools = new TreeMap<>();
676                usedPools = new TreeMap<>();
677
678                @SuppressWarnings("PMD.UseConcurrentHashMap")
679                Map<String, Integer> dupsNext = new HashMap<>();
680                for (ManagedBufferPool<?, ?> mbp : pools) {
681                    String key = mbp.name();
682                    PoolInfo infos = new PoolInfo(
683                        mbp.createdBufs.get(), mbp.queue.size(),
684                        mbp.preservedBufs, mbp.maximumBufs,
685                        mbp.bufferSize());
686                    if (allPools.containsKey(key)
687                        || dupsNext.containsKey(key)) {
688                        if (allPools.containsKey(key)) {
689                            // Found first duplicate, rename
690                            allPools.put(key + "#1", allPools.get(key));
691                            allPools.remove(key);
692                            dupsNext.put(key, 2);
693                        }
694                        allPools.put(key + "#"
695                            + (dupsNext.put(key, dupsNext.get(key) + 1)),
696                            infos);
697                    } else {
698                        allPools.put(key, infos);
699                    }
700                }
701                for (Map.Entry<String, PoolInfo> e : allPools.entrySet()) {
702                    PoolInfo infos = e.getValue();
703                    if (infos.getPooled() > 0) {
704                        nonEmptyPools.put(e.getKey(), infos);
705                    }
706                    if (infos.getCreated() > 0) {
707                        usedPools.put(e.getKey(), infos);
708                    }
709                }
710            }
