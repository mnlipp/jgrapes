059/**
060 * A queue based buffer pool. Using buffers from a pool is an important
061 * feature for limiting the computational resources for an {@link IOSubchannel}.
062 * A producer of {@link Output} events that simply creates its own buffers
063 * may produce and enqueue a large number of events that are not consumed
064 * as fast as they are produced. 
065 * 
066 * Using a buffer pool with a typical size of two synchronizes the 
067 * producer and the consumers of events nicely. The producer
068 * (thread) holds one buffer and fills it, the consumer (thread) holds 
069 * the other buffer and works with its content. If the producer finishes
070 * before the consumer, it has to stop until the consumer has processed 
071 * previous event and releases the buffer. The consumer can continue
072 * without delay, because the data has already been prepared and enqueued
073 * as the next event.
074 * 
075 * One of the biggest problems when using a pool can be to identify 
076 * leaking buffers, i.e. buffers that are not properly returned to the pool.
077 * This implementation therefore tracks all created buffers 
078 * (with a small overhead) and logs a warning if a buffer is no longer
079 * used (referenced) but has not been returned to the pool. If the
080 * log level for {@link ManagedBufferPool} is set to {@link Level#FINE},
081 * the warning also includes a stack trace of the call to {@link #acquire()}
082 * that handed out the buffer. Providing this information in addition 
083 * obviously requires a larger overhead and is therefore limited to the
084 * finer log levels.
085 *
086 * @param <W> the type of the wrapped (managed) buffer
087 * @param <T> the type of the content buffer that is wrapped
088 */
089@SuppressWarnings({ "PMD.ExcessiveImports", "PMD.NcssCount",
090    "PMD.EmptyCatchBlock" })
091public class ManagedBufferPool<W extends ManagedBuffer<T>, T extends Buffer>
092        implements BufferCollector<W> {

133    /**
134     * Create a pool that contains a varying number of (wrapped) buffers.
135     * The pool is initially empty. When buffers are requested and none 
136     * are left in the pool, new buffers are created up to the given 
137     * upper limit. Recollected buffers are put in the pool until it holds
138     * the number specified by the lower threshold. Any additional 
139     * recollected buffers are discarded. 
140     * 
141     * @param wrapper the function that converts buffers to managed buffers
142     * @param bufferFactory a function that creates a new buffer
143     * @param lowerThreshold the number of buffers kept in the pool
144     * @param upperLimit the maximum number of buffers
145     */
146    public ManagedBufferPool(BiFunction<T, BufferCollector<W>, W> wrapper,
147            Supplier<T> bufferFactory, int lowerThreshold, int upperLimit) {
148        this.wrapper = wrapper;
149        this.bufferFactory = bufferFactory;
150        preservedBufs = lowerThreshold;
151        maximumBufs = upperLimit;
152        createdBufs = new AtomicInteger();
153        queue = new ArrayBlockingQueue<W>(lowerThreshold);
154        bufferMonitor = new BufferMonitor(upperLimit);
155        MBeanView.addPool(this);
156    }

164     * @param buffers the number of buffers
165     */
166    public ManagedBufferPool(BiFunction<T, BufferCollector<W>, W> wrapper,
167            Supplier<T> bufferFactory, int buffers) {
168        this(wrapper, bufferFactory, buffers, buffers);

203    private W createBuffer() {
204        createdBufs.incrementAndGet();
205        W buffer = wrapper.apply(this.bufferFactory.get(), this);
206        bufferMonitor.put(buffer, new BufferProperties());
207        bufferSize = buffer.capacity();
208        return buffer;
209    }
210
211    /**
212     * Removes the buffer from the pool.
213     * 
214     * @param buffer the buffer to remove
215     */
216    private void removeBuffer(W buffer) {
217        createdBufs.decrementAndGet();
218        if (bufferMonitor.remove(buffer) == null) {
219            if (logger.isLoggable(Level.FINE)) {
220                logger.log(Level.WARNING,
221                    "Attempt to remove unknown buffer from pool.",
222                    new Throwable());
223            } else {
224                logger.warning("Attempt to remove unknown buffer from pool.");
225            }
226        }
227    }

229    /**
230     * Returns the size of the buffers managed by this pool.
231     * 
232     * @return the buffer size
233     */
234    public int bufferSize() {
235        if (bufferSize < 0) {
236            createBuffer().unlockBuffer();
237        }
238        return bufferSize;

241    /**
242     * Acquires a managed buffer from the pool. If the pool is empty,
243     * waits for a buffer to become available. The acquired buffer has 
244     * a lock count of one.
245     * 
246     * @return the acquired buffer
247     * @throws InterruptedException if the current thread is interrupted
248     */
249    @SuppressWarnings("PMD.GuardLogStatement")
250    public W acquire() throws InterruptedException {
251        // Stop draining, because we obviously need this kind of buffers
252        Optional.ofNullable(idleTimer.getAndSet(null)).ifPresent(
253            timer -> timer.cancel());
254        if (createdBufs.get() < maximumBufs) {
255            // Haven't reached maximum, so if no buffer is queued, create one.
256            W buffer = queue.poll();
257            if (buffer != null) {
258                buffer.lockBuffer();
259                return buffer;
260            }
261            return createBuffer();
262        }
263        // Wait for buffer to become available.
264        if (logger.isLoggable(Level.FINE)) {
265            // If configured, log message after waiting some time.
266            W buffer = queue.poll(acquireWarningLimit, TimeUnit.MILLISECONDS);
267            if (buffer != null) {
268                buffer.lockBuffer();
269                return buffer;
270            }
271            logger.fine(() -> "Waiting > " + acquireWarningLimit
272                + "ms for buffer: " + Thread.currentThread().getName());
273        }
274        W buffer = queue.take();
275        buffer.lockBuffer();
276        return buffer;
277    }

279    /**
280     * Re-adds the buffer to the pool. The buffer is cleared.
281     *
282     * @param buffer the buffer
283     * @see org.jgrapes.io.util.BufferCollector#recollect(org.jgrapes.io.util.ManagedBuffer)
284     */
285    @Override
286    public void recollect(W buffer) {
287        if (queue.size() < preservedBufs) {
288            long effectiveDrainDelay
289                = drainDelay > 0 ? drainDelay : defaultDrainDelay;
290            if (effectiveDrainDelay > 0) {
291                // Enqueue
292                buffer.clear();
293                queue.add(buffer);
294                Timer old = idleTimer.getAndSet(Components.schedule(this::drain,
295                    Duration.ofMillis(effectiveDrainDelay)));
296                if (old != null) {
297                    old.cancel();
298                }
299                return;
300            }
301        }
302        // Discard
303        removeBuffer(buffer);
304    }
305
306    @SuppressWarnings("PMD.UnusedFormalParameter")
307    private void drain(Timer timer) {
308        idleTimer.set(null);
309        while (true) {
310            W buffer = queue.poll();
311            if (buffer == null) {
312                break;
313            }
314            removeBuffer(buffer);
315        }
316    }

362    /**
363     * This is basically a WeakHashMap. We cannot use WeakHashMap
364     * because there is no "hook" into the collection of orphaned
365     * references, which is what we want here.
366     */
367    @SuppressWarnings("PMD.DataflowAnomalyAnalysis")
368    private class BufferMonitor {
369
370        private Entry<W>[] data;
371        private int indexMask;
372        private final ReferenceQueue<W> orphanedEntries
373            = new ReferenceQueue<>();
374
375        /**
376         * An Entry.
377         *
378         * @param <B> the generic type
379         */
380        private class Entry<B extends ManagedBuffer<?>> extends WeakReference<B>
381                implements Map.Entry<B, BufferProperties> {
382            /* default */ final int index;
383            /* default */ BufferProperties props;
384            /* default */ Entry<B> next;
385
386            /**
387             * Instantiates a new entry.
388             *
389             * @param buffer the buffer
390             * @param props the props
391             * @param queue the queue
392             * @param index the index
393             * @param next the next
394             */
395            /* default */ Entry(B buffer, BufferProperties props,
396                    ReferenceQueue<B> queue, int index, Entry<B> next) {
397                super(buffer, queue);
398                this.index = index;
399                this.props = props;
400                this.next = next;
401            }
402
403            @Override
404            public B getKey() {
405                return get();
406            }
407
408            @Override
409            public BufferProperties getValue() {
410                return props;
411            }
412
413            @Override
414            public BufferProperties setValue(BufferProperties props) {
415                return this.props = props;
416            }
417        }
418
419        /**
420         * @param data
421         */
422        @SuppressWarnings("unchecked")
423        public BufferMonitor(int maxBuffers) {
424            int lists = 1;
425            while (lists < maxBuffers) {
426                lists <<= 1;
427                indexMask = (indexMask << 1) + 1;
428            }
429            data = new Entry[lists];
430        }
431
432        /**
433         * Put an entry in the map.
434         *
435         * @param buffer the buffer
436         * @param properties the properties
437         * @return the buffer properties
438         */
439        @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
440        public BufferProperties put(W buffer, BufferProperties properties) {
441            check();
442            int index = buffer.hashCode() & indexMask;
443            synchronized (data) {
444                Entry<W> entry = data[index];
445                Entry<W> prev = null;
446                while (true) {
447                    if (entry == null) {
448                        // Not found, create new.
449                        entry = new Entry<>(buffer, properties,
450                            orphanedEntries, index, null);
451                        if (prev == null) {
452                            data[index] = entry; // Is first.
453                        } else {
454                            prev.next = entry; // Is next (last).
455                        }
456                        return properties;
457                    }
458                    if (entry.getKey() == buffer) {
459                        // Found, update.
460                        BufferProperties old = entry.getValue();
461                        entry.setValue(properties);
462                        return old;
463                    }
464                    prev = entry;
465                    entry = entry.next;
466                }
467            }
468        }
469
470        /**
471         * Returns the properties for the given buffer.
472         *
473         * @param buffer the buffer
474         * @return the buffer properties
475         */
476        @SuppressWarnings("unused")
477        public BufferProperties get(ManagedBuffer<?> buffer) {
478            check();
479            int index = buffer.hashCode() & indexMask;
480            synchronized (data) {
481                Entry<W> entry = data[index];
482                while (entry != null) {
483                    if (entry.getKey() == buffer) {
484                        return entry.getValue();
485                    }
486                    entry = entry.next;
487                }
488                return null;
489            }
490        }
491
492        /**
493         * Removes the given buffer.
494         *
495         * @param buffer the buffer
496         * @return the buffer properties
497         */
498        public BufferProperties remove(ManagedBuffer<?> buffer) {
499            check();
500            int index = buffer.hashCode() & indexMask;
501            synchronized (data) {
502                Entry<W> entry = data[index];
503                Entry<W> prev = null;
504                while (entry != null) {
505                    if (entry.getKey() == buffer) {
506                        if (prev == null) {
507                            data[index] = entry.next; // Was first.
508                        } else {
509                            prev.next = entry.next;
510                        }
511                        return entry.getValue();
512                    }
513                    prev = entry;
514                    entry = entry.next;
515                }
516                return null;
517            }
518        }
519
520        @SuppressWarnings("PMD.CompareObjectsWithEquals")
521        private BufferProperties remove(Entry<W> toBeRemoved) {
522            synchronized (data) {
523                Entry<W> entry = data[toBeRemoved.index];
524                Entry<W> prev = null;
525                while (entry != null) {
526                    if (entry == toBeRemoved) {
527                        if (prev == null) {
528                            data[toBeRemoved.index] = entry.next; // Was first.
529                        } else {
530                            prev.next = entry.next;
531                        }
532                        return entry.getValue();
533                    }
534                    prev = entry;
535                    entry = entry.next;
536                }
537                return null;
538            }
539        }
540
541        private void check() {
542            while (true) {
543                @SuppressWarnings("unchecked")
544                Entry<W> entry = (Entry<W>) orphanedEntries.poll();
545                if (entry == null) {
546                    return;
547                }
548                // Managed buffer has not been properly recollected, fix.
549                BufferProperties props = remove(entry);
550                if (props == null) {
551                    return;
552                }
553                createdBufs.decrementAndGet();
554                // Create warning
555                if (logger.isLoggable(Level.WARNING)) {
556                    @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
557                    final StringBuilder msg = new StringBuilder(
558                        "Orphaned buffer from pool ");
559                    msg.append(name());
560                    StackTraceElement[] trace = props.createdBy();
561                    if (trace != null) {
562                        msg.append(", created");
563                        for (StackTraceElement e : trace) {
564                            msg.append(System.lineSeparator());
565                            msg.append("\tat ");
566                            msg.append(e.toString());
567                        }
568                    }
569                    logger.warning(msg.toString());
570                }
571            }
572        }
573    }

660        /**
661         * Three views on the existing pool.
662         */
663        class PoolInfos {
664            private final SortedMap<String, PoolInfo> allPools;
665            private final SortedMap<String, PoolInfo> nonEmptyPools;
666            private final SortedMap<String, PoolInfo> usedPools;
667
668            /**
669             * Instantiates a new pool infos.
670             *
671             * @param pools the pools
672             */
673            public PoolInfos(Set<ManagedBufferPool<?, ?>> pools) {
674                allPools = new TreeMap<>();
675                nonEmptyPools = new TreeMap<>();
676                usedPools = new TreeMap<>();
677
678                @SuppressWarnings("PMD.UseConcurrentHashMap")
679                Map<String, Integer> dupsNext = new HashMap<>();
680                for (ManagedBufferPool<?, ?> mbp : pools) {
681                    String key = mbp.name();
682                    PoolInfo infos = new PoolInfo(
683                        mbp.createdBufs.get(), mbp.queue.size(),
684                        mbp.preservedBufs, mbp.maximumBufs,
685                        mbp.bufferSize());
686                    if (allPools.containsKey(key)
687                        || dupsNext.containsKey(key)) {
688                        if (allPools.containsKey(key)) {
689                            // Found first duplicate, rename
690                            allPools.put(key + "#1", allPools.get(key));
691                            allPools.remove(key);
692                            dupsNext.put(key, 2);
693                        }
694                        allPools.put(key + "#"
695                            + (dupsNext.put(key, dupsNext.get(key) + 1)),
696                            infos);
697                    } else {
698                        allPools.put(key, infos);
699                    }
700                }
701                for (Map.Entry<String, PoolInfo> e : allPools.entrySet()) {
702                    PoolInfo infos = e.getValue();
703                    if (infos.getPooled() > 0) {
704                        nonEmptyPools.put(e.getKey(), infos);
705                    }
706                    if (infos.getCreated() > 0) {
707                        usedPools.put(e.getKey(), infos);
708                    }
709                }
710            }
