019package org.jgrapes.io.events;
020
021import java.net.InetAddress;
022import java.net.InetSocketAddress;
023import org.jgrapes.core.Event;
024
025/**
026 * 
027 */
028public class OpenTcpConnection extends Event<Void> {
029
030    private final InetSocketAddress address;
031
032    /**
033     * Signals that a new TCP connection should be opened.
034     *
035     * @param address the address
036     */
037    public OpenTcpConnection(InetSocketAddress address) {
038        this.address = address;
039    }
040
041    /**
042     * Signals that a new TCP connection should be opened.
043     *
044     * @param address the address
045     * @param port the port
046     */
047    public OpenTcpConnection(InetAddress address, int port) {
048        this.address = new InetSocketAddress(address, port);
049    }
050
051    /**
052     * Gets the address.
053     *
054     * @return the address
055     */
056    public InetSocketAddress address() {
057        return address;
058    }
059}
