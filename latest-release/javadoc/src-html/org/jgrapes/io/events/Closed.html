019package org.jgrapes.io.events;
020
021import java.io.IOException;
022import java.util.Optional;
023
024import org.jgrapes.core.Event;
025
026/**
027 * This event signals that an I/O subchannel will no longer be used.
028 * Components that have allocated resources for that channel should
029 * release them in response to receiving this event.
030 */
031public class Closed extends Event<Void> {
032
033    private Throwable error;
034
035    /**
036     * Creates a new event that signals a close in response to
037     * an error (usually an {@link IOException}.
038     */
039    public Closed(Throwable error) {
040        this.error = error;
041    }
042
043    /**
044     * Creates a new event that signals a regular close.
045     */
046    public Closed() {
047        this(null);
048    }
049
050    /**
051     * Returns the cause of the {@link Closed} event in case of error.
052     *
053     * @return the optional failure
054     */
055    public Optional<Throwable> error() {
056        return Optional.ofNullable(error);
057    }
058}
