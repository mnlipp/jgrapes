096    /**
097     * On {@link ConsoleReady}, fire the {@link AddConletType}.
098     *
099     * @param event the event
100     * @param channel the channel
101     * @throws TemplateNotFoundException the template not found exception
102     * @throws MalformedTemplateNameException the malformed template name
103     *             exception
104     * @throws ParseException the parse exception
105     * @throws IOException Signals that an I/O exception has occurred.
106     */
107    @Handler
108    public void onConsoleReady(ConsoleReady event, ConsoleSession channel)
109            throws TemplateNotFoundException, MalformedTemplateNameException,
110            ParseException, IOException {
111        // Add conlet resources to page
112        channel.respond(new AddConletType(type())
113            .setDisplayNames(
114                localizations(channel.supportedLocales(), "conletName"))
115            .addScript(new ScriptResource()
116                .setScriptUri(event.renderSupport().conletResource(
117                    type(), "Bundles-functions.ftl.js"))
118                .setScriptType("module"))
119            .addCss(event.renderSupport(),
120                WebConsoleUtils.uriFromPath("Bundles-style.css")));
121    }

130    @Override
131    protected Set<RenderMode> doRenderConlet(RenderConletRequestBase<?> event,
132            ConsoleSession channel, String conletId,
133            BundleListModel conletModel) throws Exception {
134        Set<RenderMode> renderedAs = new HashSet<>();
135        if (event.renderAs().contains(RenderMode.Preview)) {
136            Template tpl
137                = freemarkerConfig().getTemplate("Bundles-preview.ftl.html");
138            channel.respond(new RenderConletFromTemplate(event, type(),
139                conletModel.getConletId(),
140                tpl, fmModel(event, channel, conletId, conletModel))
141                    .setRenderAs(
142                        RenderMode.Preview.addModifiers(event.renderAs()))
143                    .setSupportedModes(MODES));
144            List<Map<String, Object>> bundleInfos
145                = Arrays.stream(context.getBundles())
146                    .map(bndl -> createBundleInfo(bndl, channel.locale()))
147                    .collect(Collectors.toList());
148            channel.respond(new NotifyConletView(type(),
149                conletModel.getConletId(), "bundleUpdates", bundleInfos,
150                "preview", true));
151            renderedAs.add(RenderMode.Preview);
152        }
153        if (event.renderAs().contains(RenderMode.View)) {
154            Template tpl
155                = freemarkerConfig().getTemplate("Bundles-view.ftl.html");
156            channel.respond(new RenderConletFromTemplate(event,
157                type(), conletModel.getConletId(), tpl,
158                fmModel(event, channel, conletId, conletModel))
159                    .setRenderAs(
160                        RenderMode.View.addModifiers(event.renderAs())));
161            List<Map<String, Object>> bundleInfos
162                = Arrays.stream(context.getBundles())
163                    .map(bndl -> createBundleInfo(bndl, channel.locale()))
164                    .collect(Collectors.toList());
165            channel.respond(new NotifyConletView(type(),
166                conletModel.getConletId(), "bundleUpdates", bundleInfos,
167                "view", true));
168            renderedAs.add(RenderMode.View);
169        }
170        return renderedAs;
171    }

173    private Map<String, Object> createBundleInfo(Bundle bundle, Locale locale) {
174        @SuppressWarnings("PMD.UseConcurrentHashMap")
175        Map<String, Object> result = new HashMap<>();
176        result.put("id", bundle.getBundleId());
177        result.put("name",
178            Optional.ofNullable(bundle.getHeaders(locale.toString())
179                .get("Bundle-Name")).orElse(bundle.getSymbolicName()));
180        result.put("symbolicName", bundle.getSymbolicName());
181        result.put("version", bundle.getVersion().toString());
182        result.put("category",
183            Optional.ofNullable(bundle.getHeaders(locale.toString())
184                .get("Bundle-Category")).orElse(""));
185        result.put("state", "bundleState_" + bundle.getState());
186        result.put("startable", false);
187        result.put("stoppable", false);
188        if ((bundle.getState()
189            & (Bundle.RESOLVED | Bundle.INSTALLED | Bundle.ACTIVE)) != 0) {
190            boolean isFragment = (bundle.adapt(BundleRevision.class).getTypes()
191                & BundleRevision.TYPE_FRAGMENT) != 0;
192            result.put("startable", !isFragment
193                && (bundle.getState() == Bundle.INSTALLED
194                    || bundle.getState() == Bundle.RESOLVED));
195            result.put("stoppable",
196                !isFragment && bundle.getState() == Bundle.ACTIVE);
197        }
198        result.put("uninstallable", (bundle.getState()
199            & (Bundle.INSTALLED | Bundle.RESOLVED | Bundle.ACTIVE)) != 0);
200        result.put("uninstalled", bundle.getState() == Bundle.UNINSTALLED);
201        return result;
202    }

204    @Override
205    protected void doUpdateConletState(NotifyConletModel event,
206            ConsoleSession channel, BundleListModel conletState)
207            throws Exception {
208        event.stop();
209        Bundle bundle = context.getBundle(event.params().asInt(0));
210        if (bundle == null) {
211            return;
212        }
213        try {
214            switch (event.method()) {
215            case "stop":
216                bundle.stop();
217                break;
218            case "start":
219                bundle.start();
220                break;
221            case "refresh":
222                break;
223            case "update":
224                bundle.update();
225                break;
226            case "uninstall":
227                bundle.uninstall();
228                break;
229            case "sendDetails":
230                sendBundleDetails(event.conletId(), channel, bundle);
231                break;
232            default:// ignore
233                break;
234            }
235        } catch (BundleException e) {
236            // ignore
237            LOG.log(Level.WARNING, "Cannot update bundle state", e);
238        }
239    }

241    @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
242    private void sendBundleDetails(String conletId, ConsoleSession channel,
243            Bundle bundle) {
244        Locale locale = channel.locale();
245        List<Object> data = new ArrayList<>();
246        data.add(
247            new Object[] { "bundleSymbolicName", bundle.getSymbolicName() });
248        data.add(
249            new Object[] { "bundleVersion", bundle.getVersion().toString() });
250        data.add(new Object[] { "bundleLocation",
251            bundle.getLocation().replace(".", ".&#x200b;") });
252        data.add(new Object[] { "bundleLastModification",
253            Instant.ofEpochMilli(bundle.getLastModified()).toString(),
254            "dateTime" });
255        data.add(new Object[] { "bundleStartLevel",
256            bundle.adapt(BundleStartLevel.class).getStartLevel() });
257        Dictionary<String, String> dict = bundle.getHeaders(locale.toString());
258        @SuppressWarnings("PMD.UseConcurrentHashMap")
259        Map<String, String> headers = new TreeMap<>();
260        for (Enumeration<String> e = dict.keys(); e.hasMoreElements();) {
261            String key = e.nextElement();
262            headers.put(key, dict.get(key));
263        }
264        List<Object> headerList = new ArrayList<>();
265        for (Map.Entry<String, String> e : headers.entrySet()) {
266            headerList.add(new Object[] { e.getKey(),
267                e.getKey().contains("Package")
268                    ? e.getValue().replace(".", ".&#x200b;")
269                    : e.getValue() });
270        }
271        data.add(new Object[] { "manifestHeaders", headerList, "table" });
272        channel.respond(new NotifyConletView(type(),
273            conletId, "bundleDetails", bundle.getBundleId(), data));
274    }

276    /**
277     * Translates the OSGi {@link BundleEvent} to a JGrapes event and fires it
278     * on all known console session channels.
279     *
280     * @param event the event
281     */
282    @Override
283    public void bundleChanged(BundleEvent event) {
284        fire(new BundleChanged(event), trackedSessions());
285    }
286
287    /**
288     * Handles a {@link BundleChanged} event by updating the information in the
289     * console sessions.
290     *
291     * @param event the event
292     */
293    @Handler
294    @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
295    public void onBundleChanged(BundleChanged event,
296            ConsoleSession consoleSession) {
297        for (String conletId : conletIds(consoleSession)) {
298            consoleSession.respond(new NotifyConletView(type(), conletId,
299                "bundleUpdates",
300                (Object) new Object[] { createBundleInfo(
301                    event.bundleEvent().getBundle(), consoleSession.locale()) },
302                "*", false));
303        }
304    }

306    @Override
307    protected boolean doSetLocale(SetLocale event, ConsoleSession channel,
308            String conletId) throws Exception {
309        return true;
310    }

312    /**
313     * Wraps an OSGi {@link BundleEvent}.
314     */
315    public static class BundleChanged extends Event<Void> {
316        private final BundleEvent bundleEvent;
317
318        /**
319         * Instantiates a new event.
320         *
321         * @param bundleEvent the OSGi bundle event
322         */
323        public BundleChanged(BundleEvent bundleEvent) {
324            this.bundleEvent = bundleEvent;
325        }
326
327        /**
328         * Return the OSGi bundle event.
329         *
330         * @return the bundle event
331         */
332        public BundleEvent bundleEvent() {
333            return bundleEvent;
334        }
335
336    }
