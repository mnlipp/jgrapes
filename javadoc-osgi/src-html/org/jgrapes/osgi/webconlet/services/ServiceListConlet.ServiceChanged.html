059/**
060 * A conlet for inspecting the services in an OSGi runtime.
061 */
062@SuppressWarnings("PMD.DataflowAnomalyAnalysis")
063public class ServiceListConlet
064        extends FreeMarkerConlet<Serializable> implements ServiceListener {
065
066    private final ServiceComponentRuntime scr;
067    private static final Set<RenderMode> MODES = RenderMode.asSet(
068        RenderMode.Preview, RenderMode.View);
069    private final BundleContext context;
070
071    /**
072     * Creates a new component with its channel set to the given channel.
073     * 
074     * @param componentChannel the channel that the component's handlers listen
075     *            on by default and that {@link Manager#fire(Event, Channel...)}
076     *            sends the event to
077     */
078    public ServiceListConlet(Channel componentChannel, BundleContext context,
079            ServiceComponentRuntime scr) {
080        super(componentChannel);
081        this.context = context;
082        this.scr = scr;
083        context.addServiceListener(this);
084    }
085
086    /**
087     * On {@link ConsoleReady}, fire the {@link AddConletType}.
088     *
089     * @param event the event
090     * @param channel the channel
091     * @throws TemplateNotFoundException the template not found exception
092     * @throws MalformedTemplateNameException the malformed template name
093     *             exception
094     * @throws ParseException the parse exception
095     * @throws IOException Signals that an I/O exception has occurred.
096     */
097    @Handler
098    public void onConsoleReady(ConsoleReady event, ConsoleSession channel)
099            throws TemplateNotFoundException, MalformedTemplateNameException,
100            ParseException, IOException {
101        // Add conlet resources to page
102        channel.respond(new AddConletType(type())
103            .addRenderMode(RenderMode.Preview).setDisplayNames(
104                localizations(channel.supportedLocales(), "conletName"))
105            .addScript(new ScriptResource()
106                .setScriptUri(event.renderSupport().conletResource(
107                    type(), "Services-functions.ftl.js"))
108                .setScriptType("module"))
109            .addCss(event.renderSupport(),
110                WebConsoleUtils.uriFromPath("Services-style.css")));
111    }
112
113    @Override
114    protected Set<RenderMode> doRenderConlet(RenderConletRequestBase<?> event,
115            ConsoleSession channel, String conletId, Serializable conletState)
116            throws Exception {
117        Set<RenderMode> renderedAs = new HashSet<>();
118        if (event.renderAs().contains(RenderMode.Preview)) {
119            Template tpl
120                = freemarkerConfig().getTemplate("Services-preview.ftl.html");
121            channel.respond(new RenderConletFromTemplate(event,
122                type(), conletId, tpl,
123                fmModel(event, channel, conletId, conletState))
124                    .setRenderAs(
125                        RenderMode.Preview.addModifiers(event.renderAs()))
126                    .setSupportedModes(MODES));
127            List<Map<String, Object>> serviceInfos = Arrays.stream(
128                context.getAllServiceReferences(null, null))
129                .map(svc -> createServiceInfo(svc, channel.locale()))
130                .collect(Collectors.toList());
131            channel.respond(new NotifyConletView(type(),
132                conletId, "serviceUpdates", serviceInfos, "preview", true));
133            renderedAs.add(RenderMode.Preview);
134        }
135        if (event.renderAs().contains(RenderMode.View)) {
136            Template tpl
137                = freemarkerConfig().getTemplate("Services-view.ftl.html");
138            channel.respond(new RenderConletFromTemplate(event,
139                type(), conletId, tpl,
140                fmModel(event, channel, conletId, conletState))
141                    .setRenderAs(
142                        RenderMode.View.addModifiers(event.renderAs())));
143            List<Map<String, Object>> serviceInfos = Arrays.stream(
144                context.getAllServiceReferences(null, null))
145                .map(svc -> createServiceInfo(svc, channel.locale()))
146                .collect(Collectors.toList());
147            channel.respond(new NotifyConletView(type(),
148                conletId, "serviceUpdates", serviceInfos, "view", true));
149            renderedAs.add(RenderMode.View);
150        }
151        return renderedAs;
152    }

154    @SuppressWarnings({ "PMD.NcssCount", "PMD.ConfusingTernary",
155        "PMD.NPathComplexity", "PMD.AssignmentInOperand",
156        "PMD.CognitiveComplexity" })
157    private Map<String, Object>
158            createServiceInfo(ServiceReference<?> serviceRef, Locale locale) {
159        @SuppressWarnings("PMD.UseConcurrentHashMap")
160        Map<String, Object> result = new HashMap<>();
161        result.put("id", serviceRef.getProperty(Constants.SERVICE_ID));
162        String[] interfaces
163            = (String[]) serviceRef.getProperty(Constants.OBJECTCLASS);
164        result.put("type", String.join(", ", interfaces));
165        Long bundleId
166            = (Long) serviceRef.getProperty(Constants.SERVICE_BUNDLEID);
167        result.put("bundleId", bundleId.toString());
168        Bundle bundle = context.getBundle(bundleId);
169        if (bundle == null) {
170            result.put("bundleName", "");
171        } else {
172            result.put("bundleName", Optional
173                .ofNullable(bundle.getHeaders(locale.toString())
174                    .get("Bundle-Name"))
175                .orElse(bundle.getSymbolicName()));
176        }
177        String scope;
178        switch ((String) serviceRef.getProperty(Constants.SERVICE_SCOPE)) {
179        case Constants.SCOPE_BUNDLE:
180            scope = "serviceScopeBundle";
181            break;
182        case Constants.SCOPE_PROTOTYPE:
183            scope = "serviceScopePrototype";
184            break;
185        case Constants.SCOPE_SINGLETON:
186            scope = "serviceScopeSingleton";
187            break;
188        default:
189            scope = "";
190            break;
191        }
192        result.put("scope", scope);
193        Integer ranking
194            = (Integer) serviceRef.getProperty(Constants.SERVICE_RANKING);
195        result.put("ranking", ranking == null ? "" : ranking.toString());
196        String componentName
197            = (String) serviceRef.getProperty("component.name");
198        ComponentDescriptionDTO dto;
199        if (componentName != null && bundle != null
200            && (dto = scr.getComponentDescriptionDTO(bundle,
201                componentName)) != null) {
202            if (dto.scope != null) {
203                result.put("dsScope", "serviceScope"
204                    + dto.scope.substring(0, 1).toUpperCase(Locale.US)
205                    + dto.scope.substring(1));
206            }
207            result.put("implementationClass", dto.implementationClass);
208        } else {
209            Object service = context.getService(serviceRef);
210            if (service != null) {
211                result.put("implementationClass", service.getClass().getName());
212                context.ungetService(serviceRef);
213            } else {
214                result.put("implementationClass", "");
215            }
216        }
217        @SuppressWarnings("PMD.UseConcurrentHashMap")
218        Map<String, Object> properties = new HashMap<>();
219        for (String property : serviceRef.getPropertyKeys()) {
220            properties.put(property, serviceRef.getProperty(property));
221        }
222        result.put("properties", properties);
223        if (serviceRef.getUsingBundles() != null) {
224            List<String> using = new ArrayList<>();
225            for (Bundle bdl : serviceRef.getUsingBundles()) {
226                using
227                    .add(
228                        bdl.getSymbolicName() + " (" + bdl.getBundleId() + ")");
229            }
230            result.put("usingBundles", using);
231        }
232        return result;
233    }

235    /**
236     * Translates the OSGi {@link ServiceEvent} to a JGrapes event and fires it
237     * on all known console session channels.
238     *
239     * @param event the event
240     */
241    @Override
242    public void serviceChanged(ServiceEvent event) {
243        fire(new ServiceChanged(event), trackedSessions());
244    }
245
246    /**
247     * Handles a {@link ServiceChanged} event by updating the information in the
248     * console sessions.
249     *
250     * @param event the event
251     */
252    @Handler
253    @SuppressWarnings({ "PMD.AvoidInstantiatingObjectsInLoops",
254        "PMD.DataflowAnomalyAnalysis" })
255    public void onServiceChanged(ServiceChanged event,
256            ConsoleSession consoleSession) {
257        Map<String, Object> info = createServiceInfo(
258            event.serviceEvent().getServiceReference(),
259            consoleSession.locale());
260        if (event.serviceEvent().getType() == ServiceEvent.UNREGISTERING) {
261            info.put("updateType", "unregistering");
262        }
263        for (String conletId : conletIds(consoleSession)) {
264            consoleSession.respond(new NotifyConletView(
265                type(), conletId, "serviceUpdates",
266                (Object) new Object[] { info }, "*", false));
267        }
268    }
269
270    /**
271     * Wraps an OSGi {@link ServiceEvent}.
272     */
273    public static class ServiceChanged extends Event<Void> {
274        private final ServiceEvent serviceEvent;
275
276        /**
277         * Instantiates a new event.
278         *
279         * @param serviceEvent the service event
280         */
281        public ServiceChanged(ServiceEvent serviceEvent) {
282            this.serviceEvent = serviceEvent;
283        }
284
285        public ServiceEvent serviceEvent() {
286            return serviceEvent;
287        }
288    }
