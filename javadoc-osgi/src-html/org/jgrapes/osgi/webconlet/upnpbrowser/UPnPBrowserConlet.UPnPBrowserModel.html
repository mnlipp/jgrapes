098    /**
099     * On {@link ConsoleReady}, fire the {@link AddConletType}.
100     *
101     * @param event the event
102     * @param channel the channel
103     * @throws TemplateNotFoundException the template not found exception
104     * @throws MalformedTemplateNameException the malformed template name
105     *             exception
106     * @throws ParseException the parse exception
107     * @throws IOException Signals that an I/O exception has occurred.
108     */
109    @Handler
110    public void onConsoleReady(ConsoleReady event, ConsoleSession channel)
111            throws TemplateNotFoundException, MalformedTemplateNameException,
112            ParseException, IOException {
113        Reader deviceTemplate = new InputStreamReader(UPnPBrowserConlet.class
114            .getResourceAsStream("device-tree-template.html"));
115        // Add conlet resources to page
116        channel.respond(new AddConletType(type())
117            .setDisplayNames(
118                localizations(channel.supportedLocales(), "conletName"))
119            .addScript(new ScriptResource()
120                .setScriptUri(event.renderSupport().conletResource(
121                    type(), "UPnPBrowser-functions.ftl.js"))
122                .setScriptType("module"))
123            .addScript(
124                new ScriptResource()
125                    .setScriptId("upnpbrowser-device-tree-template")
126                    .setScriptType("text/x-template")
127                    .loadScriptSource(deviceTemplate))
128            .addCss(event.renderSupport(),
129                WebConsoleUtils.uriFromPath("UPnPBrowser-style.css")));
130    }

138    @Override
139    protected Set<RenderMode> doRenderConlet(RenderConletRequestBase<?> event,
140            ConsoleSession channel, String conletId,
141            UPnPBrowserModel conletState) throws Exception {
142        Set<RenderMode> renderedAs = new HashSet<>();
143        if (event.renderAs().contains(RenderMode.Preview)) {
144            Template tpl = freemarkerConfig()
145                .getTemplate("UPnPBrowser-preview.ftl.html");
146            channel.respond(new RenderConletFromTemplate(event,
147                type(), conletId, tpl,
148                fmModel(event, channel, conletId, conletState))
149                    .setRenderAs(
150                        RenderMode.Preview.addModifiers(event.renderAs()))
151                    .setSupportedModes(MODES));
152            List<Map<String, Object>> deviceInfos = Arrays.stream(
153                context.getAllServiceReferences(UPnPDevice.class.getName(),
154                    "(!(" + UPnPDevice.PARENT_UDN + "=*))"))
155                .map(svc -> createDeviceInfo(context,
156                    (ServiceReference<UPnPDevice>) svc, event.renderSupport()))
157                .collect(Collectors.toList());
158            channel.respond(new NotifyConletView(type(),
159                conletId, "deviceUpdates", deviceInfos, "preview", true));
160            renderedAs.add(RenderMode.Preview);
161        }
162        if (event.renderAs().contains(RenderMode.View)) {
163            Template tpl
164                = freemarkerConfig().getTemplate("UPnPBrowser-view.ftl.html");
165            channel.respond(new RenderConletFromTemplate(event,
166                type(), conletId, tpl,
167                fmModel(event, channel, conletId, conletState))
168                    .setRenderAs(
169                        RenderMode.View.addModifiers(event.renderAs()))
170                    .setSupportedModes(MODES));
171            @SuppressWarnings("PMD.UseConcurrentHashMap")
172            Map<String, Map<String, Object>> deviceInfos = new HashMap<>();
173            Arrays.stream(context
174                .getAllServiceReferences(UPnPDevice.class.getName(), null))
175                .map(svc -> createDeviceInfo(context,
176                    (ServiceReference<UPnPDevice>) svc, event.renderSupport()))
177                .forEach(devInfo -> deviceInfos.put((String) devInfo.get("udn"),
178                    devInfo));
179            channel.respond(new NotifyConletView(type(),
180                conletId, "deviceUpdates", treeify(deviceInfos), "view", true));
181            renderedAs.add(RenderMode.View);
182        }
183        return renderedAs;
184    }

186    @SuppressWarnings({ "PMD.NcssCount", "PMD.AvoidDuplicateLiterals" })
187    private Map<String, Object> createDeviceInfo(BundleContext context,
188            ServiceReference<UPnPDevice> deviceRef,
189            RenderSupport renderSupport) {
190        UPnPDevice device = context.getService(deviceRef);
191        if (device == null) {
192            return null;
193        }
194        try {
195            @SuppressWarnings("PMD.UseConcurrentHashMap")
196            Map<String, Object> result = new HashMap<>();
197            result.put("udn", (String) deviceRef.getProperty(UPnPDevice.UDN));
198            result.computeIfAbsent("parentUdn",
199                k -> (String) deviceRef.getProperty(UPnPDevice.PARENT_UDN));
200            result.put("friendlyName",
201                deviceRef.getProperty(UPnPDevice.FRIENDLY_NAME));
202            if (device.getIcons(null) != null) {
203                result.put("iconUrl", WebConsoleUtils.mergeQuery(
204                    renderSupport.conletResource(type(), ""),
205                    Map.of("udn", (String) deviceRef
206                        .getProperty(UPnPDevice.UDN), "resource", "icon"))
207                    .toASCIIString());
208            }
209            return result;
210        } finally {
211            context.ungetService(deviceRef);
212        }
213    }

215    @SuppressWarnings({ "unchecked", "PMD.AvoidInstantiatingObjectsInLoops" })
216    private List<Map<String, Object>>
217            treeify(Map<String, Map<String, Object>> deviceInfos) {
218        for (Map.Entry<String, Map<String, Object>> e : deviceInfos
219            .entrySet()) {
220            Optional.ofNullable(e.getValue().get("parentUdn")).ifPresent(
221                parentUdn -> ((List<Map<String, Object>>) deviceInfos
222                    .get(parentUdn).computeIfAbsent("childDevices",
223                        k -> new ArrayList<Map<String, Object>>()))
224                            .add(e.getValue()));
225        }
226        return deviceInfos.values().stream()
227            .filter(deviceInfo -> !deviceInfo.containsKey("parentUdn"))
228            .collect(Collectors.toList());
229    }

231    @Override
232    @SuppressWarnings({ "PMD.DataflowAnomalyAnalysis" })
233    protected void doGetResource(ConletResourceRequest event,
234            IOSubchannel channel) {
235        Map<String, List<String>> query
236            = WebConsoleUtils.queryAsMap(event.resourceUri());
237        if (!query.containsKey("udn")) {
238            super.doGetResource(event, channel);
239            return;
240        }
241        try {
242            Arrays.stream(context.getAllServiceReferences(null,
243                String.format("(&(%s=%s)(%s=%s))", Constants.OBJECTCLASS,
244                    UPnPDevice.class.getName(), UPnPDevice.UDN,
245                    query.get("udn").get(0))))
246                .findFirst().ifPresent(deviceRef -> {
247                    @SuppressWarnings("unchecked")
248                    UPnPDevice device = context
249                        .getService((ServiceReference<UPnPDevice>) deviceRef);
250                    if (query.getOrDefault("resource", Collections.emptyList())
251                        .contains("icon")) {
252                        provideIcon(event, device);
253                    }
254                });
255        } catch (InvalidSyntaxException e) {
256            throw new IllegalArgumentException(e);
257        }
258    }
259
260    @SuppressWarnings({ "PMD.EmptyCatchBlock", "PMD.DataflowAnomalyAnalysis" })
261    private void provideIcon(ConletResourceRequest event, UPnPDevice device) {
262        UPnPIcon[] icons = device
263            .getIcons(event.session().locale().toLanguageTag());
264        if (icons == null) {
265            icons = device.getIcons(null);
266        }
267        if (icons == null) {
268            return;
269        }
270        Arrays.sort(icons,
271            Comparator.comparingInt(UPnPIcon::getHeight).reversed());
272        UPnPIcon icon = icons[0];
273        try {
274            if (event.ifModifiedSince().isPresent()) {
275                event.setResult(new ResourceNotModified(event, Instant.now(),
276                    365 * 24 * 3600));
277            } else {
278                MediaType mediaType = null;
279                if (icon.getMimeType() != null
280                    && !icon.getMimeType().isEmpty()) {
281                    mediaType
282                        = Converters.MEDIA_TYPE
283                            .fromFieldValue(icon.getMimeType());
284                }
285                event.setResult(new ResourceByInputStream(event,
286                    icon.getInputStream(), mediaType, Instant.now(),
287                    365 * 24 * 3600));
288            }
289            event.stop();
290        } catch (IOException | java.text.ParseException e) {
291            // Handle as if no match
292        }
293    }
