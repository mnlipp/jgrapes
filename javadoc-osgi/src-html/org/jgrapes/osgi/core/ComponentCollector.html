019package org.jgrapes.osgi.core;
020
021import java.util.Arrays;
022import java.util.Collections;
023import java.util.HashMap;
024import java.util.List;
025import java.util.Map;
026import java.util.function.Function;
027import static java.util.function.Predicate.not;
028import java.util.stream.Collectors;
029import java.util.stream.Stream;
030import org.jgrapes.core.Channel;
031import org.jgrapes.core.ComponentFactory;
032import org.jgrapes.core.events.Stop;
033import org.jgrapes.util.ComponentProvider;
034import org.osgi.framework.BundleContext;
035import org.osgi.framework.ServiceReference;
036import org.osgi.util.tracker.ServiceTracker;
037import org.osgi.util.tracker.ServiceTrackerCustomizer;
038
039/**
040 * A component that collects all services from the OSGi service registry
041 * which implement the {@link ComponentFactory} interface. It uses each 
042 * to create one or more components that are then attached to the 
043 * component collector instance.
044 * 
045 * Effectively, the component collector leverages OSGi's service layer
046 * to modify the component tree at run-time.
047 * 
048 * This class uses {@link ComponentProvider#setFactories(ComponentFactory...)} 
049 * and {@link ComponentProvider#setPinned(List)} for its implementation.
050 * As it inherits from {@link ComponentProvider}, it automatically
051 * supports the provisioning of additional components through
052 * {@link ConfigurationUpdate} events. If this is not desired, invoke
053 * {@link ComponentProvider#setComponentsEntry(String)} with `null` as
054 * argument. 
055 * 
056 * @param <F> the component factory type
057 */
058@SuppressWarnings({ "PMD.DataflowAnomalyAnalysis",
059    "PMD.DataflowAnomalyAnalysis" })
060public class ComponentCollector<F extends ComponentFactory>
061        extends ComponentProvider implements ServiceTrackerCustomizer<F, F> {
062
063    private static final List<Map<Object, Object>> SINGLE_DEFAULT
064        = Arrays.asList(Collections.emptyMap());
065
066    private BundleContext context;
067    @SuppressWarnings("PMD.SingularField")
068    private ServiceTracker<F, F> serviceTracker;
069    private Function<String, List<Map<Object, Object>>> configurator;
070
071    /**
072     * Creates a collector component that uses a {@link ServiceTracker} 
073     * to monitor the addition and removal of component factories.
074     * 
075     * @see #addingService(ServiceReference)
076     * @see #removedService(ServiceReference, ComponentFactory)
077     *
078     * @param componentChannel this component's channel
079     * @param context the OSGi {@link BundleContext}
080     * @param factoryCls the factory class
081     * @param configurator the function that provides the pinned configurations
082     */
083    public ComponentCollector(
084            Channel componentChannel, BundleContext context,
085            Class<F> factoryCls,
086            Function<String, List<Map<Object, Object>>> configurator) {
087        super(componentChannel);
088        this.context = context;
089        this.configurator = configurator;
090        serviceTracker = new ServiceTracker<>(context, factoryCls, this);
091        serviceTracker.open();
092    }
093
094    /**
095     * Utility constructor that uses each factory to create a single instance,
096     * using an empty map as properties.
097     *
098     * @param componentChannel this component's channel
099     * @param context the bundle context
100     * @param factoryClass the factory class
101     */
102    public ComponentCollector(Channel componentChannel, BundleContext context,
103            Class<F> factoryClass) {
104        this(componentChannel, context, factoryClass, type -> SINGLE_DEFAULT);
105    }
106
107    /**
108     * Whenever a new factory is added, it is used to create component 
109     * instances with this component's channel. First, the `configurator`
110     * passed to the constructor is invoked with the name of the class 
111     * of the component to be created as argument. The list of maps 
112     * returned is then added to the pinned components 
113     * (see {@link ComponentProvider#setPinned(List)}). 
114     * 
115     * The map return from the `configurator` is automatically augmented
116     * with an entry with key {@link BundleContext}.class and the bundle 
117     * context passed to the constructor.
118     *
119     * @see ServiceTrackerCustomizer#addingService(ServiceReference)
120     */
121    @Override
122    public F addingService(ServiceReference<F> reference) {
123        F factory = context.getService(reference);
124        if (factories().containsKey(factory.componentType().getName())) {
125            // Factory for the new component type is already known.
126            return factory;
127        }
128
129        // Add configuration to provider before adding new factory
130        List<Map<?, ?>> newConfigs = Stream.concat(
131            // filter existing configs to avoid duplicates
132            pinned().stream().filter(not(c -> factory.componentType().getName()
133                .equals(c.get(COMPONENT_TYPE)))),
134            // add new configs
135            configurator.apply(
136                factory.componentType().getName()).stream().map(c -> {
137                    // Received may be immutable
138                    var newMap = new HashMap<>(c);
139                    newMap.put(BundleContext.class, context);
140                    if (!c.containsKey(COMPONENT_TYPE)) {
141                        newMap.put(COMPONENT_TYPE,
142                            factory.componentType().getName());
143                    }
144                    return newMap;
145                }))
146            .collect(Collectors.toList());
147        setPinned(newConfigs);
148
149        // Now add factory
150        ComponentFactory[] updatedFactories
151            = Stream.concat(factories().values().stream(), Stream.of(factory))
152                .toArray(s -> new ComponentFactory[s]);
153        setFactories(updatedFactories);
154        return factory;
155    }
156
157    /*
158     * (non-Javadoc)
159     * 
160     * @see org.osgi.util.tracker.ServiceTrackerCustomizer#modifiedService
161     */
162    @Override
163    public void modifiedService(ServiceReference<F> reference, F service) {
164        // Do nothing.
165    }
166
167    /**
168     * Deletes all child components with the type produced by the factory that 
169     * is removed. A (possibly) final {@link Stop} event is send to the
170     * detached subtrees which may be used to deactivate bundles.
171     * 
172     * @see ServiceTrackerCustomizer#removedService(ServiceReference, Object)
173     */
174    @Override
175    @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
176    public void removedService(ServiceReference<F> reference, F service) {
177        // Remove factory.
178        ComponentFactory[] updatedFactories = factories().values().stream()
179            .filter(f -> !f.componentType().getName()
180                .equals(service.componentType().getName()))
181            .toArray(s -> new ComponentFactory[s]);
182        setFactories(updatedFactories);
183
184        // Remove configuration
185        List<Map<?, ?>> updatedConfigs = pinned().stream()
186            .filter(not(c -> service.componentType().getName()
187                .equals(c.get(COMPONENT_TYPE))))
188            .collect(Collectors.toList());
189        setPinned(updatedConfigs);
190    }
191}
