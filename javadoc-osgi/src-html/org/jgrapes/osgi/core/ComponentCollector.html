019package org.jgrapes.osgi.core;
020
021import java.util.ArrayList;
022import java.util.Arrays;
023import java.util.Collections;
024import java.util.HashMap;
025import java.util.List;
026import java.util.Map;
027import java.util.ServiceLoader;
028import java.util.function.Function;
029import java.util.stream.StreamSupport;
030
031import org.jgrapes.core.Channel;
032import org.jgrapes.core.Component;
033import org.jgrapes.core.ComponentFactory;
034import org.jgrapes.core.ComponentType;
035import org.jgrapes.core.Components;
036import org.osgi.framework.BundleContext;
037import org.osgi.framework.ServiceReference;
038import org.osgi.util.tracker.ServiceTracker;
039import org.osgi.util.tracker.ServiceTrackerCustomizer;
040
041/**
042 * An advanced version of the basic {@link ComponentCollector} that is based on 
043 * the OSGi registry and not on the {@link ServiceLoader}.
044 */
045public class ComponentCollector<F extends ComponentFactory> extends Component 
046        implements ServiceTrackerCustomizer<F, F> {
047
048        private BundleContext context;
049        private ServiceTracker<F, F> serviceTracker;
050        private Function<String,List<Map<Object,Object>>> matcher;
051        
052        /**
053         * Creates a collector component that uses the {@link ServiceTracker} to
054         * monitor the addition and removal of component factories.
055         *
056         * @param componentChannel this component's channel
057         * @param context the OSGi {@link BundleContext}
058         * @param factoryCls the factory class
059         * @param matcher the matcher function
060         */
061        public ComponentCollector(
062                        Channel componentChannel, BundleContext context, Class<F> factoryCls, 
063                        Function<String,List<Map<Object,Object>>> matcher) {
064                super(componentChannel);
065                this.context = context;
066                this.matcher = matcher;
067                serviceTracker = new ServiceTracker<>(context, factoryCls, this);
068                serviceTracker.open();
069        }
070
071        private static List<Map<Object,Object>> SINGLE_DEFAULT 
072                = Arrays.asList(Collections.emptyMap());
073
074        /**
075         * Utility constructor that uses each factory to create a single
076         * instance, using an empty map as properties.
077         * 
078         * @param factoryClass the factory class
079         * @param componentChannel this component's channel
080         */
081        public ComponentCollector(Channel componentChannel, BundleContext context,
082                        Class<F> factoryClass) {
083                this(componentChannel, context, factoryClass, type -> SINGLE_DEFAULT);
084        }
085        
086        /**
087         * Whenever a new factory is added, it is used to create 
088         * component instances with this component's channel. First, the 
089         * `matcher` function passed to the constructor is invoked 
090         * with the name of the class of the component to be created 
091         * as argument. The list of maps returned is used to create 
092         * components, passing each element in the list 
093         * as parameter to {@link ComponentFactory#create(Channel, Map)}.
094         * The map return from the `matcher` is automatically augmented
095         * with an entry with key {@link BundleContext}.class and
096         * the bundle context passed to the constructor.
097         *
098         * @see ServiceTrackerCustomizer#addingService(ServiceReference)
099         */
100        @Override
101        public F addingService(ServiceReference<F> reference) {
102                F factory = context.getService(reference);
103                if (StreamSupport.stream(spliterator(), false)
104                                .filter(c -> c.getClass().equals(factory.componentType()))
105                                .count() == 0) {
106                        List<Map<Object,Object>> configs = matcher.apply(
107                                        factory.componentType().getName());
108                        for (Map<?,?> config: configs) {
109                                Map<Object,Object> props = new HashMap<>(config);
110                                props.put(BundleContext.class, context);
111                                factory.create(channel(), props).ifPresent(
112                                                component -> attach(component));
113                        }
114                }
115                return factory;
116        }
117
118        /* (non-Javadoc)
119         * @see org.osgi.util.tracker.ServiceTrackerCustomizer#modifiedService
120         */
121        @Override
122        public void modifiedService(ServiceReference<F> reference, F service) {
123        }
124
125        /**
126         * Removes all child component with the type produced by the
127         * factory that is removed.
128         * 
129         * @see ServiceTrackerCustomizer#removedService(ServiceReference, Object)
130         */
131        @Override
132        public void removedService(ServiceReference<F> reference, F service) {
133                // Avoid ConcurrentModificationException
134                List<ComponentType> toBeRemoved = new ArrayList<>();
135                for (ComponentType child: this) {
136                        if (child.getClass().equals(service.componentType())) {
137                                toBeRemoved.add(child);
138                        }
139                }
140                for (ComponentType item: toBeRemoved) {
141                        Components.manager(item).detach();
142                }
143        }
144}
