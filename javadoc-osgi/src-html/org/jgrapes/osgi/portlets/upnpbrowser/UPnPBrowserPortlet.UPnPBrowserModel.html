057/**
058 * A portlet for inspecting the services in an OSGi runtime.
059 */
060public class UPnPBrowserPortlet extends FreeMarkerPortlet {
061
062    private final ServiceComponentRuntime scr;
063    private static final Set<RenderMode> MODES = RenderMode.asSet(
064        DeleteablePreview, View);
065    private final BundleContext context;
066
067    /**
068     * Creates a new component with its channel set to the given channel.
069     * 
070     * @param componentChannel the channel that the component's handlers listen
071     *            on by default and that {@link Manager#fire(Event, Channel...)}
072     *            sends the event to
073     */
074    public UPnPBrowserPortlet(Channel componentChannel, BundleContext context,
075            ServiceComponentRuntime scr) {
076        super(componentChannel, true);
077        this.context = context;
078        this.scr = scr;
079    }
080
081    /**
082     * On {@link PortalReady}, fire the {@link AddPortletType}.
083     *
084     * @param event the event
085     * @param channel the channel
086     * @throws TemplateNotFoundException the template not found exception
087     * @throws MalformedTemplateNameException the malformed template name
088     *             exception
089     * @throws ParseException the parse exception
090     * @throws IOException Signals that an I/O exception has occurred.
091     */
092    @Handler
093    public void onPortalReady(PortalReady event, PortalSession channel)
094            throws TemplateNotFoundException, MalformedTemplateNameException,
095            ParseException, IOException {
096        ResourceBundle resourceBundle = resourceBundle(channel.locale());
097        // Add portlet resources to page
098        channel.respond(new AddPortletType(type())
099            .setDisplayName(resourceBundle.getString("portletName"))
100            .addScript(new ScriptResource()
101                .setRequires(new String[] { "vuejs.org" })
102                .setScriptUri(event.renderSupport().portletResource(
103                    type(), "UPnPBrowser-functions.ftl.js")))
104            .addCss(event.renderSupport(),
105                PortalWeblet.uriFromPath("UPnPBrowser-style.css"))
106            .setInstantiable());
107    }
108
109    /*
110     * (non-Javadoc)
111     * 
112     * @see org.jgrapes.portal.AbstractPortlet#generatePortletId()
113     */
114    @Override
115    protected String generatePortletId() {
116        return type() + "-" + super.generatePortletId();
117    }
118
119    /*
120     * (non-Javadoc)
121     * 
122     * @see org.jgrapes.portal.AbstractPortlet#modelFromSession
123     */
124    @SuppressWarnings("unchecked")
125    @Override
126    protected <T extends Serializable> Optional<T> stateFromSession(
127            Session session, String portletId, Class<T> type) {
128        if (portletId.startsWith(type() + "-")) {
129            return Optional.of((T) new UPnPBrowserModel(portletId));
130        }
131        return Optional.empty();
132    }
133
134    /*
135     * (non-Javadoc)
136     * 
137     * @see org.jgrapes.portal.AbstractPortlet#doAddPortlet
138     */
139    @Override
140    protected String doAddPortlet(AddPortletRequest event,
141            PortalSession channel)
142            throws Exception {
143        UPnPBrowserModel portletModel
144            = new UPnPBrowserModel(generatePortletId());
145        renderPortlet(event, channel, portletModel);
146        return portletModel.getPortletId();
147    }
148
149    /*
150     * (non-Javadoc)
151     * 
152     * @see org.jgrapes.portal.AbstractPortlet#doRenderPortlet
153     */
154    @Override
155    protected void doRenderPortlet(RenderPortletRequest event,
156            PortalSession channel, String portletId,
157            Serializable retrievedState)
158            throws Exception {
159        UPnPBrowserModel portletModel = (UPnPBrowserModel) retrievedState;
160        renderPortlet(event, channel, portletModel);
161    }
162
163    private void renderPortlet(RenderPortletRequestBase<?> event,
164            PortalSession channel,
165            UPnPBrowserModel portletModel) throws TemplateNotFoundException,
166            MalformedTemplateNameException, ParseException, IOException,
167            InvalidSyntaxException {
168        switch (event.renderMode()) {
169        case Preview:
170        case DeleteablePreview: {
171            Template tpl
172                = freemarkerConfig()
173                    .getTemplate("UPnPBrowser-preview.ftl.html");
174            channel.respond(new RenderPortletFromTemplate(event,
175                UPnPBrowserPortlet.class, portletModel.getPortletId(),
176                tpl, fmModel(event, channel, portletModel))
177                    .setRenderMode(DeleteablePreview).setSupportedModes(MODES)
178                    .setForeground(event.isForeground()));
179            break;
180        }
181        case View: {
182            Template tpl
183                = freemarkerConfig().getTemplate("UPnPBrowser-view.ftl.html");
184            channel.respond(new RenderPortletFromTemplate(event,
185                UPnPBrowserPortlet.class, portletModel.getPortletId(),
186                tpl, fmModel(event, channel, portletModel))
187                    .setSupportedModes(MODES)
188                    .setForeground(event.isForeground()));
189            break;
190        }
191        default:
192            break;
193        }
194    }
195
196    /*
197     * (non-Javadoc)
198     * 
199     * @see org.jgrapes.portal.AbstractPortlet#doDeletePortlet
200     */
201    @Override
202    protected void doDeletePortlet(DeletePortletRequest event,
203            PortalSession channel,
204            String portletId, Serializable portletState) throws Exception {
205        channel.respond(new DeletePortlet(portletId));
206    }
207
208    /**
209     * The portlet's model.
210     */
211    @SuppressWarnings("serial")
212    public class UPnPBrowserModel extends PortletBaseModel {
213
214        /**
215         * Instantiates a new service list model.
216         *
217         * @param portletId the portlet id
218         */
219        public UPnPBrowserModel(String portletId) {
220            super(portletId);
221        }
222
223    }
