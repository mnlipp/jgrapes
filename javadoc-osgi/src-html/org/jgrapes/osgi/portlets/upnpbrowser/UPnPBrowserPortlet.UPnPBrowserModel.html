103    /**
104     * On {@link PortalReady}, fire the {@link AddPortletType}.
105     *
106     * @param event the event
107     * @param channel the channel
108     * @throws TemplateNotFoundException the template not found exception
109     * @throws MalformedTemplateNameException the malformed template name
110     *             exception
111     * @throws ParseException the parse exception
112     * @throws IOException Signals that an I/O exception has occurred.
113     */
114    @Handler
115    public void onPortalReady(PortalReady event, PortalSession channel)
116            throws TemplateNotFoundException, MalformedTemplateNameException,
117            ParseException, IOException {
118        ResourceBundle resourceBundle = resourceBundle(channel.locale());
119        // Add portlet resources to page
120        channel.respond(new AddPortletType(type())
121            .setDisplayName(resourceBundle.getString("portletName"))
122            .addScript(new ScriptResource()
123                .setRequires(new String[] { "vuejs.org" })
124                .setScriptUri(event.renderSupport().portletResource(
125                    type(), "UPnPBrowser-functions.ftl.js")))
126            .addCss(event.renderSupport(),
127                PortalUtils.uriFromPath("UPnPBrowser-style.css"))
128            .setInstantiable());
129    }

185    @SuppressWarnings({ "PMD.AvoidDuplicateLiterals",
186        "PMD.DataflowAnomalyAnalysis" })
187    private void renderPortlet(RenderPortletRequestBase<?> event,
188            PortalSession channel,
189            UPnPBrowserModel portletModel) throws TemplateNotFoundException,
190            MalformedTemplateNameException, ParseException, IOException,
191            InvalidSyntaxException {
192        switch (event.renderMode()) {
193        case Preview:
194        case DeleteablePreview: {
195            Template tpl = freemarkerConfig()
196                .getTemplate("UPnPBrowser-preview.ftl.html");
197            channel.respond(new RenderPortletFromTemplate(event,
198                UPnPBrowserPortlet.class, portletModel.getPortletId(),
199                tpl, fmModel(event, channel, portletModel))
200                    .setRenderMode(DeleteablePreview).setSupportedModes(MODES)
201                    .setForeground(event.isForeground()));
202            @SuppressWarnings("unchecked")
203            List<Map<String, Object>> deviceInfos = Arrays.stream(
204                context.getAllServiceReferences(null,
205                    "(&" + "(" + Constants.OBJECTCLASS
206                        + "=" + UPnPDevice.class.getName() + ")"
207                        + "(!(" + UPnPDevice.PARENT_UDN + "=*)" + "))"))
208                .map(svc -> createDeviceInfo(context,
209                    (ServiceReference<UPnPDevice>) svc, event.renderSupport()))
210                .collect(Collectors.toList());
211            channel.respond(new NotifyPortletView(type(),
212                portletModel.getPortletId(), "deviceUpdates", deviceInfos,
213                "preview", true));
214            break;
215        }
216        case View: {
217            Template tpl
218                = freemarkerConfig().getTemplate("UPnPBrowser-view.ftl.html");
219            channel.respond(new RenderPortletFromTemplate(event,
220                UPnPBrowserPortlet.class, portletModel.getPortletId(),
221                tpl, fmModel(event, channel, portletModel))
222                    .setSupportedModes(MODES)
223                    .setForeground(event.isForeground()));
224            @SuppressWarnings("unchecked")
225            List<Map<String, Object>> deviceInfos = Arrays.stream(
226                context.getAllServiceReferences(UPnPDevice.class.toString(),
227                    null))
228                .map(svc -> createDeviceInfo(context,
229                    (ServiceReference<UPnPDevice>) svc, event.renderSupport()))
230                .collect(Collectors.toList());
231            channel.respond(new NotifyPortletView(type(),
232                portletModel.getPortletId(), "deviceUpdates", deviceInfos,
233                "view", true));
234            break;
235        }
236        default:
237            break;
238        }
239    }

241    @SuppressWarnings({ "PMD.NcssCount" })
242    private Map<String, Object> createDeviceInfo(BundleContext context,
243            ServiceReference<UPnPDevice> deviceRef,
244            RenderSupport renderSupport) {
245        UPnPDevice device = context.getService(deviceRef);
246        if (device == null) {
247            return null;
248        }
249        try {
250            @SuppressWarnings("PMD.UseConcurrentHashMap")
251            Map<String, Object> result = new HashMap<>();
252            result.put("friendlyName",
253                deviceRef.getProperty(UPnPDevice.FRIENDLY_NAME));
254            if (device.getIcons(null) != null) {
255                result.put("iconUrl", PortalUtils.mergeQuery(
256                    renderSupport.portletResource(type(), ""),
257                    Components.mapOf("udn", (String) deviceRef
258                        .getProperty(UPnPDevice.UDN), "resource", "icon"))
259                    .toASCIIString());
260            }
261            return result;
262        } finally {
263            context.ungetService(deviceRef);
264        }
265    }

267    @Override
268    @SuppressWarnings({ "PMD.DataflowAnomalyAnalysis" })
269    protected void doGetResource(PortletResourceRequest event,
270            IOSubchannel channel) {
271        Map<String, List<String>> query
272            = PortalUtils.queryAsMap(event.resourceUri());
273        if (!query.containsKey("udn")) {
274            super.doGetResource(event, channel);
275            return;
276        }
277        try {
278            Arrays.stream(context.getAllServiceReferences(null,
279                String.format("(&(%s=%s)(%s=%s))", Constants.OBJECTCLASS,
280                    UPnPDevice.class.getName(), UPnPDevice.UDN,
281                    query.get("udn").get(0))))
282                .findFirst().ifPresent(deviceRef -> {
283                    @SuppressWarnings("unchecked")
284                    UPnPDevice device = context
285                        .getService((ServiceReference<UPnPDevice>) deviceRef);
286                    if (query.getOrDefault("resource", Collections.emptyList())
287                        .contains("icon")) {
288                        provideIcon(event, device);
289                    }
290                });
291        } catch (InvalidSyntaxException e) {
292            throw new IllegalArgumentException(e);
293        }
294    }
295
296    @SuppressWarnings({ "PMD.EmptyCatchBlock", "PMD.DataflowAnomalyAnalysis" })
297    private void provideIcon(PortletResourceRequest event, UPnPDevice device) {
298        UPnPIcon[] icons = device
299            .getIcons(event.session().locale().toLanguageTag());
300        if (icons == null) {
301            icons = device.getIcons(null);
302        }
303        if (icons == null) {
304            return;
305        }
306        Arrays.sort(icons,
307            Comparator.comparingInt(UPnPIcon::getHeight).reversed());
308        UPnPIcon icon = icons[0];
309        try {
310            if (event.ifModifiedSince().isPresent()) {
311                event.setResult(new ResourceNotModified(event, Instant.now(),
312                    365 * 24 * 3600));
313            } else {
314                MediaType mediaType = null;
315                if (icon.getMimeType() != null
316                    && !icon.getMimeType().isEmpty()) {
317                    mediaType
318                        = Converters.MEDIA_TYPE
319                            .fromFieldValue(icon.getMimeType());
320                }
321                event.setResult(new ResourceByInputStream(event,
322                    icon.getInputStream(), mediaType, Instant.now(),
323                    365 * 24 * 3600));
324            }
325            event.stop();
326        } catch (IOException | java.text.ParseException e) {
327            // Handle as if no match
328        }
329    }
