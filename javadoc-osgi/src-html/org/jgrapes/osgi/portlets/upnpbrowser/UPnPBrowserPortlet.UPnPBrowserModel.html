070 * A portlet for inspecting the services in an OSGi runtime.

073        implements ServiceListener {

075    private static final Set<RenderMode> MODES = RenderMode.asSet(
076        DeleteablePreview, View);
077    private final BundleContext context;

079    /**
080     * Creates a new component with its channel set to the given channel.
081     * 
082     * @param componentChannel the channel that the component's handlers listen
083     *            on by default and that {@link Manager#fire(Event, Channel...)}
084     *            sends the event to
085     */
086    @SuppressWarnings("PMD.UnusedFormalParameter")
087    public UPnPBrowserPortlet(Channel componentChannel, BundleContext context,
088            ServiceComponentRuntime scr) {
089        super(componentChannel, true);
090        this.context = context;
091    }
092
093    /**
094     * On {@link PortalReady}, fire the {@link AddPortletType}.
095     *
096     * @param event the event
097     * @param channel the channel
098     * @throws TemplateNotFoundException the template not found exception
099     * @throws MalformedTemplateNameException the malformed template name
100     *             exception
101     * @throws ParseException the parse exception
102     * @throws IOException Signals that an I/O exception has occurred.
103     */
104    @Handler
105    public void onPortalReady(PortalReady event, PortalSession channel)
106            throws TemplateNotFoundException, MalformedTemplateNameException,
107            ParseException, IOException {
108        ResourceBundle resourceBundle = resourceBundle(channel.locale());
109        // Add portlet resources to page
110        channel.respond(new AddPortletType(type())
111            .setDisplayName(resourceBundle.getString("portletName"))
112            .addScript(new ScriptResource()
113                .setRequires(new String[] { "vuejs.org" })
114                .setScriptUri(event.renderSupport().portletResource(
115                    type(), "UPnPBrowser-functions.ftl.js")))
116            .addCss(event.renderSupport(),
117                PortalWeblet.uriFromPath("UPnPBrowser-style.css"))
118            .setInstantiable());
119    }

126    @Override

128        return type() + "-" + super.generatePortletId();

131    /*
132     * (non-Javadoc)
133     * 
134     * @see org.jgrapes.portal.AbstractPortlet#modelFromSession
135     */
136    @SuppressWarnings("unchecked")
137    @Override
138    protected <T extends Serializable> Optional<T> stateFromSession(
139            Session session, String portletId, Class<T> type) {
140        if (portletId.startsWith(type() + "-")) {
141            return Optional.of((T) new UPnPBrowserModel(portletId));
142        }
143        return Optional.empty();
144    }

146    /*
147     * (non-Javadoc)
148     * 
149     * @see org.jgrapes.portal.AbstractPortlet#doAddPortlet
150     */
151    @Override
152    protected String doAddPortlet(AddPortletRequest event,
153            PortalSession channel)
154            throws Exception {
155        UPnPBrowserModel portletModel
156            = new UPnPBrowserModel(generatePortletId());
157        renderPortlet(event, channel, portletModel);
158        return portletModel.getPortletId();
159    }

161    /*
162     * (non-Javadoc)

164     * @see org.jgrapes.portal.AbstractPortlet#doRenderPortlet
165     */
166    @Override
167    protected void doRenderPortlet(RenderPortletRequest event,
168            PortalSession channel, String portletId,
169            Serializable retrievedState)
170            throws Exception {
171        UPnPBrowserModel portletModel = (UPnPBrowserModel) retrievedState;
172        renderPortlet(event, channel, portletModel);
173    }
174
175    @SuppressWarnings({ "PMD.AvoidDuplicateLiterals",
176        "PMD.DataflowAnomalyAnalysis" })
177    private void renderPortlet(RenderPortletRequestBase<?> event,
178            PortalSession channel,
179            UPnPBrowserModel portletModel) throws TemplateNotFoundException,
180            MalformedTemplateNameException, ParseException, IOException,
181            InvalidSyntaxException {
182        switch (event.renderMode()) {
183        case Preview:
184        case DeleteablePreview: {
185            Template tpl = freemarkerConfig()
186                .getTemplate("UPnPBrowser-preview.ftl.html");
187            channel.respond(new RenderPortletFromTemplate(event,
188                UPnPBrowserPortlet.class, portletModel.getPortletId(),
189                tpl, fmModel(event, channel, portletModel))
190                    .setRenderMode(DeleteablePreview).setSupportedModes(MODES)
191                    .setForeground(event.isForeground()));
192            List<Map<String, Object>> serviceInfos = Arrays.stream(
193                context.getAllServiceReferences(null,
194                    "(&" + "(" + Constants.OBJECTCLASS
195                        + "=" + UPnPDevice.class.getName() + ")"
196                        + "(" + UPnPDevice.UDN + "=*)" + ")"))
197                .map(svc -> createServiceInfo(svc))
198                .collect(Collectors.toList());
199            channel.respond(new NotifyPortletView(type(),
200                portletModel.getPortletId(), "deviceUpdates", serviceInfos,
201                "preview", true));
202            break;
203        }
204        case View: {
205            Template tpl
206                = freemarkerConfig().getTemplate("UPnPBrowser-view.ftl.html");
207            channel.respond(new RenderPortletFromTemplate(event,
208                UPnPBrowserPortlet.class, portletModel.getPortletId(),
209                tpl, fmModel(event, channel, portletModel))
210                    .setSupportedModes(MODES)
211                    .setForeground(event.isForeground()));
212            List<Map<String, Object>> serviceInfos = Arrays.stream(
213                context.getAllServiceReferences(UPnPDevice.class.toString(),
214                    null))
215                .map(svc -> createServiceInfo(svc))
216                .collect(Collectors.toList());
217            channel.respond(new NotifyPortletView(type(),
218                portletModel.getPortletId(), "deviceUpdates", serviceInfos,
219                "view", true));
220            break;
221        }
222        default:
223            break;
224        }
225    }
226
227    @SuppressWarnings({ "PMD.NcssCount" })
228    private Map<String, Object>
229            createServiceInfo(ServiceReference<?> serviceRef) {
230        @SuppressWarnings("PMD.UseConcurrentHashMap")
231        Map<String, Object> result = new HashMap<>();
232        result.put("serviceId", serviceRef.getProperty(Constants.SERVICE_ID));
233        String[] props = serviceRef.getPropertyKeys();
234        result.put("friendlyName",
235            serviceRef.getProperty(UPnPDevice.FRIENDLY_NAME));
236        return result;
237    }
