076        private final static Set<RenderMode> MODES = RenderMode.asSet(

081         * Creates a new component with its channel set to the given

089                super(componentChannel, true);

094        @Handler
095        public void onPortalReady(PortalReady event, PortalSession channel) 
096                        throws TemplateNotFoundException, MalformedTemplateNameException, 
097                        ParseException, IOException {
098                ResourceBundle resourceBundle = resourceBundle(channel.locale());
099                // Add portlet resources to page
100                channel.respond(new AddPortletType(type())
101                                .setDisplayName(resourceBundle.getString("portletName"))
102                                .addScript(new ScriptResource()
103                                                .setRequires(new String[] {"datatables.net"})
104                                                .setScriptUri(event.renderSupport().portletResource(
105                                                                type(), "Bundles-functions.ftl.js")))
106                                .addCss(event.renderSupport().portletResource(
107                                                type(), "Bundles-style.css"))
108                                .setInstantiable());
109        }

119        /* (non-Javadoc)
120         * @see org.jgrapes.portal.AbstractPortlet#modelFromSession(org.jgrapes.io.IOSubchannel, java.lang.String)
121         */
122        @SuppressWarnings("unchecked")
123        @Override
124        protected <T extends Serializable> Optional<T> stateFromSession(
125                        Session session, String portletId, Class<T> type) {
126                if (portletId.startsWith(type() + "-")) {
127                        return Optional.of((T)new BundleListModel(portletId));
128                }
129                return Optional.empty();
130        }

132        /* (non-Javadoc)
133         * @see org.jgrapes.portal.AbstractPortlet#doAddPortlet(org.jgrapes.portal.events.AddPortletRequest, org.jgrapes.io.IOSubchannel, org.jgrapes.portal.AbstractPortlet.PortletModelBean)
134         */
135        @Override
136        protected String doAddPortlet(AddPortletRequest event, PortalSession channel)
137                        throws Exception {
138                BundleListModel portletModel = new BundleListModel(generatePortletId());
139                Template tpl = freemarkerConfig().getTemplate("Bundles-preview.ftl.html");
140                channel.respond(new RenderPortlet(
141                                BundleListPortlet.class, portletModel.getPortletId(),
142                                templateProcessor(tpl, fmModel(event, channel, portletModel)))
143                                .setRenderMode(DeleteablePreview).setSupportedModes(MODES)
144                                .setForeground(true));
145                List<Map<String,Object>> bundleInfos = Arrays.stream(context.getBundles())
146                                .map(b -> createBundleInfo(b, channel.locale())).collect(Collectors.toList());
147                channel.respond(new NotifyPortletView(type(),
148                                portletModel.getPortletId(), "bundleUpdates", bundleInfos, "preview", true));
149                return portletModel.getPortletId();
150        }

152        /* (non-Javadoc)
153         * @see org.jgrapes.portal.AbstractPortlet#doRenderPortlet(org.jgrapes.portal.events.RenderPortletRequest, org.jgrapes.io.IOSubchannel, org.jgrapes.portal.AbstractPortlet.PortletModelBean)
154         */
155        @Override
156        protected void doRenderPortlet(RenderPortletRequest event,
157                PortalSession channel, String portletId, Serializable retrievedState)
158                throws Exception {
159                BundleListModel portletModel = (BundleListModel)retrievedState;
160                switch (event.renderMode()) {
161                case Preview:
162                case DeleteablePreview: {
163                        Template tpl = freemarkerConfig().getTemplate("Bundles-preview.ftl.html");
164                        channel.respond(new RenderPortlet(
165                                        BundleListPortlet.class, portletId, 
166                                        templateProcessor(tpl, fmModel(event, channel, portletModel)))
167                                        .setRenderMode(DeleteablePreview).setSupportedModes(MODES)
168                                        .setForeground(event.isForeground()));
169                        List<Map<String,Object>> bundleInfos = Arrays.stream(context.getBundles())
170                                        .map(b -> createBundleInfo(b, channel.locale())).collect(Collectors.toList());
171                        channel.respond(new NotifyPortletView(type(),
172                                        event.portletId(), "bundleUpdates", bundleInfos, "preview", true));
173                        break;
174                }
175                case View: {
176                        Template tpl = freemarkerConfig().getTemplate("Bundles-view.ftl.html");
177                        channel.respond(new RenderPortlet(
178                                        BundleListPortlet.class, portletModel.getPortletId(), 
179                                        templateProcessor(tpl, fmModel(event, channel, portletModel)))
180                                        .setRenderMode(View).setSupportedModes(MODES)
181                                        .setForeground(event.isForeground()));
182                        List<Map<String,Object>> bundleInfos = Arrays.stream(context.getBundles())
183                                        .map(b -> createBundleInfo(b, channel.locale())).collect(Collectors.toList());
184                        channel.respond(new NotifyPortletView(type(),
185                                        event.portletId(), "bundleUpdates", bundleInfos, "view", true));
186                        break;
187                }
188                default:
189                        break;
190                }       
191        }

193        private Map<String,Object> createBundleInfo(Bundle bundle, Locale locale) {
194                Map<String, Object> result = new HashMap<>();
195                result.put("id", bundle.getBundleId());
196                result.put("name", Optional.ofNullable(bundle.getHeaders(locale.toString())
197                                .get("Bundle-Name")).orElse(bundle.getSymbolicName()));
198                result.put("symbolicName", bundle.getSymbolicName());
199                result.put("version", bundle.getVersion().toString());
200                result.put("category", Optional.ofNullable(bundle.getHeaders(locale.toString())
201                                .get("Bundle-Category")).orElse(""));
202                ResourceBundle rb = resourceBundle(locale);
203                result.put("state", rb.getString("bundleState_" + bundle.getState()));
204                result.put("startable", false);
205                result.put("stoppable", false);
206                if ((bundle.getState() & (Bundle.RESOLVED | Bundle.INSTALLED | Bundle.ACTIVE)) != 0) {
207                        boolean isFragment = ((bundle.adapt(BundleRevision.class).getTypes() 
208                                        & BundleRevision.TYPE_FRAGMENT) != 0);
209                        result.put("startable", !isFragment 
210                                        && (bundle.getState() == Bundle.INSTALLED
211                                                || bundle.getState() == Bundle.RESOLVED));
212                        result.put("stoppable", !isFragment && bundle.getState() == Bundle.ACTIVE);
213                }
214                result.put("uninstallable", (bundle.getState() 
215                                & (Bundle.INSTALLED | Bundle.RESOLVED | Bundle.ACTIVE)) != 0);
216                result.put("uninstalled", bundle.getState() == Bundle.UNINSTALLED);
217                return result;
218        }

230        /* (non-Javadoc)
231         * @see org.jgrapes.portal.AbstractPortlet#doNotifyPortletModel(org.jgrapes.portal.events.NotifyPortletModel, org.jgrapes.io.IOSubchannel, org.jgrapes.http.Session, java.io.Serializable)
232         */
233        @Override
234        protected void doNotifyPortletModel(NotifyPortletModel event,
235                PortalSession channel, Serializable portletState)
236                throws Exception {
237                event.stop();
238                Bundle bundle = context.getBundle(event.params().getInt(0));
239                if (bundle == null) {
240                        return;
241                }
242                try {
243                        switch (event.method()) {
244                        case "stop":
245                                bundle.stop();
246                                break;
247                        case "start":
248                                bundle.start();
249                                break;
250                        case "refresh":
251                                break;
252                        case "update":
253                                bundle.update();
254                                break;
255                        case "uninstall":
256                                bundle.uninstall();
257                                break;
258                        case "sendDetails":
259                                sendBundleDetails(event.portletId(), channel, bundle);
260                        }
261                } catch (BundleException e) {
262                        // ignore
263                }
264        }

266        private void sendBundleDetails(String portletId, PortalSession channel, Bundle bundle) {
267                Locale locale = channel.locale();
268                ResourceBundle rb = resourceBundle(locale);
269                List<Object> data = new ArrayList<>();
270                data.add(new Object[] { rb.getString("bundleSymbolicName"), bundle.getSymbolicName() });
271                data.add(new Object[] { rb.getString("bundleVersion"), bundle.getVersion().toString() });
272                data.add(new Object[] { rb.getString("bundleLocation"), 
273                                bundle.getLocation().replace(".", ".&#x200b;") });
274                data.add(new Object[] { rb.getString("bundleLastModification"),
275                                Instant.ofEpochMilli(bundle.getLastModified()).toString(),
276                                "dateTime" });
277                data.add(new Object[] { rb.getString("bundleStartLevel"), 
278                                bundle.adapt(BundleStartLevel.class).getStartLevel() });
279                Dictionary<String,String> dict = bundle.getHeaders(locale.toString());
280                Map<String,String> headers = new TreeMap<>();
281                for (Enumeration<String> e = dict.keys(); e.hasMoreElements();) {
282                        String key = e.nextElement();
283                        headers.put(key, dict.get(key));
284                }
285                List<Object> headerList = new ArrayList<>();
286                for (Map.Entry<String, String> e: headers.entrySet()) {
287                        headerList.add(new Object[] { e.getKey(), 
288                                        e.getKey().contains("Package") 
289                                        ? e.getValue().replace(".", ".&#x200b;") : e.getValue() });
290                }
291                data.add(new Object[] { rb.getString("manifestHeaders"), headerList, "table" });
292                channel.respond(new NotifyPortletView(type(),
293                                portletId, "bundleDetails", bundle.getBundleId(), data));
294        }

296        @Override
297        public void bundleChanged(BundleEvent event) {
298                for (Map.Entry<PortalSession,Set<String>> e:
299                        portletIdsByPortalSession().entrySet()) {
300                        PortalSession portalSession = e.getKey();
301                        for (String portletId: e.getValue()) {
302                                portalSession.respond(new NotifyPortletView(type(),
303                                        portletId, "bundleUpdates", (Object)new Object[]
304                                                        { createBundleInfo(event.getBundle(), portalSession.locale()) },
305                                                        "*", false));
306                        }
307                }
308        }
