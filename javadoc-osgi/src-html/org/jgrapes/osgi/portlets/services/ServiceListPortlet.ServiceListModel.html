072/**
073 * A portlet for inspecting the services in an OSGi runtime.
074 */
075public class ServiceListPortlet extends FreeMarkerPortlet
076        implements ServiceListener {
077
078        private ServiceComponentRuntime scr;
079        private final static Set<RenderMode> MODES = RenderMode.asSet(
080                        DeleteablePreview, View);
081        private BundleContext context;
082        private Map<IOSubchannel,Set<ServiceListModel>> listenersByChannel 
083                = Collections.synchronizedMap(new WeakHashMap<>());
084        
085        /**
086         * Creates a new component with its channel set to the given 
087         * channel.
088         * 
089         * @param componentChannel the channel that the component's 
090         * handlers listen on by default and that 
091         * {@link Manager#fire(Event, Channel...)} sends the event to 
092         */
093        public ServiceListPortlet(Channel componentChannel, BundleContext context,
094                        ServiceComponentRuntime scr) {
095                super(componentChannel);
096                this.context = context;
097                this.scr = scr;
098                context.addServiceListener(this);
099        }
100        
101        @Handler
102        public void onPortalReady(PortalReady event, IOSubchannel channel) 
103                        throws TemplateNotFoundException, MalformedTemplateNameException, 
104                        ParseException, IOException {
105                ResourceBundle resourceBundle = resourceBundle(locale(channel));
106                // Add portlet resources to page
107                channel.respond(new AddPortletType(type())
108                                .setDisplayName(resourceBundle.getString("portletName"))
109                                .addScript(PortalView.uriFromPath("Services-functions.ftl.js"))
110                                .addCss(PortalView.uriFromPath("Services-style.css"))
111                                .setInstantiable());
112        }
113        
114        /* (non-Javadoc)
115         * @see org.jgrapes.portal.AbstractPortlet#generatePortletId()
116         */
117        @Override
118        protected String generatePortletId() {
119                return type() + "-" + super.generatePortletId();
120        }
121
122        /* (non-Javadoc)
123         * @see org.jgrapes.portal.AbstractPortlet#modelFromSession(org.jgrapes.io.IOSubchannel, java.lang.String)
124         */
125        @SuppressWarnings("unchecked")
126        @Override
127        protected <T extends Serializable> Optional<T> stateFromSession(
128                        Session session, String portletId, Class<T> type) {
129                if (portletId.startsWith(type() + "-")) {
130                        return Optional.of((T)new ServiceListModel(portletId));
131                }
132                return Optional.empty();
133        }
134
135        /* (non-Javadoc)
136         * @see org.jgrapes.portal.AbstractPortlet#doAddPortlet(org.jgrapes.portal.events.AddPortletRequest, org.jgrapes.io.IOSubchannel, org.jgrapes.http.Session)
137         */
138        protected void doAddPortlet(AddPortletRequest event, IOSubchannel channel,
139                Session session) throws Exception {
140                ServiceListModel portletModel = new ServiceListModel(generatePortletId());
141                listenersByChannel.computeIfAbsent(
142                                channel, c -> new HashSet<>()).add(portletModel);
143                Template tpl = freemarkerConfig().getTemplate("Services-preview.ftl.html");
144                channel.respond(new RenderPortlet(
145                                ServiceListPortlet.class, portletModel.getPortletId(),
146                                DeleteablePreview, MODES, true, templateProcessor(
147                                                tpl, fmModel(event, channel, portletModel))));
148                List<Map<String,Object>> serviceInfos = Arrays.stream(
149                                context.getAllServiceReferences(null, null))
150                                .map(s -> createServiceInfo(s, locale(channel))).collect(Collectors.toList());
151                channel.respond(new NotifyPortletView(type(),
152                                portletModel.getPortletId(), "serviceUpdates", serviceInfos, "preview", true));
153        }
154
155        /* (non-Javadoc)
156         * @see org.jgrapes.portal.AbstractPortlet#doRenderPortlet(org.jgrapes.portal.events.RenderPortletRequest, org.jgrapes.io.IOSubchannel, org.jgrapes.http.Session, java.lang.String, java.io.Serializable)
157         */
158        @Override
159        protected void doRenderPortlet(RenderPortletRequest event,
160                IOSubchannel channel, Session session, 
161                String portletId, Serializable retrievedState)
162                throws Exception {
163                ServiceListModel portletModel = (ServiceListModel)retrievedState;
164                listenersByChannel.computeIfAbsent(
165                                channel, k -> new HashSet<>()).add(portletModel);
166                switch (event.renderMode()) {
167                case Preview:
168                case DeleteablePreview: {
169                        Template tpl = freemarkerConfig().getTemplate("Services-preview.ftl.html");
170                        channel.respond(new RenderPortlet(
171                                        ServiceListPortlet.class, portletId, 
172                                        DeleteablePreview, MODES, event.isForeground(), templateProcessor(
173                                                        tpl, fmModel(event, channel, portletModel))));
174                        List<Map<String,Object>> serviceInfos = Arrays.stream(
175                                        context.getAllServiceReferences(null, null))
176                                        .map(s -> createServiceInfo(s, locale(channel))).collect(Collectors.toList());
177                        channel.respond(new NotifyPortletView(type(),
178                                        portletModel.getPortletId(), "serviceUpdates", serviceInfos, "preview", true));
179                        break;
180                }
181                case View: {
182                        Template tpl = freemarkerConfig().getTemplate("Services-view.ftl.html");
183                        channel.respond(new RenderPortlet(
184                                        ServiceListPortlet.class, portletModel.getPortletId(), 
185                                        View, MODES, event.isForeground(), templateProcessor(
186                                                        tpl, fmModel(event, channel, portletModel))));
187                        List<Map<String,Object>> serviceInfos = Arrays.stream(
188                                        context.getAllServiceReferences(null, null))
189                                        .map(s -> createServiceInfo(s, locale(channel))).collect(Collectors.toList());
190                        channel.respond(new NotifyPortletView(type(),
191                                        portletModel.getPortletId(), "serviceUpdates", serviceInfos, "view", true));
192                        break;
193                }
194                default:
195                        break;
196                }       
197        }
198
199        private Map<String,Object> createServiceInfo(ServiceReference<?> serviceRef, Locale locale) {
200                Map<String, Object> result = new HashMap<>();
201                result.put("id", serviceRef.getProperty(Constants.SERVICE_ID).toString());
202                String[] interfaces = (String[])serviceRef.getProperty(Constants.OBJECTCLASS);
203                result.put("type", String.join(", ", interfaces));
204                Long bundleId = (Long)serviceRef.getProperty(Constants.SERVICE_BUNDLEID);
205                result.put("bundleId", bundleId.toString());
206                Bundle bundle = context.getBundle(bundleId);
207                result.put("bundleName", Optional.ofNullable(bundle.getHeaders(locale.toString())
208                                .get("Bundle-Name")).orElse(bundle.getSymbolicName()));
209                String scope = "";
210                switch((String)serviceRef.getProperty(Constants.SERVICE_SCOPE)) {
211                case Constants.SCOPE_BUNDLE:
212                        scope = "serviceScopeBundle";
213                        break;
214                case Constants.SCOPE_PROTOTYPE:
215                        scope = "serviceScopePrototype";
216                        break;
217                case Constants.SCOPE_SINGLETON:
218                        scope = "serviceScopeSingleton";
219                        break;
220                }
221                result.put("scope", scope);
222                Integer ranking = (Integer)serviceRef.getProperty(Constants.SERVICE_RANKING);
223                result.put("ranking", ranking == null ? "" : ranking.toString());
224                String componentName = (String)serviceRef.getProperty("component.name");
225                if (componentName != null) {
226                        ComponentDescriptionDTO dto = scr.getComponentDescriptionDTO(bundle, componentName);
227                        result.put("dsScope", "serviceScope" 
228                                        + dto.scope.substring(0, 1).toUpperCase() + dto.scope.substring(1));
229                        result.put("implementationClass", dto.implementationClass);
230                } else {
231                        Object service = context.getService(serviceRef);
232                        result.put("implementationClass", service.getClass().getName());
233                        context.ungetService(serviceRef);
234                }
235                return result;
236        }
237        
238        /* (non-Javadoc)
239         * @see org.jgrapes.portal.AbstractPortlet#doDeletePortlet(org.jgrapes.portal.events.DeletePortletRequest, org.jgrapes.io.IOSubchannel, org.jgrapes.http.Session, java.lang.String, java.io.Serializable)
240         */
241        @Override
242        protected void doDeletePortlet(DeletePortletRequest event, IOSubchannel channel, Session session, String portletId,
243                        Serializable portletState) throws Exception {
244                ServiceListModel portletModel = (ServiceListModel)portletState;
245                listenersByChannel.computeIfAbsent(
246                                channel, k -> new HashSet<>()).remove(portletModel);
247                channel.respond(new DeletePortlet(portletId));
248        }
249
250        @Override
251        public void serviceChanged(ServiceEvent event) {
252                for (Entry<IOSubchannel,Set<ServiceListModel>> e: listenersByChannel.entrySet()) {
253                        IOSubchannel channel = e.getKey();
254                        Map<String,Object> info = createServiceInfo(event.getServiceReference(), locale(channel));
255                        if (event.getType() == ServiceEvent.UNREGISTERING) {
256                                info.put("updateType", "unregistering");
257                        }
258                        for (ServiceListModel model: e.getValue()) {
259                                channel.respond(new NotifyPortletView(
260                                                type(), model.getPortletId(), "serviceUpdates", 
261                                                (Object)new Object[] { info }, "*", false));
262                        }
263                }
264        }
265
266        @SuppressWarnings("serial")
267        public class ServiceListModel extends PortletBaseModel {
268
269                public ServiceListModel(String portletId) {
270                        super(portletId);
271                }
272
273        }
