068/**
069 * A portlet for inspecting the services in an OSGi runtime.
070 */
071public class ServiceListPortlet extends FreeMarkerPortlet
072        implements ServiceListener {
073
074        private ServiceComponentRuntime scr;
075        private final static Set<RenderMode> MODES = RenderMode.asSet(
076                        DeleteablePreview, View);
077        private BundleContext context;
078        
079        /**
080         * Creates a new component with its channel set to the given 
081         * channel.
082         * 
083         * @param componentChannel the channel that the component's 
084         * handlers listen on by default and that 
085         * {@link Manager#fire(Event, Channel...)} sends the event to 
086         */
087        public ServiceListPortlet(Channel componentChannel, BundleContext context,
088                        ServiceComponentRuntime scr) {
089                super(componentChannel, true);
090                this.context = context;
091                this.scr = scr;
092                context.addServiceListener(this);
093        }
094        
095        @Handler
096        public void onPortalReady(PortalReady event, PortalSession channel) 
097                        throws TemplateNotFoundException, MalformedTemplateNameException, 
098                        ParseException, IOException {
099                ResourceBundle resourceBundle = resourceBundle(channel.locale());
100                // Add portlet resources to page
101                channel.respond(new AddPortletType(type())
102                                .setDisplayName(resourceBundle.getString("portletName"))
103                                .addScript(new ScriptResource()
104                                                .setRequires(new String[] {"datatables.net"})
105                                                .setScriptUri(event.renderSupport().portletResource(
106                                                                type(), "Services-functions.ftl.js")))
107                                .addCss(event.renderSupport().portletResource(
108                                                type(), "Services-style.css"))
109                                .setInstantiable());
110        }
111        
112        /* (non-Javadoc)
113         * @see org.jgrapes.portal.AbstractPortlet#generatePortletId()
114         */
115        @Override
116        protected String generatePortletId() {
117                return type() + "-" + super.generatePortletId();
118        }
119
120        /* (non-Javadoc)
121         * @see org.jgrapes.portal.AbstractPortlet#modelFromSession(org.jgrapes.io.IOSubchannel, java.lang.String)
122         */
123        @SuppressWarnings("unchecked")
124        @Override
125        protected <T extends Serializable> Optional<T> stateFromSession(
126                        Session session, String portletId, Class<T> type) {
127                if (portletId.startsWith(type() + "-")) {
128                        return Optional.of((T)new ServiceListModel(portletId));
129                }
130                return Optional.empty();
131        }
132
133        /* (non-Javadoc)
134         * @see org.jgrapes.portal.AbstractPortlet#doAddPortlet(org.jgrapes.portal.events.AddPortletRequest, org.jgrapes.io.IOSubchannel, org.jgrapes.http.Session)
135         */
136        protected String doAddPortlet(AddPortletRequest event, PortalSession channel)
137                        throws Exception {
138                ServiceListModel portletModel = new ServiceListModel(generatePortletId());
139                Template tpl = freemarkerConfig().getTemplate("Services-preview.ftl.html");
140                channel.respond(new RenderPortlet(
141                                ServiceListPortlet.class, portletModel.getPortletId(),
142                                templateProcessor(tpl, fmModel(event, channel, portletModel)))
143                                .setRenderMode(DeleteablePreview).setSupportedModes(MODES)
144                                .setForeground(true));
145                List<Map<String,Object>> serviceInfos = Arrays.stream(
146                                context.getAllServiceReferences(null, null))
147                                .map(s -> createServiceInfo(s, channel.locale())).collect(Collectors.toList());
148                channel.respond(new NotifyPortletView(type(),
149                                portletModel.getPortletId(), "serviceUpdates", serviceInfos, "preview", true));
150                return portletModel.getPortletId();
151        }
152
153        /* (non-Javadoc)
154         * @see org.jgrapes.portal.AbstractPortlet#doRenderPortlet(org.jgrapes.portal.events.RenderPortletRequest, org.jgrapes.io.IOSubchannel, org.jgrapes.http.Session, java.lang.String, java.io.Serializable)
155         */
156        @Override
157        protected void doRenderPortlet(RenderPortletRequest event,
158                PortalSession channel, String portletId, Serializable retrievedState)
159                throws Exception {
160                ServiceListModel portletModel = (ServiceListModel)retrievedState;
161                switch (event.renderMode()) {
162                case Preview:
163                case DeleteablePreview: {
164                        Template tpl = freemarkerConfig().getTemplate("Services-preview.ftl.html");
165                        channel.respond(new RenderPortlet(
166                                        ServiceListPortlet.class, portletId, 
167                                        templateProcessor(tpl, fmModel(event, channel, portletModel)))
168                                        .setRenderMode(DeleteablePreview).setSupportedModes(MODES)
169                                        .setForeground(event.isForeground()));
170                        List<Map<String,Object>> serviceInfos = Arrays.stream(
171                                        context.getAllServiceReferences(null, null))
172                                        .map(s -> createServiceInfo(s, channel.locale())).collect(Collectors.toList());
173                        channel.respond(new NotifyPortletView(type(),
174                                        portletModel.getPortletId(), "serviceUpdates", serviceInfos, "preview", true));
175                        break;
176                }
177                case View: {
178                        Template tpl = freemarkerConfig().getTemplate("Services-view.ftl.html");
179                        channel.respond(new RenderPortlet(
180                                        ServiceListPortlet.class, portletModel.getPortletId(), 
181                                        templateProcessor(tpl, fmModel(event, channel, portletModel)))
182                                        .setRenderMode(View).setSupportedModes(MODES)
183                                        .setForeground(event.isForeground()));
184                        List<Map<String,Object>> serviceInfos = Arrays.stream(
185                                        context.getAllServiceReferences(null, null))
186                                        .map(s -> createServiceInfo(s, channel.locale())).collect(Collectors.toList());
187                        channel.respond(new NotifyPortletView(type(),
188                                        portletModel.getPortletId(), "serviceUpdates", serviceInfos, "view", true));
189                        break;
190                }
191                default:
192                        break;
193                }       
194        }
195
196        private Map<String,Object> createServiceInfo(ServiceReference<?> serviceRef, Locale locale) {
197                Map<String, Object> result = new HashMap<>();
198                result.put("id", serviceRef.getProperty(Constants.SERVICE_ID).toString());
199                String[] interfaces = (String[])serviceRef.getProperty(Constants.OBJECTCLASS);
200                result.put("type", String.join(", ", interfaces));
201                Long bundleId = (Long)serviceRef.getProperty(Constants.SERVICE_BUNDLEID);
202                result.put("bundleId", bundleId.toString());
203                Bundle bundle = context.getBundle(bundleId);
204                result.put("bundleName", Optional.ofNullable(bundle.getHeaders(locale.toString())
205                                .get("Bundle-Name")).orElse(bundle.getSymbolicName()));
206                String scope = "";
207                switch((String)serviceRef.getProperty(Constants.SERVICE_SCOPE)) {
208                case Constants.SCOPE_BUNDLE:
209                        scope = "serviceScopeBundle";
210                        break;
211                case Constants.SCOPE_PROTOTYPE:
212                        scope = "serviceScopePrototype";
213                        break;
214                case Constants.SCOPE_SINGLETON:
215                        scope = "serviceScopeSingleton";
216                        break;
217                }
218                result.put("scope", scope);
219                Integer ranking = (Integer)serviceRef.getProperty(Constants.SERVICE_RANKING);
220                result.put("ranking", ranking == null ? "" : ranking.toString());
221                String componentName = (String)serviceRef.getProperty("component.name");
222                if (componentName != null) {
223                        ComponentDescriptionDTO dto = scr.getComponentDescriptionDTO(bundle, componentName);
224                        result.put("dsScope", "serviceScope" 
225                                        + dto.scope.substring(0, 1).toUpperCase() + dto.scope.substring(1));
226                        result.put("implementationClass", dto.implementationClass);
227                } else {
228                        Object service = context.getService(serviceRef);
229                        if (service != null) {
230                                result.put("implementationClass", service.getClass().getName());
231                                context.ungetService(serviceRef);
232                        } else {
233                                result.put("implementationClass", "");
234                        }
235                }
236                return result;
237        }

239        /* (non-Javadoc)
240         * @see org.jgrapes.portal.AbstractPortlet#doDeletePortlet(org.jgrapes.portal.events.DeletePortletRequest, org.jgrapes.io.IOSubchannel, org.jgrapes.http.Session, java.lang.String, java.io.Serializable)
241         */
242        @Override
243        protected void doDeletePortlet(DeletePortletRequest event, PortalSession channel, 
244                        String portletId, Serializable portletState) throws Exception {
245                channel.respond(new DeletePortlet(portletId));
246        }
247
248        @Override
249        public void serviceChanged(ServiceEvent event) {
250                for (Map.Entry<PortalSession,Set<String>> e:
251                        portletIdsByPortalSession().entrySet()) {
252                        PortalSession portalSession = e.getKey();
253                        Map<String,Object> info = createServiceInfo(
254                                        event.getServiceReference(), portalSession.locale());
255                        if (event.getType() == ServiceEvent.UNREGISTERING) {
256                                info.put("updateType", "unregistering");
257                        }
258                        for (String portletId: e.getValue()) {
259                                portalSession.respond(new NotifyPortletView(
260                                                type(), portletId, "serviceUpdates", 
261                                                (Object)new Object[] { info }, "*", false));
262                        }
263                }
264        }

270                        super(portletId);
271                }
