069/**
070 * A portlet for inspecting the services in an OSGi runtime.
071 */
072public class ServiceListPortlet extends FreeMarkerPortlet
073        implements ServiceListener {
074
075        private ServiceComponentRuntime scr;
076        private static final Set<RenderMode> MODES = RenderMode.asSet(
077                        DeleteablePreview, View);
078        private BundleContext context;
079        
080        /**
081         * Creates a new component with its channel set to the given 
082         * channel.
083         * 
084         * @param componentChannel the channel that the component's 
085         * handlers listen on by default and that 
086         * {@link Manager#fire(Event, Channel...)} sends the event to 
087         */
088        public ServiceListPortlet(Channel componentChannel, BundleContext context,
089                        ServiceComponentRuntime scr) {
090                super(componentChannel, true);
091                this.context = context;
092                this.scr = scr;
093                context.addServiceListener(this);
094        }
095        
096        @Handler
097        public void onPortalReady(PortalReady event, PortalSession channel) 
098                        throws TemplateNotFoundException, MalformedTemplateNameException, 
099                        ParseException, IOException {
100                ResourceBundle resourceBundle = resourceBundle(channel.locale());
101                // Add portlet resources to page
102                channel.respond(new AddPortletType(type())
103                                .setDisplayName(resourceBundle.getString("portletName"))
104                                .addScript(new ScriptResource()
105                                                .setRequires(new String[] {"datatables.net"})
106                                                .setScriptUri(event.renderSupport().portletResource(
107                                                                type(), "Services-functions.ftl.js")))
108                                .addCss(event.renderSupport().portletResource(
109                                                type(), "Services-style.css"))
110                                .setInstantiable());
111        }
112        
113        /* (non-Javadoc)
114         * @see org.jgrapes.portal.AbstractPortlet#generatePortletId()
115         */
116        @Override
117        protected String generatePortletId() {
118                return type() + "-" + super.generatePortletId();
119        }
120
121        /* (non-Javadoc)
122         * @see org.jgrapes.portal.AbstractPortlet#modelFromSession
123         */
124        @SuppressWarnings("unchecked")
125        @Override
126        protected <T extends Serializable> Optional<T> stateFromSession(
127                        Session session, String portletId, Class<T> type) {
128                if (portletId.startsWith(type() + "-")) {
129                        return Optional.of((T)new ServiceListModel(portletId));
130                }
131                return Optional.empty();
132        }
133
134        /* (non-Javadoc)
135         * @see org.jgrapes.portal.AbstractPortlet#doAddPortlet
136         */
137        protected String doAddPortlet(AddPortletRequest event, PortalSession channel)
138                        throws Exception {
139                ServiceListModel portletModel = new ServiceListModel(generatePortletId());
140                Template tpl = freemarkerConfig().getTemplate("Services-preview.ftl.html");
141                channel.respond(new RenderPortlet(
142                                ServiceListPortlet.class, portletModel.getPortletId(),
143                                templateProcessor(tpl, fmModel(event, channel, portletModel)))
144                                .setRenderMode(DeleteablePreview).setSupportedModes(MODES)
145                                .setForeground(true));
146                List<Map<String,Object>> serviceInfos = Arrays.stream(
147                                context.getAllServiceReferences(null, null))
148                                .map(s -> createServiceInfo(s, channel.locale())).collect(Collectors.toList());
149                channel.respond(new NotifyPortletView(type(),
150                                portletModel.getPortletId(), "serviceUpdates", serviceInfos, "preview", true));
151                return portletModel.getPortletId();
152        }
153
154        /* (non-Javadoc)
155         * @see org.jgrapes.portal.AbstractPortlet#doRenderPortlet
156         */
157        @Override
158        protected void doRenderPortlet(RenderPortletRequest event,
159                PortalSession channel, String portletId, Serializable retrievedState)
160                throws Exception {
161                ServiceListModel portletModel = (ServiceListModel)retrievedState;
162                switch (event.renderMode()) {
163                case Preview:
164                case DeleteablePreview: {
165                        Template tpl = freemarkerConfig().getTemplate("Services-preview.ftl.html");
166                        channel.respond(new RenderPortlet(
167                                        ServiceListPortlet.class, portletId, 
168                                        templateProcessor(tpl, fmModel(event, channel, portletModel)))
169                                        .setRenderMode(DeleteablePreview).setSupportedModes(MODES)
170                                        .setForeground(event.isForeground()));
171                        List<Map<String,Object>> serviceInfos = Arrays.stream(
172                                        context.getAllServiceReferences(null, null))
173                                        .map(s -> createServiceInfo(s, channel.locale())).collect(Collectors.toList());
174                        channel.respond(new NotifyPortletView(type(),
175                                        portletModel.getPortletId(), "serviceUpdates", serviceInfos, "preview", true));
176                        break;
177                }
178                case View: {
179                        Template tpl = freemarkerConfig().getTemplate("Services-view.ftl.html");
180                        channel.respond(new RenderPortlet(
181                                        ServiceListPortlet.class, portletModel.getPortletId(), 
182                                        templateProcessor(tpl, fmModel(event, channel, portletModel)))
183                                        .setRenderMode(View).setSupportedModes(MODES)
184                                        .setForeground(event.isForeground()));
185                        List<Map<String,Object>> serviceInfos = Arrays.stream(
186                                        context.getAllServiceReferences(null, null))
187                                        .map(s -> createServiceInfo(s, channel.locale())).collect(Collectors.toList());
188                        channel.respond(new NotifyPortletView(type(),
189                                        portletModel.getPortletId(), "serviceUpdates", serviceInfos, "view", true));
190                        break;
191                }
192                default:
193                        break;
194                }       
195        }
196
197        private Map<String,Object> createServiceInfo(ServiceReference<?> serviceRef, Locale locale) {
198                Map<String, Object> result = new HashMap<>();
199                result.put("id", serviceRef.getProperty(Constants.SERVICE_ID).toString());
200                String[] interfaces = (String[])serviceRef.getProperty(Constants.OBJECTCLASS);
201                result.put("type", String.join(", ", interfaces));
202                Long bundleId = (Long)serviceRef.getProperty(Constants.SERVICE_BUNDLEID);
203                result.put("bundleId", bundleId.toString());
204                Bundle bundle = context.getBundle(bundleId);
205                result.put("bundleName", Optional.ofNullable(bundle.getHeaders(locale.toString())
206                                .get("Bundle-Name")).orElse(bundle.getSymbolicName()));
207                String scope = "";
208                switch((String)serviceRef.getProperty(Constants.SERVICE_SCOPE)) {
209                case Constants.SCOPE_BUNDLE:
210                        scope = "serviceScopeBundle";
211                        break;
212                case Constants.SCOPE_PROTOTYPE:
213                        scope = "serviceScopePrototype";
214                        break;
215                case Constants.SCOPE_SINGLETON:
216                        scope = "serviceScopeSingleton";
217                        break;
218                }
219                result.put("scope", scope);
220                Integer ranking = (Integer)serviceRef.getProperty(Constants.SERVICE_RANKING);
221                result.put("ranking", ranking == null ? "" : ranking.toString());
222                String componentName = (String)serviceRef.getProperty("component.name");
223                if (componentName != null) {
224                        ComponentDescriptionDTO dto = scr.getComponentDescriptionDTO(bundle, componentName);
225                        result.put("dsScope", "serviceScope" 
226                                        + dto.scope.substring(0, 1).toUpperCase() + dto.scope.substring(1));
227                        result.put("implementationClass", dto.implementationClass);
228                } else {
229                        Object service = context.getService(serviceRef);
230                        if (service != null) {
231                                result.put("implementationClass", service.getClass().getName());
232                                context.ungetService(serviceRef);
233                        } else {
234                                result.put("implementationClass", "");
235                        }
236                }
237                return result;
238        }

240        /* (non-Javadoc)
241         * @see org.jgrapes.portal.AbstractPortlet#doDeletePortlet
242         */
243        @Override
244        protected void doDeletePortlet(DeletePortletRequest event, PortalSession channel, 
245                        String portletId, Serializable portletState) throws Exception {
246                channel.respond(new DeletePortlet(portletId));
247        }
248
249        @Override
250        public void serviceChanged(ServiceEvent event) {
251                for (Map.Entry<PortalSession,Set<String>> e:
252                        portletIdsByPortalSession().entrySet()) {
253                        PortalSession portalSession = e.getKey();
254                        Map<String,Object> info = createServiceInfo(
255                                        event.getServiceReference(), portalSession.locale());
256                        if (event.getType() == ServiceEvent.UNREGISTERING) {
257                                info.put("updateType", "unregistering");
258                        }
259                        for (String portletId: e.getValue()) {
260                                portalSession.respond(new NotifyPortletView(
261                                                type(), portletId, "serviceUpdates", 
262                                                (Object)new Object[] { info }, "*", false));
263                        }
264                }
265        }

269
270                public ServiceListModel(String portletId) {
271                        super(portletId);
272                }

274        }
