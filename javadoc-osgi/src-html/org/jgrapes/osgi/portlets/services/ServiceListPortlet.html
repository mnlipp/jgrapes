071/**
072 * A portlet for inspecting the services in an OSGi runtime.
073 */
074public class ServiceListPortlet extends FreeMarkerPortlet
075        implements ServiceListener {
076
077        private ServiceComponentRuntime scr;
078        private static final Set<RenderMode> MODES = RenderMode.asSet(
079                        DeleteablePreview, View);
080        private BundleContext context;
081        
082        /**
083         * Creates a new component with its channel set to the given 
084         * channel.
085         * 
086         * @param componentChannel the channel that the component's 
087         * handlers listen on by default and that 
088         * {@link Manager#fire(Event, Channel...)} sends the event to 
089         */
090        public ServiceListPortlet(Channel componentChannel, BundleContext context,
091                        ServiceComponentRuntime scr) {
092                super(componentChannel, true);
093                this.context = context;
094                this.scr = scr;
095                context.addServiceListener(this);
096        }
097        
098        @Handler
099        public void onPortalReady(PortalReady event, PortalSession channel) 
100                        throws TemplateNotFoundException, MalformedTemplateNameException, 
101                        ParseException, IOException {
102                ResourceBundle resourceBundle = resourceBundle(channel.locale());
103                // Add portlet resources to page
104                channel.respond(new AddPortletType(type())
105                                .setDisplayName(resourceBundle.getString("portletName"))
106                                .addScript(new ScriptResource()
107                                                .setRequires(new String[] {"datatables.net"})
108                                                .setScriptUri(event.renderSupport().portletResource(
109                                                                type(), "Services-functions.ftl.js")))
110                                .addCss(event.renderSupport(), PortalWeblet.uriFromPath("Services-style.css"))
111                                .setInstantiable());
112        }
113        
114        /* (non-Javadoc)
115         * @see org.jgrapes.portal.AbstractPortlet#generatePortletId()
116         */
117        @Override
118        protected String generatePortletId() {
119                return type() + "-" + super.generatePortletId();
120        }
121
122        /* (non-Javadoc)
123         * @see org.jgrapes.portal.AbstractPortlet#modelFromSession
124         */
125        @SuppressWarnings("unchecked")
126        @Override
127        protected <T extends Serializable> Optional<T> stateFromSession(
128                        Session session, String portletId, Class<T> type) {
129                if (portletId.startsWith(type() + "-")) {
130                        return Optional.of((T)new ServiceListModel(portletId));
131                }
132                return Optional.empty();
133        }
134
135        /* (non-Javadoc)
136         * @see org.jgrapes.portal.AbstractPortlet#doAddPortlet
137         */
138        protected String doAddPortlet(AddPortletRequest event, PortalSession channel)
139                        throws Exception {
140                ServiceListModel portletModel = new ServiceListModel(generatePortletId());
141                renderPortlet(event, channel, portletModel);    
142                return portletModel.getPortletId();
143        }
144
145        /* (non-Javadoc)
146         * @see org.jgrapes.portal.AbstractPortlet#doRenderPortlet
147         */
148        @Override
149        protected void doRenderPortlet(RenderPortletRequest event,
150                PortalSession channel, String portletId, Serializable retrievedState)
151                throws Exception {
152                ServiceListModel portletModel = (ServiceListModel)retrievedState;
153                renderPortlet(event, channel, portletModel);    
154        }
155
156        private void renderPortlet(RenderPortletRequestBase<?> event, PortalSession channel, 
157                        ServiceListModel portletModel) throws TemplateNotFoundException, 
158                        MalformedTemplateNameException, ParseException, IOException, InvalidSyntaxException {
159                switch (event.renderMode()) {
160                case Preview:
161                case DeleteablePreview: {
162                        Template tpl = freemarkerConfig().getTemplate("Services-preview.ftl.html");
163                        channel.respond(new RenderPortletFromTemplate(event,
164                                        ServiceListPortlet.class, portletModel.getPortletId(), 
165                                        tpl, fmModel(event, channel, portletModel))
166                                        .setRenderMode(DeleteablePreview).setSupportedModes(MODES)
167                                        .setForeground(event.isForeground()));
168                        List<Map<String,Object>> serviceInfos = Arrays.stream(
169                                        context.getAllServiceReferences(null, null))
170                                        .map(s -> createServiceInfo(s, channel.locale())).collect(Collectors.toList());
171                        channel.respond(new NotifyPortletView(type(),
172                                        portletModel.getPortletId(), "serviceUpdates", serviceInfos, "preview", true));
173                        break;
174                }
175                case View: {
176                        Template tpl = freemarkerConfig().getTemplate("Services-view.ftl.html");
177                        channel.respond(new RenderPortletFromTemplate(event,
178                                        ServiceListPortlet.class, portletModel.getPortletId(), 
179                                        tpl, fmModel(event, channel, portletModel))
180                                        .setSupportedModes(MODES).setForeground(event.isForeground()));
181                        List<Map<String,Object>> serviceInfos = Arrays.stream(
182                                        context.getAllServiceReferences(null, null))
183                                        .map(s -> createServiceInfo(s, channel.locale())).collect(Collectors.toList());
184                        channel.respond(new NotifyPortletView(type(),
185                                        portletModel.getPortletId(), "serviceUpdates", serviceInfos, "view", true));
186                        break;
187                }
188                default:
189                        break;
190                }
191        }

193        private Map<String,Object> createServiceInfo(ServiceReference<?> serviceRef, Locale locale) {
194                Map<String, Object> result = new HashMap<>();
195                result.put("id", serviceRef.getProperty(Constants.SERVICE_ID).toString());
196                String[] interfaces = (String[])serviceRef.getProperty(Constants.OBJECTCLASS);
197                result.put("type", String.join(", ", interfaces));
198                Long bundleId = (Long)serviceRef.getProperty(Constants.SERVICE_BUNDLEID);
199                result.put("bundleId", bundleId.toString());
200                Bundle bundle = context.getBundle(bundleId);
201                if (bundle == null) {
202                        result.put("bundleName", "");
203                } else {
204                        result.put("bundleName", Optional.ofNullable(bundle.getHeaders(locale.toString())
205                                        .get("Bundle-Name")).orElse(bundle.getSymbolicName()));
206                }
207                String scope = "";
208                switch((String)serviceRef.getProperty(Constants.SERVICE_SCOPE)) {
209                case Constants.SCOPE_BUNDLE:
210                        scope = "serviceScopeBundle";
211                        break;
212                case Constants.SCOPE_PROTOTYPE:
213                        scope = "serviceScopePrototype";
214                        break;
215                case Constants.SCOPE_SINGLETON:
216                        scope = "serviceScopeSingleton";
217                        break;
218                }
219                result.put("scope", scope);
220                Integer ranking = (Integer)serviceRef.getProperty(Constants.SERVICE_RANKING);
221                result.put("ranking", ranking == null ? "" : ranking.toString());
222                String componentName = (String)serviceRef.getProperty("component.name");
223                if (componentName != null && bundle != null) {
224                        ComponentDescriptionDTO dto = scr.getComponentDescriptionDTO(bundle, componentName);
225                        result.put("dsScope", "serviceScope" 
226                                        + dto.scope.substring(0, 1).toUpperCase() + dto.scope.substring(1));
227                        result.put("implementationClass", dto.implementationClass);
228                } else {
229                        Object service = context.getService(serviceRef);
230                        if (service != null) {
231                                result.put("implementationClass", service.getClass().getName());
232                                context.ungetService(serviceRef);
233                        } else {
234                                result.put("implementationClass", "");
235                        }
236                }
237                return result;
238        }

242         */
243        @Override
244        protected void doDeletePortlet(DeletePortletRequest event, PortalSession channel, 
245                        String portletId, Serializable portletState) throws Exception {
246                channel.respond(new DeletePortlet(portletId));
247        }

249        /**
250         * Translates the OSGi {@link ServiceEvent} to a JGrapes event and fires it
251         * on all known portal session channels.
252         *
253         * @param event the event
254         */
255        @Override
256        public void serviceChanged(ServiceEvent event) {
257                fire(new ServiceChanged(event), trackedSessions());
258        }
259        
260        /**
261         * Handles a {@link ServiceChanged} event by updating the information in the portal
262         * sessions.
263         *
264         * @param event the event
265         */
266        @Handler
267        public void onServiceChanged(ServiceChanged event, PortalSession portalSession) {
268                Map<String,Object> info = createServiceInfo(
269                                event.serviceEvent().getServiceReference(), portalSession.locale());
270                if (event.serviceEvent().getType() == ServiceEvent.UNREGISTERING) {
271                        info.put("updateType", "unregistering");
272                }
273                for (String portletId: portletIds(portalSession)) {
274                        portalSession.respond(new NotifyPortletView(
275                                        type(), portletId, "serviceUpdates", 
276                                        (Object)new Object[] { info }, "*", false));
277                }
278        }
279
280        /**
281         * Wraps an OSGi {@link ServiceEvent}.
282         */
283        public static class ServiceChanged extends Event<Void> {
284                private ServiceEvent serviceEvent;
285
286                public ServiceChanged(ServiceEvent serviceEvent) {
287                        this.serviceEvent = serviceEvent;
288                }
289
290                public ServiceEvent serviceEvent() {
291                        return serviceEvent;
292                }
293        }

297
298                public ServiceListModel(String portletId) {
299                        super(portletId);
300                }
