019package org.jgrapes.osgi.portal;
020
021import java.util.HashMap;
022import java.util.Map;
023import java.util.stream.StreamSupport;
024
025import org.jgrapes.core.Channel;
026import org.jgrapes.core.Component;
027import org.jgrapes.core.ComponentType;
028import org.jgrapes.core.Components;
029import org.jgrapes.osgi.factory.portal.provider.PageResourceProviderFactory;
030import org.jgrapes.osgi.factory.portlet.PortletFactory;
031import org.osgi.framework.BundleContext;
032import org.osgi.framework.ServiceReference;
033import org.osgi.util.tracker.ServiceTracker;
034import org.osgi.util.tracker.ServiceTrackerCustomizer;
035
036/**
037 * 
038 */
039public class PortalComponentsCollector extends Component {
040
041        private ServiceTracker<PageResourceProviderFactory<?>, 
042                PageResourceProviderFactory<?>> providerTracker;
043        private ServiceTracker<PortletFactory<?>, PortletFactory<?>> portletTracker;
044        
045        /**
046         * @param componentChannel
047         */
048        public PortalComponentsCollector(Channel componentChannel, BundleContext context) {
049                super(componentChannel);
050                
051                // Tracker for page resource providers
052                @SuppressWarnings("unchecked")
053                Class<PageResourceProviderFactory<?>> cls 
054                        = (Class<PageResourceProviderFactory<?>>)(Class<?>)PageResourceProviderFactory.class;
055                providerTracker = new ServiceTracker<>(context, cls,
056                                new ServiceTrackerCustomizer<PageResourceProviderFactory<?>, PageResourceProviderFactory<?>> () {
057                        /* (non-Javadoc)
058                         * @see org.osgi.util.tracker.ServiceTrackerCustomizer#addingService(org.osgi.framework.ServiceReference)
059                         */
060                        @Override
061                        public PageResourceProviderFactory<?> addingService(
062                                ServiceReference<PageResourceProviderFactory<?>> reference) {
063                                PageResourceProviderFactory<?> providerFactory = context.getService(reference);
064                                if (StreamSupport.stream(spliterator(), false)
065                                                .filter(c -> c.getClass()
066                                                                .equals(providerFactory.componentType()))
067                                                .count() == 0) {
068                                        Map<Object,Object> props = new HashMap<>();
069                                        props.put(BundleContext.class, context);
070                                        attach(providerFactory.create(channel(), props));
071                                }
072                                return providerFactory;
073                        }
074
075                        /* (non-Javadoc)
076                         * @see org.osgi.util.tracker.ServiceTrackerCustomizer#modifiedService(org.osgi.framework.ServiceReference, java.lang.Object)
077                         */
078                        @Override
079                        public void modifiedService(ServiceReference<PageResourceProviderFactory<?>> reference,
080                                        PageResourceProviderFactory<?> service) {
081                        }
082
083                        /* (non-Javadoc)
084                         * @see org.osgi.util.tracker.ServiceTrackerCustomizer#removedService(org.osgi.framework.ServiceReference, java.lang.Object)
085                         */
086                        @Override
087                        public void removedService(ServiceReference<PageResourceProviderFactory<?>> reference,
088                                        PageResourceProviderFactory<?> service) {
089                                for (ComponentType child: PortalComponentsCollector.this) {
090                                        if (child.getClass().equals(service.componentType())) {
091                                                Components.manager(child).detach();
092                                        }
093                                }
094                        }
095                        
096                });
097                providerTracker.open();
098                
099                // Tracker for portlets
100                @SuppressWarnings("unchecked")
101                Class<PortletFactory<?>> portletFactoryCls 
102                        = (Class<PortletFactory<?>>)(Class<?>)PortletFactory.class;
103                portletTracker = new ServiceTracker<>(context, portletFactoryCls, 
104                                new ServiceTrackerCustomizer<PortletFactory<?>, PortletFactory<?>> () {
105                        
106                        /* (non-Javadoc)
107                         * @see org.osgi.util.tracker.ServiceTrackerCustomizer#addingService(org.osgi.framework.ServiceReference)
108                         */
109                        @Override
110                        public PortletFactory<?> addingService(
111                                ServiceReference<PortletFactory<?>> reference) {
112                                PortletFactory<?> portletFactory = context.getService(reference);
113                                if (StreamSupport.stream(spliterator(), false)
114                                                .filter(c -> c.getClass()
115                                                                .equals(portletFactory.componentType()))
116                                                .count() == 0) {
117                                        Map<Object,Object> props = new HashMap<>();
118                                        props.put(BundleContext.class, context);
119                                        attach(portletFactory.create(channel(), props));
120                                }
121                                return portletFactory;
122                        }
123
124                        /* (non-Javadoc)
125                         * @see org.osgi.util.tracker.ServiceTrackerCustomizer#modifiedService(org.osgi.framework.ServiceReference, java.lang.Object)
126                         */
127                        @Override
128                        public void modifiedService(ServiceReference<PortletFactory<?>> reference,
129                                PortletFactory<?> service) {
130                        }
131
132                        /* (non-Javadoc)
133                         * @see org.osgi.util.tracker.ServiceTrackerCustomizer#removedService(org.osgi.framework.ServiceReference, java.lang.Object)
134                         */
135                        @Override
136                        public void removedService(ServiceReference<PortletFactory<?>> reference,
137                                PortletFactory<?> service) {
138                                for (ComponentType child: PortalComponentsCollector.this) {
139                                        if (child.getClass().equals(service.componentType())) {
140                                                Components.manager(child).detach();
141                                        }
142                                }
143                        }
144                        
145                });
146                portletTracker.open();
147        }
148
149
150}
