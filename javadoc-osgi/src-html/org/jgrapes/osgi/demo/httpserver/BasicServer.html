019package org.jgrapes.osgi.demo.httpserver;
020
021import java.io.InputStream;
022import java.net.InetSocketAddress;
023import java.security.KeyStore;
024import java.security.SecureRandom;
025
026import javax.net.ssl.KeyManagerFactory;
027import javax.net.ssl.SSLContext;
028
029import org.jgrapes.core.Channel;
030import org.jgrapes.core.Component;
031import org.jgrapes.core.Components;
032import org.jgrapes.core.NamedChannel;
033import org.jgrapes.core.events.Stop;
034import org.jgrapes.http.HttpRequestHandlerFactory;
035import org.jgrapes.http.HttpServer;
036import org.jgrapes.http.InMemorySessionManager;
037import org.jgrapes.http.LanguageSelector;
038import org.jgrapes.http.events.GetRequest;
039import org.jgrapes.http.events.PostRequest;
040import org.jgrapes.io.NioDispatcher;
041import org.jgrapes.io.util.PermitsPool;
042import org.jgrapes.net.SslServer;
043import org.jgrapes.net.TcpServer;
044import org.jgrapes.osgi.core.ComponentCollector;
045import org.osgi.framework.BundleActivator;
046import org.osgi.framework.BundleContext;
047
048/**
049 * A basic server.
050 */
051public class BasicServer extends Component implements BundleActivator {
052
053    private static BundleContext context;
054    private BasicServer app;
055
056    /**
057     * Returns the context passed to the application.
058     *
059     * @return the bundle context
060     */
061    public static BundleContext context() {
062        return context;
063    }
064
065    /*
066     * (non-Javadoc)
067     * 
068     * @see org.osgi.framework.BundleActivator#start(org.osgi.framework.
069     * BundleContext)
070     */
071    @Override
072    public void start(BundleContext context) throws Exception {
073        BasicServer.context = context;
074        // The demo component is the application
075        app = new BasicServer();
076        // Attach a general nio dispatcher
077        app.attach(new NioDispatcher());
078
079        // Create TLS "converter"
080        KeyStore serverStore = KeyStore.getInstance("JKS");
081        try (InputStream storeData
082            = BasicServer.class.getResourceAsStream("/localhost.jks")) {
083            serverStore.load(storeData, "nopass".toCharArray());
084        }
085        KeyManagerFactory kmf = KeyManagerFactory.getInstance(
086            KeyManagerFactory.getDefaultAlgorithm());
087        kmf.init(serverStore, "nopass".toCharArray());
088        SSLContext sslContext = SSLContext.getInstance("TLS");
089        sslContext.init(kmf.getKeyManagers(), null, new SecureRandom());
090        // Create a TCP server for SSL
091        Channel securedNetwork = app.attach(
092            new TcpServer().setServerAddress(new InetSocketAddress(5443))
093                .setBacklog(3000).setConnectionLimiter(new PermitsPool(50)));
094        // Network level unencrypted channel.
095        Channel httpTransport = new NamedChannel("httpTransport");
096        app.attach(new SslServer(httpTransport, securedNetwork, sslContext));
097
098        // Create an HTTP server as converter between transport and application
099        // layer.
100        app.attach(new HttpServer(app,
101            httpTransport, GetRequest.class, PostRequest.class));
102
103        // Build application layer
104        app.attach(new InMemorySessionManager(app.channel()));
105        app.attach(new LanguageSelector(app.channel()));
106        app.attach(new ComponentCollector<>(
107            app.channel(), context, HttpRequestHandlerFactory.class));
108        Components.start(app);
109    }
110
111    /*
112     * (non-Javadoc)
113     * 
114     * @see
115     * org.osgi.framework.BundleActivator#stop(org.osgi.framework.BundleContext)
116     */
117    @Override
118    public void stop(BundleContext context) throws Exception {
119        app.fire(new Stop(), Channel.BROADCAST);
120        Components.awaitExhaustion();
121    }
122
123}
