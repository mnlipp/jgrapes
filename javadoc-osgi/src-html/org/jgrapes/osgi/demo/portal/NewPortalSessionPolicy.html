019package org.jgrapes.osgi.demo.portal;
020
021import java.util.Optional;
022
023import org.jgrapes.core.Channel;
024import org.jgrapes.core.Component;
025import org.jgrapes.core.annotation.Handler;
026import org.jgrapes.http.Session;
027import org.jgrapes.io.IOSubchannel;
028import org.jgrapes.portal.events.PortalConfigured;
029import org.jgrapes.portal.events.RenderPortlet;
030
031/**
032 * 
033 */
034public class NewPortalSessionPolicy extends Component {
035
036        private final String renderedFlagName = getClass().getName() + ".rendered";
037        
038        /**
039         * Creates a new component with its channel set to
040         * itself.
041         */
042        public NewPortalSessionPolicy() {
043        }
044
045        /**
046         * Creates a new component with its channel set to the given channel.
047         * 
048         * @param componentChannel
049         */
050        public NewPortalSessionPolicy(Channel componentChannel) {
051                super(componentChannel);
052        }
053
054        @Handler
055        public void onRenderPortlet(RenderPortlet event, IOSubchannel channel) {
056                channel.associated(Session.class).ifPresent(session -> 
057                        session.put(renderedFlagName, true));
058        }
059        
060        @Handler
061        public void onPortalConfigured(PortalConfigured event, IOSubchannel channel) 
062                        throws InterruptedException {
063                Optional<Session> optSession = channel.associated(Session.class);
064                if (optSession.isPresent()) {
065                        if ((Boolean)optSession.get().getOrDefault(
066                                        renderedFlagName, false)) {
067                                return;
068                        }
069//                      fire(new AddPortletRequest(event.event().event().renderSupport(), 
070//                                      HelloWorldPortlet.class.getName(), 
071//                                      Portlet.RenderMode.DeleteablePreview), channel);
072                };
073        }
074
075}
