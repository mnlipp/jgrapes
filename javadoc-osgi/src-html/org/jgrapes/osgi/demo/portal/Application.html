019package org.jgrapes.osgi.demo.portal;
020
021import java.io.InputStream;
022import java.net.InetSocketAddress;
023import java.net.URI;
024import java.security.KeyStore;
025import java.security.SecureRandom;
026import java.util.ResourceBundle;
027
028import javax.net.ssl.KeyManagerFactory;
029import javax.net.ssl.SSLContext;
030
031import org.jgrapes.core.Channel;
032import org.jgrapes.core.Component;
033import org.jgrapes.core.Components;
034import org.jgrapes.core.NamedChannel;
035import org.jgrapes.core.events.Stop;
036import org.jgrapes.http.HttpServer;
037import org.jgrapes.http.InMemorySessionManager;
038import org.jgrapes.http.LanguageSelector;
039import org.jgrapes.http.events.GetRequest;
040import org.jgrapes.http.events.PostRequest;
041import org.jgrapes.io.FileStorage;
042import org.jgrapes.io.NioDispatcher;
043import org.jgrapes.io.util.PermitsPool;
044import org.jgrapes.net.SslServer;
045import org.jgrapes.net.TcpServer;
046import org.jgrapes.osgi.core.ComponentCollector;
047import org.jgrapes.portal.KVStoreBasedPortalPolicy;
048import org.jgrapes.portal.PageResourceProviderFactory;
049import org.jgrapes.portal.Portal;
050import org.jgrapes.portal.PortalLocalBackedKVStore;
051import org.jgrapes.portal.PortletComponentFactory;
052import org.osgi.framework.BundleActivator;
053import org.osgi.framework.BundleContext;
054
055/**
056 *
057 */
058public class Application extends Component implements BundleActivator {
059
060        private static BundleContext context;
061        private Application app;
062        
063        public static BundleContext context() {
064                return context;
065        }
066        
067        /* (non-Javadoc)
068         * @see org.osgi.framework.BundleActivator#start(org.osgi.framework.BundleContext)
069         */
070        @Override
071        public void start(BundleContext context) throws Exception {
072                Application.context = context;
073                // The demo component is the application
074                app = new Application();
075                // Attach a general nio dispatcher
076                app.attach(new NioDispatcher());
077
078                // Create TLS "converter"
079                KeyStore serverStore = KeyStore.getInstance("JKS");
080                try (InputStream kf 
081                                = Application.class.getResourceAsStream("/localhost.jks")) {
082                        serverStore.load(kf, "nopass".toCharArray());
083                }
084                KeyManagerFactory kmf = KeyManagerFactory.getInstance(
085                                KeyManagerFactory.getDefaultAlgorithm());
086        kmf.init(serverStore, "nopass".toCharArray());
087                SSLContext sslContext = SSLContext.getInstance("TLS");
088                sslContext.init(kmf.getKeyManagers(), null, new SecureRandom());
089                // Create a TCP server for SSL
090                Channel securedNetwork = app.attach(
091                                new TcpServer().setServerAddress(new InetSocketAddress(6443))
092                                .setBacklog(3000).setConnectionLimiter(new PermitsPool(50)));
093                // Network level unencrypted channel.
094                Channel httpTransport = new NamedChannel("httpTransport");
095                app.attach(new SslServer(httpTransport, securedNetwork, sslContext));
096
097                // Create an HTTP server as converter between transport and application
098                // layer.
099                app.attach(new HttpServer(app, 
100                        httpTransport, GetRequest.class, PostRequest.class));
101                
102                // Build application layer
103                app.attach(new InMemorySessionManager(app.channel()));
104                app.attach(new LanguageSelector(app.channel()));
105                app.attach(new FileStorage(app.channel(), 65536));
106                Portal portal = app.attach(new Portal(Channel.SELF, app.channel(), 
107                                new URI("/portal/")))
108                                .setResourceBundleSupplier(l -> ResourceBundle.getBundle(
109                                        getClass().getPackage().getName() + ".portal-l10n", l,
110                                        ResourceBundle.Control.getNoFallbackControl(
111                                                        ResourceBundle.Control.FORMAT_DEFAULT)))
112                                .setFallbackResourceSupplier((themeProvider, resource) -> {
113                                        return Application.class.getResourceAsStream(resource);
114                                });
115                portal.attach(new PortalLocalBackedKVStore(
116                                portal, portal.prefix().getPath()));
117                portal.attach(new KVStoreBasedPortalPolicy(portal));
118                portal.attach(new NewPortalSessionPolicy(portal));
119                portal.attach(new ComponentCollector<>(
120                                portal, context, PageResourceProviderFactory.class));
121                portal.attach(new ComponentCollector<>(
122                                portal, context, PortletComponentFactory.class));
123                Components.start(app);
124        }
125
126        /* (non-Javadoc)
127         * @see org.osgi.framework.BundleActivator#stop(org.osgi.framework.BundleContext)
128         */
129        @Override
130        public void stop(BundleContext context) throws Exception {
131                app.fire(new Stop(), Channel.BROADCAST);
132                Components.awaitExhaustion();
133        }
134}
