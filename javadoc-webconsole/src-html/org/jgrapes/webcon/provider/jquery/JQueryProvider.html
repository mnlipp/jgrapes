019package org.jgrapes.webcon.provider.jquery;
020
021import freemarker.core.ParseException;
022import freemarker.template.MalformedTemplateNameException;
023import freemarker.template.TemplateNotFoundException;
024
025import java.io.IOException;
026
027import org.jgrapes.core.Channel;
028import org.jgrapes.core.Event;
029import org.jgrapes.core.Manager;
030import org.jgrapes.core.annotation.Handler;
031import org.jgrapes.webcon.base.ConsoleSession;
032import org.jgrapes.webcon.base.PageResourceProvider;
033import org.jgrapes.webcon.base.events.AddPageResources;
034import org.jgrapes.webcon.base.events.AddPageResources.ScriptResource;
035import org.jgrapes.webcon.base.events.ConsoleReady;
036
037/**
038 * Provider for the [JQuery](https://jquery.com/) library.
039 */
040public class JQueryProvider extends PageResourceProvider {
041
042    /**
043     * Creates a new component with its channel set to the given 
044     * channel.
045     * 
046     * @param componentChannel the channel that the component's 
047     * handlers listen on by default and that 
048     * {@link Manager#fire(Event, Channel...)} sends the event to 
049     */
050    public JQueryProvider(Channel componentChannel) {
051        super(componentChannel);
052    }
053
054    /**
055     * On {@link ConsoleReady}, fire the appropriate {@link AddPageResources}.
056     *
057     * @param event the event
058     * @param portalSession the portal session
059     * @throws TemplateNotFoundException the template not found exception
060     * @throws MalformedTemplateNameException the malformed template name exception
061     * @throws ParseException the parse exception
062     * @throws IOException Signals that an I/O exception has occurred.
063     */
064    @Handler(priority = 100)
065    public void onPortalReady(ConsoleReady event, ConsoleSession portalSession)
066            throws TemplateNotFoundException, MalformedTemplateNameException,
067            ParseException, IOException {
068        String minExt = event.renderSupport()
069            .useMinifiedResources() ? ".min" : "";
070        portalSession.respond(new AddPageResources()
071            .addScriptResource(new ScriptResource()
072                .setProvides(new String[] { "jquery" })
073                .setScriptUri(event.renderSupport().pageResource(
074                    "jquery/jquery" + minExt + ".js"))));
075    }
076}
