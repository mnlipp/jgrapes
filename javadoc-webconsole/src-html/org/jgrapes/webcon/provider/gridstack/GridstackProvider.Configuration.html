019package org.jgrapes.webcon.provider.gridstack;
020
021import freemarker.core.ParseException;
022import freemarker.template.MalformedTemplateNameException;
023import freemarker.template.TemplateNotFoundException;
024
025import java.io.IOException;
026import java.util.Map;
027import java.util.logging.Logger;
028
029import org.jgrapes.core.Channel;
030import org.jgrapes.core.Event;
031import org.jgrapes.core.Manager;
032import org.jgrapes.core.annotation.Handler;
033import org.jgrapes.webcon.base.PageResourceProvider;
034import org.jgrapes.webcon.base.PortalSession;
035import org.jgrapes.webcon.base.events.AddPageResources;
036import org.jgrapes.webcon.base.events.PortalReady;
037import org.jgrapes.webcon.base.events.AddPageResources.ScriptResource;
038
039/**
040 * Provider for the [Gridstack.js](http://gridstackjs.com/) library.
041 */
042public class GridstackProvider extends PageResourceProvider {
043
044    private static Logger LOG
045        = Logger.getLogger(GridstackProvider.class.getName());
046
047    public enum Configuration {
048        CoreOnly, CoreWithJQueryUI, All
049    }
050
051    private Configuration configuration = Configuration.All;
052
053    /**
054     * Creates a new component with its channel set to the given 
055     * channel.
056     *
057     * @param componentChannel the channel that the component's
058     * handlers listen on by default and that 
059     * {@link Manager#fire(Event, Channel...)} sends the event to
060     * @param properties the properties used to configure the component
061     */
062    public GridstackProvider(Channel componentChannel,
063            Map<Object, Object> properties) {
064        super(componentChannel);
065        if (properties.containsKey("configuration")) {
066            String config = (String) properties.get("configuration");
067            try {
068                configuration = Configuration.valueOf(config);
069            } catch (IllegalArgumentException e) {
070                LOG.severe("Illegal value for \"configuration\": " + config);
071            }
072        }
073    }
074
075    /**
076     * On {@link PortalReady}, fire the appropriate {@link AddPageResources}.
077     *
078     * @param event the event
079     * @param portalSession the portal session
080     * @throws TemplateNotFoundException the template not found exception
081     * @throws MalformedTemplateNameException the malformed template name exception
082     * @throws ParseException the parse exception
083     * @throws IOException Signals that an I/O exception has occurred.
084     */
085    @Handler(priority = 100)
086    public void onPortalReady(PortalReady event, PortalSession portalSession)
087            throws TemplateNotFoundException, MalformedTemplateNameException,
088            ParseException, IOException {
089        String minExt = event.renderSupport()
090            .useMinifiedResources() ? ".min" : "";
091        AddPageResources addRequest = new AddPageResources()
092            .addCss(event.renderSupport().pageResource(
093                "gridstack/gridstack" + minExt + ".css"));
094        switch (configuration) {
095        case CoreOnly:
096            addRequest
097                .addScriptResource(new ScriptResource()
098                    .setRequires("jquery")
099                    .setProvides("gridstack")
100                    .setScriptUri(event.renderSupport().pageResource(
101                        "gridstack/gridstack" + minExt + ".js")));
102            break;
103        case CoreWithJQueryUI:
104            addRequest.addScriptResource(new ScriptResource()
105                .setRequires("jquery", "jquery-ui")
106                .setProvides("gridstack")
107                .setScriptUri(event.renderSupport().pageResource(
108                    "gridstack/gridstack" + minExt + ".js")));
109            addRequest.addScriptResource(new ScriptResource()
110                .setProvides("gridstack.jQueryUI")
111                .setScriptUri(event.renderSupport().pageResource(
112                    "gridstack/gridstack.jQueryUI" + minExt + ".js")));
113            break;
114        case All:
115            addRequest.addScriptResource(new ScriptResource()
116                .setRequires("jquery")
117                .setProvides("gridstack", "gridstack.all")
118                .setScriptUri(event.renderSupport().pageResource(
119                    "gridstack/gridstack.all.js")));
120        }
121        portalSession.respond(addRequest);
122    }
123}
