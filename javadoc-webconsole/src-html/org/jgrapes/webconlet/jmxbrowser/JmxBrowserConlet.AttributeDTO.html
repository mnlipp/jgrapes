100    @Handler
101    public void onConsoleReady(ConsoleReady event, ConsoleSession channel)
102            throws TemplateNotFoundException, MalformedTemplateNameException,
103            ParseException, IOException {
104        // Add conlet resources to page
105        channel.respond(new AddConletType(type())
106            .setDisplayNames(
107                localizations(channel.supportedLocales(), "conletName"))
108            .addScript(new ScriptResource()
109                .setScriptUri(event.renderSupport().conletResource(
110                    type(), "jmxbrowser.min.js"))
111                .setScriptType("module")));
112    }

146    private Set<RenderMode> renderConlet(RenderConletRequestBase<?> event,
147            ConsoleSession channel, JmxBrowserModel conletModel)
148            throws TemplateNotFoundException,
149            MalformedTemplateNameException, ParseException, IOException {
150        Set<RenderMode> renderedAs = new HashSet<>();
151        if (event.renderAs().contains(RenderMode.Preview)) {
152            Template tpl = freemarkerConfig()
153                .getTemplate("JmxBrowser-preview.ftl.html");
154            channel.respond(new RenderConletFromTemplate(event,
155                type(), conletModel.getConletId(), tpl,
156                fmModel(event, channel, conletModel))
157                    .setRenderAs(
158                        RenderMode.Preview.addModifiers(event.renderAs()))
159                    .setSupportedModes(MODES));
160            channel.respond(new NotifyConletView(type(),
161                conletModel.getConletId(), "mbeansTree", genMBeansTree(),
162                "preview", true));
163            renderedAs.add(RenderMode.Preview);
164        }
165        if (event.renderAs().contains(RenderMode.View)) {
166            Template tpl
167                = freemarkerConfig().getTemplate("JmxBrowser-view.ftl.html");
168            channel.respond(new RenderConletFromTemplate(event,
169                type(), conletModel.getConletId(), tpl,
170                fmModel(event, channel, conletModel))
171                    .setRenderAs(
172                        RenderMode.View.addModifiers(event.renderAs()))
173                    .setSupportedModes(MODES));
174            channel.respond(new NotifyConletView(type(),
175                conletModel.getConletId(), "mbeansTree", genMBeansTree(),
176                "view", true));
177            renderedAs.add(RenderMode.View);
178        }
179        return renderedAs;
180    }
181
182    @Override
183    protected void doNotifyConletModel(NotifyConletModel event,
184            ConsoleSession channel, JmxBrowserModel conletModel)
185            throws Exception {
186        event.stop();
187        if (event.method().equals("sendMBean")) {
188            JsonArray segments = (JsonArray) event.params().get(0);
189            String domain = segments.asString(0);
190            Hashtable<String, String> props = new Hashtable<>();
191            for (int i = 1; i < segments.size(); i++) {
192                String[] keyProp = segments.asString(i).split("=", 2);
193                props.put(keyProp[0], keyProp[1]);
194            }
195            Set<ObjectName> mbeanNames
196                = mbeanServer.queryNames(new ObjectName(domain, props), null);
197            if (mbeanNames.size() == 0) {
198                return;
199            }
200            ObjectName mbeanName = mbeanNames.iterator().next();
201            MBeanInfo info = mbeanServer.getMBeanInfo(mbeanName);
202            channel.respond(new NotifyConletView(type(),
203                conletModel.getConletId(), "mbeanDetails",
204                new Object[] { genAttributesInfo(mbeanName, info), null }));
205        }
206    }

208    public static class NodeDTO {
209        public String segment;
210        public String label;
211        public Set<NodeDTO> children;
212
213        public NodeDTO(String segment, String label, Set<NodeDTO> children) {
214            this.segment = segment;
215            this.label = label;
216            this.children = children;
217        }
218
219        public NodeDTO(String segment, String label) {
220            this(segment, label,
221                new TreeSet<>(Comparator.comparing((node) -> node.label)));
222        }
223
224        public String getSegment() {
225            return segment;
226        }
227
228        public String getLabel() {
229            return label;
230        }
231
232        public Set<NodeDTO> getChildren() {
233            return children;
234        }
235
236        @Override
237        public int hashCode() {
238            final int prime = 31;
239            int result = 1;
240            result
241                = prime * result + ((segment == null) ? 0 : segment.hashCode());
242            return result;
243        }
244
245        @Override
246        public boolean equals(Object obj) {
247            if (this == obj) {
248                return true;
249            }
250            if (obj == null) {
251                return false;
252            }
253            if (getClass() != obj.getClass()) {
254                return false;
255            }
256            NodeDTO other = (NodeDTO) obj;
257            if (segment == null) {
258                if (other.segment != null) {
259                    return false;
260                }
261            } else if (!segment.equals(other.segment)) {
262                return false;
263            }
264            return true;
265        }
266    }

268    private List<NodeDTO> genMBeansTree() {
269        Map<String, NodeDTO> trees = new TreeMap<>();
270        Set<ObjectName> mbeanNames = mbeanServer.queryNames(null, null);
271        for (ObjectName mbn : mbeanNames) {
272            NodeDTO domainGroup = trees.computeIfAbsent(mbn.getDomain(),
273                key -> new NodeDTO(mbn.getDomain(), mbn.getDomain()));
274            appendToGroup(domainGroup, "type",
275                new Hashtable<>(mbn.getKeyPropertyList()), mbn);
276        }
277        List<NodeDTO> roots = new ArrayList<>(trees.values());
278        return roots;
279    }
280
281    private void appendToGroup(NodeDTO parent, String property,
282            Hashtable<String, String> propsLeft, ObjectName mbn) {
283        if (!propsLeft.keySet().contains(property)) {
284            try {
285                String left = ObjectName.getInstance("tmp", propsLeft)
286                    .getCanonicalKeyPropertyListString();
287                parent.children.add(new NodeDTO(left, left, null));
288            } catch (MalformedObjectNameException e) {
289                // Shoudn't happen
290            }
291            return;
292        }
293        Set<NodeDTO> candidates = parent.children;
294        String partition = property + "=" + propsLeft.get(property);
295        NodeDTO match = candidates.stream()
296            .filter(n -> n.segment.equals(partition)).findFirst()
297            .orElseGet(() -> {
298                NodeDTO node = new NodeDTO(partition, propsLeft.get(property));
299                candidates.add(node);
300                return node;
301            });
302        propsLeft.remove(property);
303        appendToGroup(match, property.equals("type") ? "name" : "", propsLeft,
304            mbn);
305    }

351    private List<AttributeDTO> genAttributesInfo(ObjectName mbeanName,
352            MBeanInfo info) {
353        List<AttributeDTO> result = new ArrayList<>();
354        for (MBeanAttributeInfo attr : info.getAttributes()) {
355            try {
356                Object value
357                    = mbeanServer.getAttribute(mbeanName, attr.getName());
358                result.add(new AttributeDTO(attr.getName(), value,
359                    attr.isWritable()));
360            } catch (InstanceNotFoundException | RuntimeMBeanException
361                    | AttributeNotFoundException | ReflectionException
362                    | MBeanException | IllegalArgumentException e) {
363                // Ignore (shouldn't happen)
364            }
365        }
366        return result;
367    }
