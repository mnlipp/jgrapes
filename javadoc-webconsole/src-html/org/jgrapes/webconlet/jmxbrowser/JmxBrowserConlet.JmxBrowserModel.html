058    private static final Set<RenderMode> MODES = RenderMode.asSet(
059        RenderMode.Preview, RenderMode.View);

062        = ManagementFactory.getPlatformMBeanServer();

064    /**
065     * Creates a new component with its channel set to the given channel.
066     * 
067     * @param componentChannel the channel that the component's handlers listen
068     *            on by default and that {@link Manager#fire(Event, Channel...)}
069     *            sends the event to
070     */
071    public JmxBrowserConlet(Channel componentChannel) {
072        super(componentChannel);
073    }
074
075    /**
076     * On {@link ConsoleReady}, fire the {@link AddConletType}.
077     *
078     * @param event the event
079     * @param channel the channel
080     * @throws TemplateNotFoundException the template not found exception
081     * @throws MalformedTemplateNameException the malformed template name
082     *             exception
083     * @throws ParseException the parse exception
084     * @throws IOException Signals that an I/O exception has occurred.
085     */
086    @Handler
087    public void onConsoleReady(ConsoleReady event, ConsoleSession channel)
088            throws TemplateNotFoundException, MalformedTemplateNameException,
089            ParseException, IOException {
090        // Add conlet resources to page
091        channel.respond(new AddConletType(type())
092            .setDisplayNames(
093                localizations(channel.supportedLocales(), "conletName"))
094            .addScript(new ScriptResource()
095                .setScriptUri(event.renderSupport().conletResource(
096                    type(), "JmxBrowser-functions.ftl.js"))
097                .setScriptType("module"))
098            .addCss(event.renderSupport(),
099                WebConsoleUtils.uriFromPath("JmxBrowser-style.css")));
100    }
101
102    @Override
103    protected String generateConletId() {
104        return type() + "-" + super.generateConletId();
105    }
106
107    @Override
108    protected Optional<JmxBrowserModel> stateFromSession(
109            Session session, String conletId) {
110        if (conletId.startsWith(type() + "-")) {
111            return Optional.of(new JmxBrowserModel(conletId));
112        }
113        return Optional.empty();
114    }
115
116    @Override
117    protected ConletTrackingInfo doAddConlet(AddConletRequest event,
118            ConsoleSession channel)
119            throws Exception {
120        JmxBrowserModel conletModel
121            = new JmxBrowserModel(generateConletId());
122        return new ConletTrackingInfo(conletModel.getConletId())
123            .addModes(renderConlet(event, channel, conletModel));
124    }
125
126    @Override
127    protected Set<RenderMode> doRenderConlet(RenderConletRequest event,
128            ConsoleSession channel, String conletId,
129            JmxBrowserModel conletModel)
130            throws Exception {
131        return renderConlet(event, channel, conletModel);
132    }
133
134    private Set<RenderMode> renderConlet(RenderConletRequestBase<?> event,
135            ConsoleSession channel, JmxBrowserModel conletModel)
136            throws TemplateNotFoundException,
137            MalformedTemplateNameException, ParseException, IOException {
138        Set<RenderMode> renderedAs = new HashSet<>();
139        if (event.renderAs().contains(RenderMode.Preview)) {
140            Template tpl = freemarkerConfig()
141                .getTemplate("JmxBrowser-preview.ftl.html");
142            channel.respond(new RenderConletFromTemplate(event,
143                type(), conletModel.getConletId(), tpl,
144                fmModel(event, channel, conletModel))
145                    .setRenderAs(
146                        RenderMode.Preview.addModifiers(event.renderAs()))
147                    .setSupportedModes(MODES));
148            channel.respond(new NotifyConletView(type(),
149                conletModel.getConletId(), "mbeansTree", genMBeansTree(),
150                "preview", true));
151            renderedAs.add(RenderMode.Preview);
152        }
153        if (event.renderAs().contains(RenderMode.View)) {
154            Template tpl
155                = freemarkerConfig().getTemplate("JmxBrowser-view.ftl.html");
156            channel.respond(new RenderConletFromTemplate(event,
157                type(), conletModel.getConletId(), tpl,
158                fmModel(event, channel, conletModel))
159                    .setRenderAs(
160                        RenderMode.View.addModifiers(event.renderAs()))
161                    .setSupportedModes(MODES));
162            renderedAs.add(RenderMode.View);
163        }
164        return renderedAs;
165    }
166
167    public static class NodeDTO {
168        public String segment;
169        public String label;
170        List<NodeDTO> children;
171
172        public NodeDTO(String segment, String label, List<NodeDTO> children) {
173            this.segment = segment;
174            this.label = label;
175            this.children = children;
176        }
177
178        public String getSegment() {
179            return segment;
180        }
181
182        public String getLabel() {
183            return label;
184        }
185
186        public List<NodeDTO> getChildren() {
187            return children;
188        }
189    }
190
191    private List<NodeDTO> genMBeansTree() {
192        Map<String, NodeDTO> trees = new TreeMap<>();
193        Set<ObjectName> mbeanNames = mbeanServer.queryNames(null, null);
194        for (ObjectName mbn : mbeanNames) {
195            NodeDTO domain = trees.computeIfAbsent(mbn.getDomain(),
196                key -> new NodeDTO(mbn.getDomain(), mbn.getDomain(),
197                    new ArrayList<>()));
198            List<NodeDTO> types = domain.children;
199            String typeName
200                = Optional.ofNullable(mbn.getKeyProperty("type")).orElse("");
201            String typeProp = "type=" + typeName;
202            NodeDTO typeNode = types.stream()
203                .filter(n -> n.segment.equals(typeProp)).findFirst()
204                .orElseGet(() -> {
205                    NodeDTO node = new NodeDTO(typeProp, typeName, null);
206                    types.add(node);
207                    return node;
208                });
209        }
210        List<NodeDTO> roots = new ArrayList<>(trees.values());
211        roots.sort(Comparator.comparing(node -> node.label));
212        return roots;
213    }
