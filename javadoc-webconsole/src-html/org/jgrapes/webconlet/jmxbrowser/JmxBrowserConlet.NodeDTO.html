084     *
085     * @param event the event
086     * @param channel the channel
087     * @throws TemplateNotFoundException the template not found exception

093    @Handler
094    public void onConsoleReady(ConsoleReady event, ConsoleSession channel)
095            throws TemplateNotFoundException, MalformedTemplateNameException,
096            ParseException, IOException {
097        // Add conlet resources to page
098        channel.respond(new AddConletType(type())
099            .setDisplayNames(
100                localizations(channel.supportedLocales(), "conletName"))
101            .addScript(new ScriptResource()
102                .setScriptUri(event.renderSupport().conletResource(
103                    type(), "JmxBrowser-functions.ftl.js"))
104                .setScriptType("module"))
105            .addCss(event.renderSupport(),
106                WebConsoleUtils.uriFromPath("JmxBrowser-style.css")));
107    }

114    @Override
115    protected Optional<JmxBrowserModel> stateFromSession(
116            Session session, String conletId) {
117        if (conletId.startsWith(type() + "-")) {

119        }
120        return Optional.empty();

123    @Override
124    protected ConletTrackingInfo doAddConlet(AddConletRequest event,
125            ConsoleSession channel)
126            throws Exception {
127        JmxBrowserModel conletModel
128            = new JmxBrowserModel(generateConletId());
129        return new ConletTrackingInfo(conletModel.getConletId())
130            .addModes(renderConlet(event, channel, conletModel));
131    }

136            JmxBrowserModel conletModel)
137            throws Exception {
138        return renderConlet(event, channel, conletModel);
139    }

141    private Set<RenderMode> renderConlet(RenderConletRequestBase<?> event,
142            ConsoleSession channel, JmxBrowserModel conletModel)
143            throws TemplateNotFoundException,
144            MalformedTemplateNameException, ParseException, IOException {
145        Set<RenderMode> renderedAs = new HashSet<>();
146        if (event.renderAs().contains(RenderMode.Preview)) {
147            Template tpl = freemarkerConfig()
148                .getTemplate("JmxBrowser-preview.ftl.html");
149            channel.respond(new RenderConletFromTemplate(event,
150                type(), conletModel.getConletId(), tpl,
151                fmModel(event, channel, conletModel))
152                    .setRenderAs(
153                        RenderMode.Preview.addModifiers(event.renderAs()))
154                    .setSupportedModes(MODES));
155            channel.respond(new NotifyConletView(type(),
156                conletModel.getConletId(), "mbeansTree", genMBeansTree(),
157                "preview", true));
158            renderedAs.add(RenderMode.Preview);
159        }
160        if (event.renderAs().contains(RenderMode.View)) {
161            Template tpl
162                = freemarkerConfig().getTemplate("JmxBrowser-view.ftl.html");
163            channel.respond(new RenderConletFromTemplate(event,
164                type(), conletModel.getConletId(), tpl,
165                fmModel(event, channel, conletModel))
166                    .setRenderAs(
167                        RenderMode.View.addModifiers(event.renderAs()))
168                    .setSupportedModes(MODES));
169            channel.respond(new NotifyConletView(type(),
170                conletModel.getConletId(), "mbeansTree", genMBeansTree(),
171                "view", true));
172            renderedAs.add(RenderMode.View);
173        }
174        return renderedAs;
175    }

197    public static class NodeDTO {
198        public String segment;
199        public String label;
200        public Set<NodeDTO> children;
201
202        public NodeDTO(String segment, String label, Set<NodeDTO> children) {
203            this.segment = segment;
204            this.label = label;
205            this.children = children;
206        }
207
208        public NodeDTO(String segment, String label) {
209            this(segment, label,
210                new TreeSet<>(Comparator.comparing((node) -> node.label)));
211        }
212
213        public String getSegment() {
214            return segment;
215        }
216
217        public String getLabel() {
218            return label;
219        }
220
221        public Set<NodeDTO> getChildren() {
222            return children;
223        }
224
225        @Override
226        public int hashCode() {
227            final int prime = 31;
228            int result = 1;
229            result
230                = prime * result + ((segment == null) ? 0 : segment.hashCode());
231            return result;
232        }
233
234        @Override
235        public boolean equals(Object obj) {
236            if (this == obj) {
237                return true;
238            }
239            if (obj == null) {
240                return false;
241            }
242            if (getClass() != obj.getClass()) {
243                return false;
244            }
245            NodeDTO other = (NodeDTO) obj;
246            if (segment == null) {
247                if (other.segment != null) {
248                    return false;
249                }
250            } else if (!segment.equals(other.segment)) {
251                return false;
252            }
253            return true;
254        }
255    }
256
257    private List<NodeDTO> genMBeansTree() {
258        Map<String, NodeDTO> trees = new TreeMap<>();
259        Set<ObjectName> mbeanNames = mbeanServer.queryNames(null, null);
260        for (ObjectName mbn : mbeanNames) {
261            NodeDTO domainGroup = trees.computeIfAbsent(mbn.getDomain(),
262                key -> new NodeDTO(mbn.getDomain(), mbn.getDomain()));
263            appendToGroup(domainGroup, "type",
264                new Hashtable<>(mbn.getKeyPropertyList()), mbn);
265        }
266        List<NodeDTO> roots = new ArrayList<>(trees.values());
267        return roots;
268    }
269
270    private void appendToGroup(NodeDTO parent, String property,
271            Hashtable<String, String> propsLeft, ObjectName mbn) {
272        if (!propsLeft.keySet().contains(property)) {
273            try {
274                String left = ObjectName.getInstance("tmp", propsLeft)
275                    .getCanonicalKeyPropertyListString();
276                parent.children.add(new NodeDTO(left, left, null));
277            } catch (MalformedObjectNameException e) {
278                // Shoudn't happen
279            }
280            return;
281        }
282        Set<NodeDTO> candidates = parent.children;
283        String partition = property + "=" + propsLeft.get(property);
284        NodeDTO match = candidates.stream()
285            .filter(n -> n.segment.equals(partition)).findFirst()
286            .orElseGet(() -> {
287                NodeDTO node = new NodeDTO(partition, propsLeft.get(property));
288                candidates.add(node);
289                return node;
290            });
291        propsLeft.remove(property);
292        appendToGroup(match, property.equals("type") ? "name" : "", propsLeft,
293            mbn);
294    }
