019package org.jgrapes.webconlet.sysinfo;
020
021import java.util.Map;
022import java.util.Optional;
023import org.jgrapes.core.Channel;
024import org.jgrapes.core.ComponentType;
025import org.jgrapes.webconsole.base.ConletComponentFactory;
026
027/**
028 * A factory for creating {@link SysInfoConlet}s.
029 */
030public class SysInfoConletFactory implements ConletComponentFactory {
031
032    /*
033     * (non-Javadoc)
034     * 
035     * @see org.jgrapes.core.ComponentFactory#componentType()
036     */
037    @Override
038    public Class<? extends ComponentType> componentType() {
039        return SysInfoConlet.class;
040    }
041
042    /*
043     * (non-Javadoc)
044     * 
045     * @see org.jgrapes.core.ComponentFactory#create(org.jgrapes.core.Channel,
046     * java.util.Map)
047     */
048    @Override
049    public Optional<ComponentType> create(
050            Channel componentChannel, Map<Object, Object> properties) {
051        return Optional.of(new SysInfoConlet(componentChannel));
052    }
053
054}
