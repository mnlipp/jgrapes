054/**
055 * 
056 */
057public class SysInfoConlet
058        extends FreeMarkerComponent<SysInfoConlet.SysInfoModel> {
059
060    private static final Set<RenderMode> MODES = RenderMode.asSet(
061        RenderMode.DeleteablePreview, RenderMode.View);
062
063    /**
064     * The periodically generated update event.
065     */
066    public static class Update extends Event<Void> {
067    }
068
069    /**
070     * Creates a new component with its channel set to the given channel.
071     * 
072     * @param componentChannel the channel that the component's handlers listen
073     *            on by default and that {@link Manager#fire(Event, Channel...)}
074     *            sends the event to
075     */
076    public SysInfoConlet(Channel componentChannel) {
077        super(componentChannel);
078        setPeriodicRefresh(Duration.ofSeconds(1), () -> new Update());
079    }
080
081    /**
082     * On {@link ConsoleReady}, fire the {@link AddConletType}.
083     *
084     * @param event the event
085     * @param portalSession the portal session
086     * @throws TemplateNotFoundException the template not found exception
087     * @throws MalformedTemplateNameException the malformed template name
088     *             exception
089     * @throws ParseException the parse exception
090     * @throws IOException Signals that an I/O exception has occurred.
091     */
092    @Handler
093    public void onPortalReady(ConsoleReady event, ConsoleSession portalSession)
094            throws TemplateNotFoundException, MalformedTemplateNameException,
095            ParseException, IOException {
096        // Add SysInfoConlet resources to page
097        portalSession.respond(new AddConletType(type())
098            .setDisplayNames(
099                displayNames(portalSession.supportedLocales(), "portletName"))
100            .addScript(new ScriptResource()
101                .setRequires("chart.js")
102                .setScriptUri(event.renderSupport().portletResource(
103                    type(), "SysInfo-functions.ftl.js")))
104            .addCss(event.renderSupport(), WebConsoleUtils.uriFromPath(
105                "SysInfo-style.css")));
106    }
107
108    /*
109     * (non-Javadoc)
110     * 
111     * @see org.jgrapes.portal.AbstractPortlet#generatePortletId()
112     */
113    @Override
114    protected String generatePortletId() {
115        return type() + "-" + super.generatePortletId();
116    }
117
118    /*
119     * (non-Javadoc)
120     * 
121     * @see org.jgrapes.portal.AbstractPortlet#modelFromSession
122     */
123    @Override
124    protected Optional<SysInfoModel> stateFromSession(
125            Session session, String portletId) {
126        if (portletId.startsWith(type() + "-")) {
127            return Optional.of(new SysInfoModel(portletId));
128        }
129        return Optional.empty();
130    }
131
132    @Override
133    public String doAddPortlet(AddConletRequest event,
134            ConsoleSession portalSession) throws Exception {
135        String portletId = generatePortletId();
136        SysInfoModel portletModel = putInSession(
137            portalSession.browserSession(), new SysInfoModel(portletId));
138        renderPortlet(event, portalSession, portletModel);
139        return portletId;
140    }
141
142    /*
143     * (non-Javadoc)
144     * 
145     * @see org.jgrapes.portal.AbstractPortlet#doRenderPortlet
146     */
147    @Override
148    protected void doRenderPortlet(RenderConletRequest event,
149            ConsoleSession portalSession, String portletId,
150            SysInfoModel portletModel) throws Exception {
151        renderPortlet(event, portalSession, portletModel);
152    }
153
154    @SuppressWarnings("PMD.DataflowAnomalyAnalysis")
155    private void renderPortlet(RenderConletRequestBase<?> event,
156            ConsoleSession portalSession, SysInfoModel portletModel)
157            throws TemplateNotFoundException, MalformedTemplateNameException,
158            ParseException, IOException {
159        if (event.renderPreview()) {
160            Template tpl
161                = freemarkerConfig().getTemplate("SysInfo-preview.ftl.html");
162            portalSession.respond(new RenderPortletFromTemplate(event,
163                SysInfoConlet.class, portletModel.getPortletId(),
164                tpl, fmModel(event, portalSession, portletModel))
165                    .setRenderMode(RenderMode.DeleteablePreview)
166                    .setSupportedModes(MODES)
167                    .setForeground(event.isForeground()));
168            updateView(portalSession, portletModel.getPortletId());
169        }
170        if (event.renderModes().contains(RenderMode.View)) {
171            Template tpl
172                = freemarkerConfig().getTemplate("SysInfo-view.ftl.html");
173            portalSession.respond(new RenderPortletFromTemplate(event,
174                SysInfoConlet.class, portletModel.getPortletId(),
175                tpl, fmModel(event, portalSession, portletModel))
176                    .setRenderMode(RenderMode.View)
177                    .setSupportedModes(MODES)
178                    .setForeground(event.isForeground()));
179        }
180    }
181
182    private void updateView(ConsoleSession portalSession, String portletId) {
183        if (!portalSession.isConnected()) {
184            return;
185        }
186        Runtime runtime = Runtime.getRuntime();
187        portalSession.respond(new NotifyConletView(type(),
188            portletId, "updateMemorySizes",
189            System.currentTimeMillis(), runtime.maxMemory(),
190            runtime.totalMemory(),
191            runtime.totalMemory() - runtime.freeMemory()));
192    }
193
194    /*
195     * (non-Javadoc)
196     * 
197     * @see org.jgrapes.portal.AbstractPortlet#doDeletePortlet
198     */
199    @Override
200    protected void doDeletePortlet(DeleteConletRequest event,
201            ConsoleSession portalSession, String portletId,
202            SysInfoModel retrievedState) throws Exception {
203        portalSession.respond(new DeleteConlet(portletId));
204    }
205
206    /**
207     * Handle the periodic update event by sending {@link NotifyConletView}
208     * events.
209     *
210     * @param event the event
211     * @param portalSession the portal session
212     */
213    @Handler
214    public void onUpdate(Update event, ConsoleSession portalSession) {
215        for (String portletId : portletIds(portalSession)) {
216            updateView(portalSession, portletId);
217        }
218    }
219
220    @Override
221    @SuppressWarnings("PMD.DoNotCallGarbageCollectionExplicitly")
222    protected void doNotifyPortletModel(NotifyConletModel event,
223            ConsoleSession portalSession, SysInfoModel portletState)
224            throws Exception {
225        event.stop();
226        System.gc();
227        for (String portletId : portletIds(portalSession)) {
228            updateView(portalSession, portletId);
229        }
230    }
231
232    /**
233     * The portlet's model.
234     */
235    @SuppressWarnings("serial")
236    public static class SysInfoModel
237            extends AbstractConlet.PortletBaseModel {
238
239        /**
240         * Creates a new model with the given type and id.
241         * 
242         * @param portletId the portlet id
243         */
244        @ConstructorProperties({ "portletId" })
245        public SysInfoModel(String portletId) {
246            super(portletId);
247        }
248
249        /**
250         * Return the system properties.
251         *
252         * @return the properties
253         */
254        public Properties systemProperties() {
255            return System.getProperties();
256        }
257
258        /**
259         * Return the {@link Runtime}.
260         *
261         * @return the runtime
262         */
263        public Runtime runtime() {
264            return Runtime.getRuntime();
265        }
266    }
