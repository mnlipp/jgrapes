019package org.jgrapes.webconsole.bootstrap4;
020
021import java.net.URI;
022import org.jgrapes.core.Channel;
023import org.jgrapes.http.events.Request;
024import org.jgrapes.http.events.Response;
025import org.jgrapes.webconsole.base.WebConsole;
026import org.jgrapes.webconsole.base.freemarker.FreeMarkerConsoleWeblet;
027
028/**
029 * Provides resources using {@link Request}/{@link Response}
030 * events. Some resource requests (page resource, conlet resource)
031 * are forwarded via the {@link WebConsole} component to the 
032 * web console components.
033 */
034@SuppressWarnings({ "PMD.ExcessiveImports", "PMD.NcssCount",
035    "PMD.TooManyMethods" })
036public class Bootstrap4Weblet extends FreeMarkerConsoleWeblet {
037
038    /**
039     * Instantiates a new Bootstrap 3 UI weblet.
040     *
041     * @param webletChannel the weblet channel
042     * @param consoleChannel the web console channel
043     * @param consolePrefix the web console prefix
044     */
045    public Bootstrap4Weblet(Channel webletChannel, Channel consoleChannel,
046            URI consolePrefix) {
047        super(webletChannel, consoleChannel, consolePrefix);
048    }
049
050    @Override
051    public String styling() {
052        return "bootstrap4";
053    }
054
055}
