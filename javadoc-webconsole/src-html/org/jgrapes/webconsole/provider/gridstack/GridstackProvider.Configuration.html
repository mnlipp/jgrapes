019package org.jgrapes.webconsole.provider.gridstack;
020
021import freemarker.core.ParseException;
022import freemarker.template.MalformedTemplateNameException;
023import freemarker.template.TemplateNotFoundException;
024import java.io.IOException;
025import java.util.Map;
026import java.util.logging.Logger;
027import org.jgrapes.core.Channel;
028import org.jgrapes.core.Event;
029import org.jgrapes.core.Manager;
030import org.jgrapes.core.annotation.Handler;
031import org.jgrapes.webconsole.base.ConsoleSession;
032import org.jgrapes.webconsole.base.PageResourceProvider;
033import org.jgrapes.webconsole.base.events.AddPageResources;
034import org.jgrapes.webconsole.base.events.AddPageResources.ScriptResource;
035import org.jgrapes.webconsole.base.events.ConsoleReady;
036
037/**
038 * Provider for the [Gridstack.js](http://gridstackjs.com/) library.
039 */
040@SuppressWarnings("PMD.GuardLogStatement")
041public class GridstackProvider extends PageResourceProvider {
042
043    @SuppressWarnings({ "PMD.FieldNamingConventions",
044        "PMD.VariableNamingConventions" })
045    private static Logger LOG
046        = Logger.getLogger(GridstackProvider.class.getName());
047
048    /**
049     * The Configuration.
050     */
051    @SuppressWarnings("PMD.FieldNamingConventions")
052    public enum Configuration {
053        CoreOnly, CoreWithJQUiPlugin, All
054    }
055
056    private Configuration configuration = Configuration.CoreOnly;
057    private final boolean requireTouchPunch;
058
059    /**
060     * Creates a new component with its channel set to the given 
061     * channel.
062     *
063     * Supported properties are:
064     * 
065     *  * *configuration*: the string representation of one of the
066     *    {@link Configuration} values. Determines the gridstack 
067     *    configuration to provide. defaults to "All".
068     *    
069     *  * *requireTouchPunch*: a boolean indicating whether
070     *    "jquery-ui.touch-punch" should be required to ensure that
071     *    it is loaded before gridstack (see {@link AddPageResources}).
072     *    Defaults to `false`. If set to `true`, forces configuration
073     *    `All` to be replaced by `CoreWithJQUiPlugin` because the 
074     *    touch punch provider requires "jquery" and thus the full 
075     *    JQuery library is available anyway.
076     *
077     * @param componentChannel the channel that the component's
078     * handlers listen on by default and that 
079     * {@link Manager#fire(Event, Channel...)} sends the event to
080     * @param properties the properties used to configure the component
081     */
082    public GridstackProvider(Channel componentChannel,
083            Map<Object, Object> properties) {
084        super(componentChannel);
085        requireTouchPunch
086            = (boolean) properties.getOrDefault("requireTouchPunch", false);
087        if (properties.containsKey("configuration")) {
088            String config = (String) properties.get("configuration");
089            try {
090                configuration = Configuration.valueOf(config);
091            } catch (IllegalArgumentException e) {
092                LOG.severe(
093                    "Illegal value for \"configuration\": " + config);
094            }
095        }
096        if (requireTouchPunch && configuration == Configuration.All) {
097            configuration = Configuration.CoreWithJQUiPlugin;
098        }
099    }
100
101    /**
102     * On {@link ConsoleReady}, fire the appropriate {@link AddPageResources}.
103     *
104     * @param event the event
105     * @param consoleSession the web console session
106     * @throws TemplateNotFoundException the template not found exception
107     * @throws MalformedTemplateNameException the malformed template name exception
108     * @throws ParseException the parse exception
109     * @throws IOException Signals that an I/O exception has occurred.
110     */
111    @Handler(priority = 100)
112    @SuppressWarnings("PMD.AvoidDuplicateLiterals")
113    public void onConsoleReady(ConsoleReady event,
114            ConsoleSession consoleSession)
115            throws TemplateNotFoundException, MalformedTemplateNameException,
116            ParseException, IOException {
117        String minExt = event.renderSupport()
118            .useMinifiedResources() ? ".min" : "";
119        AddPageResources addRequest = new AddPageResources()
120            .addCss(event.renderSupport().pageResource(
121                "gridstack/gridstack" + minExt + ".css"));
122        switch (configuration) {
123        case CoreOnly:
124            addRequest
125                .addScriptResource(new ScriptResource()
126                    .setRequires(requireTouchPunch
127                        ? "jquery-ui.touch-punch"
128                        : "jquery")
129                    .setProvides("gridstack")
130                    .setScriptUri(event.renderSupport().pageResource(
131                        "gridstack/gridstack" + minExt + ".js")));
132            break;
133        case CoreWithJQUiPlugin:
134            addRequest.addScriptResource(new ScriptResource()
135                .setRequires(requireTouchPunch
136                    ? "jquery-ui.touch-punch"
137                    : "jquery")
138                .setProvides("gridstack")
139                .setScriptUri(event.renderSupport().pageResource(
140                    "gridstack/gridstack" + minExt + ".js")));
141            addRequest.addScriptResource(new ScriptResource()
142                .setRequires("gridstack")
143                .setProvides("gridstack.jQueryUI")
144                .setScriptUri(event.renderSupport().pageResource(
145                    "gridstack/gridstack.jQueryUI" + minExt + ".js")));
146            break;
147        case All:
148            addRequest.addScriptResource(new ScriptResource()
149                .setProvides("jquery", "jquery-ui",
150                    "gridstack", "gridstack.all")
151                .setScriptUri(event.renderSupport().pageResource(
152                    "gridstack/gridstack.all.js")));
153            break;
154        default:
155            break;
156        }
157        consoleSession.respond(addRequest);
158    }
159}
