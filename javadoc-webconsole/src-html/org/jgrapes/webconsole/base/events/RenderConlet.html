019package org.jgrapes.webconsole.base.events;
020
021import java.io.IOException;
022import java.io.Writer;
023import java.util.Collections;
024import java.util.HashSet;
025import java.util.Set;
026import java.util.concurrent.ExecutionException;
027import java.util.concurrent.Future;
028import org.jgrapes.webconsole.base.Conlet.RenderMode;
029
030/**
031 * A base class for web console rendering. Sent to the web console page for 
032 * adding or updating a complete web console component representation. 
033 * This class maintains all required information except the actual content
034 * (the HTML) which must be provided by the derived classes.
035 * 
036 * The HTML provided is searched for attributes `data-jgwc-on-load`
037 * and `data-jgwc-on-unload` which must have as value the name of a 
038 * function. When the HTML has been loaded or unloaded (i.e. added to
039 * the DOM or removed from the DOM), the respective functions are 
040 * invoked with the element containing the attribute as their first 
041 * parameter. A second boolean parameter is `true` if the on-load 
042 * function is called due to an update of an already existing container.
043 * 
044 * The HTML elements of edit dialogs ({@link RenderMode#Edit})
045 * can have an additional attribute `data-jgwc-on-apply`
046 * which must have as its value the name of a function. This
047 * function is invoked when changes made in the form must be
048 * applied (e.g. before the dialog is closed).
049 */
050public abstract class RenderConlet extends ConsoleCommand {
051
052    private static final Set<RenderMode> DEFAULT_SUPPORTED
053        = Collections.unmodifiableSet(RenderMode.asSet(RenderMode.Preview));
054
055    private final Class<?> conletClass;
056    private final String conletId;
057    private Set<RenderMode> renderAs = RenderMode.asSet(RenderMode.Preview);
058    private Set<RenderMode> supportedModes = DEFAULT_SUPPORTED;
059
060    /**
061     * Creates a new event.
062     * 
063     * @param conletClass the web console component class
064     * @param conletId the id of the web console component
065     */
066    public RenderConlet(Class<?> conletClass, String conletId) {
067        this.conletClass = conletClass;
068        this.conletId = conletId;
069    }
070
071    /**
072     * Returns the web console component class as specified on creation.
073     *
074     * @return the class
075     */
076    public Class<?> conletClass() {
077        return conletClass;
078    }
079
080    /**
081     * Returns the web console component id as specified on creation.
082     * 
083     * @return the web console component id
084     */
085    public String conletId() {
086        return conletId;
087    }
088
089    /**
090     * Set the render mode. The default value is {@link RenderMode#Preview}.
091     * 
092     * @param renderMode the render mode to set
093     * @return the event for easy chaining
094     */
095    public RenderConlet setRenderAs(RenderMode renderMode) {
096        this.renderAs = RenderMode.asSet(renderMode);
097        return this;
098    }
099
100    /**
101     * Set the render mode (including modifier).
102     * 
103     * @param renderAs the render mode to use
104     * @return the event for easy chaining
105     */
106    public RenderConlet setRenderAs(Set<RenderMode> renderAs) {
107        this.renderAs = new HashSet<>(renderAs);
108        return this;
109    }
110
111    /**
112     * Returns the render mode.
113     * 
114     * @return the render mode
115     */
116    public Set<RenderMode> renderAs() {
117        return Collections.unmodifiableSet(renderAs);
118    }
119
120    /**
121     * Set the supported render modes. The default value is 
122     * {@link RenderMode#Preview}.
123     * 
124     * @param supportedModes the supported render modes to set
125     * @return the event for easy chaining
126     */
127    public RenderConlet setSupportedModes(Set<RenderMode> supportedModes) {
128        this.supportedModes = supportedModes;
129        return this;
130    }
131
132    /**
133     * Add the given render mode to the supported render modes.
134     * 
135     * @param supportedMode the supported render modes to add
136     * @return the event for easy chaining
137     */
138    public RenderConlet addSupportedMode(RenderMode supportedMode) {
139        if (supportedModes == DEFAULT_SUPPORTED) { // NOPMD, check identity
140            supportedModes = new HashSet<>(DEFAULT_SUPPORTED);
141        }
142        supportedModes.add(supportedMode);
143        return this;
144    }
145
146    /**
147     * Returns the supported modes.
148     * 
149     * @return the supported modes
150     */
151    public Set<RenderMode> supportedRenderModes() {
152        return supportedModes;
153    }
154
155    /**
156     * Provides the HTML that displays the web console component 
157     * on the page.
158     * 
159     * @return the HTML
160     */
161    public abstract Future<String> content();
162
163    /**
164     * Writes the JSON notification to the given writer.
165     *
166     * @param writer the writer
167     * @throws ExecutionException 
168     * @throws InterruptedException 
169     */
170    @Override
171    public void toJson(Writer writer)
172            throws InterruptedException, IOException {
173        try {
174            toJson(writer, "updateConlet", conletId(),
175                renderAs().stream().map(RenderMode::name)
176                    .toArray(size -> new String[size]),
177                supportedRenderModes().stream().map(RenderMode::name)
178                    .toArray(size -> new String[size]),
179                content().get());
180        } catch (ExecutionException e) {
181            throw new IOException(e);
182        }
183    }
184}
