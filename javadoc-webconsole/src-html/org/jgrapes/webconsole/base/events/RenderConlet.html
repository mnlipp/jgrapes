019package org.jgrapes.webconsole.base.events;
020
021import java.io.IOException;
022import java.io.Writer;
023import java.util.Collections;
024import java.util.HashSet;
025import java.util.Set;
026import java.util.concurrent.ExecutionException;
027import java.util.concurrent.Future;
028import org.jgrapes.webconsole.base.Conlet.RenderMode;
029
030/**
031 * A base class for web console rendering. Sent to the web console page for 
032 * adding or updating a complete web console component representation. 
033 * This class contains all required information except the actual content
034 * (the HTML) which must be provided by the derived classes.
035 * 
036 * The content must be valid HTML, usable as inner HTML of a block level
037 * element (typically a `div`). If the root element has an attribute
038 * `data-conlet-title`, its value overrides the default title (the display
039 * name of the conlet type, see {@link AddConletType#displayNames()}).
040 * 
041 * The content provided is searched for attributes `data-jgwc-on-load`
042 * and `data-jgwc-on-unload` which must have as value the name of a 
043 * function. When the HTML has been loaded or unloaded (i.e. added to
044 * the DOM or removed from the DOM), the respective functions are 
045 * invoked with the element containing the attribute as their first 
046 * parameter. A second boolean parameter is `true` if the on-load 
047 * function is called due to an update of an already existing container.
048 * 
049 * The HTML elements of edit dialogs ({@link RenderMode#Edit})
050 * can have an additional attribute `data-jgwc-on-apply`
051 * which must have as its value the name of a function. This
052 * function is invoked when changes made in the form must be
053 * applied (e.g. before the dialog is closed).
054 */
055@SuppressWarnings("PMD.LinguisticNaming")
056public abstract class RenderConlet extends ConsoleCommand {
057
058    private static final Set<RenderMode> DEFAULT_SUPPORTED
059        = Collections.unmodifiableSet(RenderMode.asSet(RenderMode.Preview));
060
061    private final String conletType;
062    private final String conletId;
063    private Set<RenderMode> renderAs = RenderMode.asSet(RenderMode.Preview);
064    private Set<RenderMode> supportedModes = DEFAULT_SUPPORTED;
065
066    /**
067     * Creates a new event.
068     *
069     * @param conletType the conlet type
070     * @param conletId the id of the web console component
071     */
072    public RenderConlet(String conletType, String conletId) {
073        this.conletType = conletType;
074        this.conletId = conletId;
075    }
076
077    /**
078     * Returns the web console component type as specified on creation.
079     *
080     * @return the type
081     */
082    public String conletType() {
083        return conletType;
084    }
085
086    /**
087     * Returns the web console component id as specified on creation.
088     * 
089     * @return the web console component id
090     */
091    public String conletId() {
092        return conletId;
093    }
094
095    /**
096     * Set the render mode. The default value is {@link RenderMode#Preview}.
097     * 
098     * @param renderMode the render mode to set
099     * @return the event for easy chaining
100     */
101    public RenderConlet setRenderAs(RenderMode renderMode) {
102        this.renderAs = RenderMode.asSet(renderMode);
103        return this;
104    }
105
106    /**
107     * Set the render mode (including modifier).
108     * 
109     * @param renderAs the render mode to use
110     * @return the event for easy chaining
111     */
112    public RenderConlet setRenderAs(Set<RenderMode> renderAs) {
113        this.renderAs = new HashSet<>(renderAs);
114        return this;
115    }
116
117    /**
118     * Returns the render mode.
119     * 
120     * @return the render mode
121     */
122    public Set<RenderMode> renderAs() {
123        return Collections.unmodifiableSet(renderAs);
124    }
125
126    /**
127     * Set the supported render modes. The default value is 
128     * {@link RenderMode#Preview}.
129     * 
130     * @param supportedModes the supported render modes to set
131     * @return the event for easy chaining
132     */
133    public RenderConlet setSupportedModes(Set<RenderMode> supportedModes) {
134        this.supportedModes = supportedModes;
135        return this;
136    }
137
138    /**
139     * Add the given render mode to the supported render modes.
140     * 
141     * @param supportedMode the supported render modes to add
142     * @return the event for easy chaining
143     */
144    public RenderConlet addSupportedMode(RenderMode supportedMode) {
145        if (supportedModes == DEFAULT_SUPPORTED) { // NOPMD, check identity
146            supportedModes = new HashSet<>(DEFAULT_SUPPORTED);
147        }
148        supportedModes.add(supportedMode);
149        return this;
150    }
151
152    /**
153     * Returns the supported modes.
154     * 
155     * @return the supported modes
156     */
157    public Set<RenderMode> supportedRenderModes() {
158        return supportedModes;
159    }
160
161    /**
162     * Provides the HTML that displays the web console component 
163     * on the page.
164     * 
165     * @return the HTML
166     */
167    public abstract Future<String> content();
168
169    /**
170     * Writes the JSON notification to the given writer.
171     *
172     * @param writer the writer
173     * @throws ExecutionException 
174     * @throws InterruptedException 
175     */
176    @Override
177    public void toJson(Writer writer)
178            throws InterruptedException, IOException {
179        try {
180            toJson(writer, "updateConlet", conletType(), conletId(),
181                renderAs().stream().map(RenderMode::name)
182                    .toArray(size -> new String[size]),
183                supportedRenderModes().stream().map(RenderMode::name)
184                    .toArray(size -> new String[size]),
185                content().get());
186        } catch (ExecutionException e) {
187            throw new IOException(e);
188        }
189    }
190}
