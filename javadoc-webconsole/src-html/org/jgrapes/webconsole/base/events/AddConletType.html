019package org.jgrapes.webconsole.base.events;
020
021import java.io.IOException;
022import java.io.Writer;
023import java.net.URI;
024import java.util.ArrayList;
025import java.util.Arrays;
026import java.util.Collections;
027import java.util.List;
028import java.util.Locale;
029import java.util.Map;
030import java.util.stream.Collectors;
031
032import org.jdrupes.json.JsonArray;
033import org.jgrapes.webconsole.base.Conlet.RenderMode;
034import org.jgrapes.webconsole.base.RenderSupport;
035import org.jgrapes.webconsole.base.events.AddPageResources.ScriptResource;
036
037/**
038 * Adds a web console component type with its global resources 
039 * (JavaScript and/or CSS) to the console page. Specifying global 
040 * resources result in the respective
041 * `<link .../>` or `<script ...></script>` nodes
042 * being added to the page's `<head>` node.
043 * 
044 * This in turn causes the browser to issue `GET` requests that
045 * (usually) refer to the web console component's resources. These requests are
046 * converted to {@link ConletResourceRequest}s by the portal and
047 * sent to the web console components, which must respond to the requests.
048 * 
049 * The sequence of events is shown in the diagram.
050 * 
051 * ![WebConsole Ready Event Sequence](AddConletTypeSeq.svg)
052 * 
053 * See {@link ResourceRequest} for details about the processing
054 * of the {@link ConletResourceRequest}.
055 * 
056 * A portelt's JavaScript may (and probably must) make use of
057 * the functions provided by the portal page. See the 
058 * <a href="../jsdoc/module-jgportal.html">JavaScript
059 * documentation of these functions</a> for details.
060 * 
061 * @startuml AddConletTypeSeq.svg
062 * hide footbox
063 * 
064 * activate Browser
065 * Browser -> WebConsole: "portalReady"
066 * deactivate Browser
067 * activate WebConsole
068 * WebConsole -> ConletX: ConsoleReady 
069 * deactivate WebConsole
070 * activate ConletX
071 * ConletX -> WebConsole: AddConletType 
072 * deactivate ConletX
073 * activate WebConsole
074 * WebConsole -> Browser: "addConletType"
075 * activate Browser
076 * deactivate WebConsole
077 * Browser -> WebConsole: "GET <conlet resource URI>"
078 * activate WebConsole
079 * WebConsole -> ConletX: ConletResourceRequest
080 * deactivate Browser
081 * activate ConletX
082 * deactivate ConletX
083 * 
084 * @enduml
085 */
086public class AddConletType extends ConsoleCommand {
087
088    private final String conletType;
089    private Map<Locale, String> displayNames = Collections.emptyMap();
090    private final List<URI> cssUris = new ArrayList<>();
091    private final List<ScriptResource> scriptResources = new ArrayList<>();
092    private List<RenderMode> renderModes;
093
094    /**
095     * Create a new event for the given web console component type.
096     * 
097     * @param conletType a unique id for the web console component type 
098     * (usually the class name)
099     */
100    public AddConletType(String conletType) {
101        this.conletType = conletType;
102    }
103
104    /**
105     * Return the web console component type.
106     * 
107     * @return the web console component type
108     */
109    public String conletType() {
110        return conletType;
111    }
112
113    /**
114     * Sets the display names.
115     * 
116     * @param displayNames the display names
117     * @return the event for easy chaining
118     */
119    public AddConletType setDisplayNames(Map<Locale, String> displayNames) {
120        this.displayNames = displayNames;
121        return this;
122    }
123
124    /**
125     * Return the display names.
126     * 
127     * @return the displayNames
128     */
129    public Map<Locale, String> displayNames() {
130        return displayNames;
131    }
132
133    /**
134     * Add a render mode. The render mode determines how the web console 
135     * component is initially rendered (i.e. when added). Several modes may be
136     * added in order of preference. The default mode (i.e. none 
137     * specified) is {@link RenderMode#Preview}.
138     *
139     * @param mode the mode
140     * @return the event for easy chaining
141     */
142    public AddConletType addRenderMode(RenderMode mode) {
143        if (renderModes == null) {
144            renderModes = new ArrayList<>();
145        }
146        renderModes.add(mode);
147        return this;
148    }
149
150    /**
151     * Return the render modes.
152     * 
153     * @return the result
154     */
155    public List<RenderMode> renderModes() {
156        if (renderModes == null) {
157            return Arrays.asList(RenderMode.Preview);
158        }
159        return renderModes;
160    }
161
162    /**
163     * Add a script resource to be requested by the browser.
164     * 
165     * @param scriptResource the script resource
166     * @return the event for easy chaining
167     */
168    public AddConletType addScript(ScriptResource scriptResource) {
169        scriptResources.add(scriptResource);
170        return this;
171    }
172
173    /**
174     * Add the URI of a CSS resource that is to be added to the
175     * header section of the portal page.
176     *
177     * @param renderSupport the render support for mapping the `uri`
178     * @param uri the URI
179     * @return the event for easy chaining
180     */
181    public AddConletType addCss(RenderSupport renderSupport, URI uri) {
182        cssUris.add(renderSupport.conletResource(conletType(), uri));
183        return this;
184    }
185
186    /**
187     * Return all script resources.
188     * 
189     * @return the result
190     */
191    public ScriptResource[] scriptResources() {
192        return scriptResources.toArray(new ScriptResource[0]);
193    }
194
195    /**
196     * Return all CSS URIs.
197     * 
198     * @return the result
199     */
200    public URI[] cssUris() {
201        return cssUris.toArray(new URI[0]);
202    }
203
204    @Override
205    public void toJson(Writer writer) throws IOException {
206        JsonArray strArray = JsonArray.create();
207        for (ScriptResource scriptResource : scriptResources()) {
208            strArray.append(scriptResource.toJsonValue());
209        }
210        toJson(writer, "addConletType", conletType(),
211            displayNames().entrySet().stream()
212                .collect(Collectors.toMap(e -> e.getKey().toLanguageTag(),
213                    e -> e.getValue())),
214            Arrays.stream(cssUris()).map(
215                uri -> uri.toString()).toArray(String[]::new),
216            strArray, renderModes().stream().map(RenderMode::name)
217                .toArray(size -> new String[size]));
218    }
219}
