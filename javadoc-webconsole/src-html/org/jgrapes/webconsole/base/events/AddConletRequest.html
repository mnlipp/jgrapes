019package org.jgrapes.webconsole.base.events;
020
021import java.util.HashMap;
022import java.util.Map;
023import java.util.Set;
024import java.util.function.BiConsumer;
025import org.jgrapes.webconsole.base.Conlet.RenderMode;
026import org.jgrapes.webconsole.base.RenderSupport;
027
028/**
029 * Sent to the console (server) if a new web console component instance of a given 
030 * type should be added to the web console page. The console server usually 
031 * responds with a {@link RenderConlet} event that has as payload the
032 * HTML that displays the web console component on the console page.
033 * 
034 * Properties may be passed with the event. The interpretation
035 * of the properties is completely dependent on the handling web console 
036 * component. It is recommended to use {@link String}s as keys and JDK types
037 * as values. This avoids classpath dependencies on the web console component
038 * that is to be added. 
039 * 
040 * The event's result is the web console component id of the new 
041 * web console component instance.
042 * 
043 * ![Event Sequence](AddConletRequestSeq.svg)
044 * 
045 * @startuml AddConletRequestSeq.svg
046 * hide footbox
047 * 
048 * Browser -> WebConsole: "addConlet"
049 * activate WebConsole
050 * WebConsole -> Conlet: AddConletRequest
051 * deactivate WebConsole
052 * activate Conlet
053 * Conlet -> WebConsole: RenderConlet
054 * deactivate Conlet
055 * activate WebConsole
056 * WebConsole -> Browser: "renderConlet"
057 * deactivate WebConsole
058 * 
059 * @enduml
060 * 
061 */
062public class AddConletRequest extends RenderConletRequestBase<String> {
063
064    private final String conletType;
065    private boolean foreground = true;
066    private Map<? extends Object, ? extends Object> properties;
067
068    /**
069     * Creates a new event.
070     *
071     * @param renderSupport the render support
072     * @param conletType the type of the web console component
073     * @param renderModes the render modes
074     */
075    public AddConletRequest(RenderSupport renderSupport, String conletType,
076            Set<RenderMode> renderModes) {
077        super(renderSupport, renderModes);
078        this.conletType = conletType;
079    }
080
081    /**
082     * Creates a new event.
083     *
084     * @param renderSupport the render support
085     * @param conletType the type of the web console component
086     * @param renderModes the render modes
087     * @param properties optional values for properties of the 
088     * web console component instance
089     */
090    public AddConletRequest(RenderSupport renderSupport, String conletType,
091            Set<RenderMode> renderModes, Map<?, ?> properties) {
092        super(renderSupport, renderModes);
093        this.conletType = conletType;
094        @SuppressWarnings("unchecked")
095        Map<Object, Object> props = (Map<Object, Object>) properties;
096        this.properties = props;
097    }
098
099    /**
100     * Determines if the web console component will be put in the foreground.
101     * Defaults to `true` for added web console components as they are 
102     * most likely supposed to be seen. 
103     *
104     * @param foreground the foreground
105     * @return the event for easy chaining
106     */
107    public AddConletRequest setForeground(boolean foreground) {
108        this.foreground = foreground;
109        return this;
110    }
111
112    /**
113     * Returns the web console component type
114     * 
115     * @return the web console component type
116     */
117    public String conletType() {
118        return conletType;
119    }
120
121    /**
122     * Returns the properties. Every event returns a mutable map,
123     * thus allowing event handlers to modify the map even if
124     * none was passed to the constructor.
125     */
126    public Map<Object, Object> properties() {
127        if (properties == null) {
128            properties = new HashMap<>();
129        }
130        @SuppressWarnings("unchecked")
131        Map<Object, Object> props = (Map<Object, Object>) properties;
132        return props;
133    }
134
135    /**
136     * Convenience method for adding properties one-by-one.
137     * 
138     * @param key the property key
139     * @param value the property value
140     * @return the event for easy chaining
141     */
142    public AddConletRequest addProperty(Object key, Object value) {
143        properties().put(key, value);
144        return this;
145    }
146
147    /**
148     * Convenience method that performs the given action if a property
149     * with the given key exists.
150     * 
151     * @param key the property key
152     * @param action the action to perform
153     */
154    public AddConletRequest ifPresent(
155            Object key, BiConsumer<Object, Object> action) {
156        if (properties().containsKey(key)) {
157            action.accept(key, properties().get(key));
158        }
159        return this;
160    }
161
162    @Override
163    public boolean isForeground() {
164        return foreground;
165    }
166
167}
