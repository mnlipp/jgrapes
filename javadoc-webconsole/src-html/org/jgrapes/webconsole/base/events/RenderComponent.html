019package org.jgrapes.webconsole.base.events;
020
021import java.io.IOException;
022import java.io.Writer;
023import java.util.Arrays;
024import java.util.Collections;
025import java.util.HashSet;
026import java.util.Set;
027import java.util.concurrent.ExecutionException;
028import java.util.concurrent.Future;
029
030import org.jgrapes.webconsole.base.ConsoleComponent.RenderMode;
031
032/**
033 * A base class for portal rendering. Sent to the portal page for 
034 * adding or updating a complete portlet representation. This class
035 * maintains all required information except the actual content
036 * (the HTML) which must be provided by the derived classes.
037 * 
038 * The HTML provided is searched for attributes `data-jgp-on-load`
039 * which must have as value the name of a function. When the
040 * HTML has been loaded, this function is invoked with the element
041 * containing the attribute as parameter.
042 * 
043 * The HTML elements of edit dialogs ({@link RenderMode#Edit})
044 * can have an additional attribute `data-jgp-on-apply`
045 * which must have as its value the name of a function. This
046 * function is invoked when changes made in the form must be
047 * applied (e.g. before the dialog is closed).
048 */
049public abstract class RenderComponent extends ConsoleCommand {
050
051    private static final Set<RenderMode> DEFAULT_SUPPORTED
052        = Collections.unmodifiableSet(new HashSet<>(
053            Arrays.asList(new RenderMode[] { RenderMode.Preview })));
054
055    private final Class<?> portletClass;
056    private final String portletId;
057    private RenderMode renderMode = RenderMode.Preview;
058    private Set<RenderMode> supportedModes = DEFAULT_SUPPORTED;
059    private boolean foreground;
060
061    /**
062     * Creates a new event.
063     * 
064     * @param portletClass the portlet class
065     * @param portletId the id of the portlet
066     */
067    public RenderComponent(Class<?> portletClass, String portletId) {
068        this.portletClass = portletClass;
069        this.portletId = portletId;
070    }
071
072    /**
073     * Returns the portlet class as specified on creation.
074     *
075     * @return the class
076     */
077    public Class<?> portletClass() {
078        return portletClass;
079    }
080
081    /**
082     * Returns the portlet id as specified on creation.
083     * 
084     * @return the portlet id
085     */
086    public String portletId() {
087        return portletId;
088    }
089
090    /**
091     * Set the render mode. The default value is {@link RenderMode#Preview}.
092     * 
093     * @param renderMode the render mode to set
094     * @return the event for easy chaining
095     */
096    public RenderComponent setRenderMode(RenderMode renderMode) {
097        this.renderMode = renderMode;
098        return this;
099    }
100
101    /**
102     * Returns the render mode.
103     * 
104     * @return the render mode
105     */
106    public RenderMode renderMode() {
107        return renderMode;
108    }
109
110    /**
111     * Set the supported render modes. The default value is 
112     * {@link RenderMode#Preview}.
113     * 
114     * @param supportedModes the supported render modes to set
115     * @return the event for easy chaining
116     */
117    public RenderComponent setSupportedModes(Set<RenderMode> supportedModes) {
118        this.supportedModes = supportedModes;
119        return this;
120    }
121
122    /**
123     * Add the given render mode to the supported render modes.
124     * 
125     * @param supportedMode the supported render modes to add
126     * @return the event for easy chaining
127     */
128    public RenderComponent addSupportedMode(RenderMode supportedMode) {
129        if (supportedModes == DEFAULT_SUPPORTED) { // NOPMD, check identity
130            supportedModes = new HashSet<>(DEFAULT_SUPPORTED);
131        }
132        supportedModes.add(supportedMode);
133        return this;
134    }
135
136    /**
137     * Returns the supported modes.
138     * 
139     * @return the supported modes
140     */
141    public Set<RenderMode> supportedRenderModes() {
142        return supportedModes;
143    }
144
145    /**
146     * Id set, the tab with the portlet is put in the foreground
147     * when the portlet is rendered. The default value is `false`.
148     * 
149     * @param foreground if set, the portlet is put in foreground
150     * @return the event for easy chaining
151     */
152    public RenderComponent setForeground(boolean foreground) {
153        this.foreground = foreground;
154        return this;
155    }
156
157    /**
158     * Indicates if portelt is to be put in foreground.
159     * 
160     * @return the result
161     */
162    public boolean isForeground() {
163        return foreground;
164    }
165
166    /**
167     * Provides the HTML that displays the portlet 
168     * on the page.
169     * 
170     * @return the HTML
171     */
172    public abstract Future<String> content();
173
174    /**
175     * Writes the JSON notification to the given writer.
176     *
177     * @param writer the writer
178     * @throws ExecutionException 
179     * @throws InterruptedException 
180     */
181    @Override
182    public void toJson(Writer writer)
183            throws InterruptedException, IOException {
184        try {
185            toJson(writer, "updatePortlet", portletId(), renderMode().name(),
186                supportedRenderModes().stream().map(RenderMode::name)
187                    .toArray(size -> new String[size]),
188                content().get(), isForeground());
189        } catch (ExecutionException e) {
190            throw new IOException(e);
191        }
192    }
193}
