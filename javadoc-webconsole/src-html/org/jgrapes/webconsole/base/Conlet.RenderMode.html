019package org.jgrapes.webconsole.base;
020
021import java.util.Arrays;
022import java.util.Collections;
023import java.util.HashSet;
024import java.util.Set;
025
026/**
027 * This interface provides web console component related constants. 
028 * Web console components need not implement this interface, not even 
029 * as a marker interface. They only have to expose a specific response 
030 * behavior to certain events. An overview of the event sequences is 
031 * provided by the package description. Make sure to read this 
032 * description first.
033 */
034public interface Conlet {
035
036    /**
037     * The render modes.
038     */
039    @SuppressWarnings("PMD.FieldNamingConventions")
040    enum RenderMode {
041        Preview, View, Edit, Help,
042        /** Modifier, indicates that a {@link #Preview} may not be removed. */
043        StickyPreview,
044        /** Modifier, forces rendered view to be put in foreground. */
045        Foreground;
046
047        /**
048         * The basic modes (the modes without modifiers).
049         */
050        @SuppressWarnings("PMD.VariableNamingConventions")
051        public static final Set<RenderMode> basicModes
052            = Collections.unmodifiableSet(
053                asSet(RenderMode.Preview, RenderMode.View, RenderMode.Edit,
054                    RenderMode.Help));
055
056        /**
057         * Utility method that creates a {@link Set} of render modes
058         * from enumerated values.
059         *
060         * @param modes the modes
061         * @return the sets the
062         */
063        public static Set<RenderMode> asSet(RenderMode... modes) {
064            return new HashSet<>(Arrays.asList(modes));
065        }
066
067        /**
068         * Retrieves the modifiers.
069         */
070        public static Set<RenderMode> modifiers(Set<RenderMode> modes) {
071            Set<RenderMode> result = new HashSet<>(modes);
072            result.removeAll(basicModes);
073            return result;
074        }
075
076        /**
077         * Adds the modifiers from the given set to this basic mode.
078         *
079         * @param modifiers the set with modifiers
080         * @return the augmented basic mode
081         */
082        public Set<RenderMode> addModifiers(Set<RenderMode> modifiers) {
083            Set<RenderMode> result = modifiers(modifiers);
084            result.add(this);
085            return result;
086        }
087    }
088
089}
