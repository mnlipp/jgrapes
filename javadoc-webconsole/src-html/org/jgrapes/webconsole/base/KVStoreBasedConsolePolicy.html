019package org.jgrapes.webconsole.base;
020

022import java.util.Collections;
023import java.util.HashMap;
024import java.util.List;
025import java.util.Map;
026import java.util.Optional;

029import org.jdrupes.json.JsonBeanEncoder;

047/**
048 * A component that restores the console layout
049 * using key/value events for persisting the data between sessions.
050 * 
051 * ![Boot Event Sequence](KVPPBootSeq.svg)
052 * 
053 * This component requires another component that handles the key/value
054 * store events ({@link KeyValueStoreUpdate}, {@link KeyValueStoreQuery})
055 * used by this component for implementing persistence. When the portal becomes
056 * ready, this policy sends a query for the persisted data.
057 * 
058 * When the portal has been prepared, the policy sends the last layout
059 * as retrieved from persistent storage to the portal and then generates
060 * render events for all portlets contained in this layout.
061 * 
062 * Each time the layout is changed in the portal, the portal sends the
063 * new layout data and this component updates the persistent storage
064 * accordingly.
065 * 
066 * @startuml KVPPBootSeq.svg
067 * hide footbox
068 * 
069 * Browser -> WebConsole: "portalReady"
070 * activate WebConsole
071 * WebConsole -> KVStoreBasedConsolePolicy: ConsoleReady
072 * deactivate WebConsole
073 * activate KVStoreBasedConsolePolicy
074 * KVStoreBasedConsolePolicy -> "KV Store": KeyValueStoreQuery
075 * activate "KV Store"
076 * "KV Store" -> KVStoreBasedConsolePolicy: KeyValueStoreData
077 * deactivate "KV Store"
078 * deactivate KVStoreBasedConsolePolicy
079 * 
080 * actor System
081 * System -> KVStoreBasedConsolePolicy: ConsolePrepared
082 * activate KVStoreBasedConsolePolicy
083 * KVStoreBasedConsolePolicy -> WebConsole: LastConsoleLayout
084 * activate WebConsole
085 * WebConsole -> Browser: "lastPortalLayout"
086 * deactivate WebConsole
087 * loop for all portlets to be displayed
088 *     KVStoreBasedConsolePolicy -> PortletX: RenderConletRequest
089 *     activate PortletX
090 *     PortletX -> WebConsole: RenderConlet
091 *     deactivate PortletX
092 *     activate WebConsole
093 *     WebConsole -> Browser: "renderPortlet"
094 *     deactivate WebConsole
095 * end
096 * deactivate KVStoreBasedConsolePolicy
097 * 
098 * Browser -> WebConsole: "portalLayout"
099 * activate WebConsole
100 * WebConsole -> KVStoreBasedConsolePolicy: ConsoleLayoutChanged
101 * deactivate WebConsole
102 * activate KVStoreBasedConsolePolicy
103 * KVStoreBasedConsolePolicy -> "KV Store": KeyValueStoreUpdate
104 * deactivate KVStoreBasedConsolePolicy
105 * 
106 * @enduml
107 */
108public class KVStoreBasedConsolePolicy extends Component {
109
110    /**
111     * Creates a new component with its channel set to
112     * itself.
113     */
114    public KVStoreBasedConsolePolicy() {
115        // Everything done by super.
116    }
117
118    /**
119     * Creates a new component with its channel set to the given channel.
120     * 
121     * @param componentChannel
122     */
123    public KVStoreBasedConsolePolicy(Channel componentChannel) {
124        super(componentChannel);
125    }
126
127    /**
128     * Intercept the {@link ConsoleReady} event. Request the 
129     * session data from the key/value store and resume.
130     * 
131     * @param event
132     * @param channel
133     * @throws InterruptedException
134     */
135    @Handler
136    public void onConsoleReady(ConsoleReady event, ConsoleSession channel)
137            throws InterruptedException {
138        ConsoleSessionDataStore sessionDs = channel.associated(
139            ConsoleSessionDataStore.class,
140            () -> new ConsoleSessionDataStore(channel.browserSession()));
141        sessionDs.onConsoleReady(event, channel);
142    }
143
144    /**
145     * Handle returned data.
146     *
147     * @param event the event
148     * @param channel the channel
149     * @throws JsonDecodeException the json decode exception
150     */
151    @Handler
152    public void onKeyValueStoreData(
153            KeyValueStoreData event, ConsoleSession channel)
154            throws JsonDecodeException {
155        Optional<ConsoleSessionDataStore> optSessionDs
156            = channel.associated(ConsoleSessionDataStore.class);
157        if (optSessionDs.isPresent()) {
158            optSessionDs.get().onKeyValueStoreData(event, channel);
159        }
160    }
161
162    /**
163     * Handle portal page loaded.
164     *
165     * @param event the event
166     * @param channel the channel
167     */
168    @Handler
169    public void onConsolePrepared(
170            ConsolePrepared event, ConsoleSession channel) {
171        channel.associated(ConsoleSessionDataStore.class).ifPresent(
172            psess -> psess.onConsolePrepared(event, channel));
173    }
174
175    /**
176     * Handle changed layout.
177     *
178     * @param event the event
179     * @param channel the channel
180     * @throws IOException Signals that an I/O exception has occurred.
181     */
182    @Handler
183    public void onConsoleLayoutChanged(ConsoleLayoutChanged event,
184            ConsoleSession channel) throws IOException {
185        Optional<ConsoleSessionDataStore> optDs = channel.associated(
186            ConsoleSessionDataStore.class);
187        if (optDs.isPresent()) {
188            optDs.get().onConsoleLayoutChanged(event, channel);
189        }
190    }
191
192    /**
193     * Stores the data for the portal session.
194     */
195    @SuppressWarnings("PMD.CommentRequired")
196    private class ConsoleSessionDataStore {
197
198        private final String storagePath;
199        private Map<String, Object> persisted;
200
201        public ConsoleSessionDataStore(Session session) {
202            storagePath = "/"
203                + WebConsoleUtils.userFromSession(session)
204                    .map(UserPrincipal::toString).orElse("")
205                + "/" + KVStoreBasedConsolePolicy.class.getName();
206        }
207
208        public void onConsoleReady(ConsoleReady event, IOSubchannel channel)
209                throws InterruptedException {
210            if (persisted != null) {
211                return;
212            }
213            KeyValueStoreQuery query = new KeyValueStoreQuery(
214                storagePath, channel);
215            fire(query, channel);
216        }
217
218        public void onKeyValueStoreData(
219                KeyValueStoreData event, IOSubchannel channel)
220                throws JsonDecodeException {
221            if (!event.event().query().equals(storagePath)) {
222                return;
223            }
224            String data = event.data().get(storagePath);
225            if (data != null) {
226                JsonBeanDecoder decoder = JsonBeanDecoder.create(data);
227                @SuppressWarnings({ "unchecked", "PMD.LooseCoupling" })
228                Class<Map<String, Object>> cls
229                    = (Class<Map<String, Object>>) (Class<?>) HashMap.class;
230                persisted = decoder.readObject(cls);
231            }
232        }
233
234        @SuppressWarnings({ "PMD.DataflowAnomalyAnalysis",
235            "PMD.AvoidInstantiatingObjectsInLoops" })
236        public void onConsolePrepared(
237                ConsolePrepared event, IOSubchannel channel) {
238            if (persisted == null) {
239                // Retrieval was not successful
240                persisted = new HashMap<>();
241            }
242            // Make sure data is consistent
243            @SuppressWarnings("unchecked")
244            List<String> previewLayout = (List<String>) persisted
245                .computeIfAbsent("previewLayout",
246                    newKey -> {
247                        return Collections.emptyList();
248                    });
249            @SuppressWarnings("unchecked")
250            List<String> tabsLayout = (List<String>) persisted.computeIfAbsent(
251                "tabsLayout", newKey -> {
252                    return Collections.emptyList();
253                });
254            JsonObject xtraInfo = (JsonObject) persisted.computeIfAbsent(
255                "xtraInfo", newKey -> {
256                    return JsonObject.create();
257                });
258
259            // Update layout
260            channel.respond(new LastConsoleLayout(
261                previewLayout, tabsLayout, xtraInfo));
262
263            // Restore portlets
264            for (String portletId : tabsLayout) {
265                fire(new RenderConletRequest(
266                    event.event().renderSupport(), portletId,
267                    RenderMode.asSet(RenderMode.View)), channel);
268            }
269            for (String portletId : previewLayout) {
270                fire(new RenderConletRequest(
271                    event.event().renderSupport(), portletId,
272                    RenderMode.asSet(RenderMode.Preview,
273                        RenderMode.Foreground)),
274                    channel);
275            }
276        }
277
278        public void onConsoleLayoutChanged(ConsoleLayoutChanged event,
279                IOSubchannel channel) throws IOException {
280            persisted.put("previewLayout", event.previewLayout());
281            persisted.put("tabsLayout", event.tabsLayout());
282            persisted.put("xtraInfo", event.xtraInfo());
283
284            // Now store.
285            JsonBeanEncoder encoder = JsonBeanEncoder.create();
286            encoder.writeObject(persisted);
287            fire(new KeyValueStoreUpdate()
288                .update(storagePath, encoder.toJson()), channel);
289        }
290
291    }
292}
