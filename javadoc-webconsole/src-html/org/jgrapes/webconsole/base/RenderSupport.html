019package org.jgrapes.webconsole.base;
020
021import java.net.URI;
022
023import org.jgrapes.webconsole.base.events.ConletResourceRequest;
024import org.jgrapes.webconsole.base.events.PageResourceRequest;
025
026/**
027 * Provides support for creating URIs in the web console scope that
028 * are forwarded to components listening on the web console channel.  
029 */
030public interface RenderSupport {
031
032    /**
033     * Returns the web console library URI.
034     *
035     * @param uri the uri
036     * @return the uri
037     */
038    URI consoleBaseResource(URI uri);
039
040    /**
041     * Convenience method that converts the path to an URI
042     * before calling {@link #consoleBaseResource(URI)}.
043     * 
044     * @param path the path 
045     * @return the resulting URI
046     */
047    default URI consoleBaseResource(String path) {
048        return consoleBaseResource(WebConsoleUtils.uriFromPath(path));
049    }
050
051    /**
052     * Create a reference to a resource provided by the web console.
053     * 
054     * @param uri the URI  
055     * @return the resulting URI
056     */
057    URI consoleResource(URI uri);
058
059    /**
060     * Convenience method that converts the path to an URI
061     * before calling {@link #consoleResource(URI)}.
062     * 
063     * @param path the path 
064     * @return the resulting URI
065     */
066    default URI consoleResource(String path) {
067        return consoleResource(WebConsoleUtils.uriFromPath(path));
068    }
069
070    /**
071     * Create a reference to a resource provided by a page resource
072     * provider. Requesting the resulting URI results in a 
073     * {@link PageResourceRequest}.
074     * 
075     * @param uri the URI made available as
076     * {@link PageResourceRequest#resourceUri()}  
077     * @return the resulting URI
078     */
079    URI pageResource(URI uri);
080
081    /**
082     * Convenience method that converts the path to an URI
083     * before calling {@link #pageResource(URI)}.
084     * 
085     * @param path the path 
086     * @return the resulting URI
087     */
088    default URI pageResource(String path) {
089        return pageResource(WebConsoleUtils.uriFromPath(path));
090    }
091
092    /**
093     * Create a reference to a resource provided by a portlet
094     * of the given type. Requesting the resulting URI results
095     * in a {@link ConletResourceRequest}.
096     * 
097     * @param portletType the portlet type
098     * @param uri the URI made available as 
099     * {@link ConletResourceRequest#resourceUri()}
100     * @return the resulting URI
101     */
102    URI conletResource(String portletType, URI uri);
103
104    /**
105     * Convenience method that converts the path to an URI
106     * before calling {@link #conletResource(String, URI)}.
107     * 
108     * @param conletType the portlet type
109     * @param path the path 
110     * @return the resulting URI
111     */
112    default URI conletResource(String conletType, String path) {
113        return conletResource(conletType, WebConsoleUtils.uriFromPath(path));
114    }
115
116    /**
117     * Indicates if minified resources should be used.
118     * 
119     * @return the setting
120     */
121    boolean useMinifiedResources();
122}
