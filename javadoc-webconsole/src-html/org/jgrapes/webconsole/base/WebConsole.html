063/**
064 * Provides the web console component related part of the console.
065 */
066public class WebConsole extends Component {
067
068    private static final Logger logger
069        = Logger.getLogger(WebConsole.class.getName());
070
071    private ConsoleWeblet view;
072
073    /**
074     * @param componentChannel
075     */
076    WebConsole(Channel componentChannel) {
077        super(componentChannel);
078    }
079
080    /* default */ void setView(ConsoleWeblet view) {
081        this.view = view;
082        MBeanView.addConsole(this);
083    }
084
085    /**
086     * Handle JSON input.
087     *
088     * @param event the event
089     * @param channel the channel
090     * @throws InterruptedException the interrupted exception
091     * @throws IOException Signals that an I/O exception has occurred.
092     */
093    @Handler
094    @SuppressWarnings("PMD.DataflowAnomalyAnalysis")
095    public void onJsonInput(JsonInput event, ConsoleSession channel)
096            throws InterruptedException, IOException {
097        // Send events to web console components on console's channel
098        JsonArray params = event.request().params();
099        switch (event.request().method()) {
100        case "consoleReady": {
101            fire(new ConsoleReady(view.renderSupport()), channel);
102            break;
103        }
104        case "addConlet": {
105            fire(new AddConletRequest(view.renderSupport(),
106                params.asString(0), params.asArray(1).stream().map(
107                    value -> RenderMode.valueOf((String) value))
108                    .collect(Collectors.toSet())),
109                channel);
110            break;
111        }
112        case "deleteConlet": {
113            fire(new DeleteConletRequest(
114                view.renderSupport(), params.asString(0)), channel);
115            break;
116        }
117        case "consoleLayout": {
118            List<String> previewLayout = params.asArray(0).stream().map(
119                value -> (String) value).collect(Collectors.toList());
120            List<String> tabsLayout = params.asArray(1).stream().map(
121                value -> (String) value).collect(Collectors.toList());
122            JsonObject xtraInfo = (JsonObject) params.get(2);
123            fire(new ConsoleLayoutChanged(
124                previewLayout, tabsLayout, xtraInfo), channel);
125            break;
126        }
127        case "renderConlet": {
128            fire(new RenderConletRequest(view.renderSupport(),
129                params.asString(0),
130                params.asArray(1).stream().map(
131                    value -> RenderMode.valueOf((String) value))
132                    .collect(Collectors.toSet())),
133                channel);
134            break;
135        }
136        case "setLocale": {
137            fire(new SetLocale(view.renderSupport(),
138                Locale.forLanguageTag(params.asString(0)),
139                params.asBoolean(1)), channel);
140            break;
141        }
142        case "notifyConletModel": {
143            fire(new NotifyConletModel(view.renderSupport(),
144                params.asString(0), params.asString(1),
145                params.size() <= 2
146                    ? JsonArray.EMPTY_ARRAY
147                    : params.asArray(2)),
148                channel);
149            break;
150        }
151        default:
152            // Ignore unknown
153            break;
154        }
155    }
156
157    /**
158     * Handle network configured condition.
159     *
160     * @param event the event
161     * @param channel the channel
162     * @throws InterruptedException the interrupted exception
163     * @throws IOException Signals that an I/O exception has occurred.
164     */
165    @Handler
166    public void onConsoleConfigured(
167            ConsoleConfigured event, ConsoleSession channel)
168            throws InterruptedException, IOException {
169        channel.respond(new SimpleConsoleCommand("consoleConfigured"));
170    }
171
172    /**
173     * Fallback handler that sends a {@link DeleteConlet} event 
174     * if the {@link RenderConletRequest} event has not been handled
175     * successfully.
176     *
177     * @param event the event
178     * @param channel the channel
179     */
180    @Handler(priority = -1000000)
181    public void onRenderConlet(
182            RenderConletRequest event, ConsoleSession channel) {
183        if (!event.hasBeenRendered()) {
184            channel.respond(new DeleteConlet(event.conletId()));
185        }
186    }
187
188    /**
189     * Discard all console sessions on stop.
190     *
191     * @param event the event
192     */
193    @Handler
194    public void onStop(Stop event) {
195        for (ConsoleSession ps : ConsoleSession.byConsole(this)) {
196            ps.discard();
197        }
198    }

200    /**
201     * The MBeans view of a console.
202     */
203    @SuppressWarnings({ "PMD.CommentRequired", "PMD.AvoidDuplicateLiterals" })
204    public interface ConsoleMXBean {
205
206        @SuppressWarnings("PMD.CommentRequired")
207        class ConsoleSessionInfo {
208
209            private final ConsoleSession session;
210
211            public ConsoleSessionInfo(ConsoleSession session) {
212                super();
213                this.session = session;
214            }
215
216            public String getChannel() {
217                return session.upstreamChannel().toString();
218            }
219
220            public String getExpiresAt() {
221                return session.expiresAt().atZone(ZoneId.systemDefault())
222                    .toString();
223            }
224        }
225
226        String getComponentPath();
227
228        String getPrefix();
229
230        boolean isUseMinifiedResources();
231
232        void setUseMinifiedResources(boolean useMinifiedResources);
233
234        SortedMap<String, ConsoleSessionInfo> getConsoleSessions();
235    }
236
237    @SuppressWarnings("PMD.CommentRequired")
238    public static class WebConsoleInfo implements ConsoleMXBean {
239
240        private static MBeanServer mbs
241            = ManagementFactory.getPlatformMBeanServer();
242
243        private ObjectName mbeanName;
244        private final WeakReference<WebConsole> consoleRef;
245
246        public WebConsoleInfo(WebConsole console) {
247            try {
248                mbeanName = new ObjectName("org.jgrapes.webconsole:type="
249                    + WebConsole.class.getSimpleName() + ",name="
250                    + ObjectName.quote(Components.simpleObjectName(console)
251                        + " (" + console.view.prefix().toString() + ")"));
252            } catch (MalformedObjectNameException e) {
253                // Should not happen
254                logger.log(Level.WARNING, e.getMessage(), e);
255            }
256            consoleRef = new WeakReference<>(console);
257            try {
258                mbs.unregisterMBean(mbeanName);
259            } catch (Exception e) { // NOPMD
260                // Just in case, should not work
261            }
262            try {
263                mbs.registerMBean(this, mbeanName);
264            } catch (InstanceAlreadyExistsException | MBeanRegistrationException
265                    | NotCompliantMBeanException e) {
266                // Should not happen
267                logger.log(Level.WARNING, e.getMessage(), e);
268            }
269        }
270
271        public Optional<WebConsole> console() {
272            WebConsole console = consoleRef.get();
273            if (console == null) {
274                try {
275                    mbs.unregisterMBean(mbeanName);
276                } catch (Exception e) { // NOPMD
277                    // Should work.
278                }
279            }
280            return Optional.ofNullable(console);
281        }
282
283        @Override
284        public String getComponentPath() {
285            return console().map(mgr -> mgr.componentPath())
286                .orElse("<removed>");
287        }
288
289        @Override
290        public String getPrefix() {
291            return console().map(
292                console -> console.view.prefix().toString())
293                .orElse("<unknown>");
294        }
295
296        @Override
297        public boolean isUseMinifiedResources() {
298            return console().map(
299                console -> console.view.useMinifiedResources())
300                .orElse(false);
301        }
302
303        @Override
304        public void setUseMinifiedResources(boolean useMinifiedResources) {
305            console().ifPresent(console -> console.view.setUseMinifiedResources(
306                useMinifiedResources));
307        }
308
309        @Override
310        @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
311        public SortedMap<String, ConsoleSessionInfo> getConsoleSessions() {
312            SortedMap<String, ConsoleSessionInfo> result = new TreeMap<>();
313            console().ifPresent(console -> {
314                for (ConsoleSession ps : ConsoleSession.byConsole(console)) {
315                    result.put(Components.simpleObjectName(ps),
316                        new ConsoleSessionInfo(ps));
317                }
318            });
319            return result;
320        }
321    }

323    /**
324     * An MBean interface for getting information about all consoles.
325     * 
326     * There is currently no summary information. However, the (periodic)
327     * invocation of {@link WebConsoleSummaryMXBean#getConsoles()} ensures
328     * that entries for removed {@link WebConsole}s are unregistered.
329     */
330    @SuppressWarnings("PMD.CommentRequired")
331    public interface WebConsoleSummaryMXBean {
332
333        Set<ConsoleMXBean> getConsoles();
334
335    }
336
337    /**
338     * Provides an MBean view of the console.
339     */
340    @SuppressWarnings("PMD.CommentRequired")
341    private static class MBeanView implements WebConsoleSummaryMXBean {
342
343        private static Set<WebConsoleInfo> consoleInfos = new HashSet<>();
344
345        public static void addConsole(WebConsole console) {
346            synchronized (consoleInfos) {
347                consoleInfos.add(new WebConsoleInfo(console));
348            }
349        }
350
351        @Override
352        public Set<ConsoleMXBean> getConsoles() {
353            Set<WebConsoleInfo> expired = new HashSet<>();
354            synchronized (consoleInfos) {
355                for (WebConsoleInfo consoleInfo : consoleInfos) {
356                    if (!consoleInfo.console().isPresent()) {
357                        expired.add(consoleInfo);
358                    }
359                }
360                consoleInfos.removeAll(expired);
361            }
362            @SuppressWarnings("unchecked")
363            Set<ConsoleMXBean> result
364                = (Set<ConsoleMXBean>) (Object) consoleInfos;
365            return result;
366        }
367    }
368
369    static {
370        try {
371            MBeanServer mbs = ManagementFactory.getPlatformMBeanServer();
372            ObjectName mxbeanName
373                = new ObjectName("org.jgrapes.webconsole:type="
374                    + WebConsole.class.getSimpleName() + "s");
375            mbs.registerMBean(new MBeanView(), mxbeanName);
376        } catch (MalformedObjectNameException | InstanceAlreadyExistsException
377                | MBeanRegistrationException | NotCompliantMBeanException e) {
378            // Should not happen
379            logger.log(Level.WARNING, e.getMessage(), e);
380        }
381    }
