019package org.jgrapes.examples.io.consoleinput;
020
021import org.jgrapes.core.Component;
022import org.jgrapes.core.Components;
023import org.jgrapes.core.annotation.Handler;
024import org.jgrapes.core.events.Stop;
025import org.jgrapes.io.InputStreamMonitor;
026import org.jgrapes.io.events.Input;
027import org.jgrapes.io.util.ManagedByteBuffer;
028
029/**
030 * An application that echos data from the console until a line
031 * with "QUIT" is received.
032 * 
033 * @startuml ConsoleEchoApp.svg
034 * 
035 * object "app: EchoUntilQuit" as app
036 * object "console: InputStreamMonitor" as console
037 * object "channel: Channel" as channel
038 * 
039 * app -- console
040 * app .. channel
041 * console .up. channel
042 * 
043 * note "Besides the (structural) parent-child \n\
044 *   association the two components are \n\
045 *   logically connected by the channel" as ChannelNote
046 * 
047 * channel .. ChannelNote
048 * 
049 * @enduml
050 */
051public class EchoUntilQuit extends Component {
052
053        @Handler
054        public void onInput(Input<ManagedByteBuffer> event) {
055                byte[] bytes = new byte[event.remaining()];
056                event.buffer().backingBuffer().get(bytes);
057                String data = new String(bytes);
058                System.out.print(data);
059                if (data.trim().equals("QUIT")) {
060                        fire (new Stop());
061                }
062        }
063        
064        /**
065         * The main method.
066         *
067         * @param args the arguments
068         * @throws InterruptedException the interrupted exception
069         */
070        public static void main(String[] args) throws InterruptedException {
071                EchoUntilQuit app = new EchoUntilQuit();
072                app.attach(new InputStreamMonitor(app.channel(), System.in));
073                Components.start(app);
074                Components.awaitExhaustion();
075                System.exit(0);
076        }
077}
