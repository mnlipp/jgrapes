019package org.jgrapes.http;
020
021import java.time.Duration;
022import java.time.Instant;
023import java.util.Iterator;
024import java.util.LinkedHashMap;
025import java.util.Map.Entry;
026import java.util.Optional;
027
028import org.jgrapes.core.Channel;
029
030/**
031 * A in memory session manager. 
032 */
033public class InMemorySessionManager extends SessionManager {
034
035        @SuppressWarnings("serial")
036        private LinkedHashMap<String,Session> sessionsById 
037                = new LinkedHashMap<String,Session>(16, 0.75f, true) {
038
039                        @Override
040                        protected boolean removeEldestEntry(Entry<String, Session> eldest) {
041                                return maxSessions() > 0 && size() > maxSessions();
042                        }
043        };
044        
045        /**
046         * Creates a new session manager with its channel set to
047         * itself and the scope set to "/".
048         */
049        public InMemorySessionManager() {
050                this("/");
051        }
052
053        /**
054         * Creates a new session manager with its channel set to
055         * itself and the scope to scope.
056         * 
057         * @param scope the scope
058         */
059        public InMemorySessionManager(String scope) {
060                super(scope);
061        }
062
063        /**
064         * Creates a new session manager with its channel set to
065         * the given channel and the scope to "/".
066         * 
067         * @param componentChannel the component channel
068         */
069        public InMemorySessionManager(Channel componentChannel) {
070                super(componentChannel);
071        }
072
073        protected Session createSession(String sessionId) {
074                Session session = new InMemorySession(sessionId);
075                Instant now = Instant.now();
076                synchronized (this) {
077                        if (absoluteTimeout() > 0) {
078                                // Quick check for sessions that are too old
079                                for (Iterator<Entry<String, Session>> iter = sessionsById
080                                        .entrySet().iterator(); iter.hasNext();) {
081                                        Entry<String, Session> entry = iter.next();
082                                        if (Duration.between(entry.getValue().createdAt(), now)
083                                                .getSeconds() > absoluteTimeout()) {
084                                                iter.remove();
085                                        } else {
086                                                break;
087                                        }
088                                }
089                        }
090                        if (sessionsById.size() >= maxSessions() && absoluteTimeout() > 0) {
091                                // Thorough search for sessions that are too old
092                                sessionsById.entrySet().removeIf(entry -> {
093                                        return Duration.between(entry.getValue().createdAt(),
094                                                                now).getSeconds() > absoluteTimeout()
095                                                        || Duration.between(entry.getValue().lastUsedAt(),
096                                                                now).getSeconds() > idleTimeout();
097                                });
098                        }
099                        sessionsById.put(sessionId, session);
100                }
101                return session;
102        }
103        
104        protected Optional<Session> lookupSession(String sessionId) {
105                return Optional.ofNullable(sessionsById.get(sessionId));
106        }
107        
108        protected void removeSession(String sessionId) {
109                synchronized (this) {
110                        sessionsById.remove(sessionId);
111                }
112        }
113}
