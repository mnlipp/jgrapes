019package org.jgrapes.http;
020
021import java.net.HttpCookie;
022import java.security.SecureRandom;
023import java.time.Duration;
024import java.time.Instant;
025import java.util.Iterator;
026import java.util.LinkedHashMap;
027import java.util.Map.Entry;
028import java.util.Optional;
029
030import org.jdrupes.httpcodec.protocols.http.HttpField;
031import org.jdrupes.httpcodec.protocols.http.HttpRequest;
032import org.jdrupes.httpcodec.protocols.http.HttpResponse;
033import org.jdrupes.httpcodec.types.CacheControlDirectives;
034import org.jdrupes.httpcodec.types.Converters;
035import org.jdrupes.httpcodec.types.CookieList;
036import org.jdrupes.httpcodec.types.Directive;
037import org.jgrapes.core.Channel;
038import org.jgrapes.core.Component;
039import org.jgrapes.core.annotation.Handler;
040import org.jgrapes.core.internal.EventBase;
041import org.jgrapes.http.events.DiscardSession;
042import org.jgrapes.http.events.Request;
043
044/**
045 * A in memory session manager. The event is associated with a 
046 * {@link Session} object using `Session.class` as association identifier.
047 * 
048 * @see EventBase#setAssociated(Object, Object)
049 * @see "[OWASP Session Management Cheat Sheet](https://www.owasp.org/index.php/Session_Management_Cheat_Sheet)"
050 */
051public class InMemorySessionManager extends Component {
052
053        private static SecureRandom secureRandom = new SecureRandom();
054        
055        private String idName = "id";
056        @SuppressWarnings("serial")
057        private LinkedHashMap<String,Session> sessionsById 
058                = new LinkedHashMap<String,Session>(16, 0.75f, true) {
059
060                        @Override
061                        protected boolean removeEldestEntry(Entry<String, Session> eldest) {
062                                return InMemorySessionManager.this.maxSessions > 0
063                                                && size() > InMemorySessionManager.this.maxSessions;
064                        }
065        };
066        private int absoluteTimeout = 9*60*60;
067        private int idleTimeout = 30*60;
068        private int maxSessions = 1000;
069        
070        /**
071         * Creates a new session manager with its channel set to
072         * itself.
073         */
074        public InMemorySessionManager() {
075        }
076
077        /**
078         * Creates a new session manager with its channel set to the given 
079         * channel.
080         * 
081         * @param componentChannel the channel
082         */
083        public InMemorySessionManager(Channel componentChannel) {
084                super(componentChannel);
085        }
086
087        /**
088         * The name used for the session id cookie. Defaults to "`id`".
089         * 
090         * @return the id name
091         */
092        public String idName() {
093                return idName;
094        }
095
096        /**
097         * @param idName the id name to set
098         * 
099         * @return the session manager for easy chaining
100         */
101        public InMemorySessionManager setIdName(String idName) {
102                this.idName = idName;
103                return this;
104        }
105
106        /**
107         * Set the maximum number of sessions. If the value is zero or less,
108         * an unlimited number of sessions is supported. The default value
109         * is 1000.
110         * 
111         * If adding a new session would exceed the limit, first all
112         * sessions older than {@link #absoluteTimeout()} are removed.
113         * If this doesn't free a slot, the least recently used session
114         * is removed.
115         * 
116         * @param maxSessions the maxSessions to set
117         * @return the session manager for easy chaining
118         */
119        public InMemorySessionManager setMaxSessions(int maxSessions) {
120                this.maxSessions = maxSessions;
121                return this;
122        }
123
124        /**
125         * @return the maxSessions
126         */
127        public int maxSessions() {
128                return maxSessions;
129        }
130
131        /**
132         * Sets the absolute timeout for a session in seconds. The absolute
133         * timeout is the time after which a session is invalidated (relative
134         * to its creation time). Defaults to 9 hours. Zero or less disables
135         * the timeout.
136         * 
137         * @param absoluteTimeout the absolute timeout
138         * @return the session manager for easy chaining
139         */
140        public InMemorySessionManager setAbsoluteTimeout(int absoluteTimeout) {
141                this.absoluteTimeout = absoluteTimeout;
142                return this;
143        }
144
145        /**
146         * @return the absolute session timeout (in seconds)
147         */
148        public int absoluteTimeout() {
149                return absoluteTimeout;
150        }
151
152        /**
153         * Sets the idle timeout for a session in seconds. Defaults to 30 minutes.
154         * Zero or less disables the timeout. 
155         * 
156         * @param idleTimeout the absolute timeout
157         * @return the session manager for easy chaining
158         */
159        public InMemorySessionManager setIdleTimeout(int idleTimeout) {
160                this.idleTimeout = idleTimeout;
161                return this;
162        }
163
164        /**
165         * @return the idle timeout (in seconds)
166         */
167        public int idleTimeout() {
168                return idleTimeout;
169        }
170
171        /**
172         * Associates the event with a {@link Session} object
173         * using `Session.class` as association identifier.
174         * 
175         * @param event the event
176         */
177        @Handler(priority=1000)
178        public void onRequest(Request event) {
179                final HttpRequest request = event.request();
180                Optional<String> requestedSessionId = request.findValue(
181                        HttpField.COOKIE, Converters.COOKIE_LIST)
182                        .flatMap(cookies -> cookies.stream().filter(
183                                cookie -> cookie.getName().equals(idName))
184                                .findFirst().map(HttpCookie::getValue));
185                if (requestedSessionId.isPresent()) {
186                        String sessionId = requestedSessionId.get();
187                        synchronized(InMemorySessionManager.this) {
188                                Session session = sessionsById.get(sessionId);
189                                Instant now = Instant.now();
190                                if (session != null) {
191                                        if ((absoluteTimeout <= 0
192                                                        || Duration.between(session.createdAt(), 
193                                                                        now).getSeconds() < absoluteTimeout)
194                                                && (idleTimeout <= 0
195                                                        || Duration.between(session.lastUsedAt(),
196                                                                        now).getSeconds() < idleTimeout)) {
197                                                event.setAssociated(Session.class, session);
198                                                session.updateLastUsedAt();
199                                                return;
200                                        }
201                                        // Invalidate, too old 
202                                        sessionsById.remove(sessionId);
203                                }
204                        }
205                }
206                String sessionId = createSessionId(request.response().get());
207                Session session = createSession(sessionId);
208                event.setAssociated(Session.class, session);
209        }
210
211        private Session createSession(String sessionId) {
212                Session session = new Session(sessionId);
213                Instant now = Instant.now();
214                synchronized (this) {
215                        if (absoluteTimeout > 0) {
216                                // Quick check for sessions that are too old
217                                for (Iterator<Entry<String, Session>> iter = sessionsById
218                                        .entrySet().iterator(); iter.hasNext();) {
219                                        Entry<String, Session> entry = iter.next();
220                                        if (Duration.between(entry.getValue().createdAt(), now)
221                                                .getSeconds() > absoluteTimeout) {
222                                                iter.remove();
223                                        } else {
224                                                break;
225                                        }
226                                }
227                        }
228                        if (sessionsById.size() >= maxSessions && absoluteTimeout > 0) {
229                                // Thorough search for sessions that are too old
230                                sessionsById.entrySet().removeIf(entry -> {
231                                        return Duration.between(entry.getValue().createdAt(),
232                                                                now).getSeconds() > absoluteTimeout
233                                                        || Duration.between(entry.getValue().lastUsedAt(),
234                                                                now).getSeconds() > idleTimeout;
235                                });
236                        }
237                        sessionsById.put(sessionId, session);
238                }
239                return session;
240        }
241        
242        private String createSessionId(HttpResponse response) {
243                StringBuilder sessionIdBuilder = new StringBuilder();
244                byte[] bytes = new byte[16];
245                secureRandom.nextBytes(bytes);
246                for (byte b: bytes) {
247                        sessionIdBuilder.append(Integer.toHexString(b & 0xff));
248                }
249                String sessionId = sessionIdBuilder.toString();
250                HttpCookie sessionCookie = new HttpCookie(idName, sessionId);
251                sessionCookie.setHttpOnly(true);
252                response.computeIfAbsent(HttpField.SET_COOKIE, CookieList::new)
253                        .value().add(sessionCookie);
254                response.computeIfAbsent(
255                                HttpField.CACHE_CONTROL, CacheControlDirectives::new)
256                        .value().add(new Directive("no-cache", "SetCookie, Set-Cookie2"));
257                return sessionId;
258        }
259        
260        /**
261         * Discards the given session.
262         * 
263         * @param event the event
264         */
265        @Handler(channels=Channel.class)
266        public void discard(DiscardSession event) {
267                synchronized (this) {
268                        sessionsById.remove(event.session().id());
269                }
270        }
271}
