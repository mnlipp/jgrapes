019package org.jgrapes.http.events;
020
021import java.net.URI;
022
023import org.jdrupes.httpcodec.protocols.http.HttpRequest;
024import org.jgrapes.http.HttpServer;
025import org.jgrapes.io.events.Opened;
026
027/**
028 * Indicates that a component has accepted a {@link GetRequest} with
029 * a header that requested an upgrade to the WebSocket protocol.
030 * 
031 * The {@link HttpServer} component listens for such events and 
032 * automatically creates the required {@link Response} event.
033 */
034public class WebSocketAccepted extends Opened {
035
036        private URI resourceName;
037        private Request requestEvent;
038        
039        /**
040         * Creates a new event. The request event passed in as parameter
041         * is used by the {@link HttpServer} to build the response message
042         * and link an existing session to the web socket.
043         * 
044         * To be precise, the {@link HttpServer} retrieves the {@link HttpRequest}
045         * from the request event and uses the prepared response provided by 
046         * {@link HttpRequest#response()} to build the response. The default
047         * information contained in this prepared response is sufficient to
048         * build the actual response. If required, the accepting component 
049         * can add special header fields to the prepared response.
050         * 
051         * @param request the base response data
052         */
053        public WebSocketAccepted(Request request) {
054                this.requestEvent = request;
055        }
056
057        /**
058         * Returns the resource for which the socket was opened.
059         * 
060         * @return the value
061         */
062        public URI resourceName() {
063                return resourceName;
064        }
065
066        /**
067         * Returns the original request. 
068         * 
069         * @return the value
070         */
071        public Request requestEvent() {
072                return requestEvent;
073        }
074}
