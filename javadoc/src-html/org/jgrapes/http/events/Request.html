019package org.jgrapes.http.events;
020
021import java.net.URI;
022import java.net.URISyntaxException;
023import java.util.StringTokenizer;
024
025import org.jdrupes.httpcodec.protocols.http.HttpRequest;
026import org.jgrapes.core.Channel;
027import org.jgrapes.core.CompletedEvent;
028import org.jgrapes.core.Components;
029import org.jgrapes.core.Event;
030import org.jgrapes.core.internal.Common;
031import org.jgrapes.http.ResourcePattern;
032
033/**
034 * @author Michael N. Lipp
035 *
036 */
037public class Request extends Event<Void> {
038
039        public static class Completed extends CompletedEvent<Request> {
040        }
041        
042        private HttpRequest request;
043        private URI uri;
044        private MatchValue matchValue;
045        private URI matchUri;
046        
047        /**
048         * Creates a new request event with the associated {@link Completed}
049         * event.
050         * 
051         * @param protocol the protocol as reported by {@link #requestUri()}
052         * @param request the request data
053         * @param matchLevels the number of elements from the request path
054         * to use in the match value (see {@link #matchValue})
055         * @param channels the channels associated with this event
056         */
057        public Request(String protocol, HttpRequest request, 
058                        int matchLevels, Channel... channels) {
059                super(new Completed(), channels);
060                this.request = request;
061                try {
062                        URI headerInfo = new URI(protocol, null, 
063                                        request.host(), request.port(), null, null, null);
064                        uri = headerInfo.resolve(request.requestUri());
065                        StringTokenizer st = new StringTokenizer(uri.getPath(), "/");
066                        StringBuilder mp = new StringBuilder();
067                        for (int i = 0; i < matchLevels && st.hasMoreTokens(); i++) {
068                                mp.append("/");
069                                mp.append(st.nextToken());
070                        }
071                        if (st.hasMoreTokens()) {
072                                mp.append("/**");
073                        }
074                        String matchPath = mp.toString();
075                        matchUri = new URI(uri.getScheme(), null, uri.getHost(),
076                                uri.getPort(), uri.getPath(), null, null);
077                        matchValue = new MatchValue(getClass(), 
078                                        (uri.getScheme() == null ? "" 
079                                                : (uri.getScheme() + "://"))
080                                         + (uri.getHost() == null ? "" 
081                                                : (uri.getHost() + (uri.getPort() == -1 ? "" 
082                                                        : (":" + uri.getPort()))))
083                                         + matchPath);
084                } catch (URISyntaxException e) {
085                        throw new IllegalArgumentException();
086                }
087        }
088
089        /**
090         * Returns the "raw" request as provided by the HTTP decoder.
091         * 
092         * @return the request
093         */
094        public HttpRequest request() {
095                return request;
096        }
097
098        /**
099         * Returns a URI that is built from the information provided by the
100         * decoder.
101         * 
102         * @return the URI
103         */
104        public URI requestUri() {
105                return uri;
106        }
107        
108        /**
109         * The match value consists of the event class and a URI.
110         * The URI is similar to the request URI but its path elements
111         * are shortened as specified in the constructor.
112         * 
113         * As the match value is used as key in a map that speeds up
114         * the lookup of handlers, having the complete URI in the match
115         * value would inflate this map.
116         * 
117         * @see org.jgrapes.core.Event#defaultCriterion()
118         */
119        @Override
120        public Object defaultCriterion() {
121                return matchValue;
122        }
123
124        /* (non-Javadoc)
125         * @see org.jgrapes.core.Event#isMatchedBy(java.lang.Object)
126         */
127        @Override
128        public boolean isEligibleFor(Object value) {
129                if (!(value instanceof MatchValue)) {
130                        return super.isEligibleFor(value);
131                }
132                MatchValue mv = (MatchValue)value;
133                if (!mv.type.isAssignableFrom(matchValue.type)) {
134                        return false;
135                }
136                if (mv.resource instanceof ResourcePattern) {
137                        return ((ResourcePattern)mv.resource).matches(matchUri) >= 0;
138                }
139                return mv.resource.equals(matchValue.resource);
140        }
141
142        /* (non-Javadoc)
143         * @see java.lang.Object#toString()
144         */
145        @Override
146        public String toString() {
147                StringBuilder builder = new StringBuilder();
148                builder.append(Components.objectName(this));
149                builder.append(" [\"");
150                String path = request.requestUri().getPath();
151                if (path.length() > 15) {
152                        builder.append("...");
153                        builder.append(path.substring(path.length() - 12));
154                } else {
155                        builder.append(path);
156                }
157                builder.append("\"");
158                if (channels != null) {
159                        builder.append(", channels=");
160                        builder.append(Common.channelsToString(channels));
161                }
162                builder.append("]");
163                return builder.toString();
164        }
165        
166        public static Object createMatchValue(
167                        Class<?> type, ResourcePattern resource) {
168                return new MatchValue(type, resource);
169        }
170        
171        private static class MatchValue {
172                private Class<?> type;
173                private Object resource;
174
175                /**
176                 * @param type
177                 * @param resource
178                 */
179                public MatchValue(Class<?> type, Object resource) {
180                        super();
181                        this.type = type;
182                        this.resource = resource;
183                }
184
185                /* (non-Javadoc)
186                 * @see java.lang.Object#hashCode()
187                 */
188                @Override
189                public int hashCode() {
190                        final int prime = 31;
191                        int result = 1;
192                        result = prime * result
193                                + ((resource == null) ? 0 : resource.hashCode());
194                        result = prime * result + ((type == null) ? 0 : type.hashCode());
195                        return result;
196                }
197
198                /* (non-Javadoc)
199                 * @see java.lang.Object#equals(java.lang.Object)
200                 */
201                @Override
202                public boolean equals(Object obj) {
203                        if (this == obj) {
204                                return true;
205                        }
206                        if (obj == null) {
207                                return false;
208                        }
209                        if (getClass() != obj.getClass()) {
210                                return false;
211                        }
212                        MatchValue other = (MatchValue) obj;
213                        if (resource == null) {
214                                if (other.resource != null) {
215                                        return false;
216                                }
217                        } else if (!resource.equals(other.resource)) {
218                                return false;
219                        }
220                        if (type == null) {
221                                if (other.type != null) {
222                                        return false;
223                                }
224                        } else if (!type.equals(other.type)) {
225                                return false;
226                        }
227                        return true;
228                }
229                
230        }
231}
