019package org.jgrapes.http.events;
020
021import java.net.URI;
022import java.net.URISyntaxException;
023import java.util.Iterator;
024
025import org.jdrupes.httpcodec.protocols.http.HttpRequest;
026import org.jgrapes.core.Channel;
027import org.jgrapes.core.CompletionEvent;
028import org.jgrapes.core.Components;
029import org.jgrapes.core.Event;
030import org.jgrapes.http.ResourcePattern;
031import org.jgrapes.http.ResourcePattern.PathSpliterator;
032
033/**
034 * The base class for all HTTP requests such as {@link GetRequest},
035 * {@link PostRequest} etc. A result of `true` indicates that the 
036 * request has been processed, i.e. a response has been sent or will
037 * sent.
038 */
039public class Request extends Event<Boolean> {
040
041    @SuppressWarnings("PMD.AvoidFieldNameMatchingTypeName")
042    private final HttpRequest request;
043    private URI uri;
044    private MatchValue matchValue;
045    private URI matchUri;
046
047    /**
048     * The associated completion event.
049     */
050    public static class Completed extends CompletionEvent<Request> {
051
052        /**
053         * Instantiates a new event.
054         *
055         * @param monitoredEvent the monitored event
056         * @param channels the channels
057         */
058        public Completed(Request monitoredEvent, Channel... channels) {
059            super(monitoredEvent, channels);
060        }
061    }
062
063    /**
064     * Creates a new request event with the associated {@link Completed}
065     * event.
066     * 
067     * @param protocol the protocol as reported by {@link #requestUri()}
068     * @param request the request data
069     * @param matchLevels the number of elements from the request path
070     * to use in the match value (see {@link #matchValue})
071     * @param channels the channels associated with this event
072     */
073    @SuppressWarnings("PMD.UselessParentheses")
074    public Request(String protocol, HttpRequest request,
075            int matchLevels, Channel... channels) {
076        super(channels);
077        new Completed(this);
078        this.request = request;
079        try {
080            URI headerInfo = new URI(protocol, null,
081                request.host(), request.port(), null, null, null);
082            uri = headerInfo.resolve(request.requestUri());
083            Iterator<String> segs = PathSpliterator.stream(
084                uri.getPath()).skip(1).iterator();
085            StringBuilder pattern = new StringBuilder(20);
086            for (int i = 0; i < matchLevels && segs.hasNext(); i++) {
087                pattern.append('/')
088                    .append(segs.next());
089            }
090            if (segs.hasNext()) {
091                pattern.append("/**");
092            }
093            String matchPath = pattern.toString();
094            matchUri = new URI(uri.getScheme(), null, uri.getHost(),
095                uri.getPort(), uri.getPath(), null, null);
096            matchValue = new MatchValue(getClass(),
097                (uri.getScheme() == null ? ""
098                    : (uri.getScheme() + "://"))
099                    + (uri.getHost() == null ? ""
100                        : (uri.getHost() + (uri.getPort() == -1 ? ""
101                            : (":" + uri.getPort()))))
102                    + matchPath);
103        } catch (URISyntaxException e) {
104            throw new IllegalArgumentException(e);
105        }
106    }
107
108    /**
109     * Creates the appropriate derived request event type from
110     * a given {@link HttpRequest}.
111     * 
112     * @param request the HTTP request
113     * @param secure whether the request was received over a secure channel
114     * @param matchLevels the match levels
115     * @return the request event
116     */
117    public static Request fromHttpRequest(
118            HttpRequest request, boolean secure, int matchLevels) {
119        switch (request.method()) {
120        case "OPTIONS":
121            return new OptionsRequest(request, secure, matchLevels);
122        case "GET":
123            return new GetRequest(request, secure, matchLevels);
124        case "HEAD":
125            return new HeadRequest(request, secure, matchLevels);
126        case "POST":
127            return new PostRequest(request, secure, matchLevels);
128        case "PUT":
129            return new PutRequest(request, secure, matchLevels);
130        case "DELETE":
131            return new DeleteRequest(request, secure, matchLevels);
132        case "TRACE":
133            return new TraceRequest(request, secure, matchLevels);
134        case "CONNECT":
135            return new ConnectRequest(request, secure, matchLevels);
136        default:
137            return new Request(secure ? "https" : "http", request, matchLevels);
138        }
139    }
140
141    /**
142     * Returns the "raw" request as provided by the HTTP decoder.
143     * 
144     * @return the request
145     */
146    public HttpRequest httpRequest() {
147        return request;
148    }
149
150    /**
151     * Returns an absolute URI of the request that is built from the 
152     * information provided by the decoder.
153     * 
154     * @return the URI
155     */
156    public URI requestUri() {
157        return uri;
158    }
159
160    /**
161     * The match value consists of the event class and a URI.
162     * The URI is similar to the request URI but its path elements
163     * are shortened as specified in the constructor.
164     * 
165     * As the match value is used as key in a map that speeds up
166     * the lookup of handlers, having the complete URI in the match
167     * value would inflate this map.
168     * 
169     * @see org.jgrapes.core.Event#defaultCriterion()
170     */
171    @Override
172    public Object defaultCriterion() {
173        return matchValue;
174    }
175
176    /*
177     * (non-Javadoc)
178     * 
179     * @see org.jgrapes.core.Event#isMatchedBy(java.lang.Object)
180     */
181    @Override
182    public boolean isEligibleFor(Object value) {
183        if (!(value instanceof MatchValue)) {
184            return super.isEligibleFor(value);
185        }
186        MatchValue mval = (MatchValue) value;
187        if (!mval.type.isAssignableFrom(matchValue.type)) {
188            return false;
189        }
190        if (mval.resource instanceof ResourcePattern) {
191            return ((ResourcePattern) mval.resource).matches(matchUri) >= 0;
192        }
193        return mval.resource.equals(matchValue.resource);
194    }
195
196    /*
197     * (non-Javadoc)
198     * 
199     * @see java.lang.Object#toString()
200     */
201    @Override
202    @SuppressWarnings("PMD.AvoidLiteralsInIfCondition")
203    public String toString() {
204        StringBuilder builder = new StringBuilder();
205        builder.append(Components.objectName(this))
206            .append(" [\"");
207        String path = request.requestUri().getPath();
208        if (path.length() > 15) {
209            builder.append("...")
210                .append(path.substring(path.length() - 12));
211        } else {
212            builder.append(path);
213        }
214        builder.append('\"');
215        if (channels().length > 0) {
216            builder.append(", channels=");
217            builder.append(Channel.toString(channels()));
218        }
219        builder.append(']');
220        return builder.toString();
221    }
222
223    /**
224     * Creates the match value.
225     *
226     * @param type the type
227     * @param resource the resource
228     * @return the object
229     */
230    public static Object createMatchValue(
231            Class<?> type, ResourcePattern resource) {
232        return new MatchValue(type, resource);
233    }
234
235    /**
236     * Represents a match value.
237     */
238    private static class MatchValue {
239        private final Class<?> type;
240        private final Object resource;
241
242        /**
243         * @param type
244         * @param resource
245         */
246        public MatchValue(Class<?> type, Object resource) {
247            super();
248            this.type = type;
249            this.resource = resource;
250        }
251
252        /*
253         * (non-Javadoc)
254         * 
255         * @see java.lang.Object#hashCode()
256         */
257        @Override
258        @SuppressWarnings("PMD.DataflowAnomalyAnalysis")
259        public int hashCode() {
260            @SuppressWarnings("PMD.AvoidFinalLocalVariable")
261            final int prime = 31;
262            int result = 1;
263            result = prime * result
264                + ((resource == null) ? 0 : resource.hashCode());
265            result = prime * result + ((type == null) ? 0 : type.hashCode());
266            return result;
267        }
268
269        /*
270         * (non-Javadoc)
271         * 
272         * @see java.lang.Object#equals(java.lang.Object)
273         */
274        @Override
275        public boolean equals(Object obj) {
276            if (this == obj) {
277                return true;
278            }
279            if (obj == null) {
280                return false;
281            }
282            if (getClass() != obj.getClass()) {
283                return false;
284            }
285            MatchValue other = (MatchValue) obj;
286            if (resource == null) {
287                if (other.resource != null) {
288                    return false;
289                }
290            } else if (!resource.equals(other.resource)) {
291                return false;
292            }
293            if (type == null) {
294                if (other.type != null) {
295                    return false;
296                }
297            } else if (!type.equals(other.type)) {
298                return false;
299            }
300            return true;
301        }
302
303    }
304}
