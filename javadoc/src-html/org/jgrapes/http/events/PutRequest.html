019package org.jgrapes.http.events;
020
021import org.jdrupes.httpcodec.protocols.http.HttpRequest;
022import org.jgrapes.core.Channel;
023
024/**
025 * Represents a HTTP PUT request.
026 */
027public class PutRequest extends Request {
028
029    /**
030     * Create a new event.
031     * 
032     * @param request the request data
033     * @param secure indicates whether the request was received on a
034     * secure channel
035     * @param matchLevels the number of elements from the request path
036     * to use in the match value
037     * @param channels the channels on which the event is to be 
038     * fired (optional)
039     */
040    public PutRequest(HttpRequest request, boolean secure,
041            int matchLevels, Channel... channels) {
042        super(secure ? "https" : "http", request, matchLevels, channels);
043    }
044
045}
