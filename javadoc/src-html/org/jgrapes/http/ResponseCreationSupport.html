057/**
058 * Provides methods that support the creation of a {@link Response} events.
059 */
060@SuppressWarnings("PMD.DataflowAnomalyAnalysis")
061public abstract class ResponseCreationSupport {

063    /** A default implementation for the max-age calculator. */
064    @SuppressWarnings("PMD.LongVariable")
065    public static final MaxAgeCalculator DEFAULT_MAX_AGE_CALCULATOR
066        = new DefaultMaxAgeCalculator();

068    /**
069     * Send a response to the given request with the given status code 
070     * and reason phrase, including a `text/plain` body with the status 
071     * code and reason phrase. 
072     *
073     * @param request the request
074     * @param channel for responding; events will be sent using
075     * {@link IOSubchannel#respond(org.jgrapes.core.Event)}
076     * @param statusCode the status code to send
077     * @param reasonPhrase the reason phrase to send
078     */
079    @SuppressWarnings("PMD.EmptyCatchBlock")
080    public static void sendResponse(HttpRequest request,
081            IOSubchannel channel, int statusCode, String reasonPhrase) {
082        HttpResponse response = request.response().get();
083        response.setStatusCode(statusCode).setReasonPhrase(reasonPhrase)
084            .setHasPayload(true).setField(
085                HttpField.CONTENT_TYPE,
086                MediaType.builder().setType("text", "plain")
087                    .setParameter("charset", "utf-8").build());
088        // Act like a sub-component, i.e. generate events that are
089        // handled by this HTTP server as if sent from a sub-component.
090        channel.respond(new Response(response));
091        try {
092            channel.respond(Output.from((statusCode + " " + reasonPhrase)
093                .getBytes("utf-8"), true));
094        } catch (UnsupportedEncodingException e) {
095            // Supported by definition
096        }
097    }
098
099    /**
100     * Shorthand for invoking 
101     * {@link #sendResponse(HttpRequest, IOSubchannel, int, String)}
102     * with a predefined HTTP status.
103     *
104     * @param request the request
105     * @param channel the channel
106     * @param status the status
107     */
108    public static void sendResponse(HttpRequest request,
109            IOSubchannel channel, HttpStatus status) {
110        sendResponse(request, channel, status.statusCode(),
111            status.reasonPhrase());
112    }

114    /**
115     * Creates and sends a response with static content. The content 
116     * is looked up by invoking the resolver with the path from the request.
117     * 
118     * The response includes a max-age header with a default value of
119     * 600. The value may be modified by specifying validity infos.
120     *
121     * @param request the request
122     * @param channel the channel
123     * @param resolver the resolver
124     * @param maxAgeCalculator the max age calculator, if `null`
125     * the default calculator is used.
126     * @return `true` if a response was sent
127     */
128    @SuppressWarnings({ "PMD.NcssCount",
129        "PMD.UseStringBufferForStringAppends" })
130    public static boolean sendStaticContent(
131            HttpRequest request, IOSubchannel channel,
132            Function<String, URL> resolver, MaxAgeCalculator maxAgeCalculator) {
133        String path = request.requestUri().getPath();
134        URL resourceUrl = resolver.apply(path);
135        ResourceInfo info;
136        URLConnection resConn;
137        InputStream resIn;
138        try {
139            if (resourceUrl == null) {
140                throw new IOException();
141            }
142            info = ResponseCreationSupport.resourceInfo(resourceUrl);
143            if (Boolean.TRUE.equals(info.isDirectory())) {
144                throw new IOException();
145            }
146            resConn = resourceUrl.openConnection();
147            resIn = resConn.getInputStream();
148        } catch (IOException e1) {
149            try {
150                if (!path.endsWith("/")) {
151                    path += "/";
152                }
153                path += "index.html";
154                resourceUrl = resolver.apply(path);
155                if (resourceUrl == null) {
156                    return false;
157                }
158                info = ResponseCreationSupport.resourceInfo(resourceUrl);
159                resConn = resourceUrl.openConnection();
160                resIn = resConn.getInputStream();
161            } catch (IOException e2) {
162                return false;
163            }
164        }
165        HttpResponse response = request.response().get();
166        response.setField(HttpField.LAST_MODIFIED,
167            Optional.ofNullable(info.getLastModifiedAt())
168                .orElseGet(() -> Instant.now()));
169
170        // Get content type and derive max age
171        MediaType mediaType = HttpResponse.contentType(
172            ResponseCreationSupport.uriFromUrl(resourceUrl));
173        setMaxAge(response, maxAgeCalculator, request, mediaType);
174
175        // Check if sending is really required.
176        Optional<Instant> modifiedSince = request
177            .findValue(HttpField.IF_MODIFIED_SINCE, Converters.DATE_TIME);
178        if (modifiedSince.isPresent() && info.getLastModifiedAt() != null
179            && !info.getLastModifiedAt().isAfter(modifiedSince.get())) {
180            response.setStatus(HttpStatus.NOT_MODIFIED);
181            channel.respond(new Response(response));
182        } else {
183            response.setContentType(mediaType);
184            response.setStatus(HttpStatus.OK);
185            channel.respond(new Response(response));
186            // Start sending content (Output events as resonses)
187            (new InputStreamPipeline(resIn, channel).suppressClose()).run();
188        }
189        return true;
190    }
191
192    /**
193     * Shorthand for invoking 
194     * {@link #sendStaticContent(HttpRequest, IOSubchannel, Function, MaxAgeCalculator)}
195     * with the {@link HttpRequest} from the event. Also sets the result
196     * of the event to `true` and invokes {@link Event#stop()} 
197     * if a response was sent.
198     *
199     * @param event the event
200     * @param channel the channel
201     * @param resolver the resolver
202     * @param maxAgeCalculator the max age calculator, if `null`
203     * the default calculator is used.
204     * @return `true` if a response was sent
205     * @throws ParseException the parse exception
206     */
207    public static boolean sendStaticContent(
208            Request event, IOSubchannel channel,
209            Function<String, URL> resolver, MaxAgeCalculator maxAgeCalculator) {
210        if (sendStaticContent(
211            event.httpRequest(), channel, resolver, maxAgeCalculator)) {
212            event.setResult(true);
213            event.stop();
214            return true;
215        }
216        return false;
217    }

219    /**
220     * Combines the known information about a resource.
221     */
222    public static class ResourceInfo {
223        public Boolean isDirectory;
224        public Instant lastModifiedAt;
225
226        /**
227         * @param isDirectory
228         * @param lastModifiedAt
229         */
230        public ResourceInfo(Boolean isDirectory, Instant lastModifiedAt) {
231            this.isDirectory = isDirectory;
232            this.lastModifiedAt = lastModifiedAt;
233        }
234
235        /**
236         * @return the isDirectory
237         */
238        public Boolean isDirectory() {
239            return isDirectory;
240        }
241
242        /**
243         * @return the lastModifiedAt
244         */
245        public Instant getLastModifiedAt() {
246            return lastModifiedAt;
247        }
248    }

250    /**
251     * Attempts to lookup the additional resource information for the
252     * given URL. 
253     * 
254     * If a {@link URL} references a file, it is easy to find out if 
255     * the resource referenced is a directory and to get its last 
256     * modification time. Getting the same information
257     * for a {@link URL} that references resources in a jar is a bit
258     * more difficult. This method handles both cases.
259     *
260     * @param resource the resource URL
261     * @return the resource info
262     */
263    @SuppressWarnings("PMD.EmptyCatchBlock")
264    public static ResourceInfo resourceInfo(URL resource) {
265        try {
266            Path path = Paths.get(resource.toURI());
267            return new ResourceInfo(Files.isDirectory(path),
268                Files.getLastModifiedTime(path).toInstant()
269                    .with(ChronoField.NANO_OF_SECOND, 0));
270        } catch (FileSystemNotFoundException | IOException
271                | URISyntaxException e) {
272            // Fall through
273        }
274        if ("jar".equals(resource.getProtocol())) {
275            try {
276                JarURLConnection conn
277                    = (JarURLConnection) resource.openConnection();
278                JarEntry entry = conn.getJarEntry();
279                return new ResourceInfo(entry.isDirectory(),
280                    entry.getLastModifiedTime().toInstant()
281                        .with(ChronoField.NANO_OF_SECOND, 0));
282            } catch (IOException e) {
283                // Fall through
284            }
285        }
286        try {
287            URLConnection conn = resource.openConnection();
288            long lastModified = conn.getLastModified();
289            if (lastModified != 0) {
290                return new ResourceInfo(null, Instant.ofEpochMilli(
291                    lastModified).with(ChronoField.NANO_OF_SECOND, 0));
292            }
293        } catch (IOException e) {
294            // Fall through
295        }
296        return new ResourceInfo(null, null);
297    }

299    /**
300     * Create a {@link URI} from a path. This is similar to calling
301     * `new URI(null, null, path, null)` with the {@link URISyntaxException}
302     * converted to a {@link IllegalArgumentException}.
303     * 
304     * @param path the path
305     * @return the uri
306     * @throws IllegalArgumentException if the string violates 
307     * RFC 2396
308     */
309    public static URI uriFromPath(String path) throws IllegalArgumentException {
310        try {
311            return new URI(null, null, path, null);
312        } catch (URISyntaxException e) {
313            throw new IllegalArgumentException(e);
314        }
315    }
316
317    /**
318     * Create a {@link URI} from a {@link URL}. This is similar to calling
319     * `url.toURI()` with the {@link URISyntaxException}
320     * converted to a {@link IllegalArgumentException}.
321     * 
322     * @param url the url
323     * @return the uri
324     * @throws IllegalArgumentException if the url violates RFC 2396
325     */
326    public static URI uriFromUrl(URL url) throws IllegalArgumentException {
327        try {
328            return url.toURI();
329        } catch (URISyntaxException e) {
330            throw new IllegalArgumentException(e);
331        }
332    }

334    /**
335     * Sets the cache control header in the given response,
336     * looking up the value for the given media type or using
337     * the default value.
338     *
339     * @param response the response
340     * @param maxAgeCalculator the max age calculator
341     * @param request the request
342     * @param mediaType the media type
343     * @return the value set
344     */
345    public static long setMaxAge(HttpResponse response,
346            MaxAgeCalculator maxAgeCalculator,
347            HttpRequest request, MediaType mediaType) {
348        List<Directive> directives = new ArrayList<>();
349        int maxAge = maxAgeCalculator == null
350            ? DEFAULT_MAX_AGE_CALCULATOR.maxAge(request, mediaType)
351            : maxAgeCalculator.maxAge(request, mediaType);
352        directives.add(new Directive("max-age", maxAge));
353        response.setField(HttpField.CACHE_CONTROL, directives);
354        return maxAge;
355    }

357    /**
358     * Describes a calculator for the max-age property.
359     */
360    @FunctionalInterface
361    public interface MaxAgeCalculator {
362
363        /**
364         * Calculate a max age value for a response using the given 
365         * request and the media type of the repsonse.
366         *
367         * @param request the request, usually only the URI is
368         * considered for the calculation
369         * @param mediaType the media type of the response
370         * @return the max age value to be used in the response
371         */
372        int maxAge(HttpRequest request, MediaType mediaType);
373    }
374
375    /**
376     * DefaultMaxAgeCalculator provides an implementation that 
377     * tries to guess a good max age value by looking at the
378     * path of the requested resource. If the path contains
379     * the pattern "dash, followed by a number, followed by
380     * a dot and a number" it is assumed that the resource
381     * is versioned, i.e. its path changes if the resource
382     * changes. In this case a max age of one year is returned.
383     * In all other cases, a max age value of 60 (one minute)
384     * is returned.
385     */
386    public static class DefaultMaxAgeCalculator implements MaxAgeCalculator {
387
388        public static final Pattern VERSION_PATTERN
389            = Pattern.compile("-[0-9]+\\.[0-9]+");
390
391        @Override
392        public int maxAge(HttpRequest request, MediaType mediaType) {
393            if (VERSION_PATTERN.matcher(
394                request.requestUri().getPath()).find()) {
395                return 365 * 24 * 3600;
396            }
397            return 60;
398        }
399
400    }
