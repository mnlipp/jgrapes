064/**
065 * A base class for components that generate responses to
066 * HTTP requests which are based on a FreeMarker template.
067 */
068@SuppressWarnings("PMD.DataClass")
069public class FreeMarkerRequestHandler extends Component {
070    public static final Pattern TEMPLATE_PATTERN
071        = Pattern.compile(".*\\.ftl\\.[a-z]+$");
072
073    private ClassLoader contentLoader;
074    private String contentPath;
075    private URI prefix;
076    private ResourcePattern prefixPattern;
077    private Configuration fmConfig;
078    private MaxAgeCalculator maxAgeCalculator;
079
080    /**
081     * Instantiates a new free marker request handler.
082     * 
083     * The prefix path is removed from the request paths before resolving
084     * them against the content root. A prefix path must start with a
085     * slash and must end with a slash. If the request handler
086     * should respond to top-level requests, the prefix must be
087     * a single slash.
088     *
089     * @param componentChannel the component channel
090     * @param channelReplacements the channel replacements to apply
091     * to the `channels` elements of the {@link Handler} annotations
092     * @param contentLoader the content loader
093     * @param contentPath the content path
094     * @param prefix the prefix used in requests
095     */
096    public FreeMarkerRequestHandler(Channel componentChannel,
097            ChannelReplacements channelReplacements,
098            ClassLoader contentLoader, String contentPath, URI prefix) {
099        super(componentChannel, channelReplacements);
100        String prefixPath = prefix.getPath();
101        if (!prefixPath.startsWith("/") || !prefixPath.endsWith("/")) {
102            throw new IllegalArgumentException("Illegal prefix: " + prefix);
103        }
104        this.prefix = prefix;
105        this.contentLoader = contentLoader;
106        if (contentPath.startsWith("/")) {
107            contentPath = contentPath.substring(1);
108        }
109        if (contentPath.endsWith("/")) {
110            contentPath = contentPath.substring(0, contentPath.length() - 1);
111        }
112        this.contentPath = contentPath;
113        try {
114            this.prefixPattern = new ResourcePattern(
115                prefixPath.substring(0, prefixPath.length() - 1) + "|**");
116        } catch (ParseException e) {
117            throw new IllegalArgumentException(e);
118        }
119    }

176    /**
177     * Sets the {@link MaxAgeCalculator} for generating the `Cache-Control` 
178     * (`max-age`) header of the response. The default is `null`. This
179     * causes 0 to be provided for responses generated from templates and the 
180     * {@link ResponseCreationSupport#DEFAULT_MAX_AGE_CALCULATOR} to be
181     * used for static content.
182     * 
183     * @param maxAgeCalculator the maxAgeCalculator to set
184     */
185    public void setMaxAgeCalculator(MaxAgeCalculator maxAgeCalculator) {
186        this.maxAgeCalculator = maxAgeCalculator;
187    }
188
189    /**
190     * Removes the prefix specified in the constructor from the
191     * path in the request. Checks if the resulting path  
192     * ends with `*.ftl.*`. If so, processes the template with the
193     * {@link #sendProcessedTemplate(Request.In, IOSubchannel, String)} (which 
194     * uses {@link #fmSessionModel(Optional)}) and sends the result. 
195     * Else, tries to serve static content with the optionally 
196     * shortened path.
197     * 
198     * @param event the event
199     * @param channel the channel
200     * @throws ParseException the parse exception
201     */
202    @SuppressWarnings("PMD.DataflowAnomalyAnalysis")
203    protected void doRespond(Request.In event, IOSubchannel channel)
204            throws ParseException {
205        final HttpRequest request = event.httpRequest();
206        prefixPattern.pathRemainder(request.requestUri()).ifPresent(path -> {
207            boolean success = false;
208            if (TEMPLATE_PATTERN.matcher(path).matches()) {
209                success = sendProcessedTemplate(event, channel, path);
210            } else {
211                success = ResponseCreationSupport.sendStaticContent(
212                    request, channel, requestPath -> contentLoader
213                        .getResource(contentPath + "/" + path),
214                    maxAgeCalculator);
215            }
216            event.setResult(true);
217            event.stop();
218            if (!success) {
219                channel.respond(new Close());
220            }
221        });
222    }
223
224    /**
225     * Render a response using the given template. Send the 
226     * {@link Response} and at least one {@link Output} event.
227     *
228     * If the method does not return `true`, the invoker should close
229     * the connection because it is most likely in an undefined state.
230     *
231     * @param event the request event
232     * @param channel the channel
233     * @param tpl the template
234     * @return false if an error occurred
235     */
236    protected boolean sendProcessedTemplate(
237            Request.In event, IOSubchannel channel, Template tpl) {
238        // Prepare response
239        HttpResponse response = event.httpRequest().response().get();
240        MediaType mediaType = contentType(
241            ResponseCreationSupport.uriFromPath(tpl.getSourceName()));
242        response.setContentType(mediaType);
243
244        // Send response
245        response.setStatus(HttpStatus.OK);
246        response.setField(HttpField.LAST_MODIFIED, Instant.now());
247        if (maxAgeCalculator == null) {
248            ResponseCreationSupport.setMaxAge(response, 0);
249        } else {
250            ResponseCreationSupport.setMaxAge(response,
251                maxAgeCalculator.maxAge(event.httpRequest(), mediaType));
252        }
253        channel.respond(new Response(response));
254
255        // Send content
256        try (ByteBufferOutputStream bbos = new ByteBufferOutputStream(
257            channel, channel.responsePipeline());
258                Writer out = new OutputStreamWriter(
259                    bbos.suppressClose(), "utf-8")) {
260            Map<String, Object> model = fmSessionModel(
261                event.associated(Session.class));
262            tpl.setLocale((Locale) model.get("locale"));
263            tpl.process(model, out);
264            return true;
265        } catch (IOException | TemplateException e) {
266            // Too late to do anything about this (header was sent).
267            fire(new Error(event, e), channel);
268        }
269        return false;
270    }
271
272    /**
273     * Render a response using the template obtained from the config with
274     * {@link Configuration#getTemplate(String)} and the given path.
275     *
276     * @param event
277     *            the event
278     * @param channel
279     *            the channel
280     * @param path
281     *            the path
282     */
283    protected boolean sendProcessedTemplate(
284            Request.In event, IOSubchannel channel, String path) {
285        try {
286            // Get template (no need to continue if this fails).
287            Template tpl = freemarkerConfig().getTemplate(path);
288            return sendProcessedTemplate(event, channel, tpl);
289        } catch (IOException e) {
290            fire(new Error(event, e), channel);
291            return false;
292        }
293    }
294
295    /**
296     * Creates the configuration for freemarker template processing.
297     * 
298     * @return the configuration
299     */
300    protected Configuration freemarkerConfig() {
301        if (fmConfig == null) {
302            fmConfig = new Configuration(Configuration.VERSION_2_3_26);
303            fmConfig.setClassLoaderForTemplateLoading(
304                contentLoader, contentPath);
305            fmConfig.setDefaultEncoding("utf-8");
306            fmConfig.setTemplateExceptionHandler(
307                TemplateExceptionHandler.RETHROW_HANDLER);
308            fmConfig.setLogTemplateExceptions(false);
309        }
310        return fmConfig;
311    }
312
313    /**
314     * Build a freemarker model holding the information associated with the
315     * session.
316     * 
317     * This model provides: 
318     * 
319     * * The `locale` (of type {@link Locale}). 
320     * 
321     * * The `resourceBundle` (of type {@link ResourceBundle}). 
322     * 
323     * * A function "`_`" that looks up the given key in the 
324     *   resource bundle.
325     * 
326     * @param session
327     *            the session
328     * @return the model
329     */
330    protected Map<String, Object> fmSessionModel(Optional<Session> session) {
331        @SuppressWarnings("PMD.UseConcurrentHashMap")
332        final Map<String, Object> model = new HashMap<>();
333        Locale locale = session.map(
334            sess -> sess.locale()).orElse(Locale.getDefault());
335        model.put("locale", locale);
336        final ResourceBundle resourceBundle = resourceBundle(locale);
337        model.put("resourceBundle", resourceBundle);
338        model.put("_", new TemplateMethodModelEx() {
339            @Override
340            @SuppressWarnings("PMD.EmptyCatchBlock")
341            public Object exec(@SuppressWarnings("rawtypes") List arguments)
342                    throws TemplateModelException {
343                @SuppressWarnings("unchecked")
344                List<TemplateModel> args = (List<TemplateModel>) arguments;
345                if (!(args.get(0) instanceof SimpleScalar)) {
346                    throw new TemplateModelException("Not a string.");
347                }
348                String key = ((SimpleScalar) args.get(0)).getAsString();
349                try {
350                    return resourceBundle.getString(key);
351                } catch (MissingResourceException e) {
352                    // no luck
353                }
354                return key;
355            }
356        });
357        return model;
358    }

360    /**
361     * Used to get the content type when generating a response with
362     * {@link #sendProcessedTemplate(Request.In, IOSubchannel, Template)}. 
363     * May be overridden by derived classes. This implementation simply invokes
364     * {@link HttpResponse#contentType(URI)}.
365     *
366     * @param request the request
367     * @return the content type
368     */
369    protected MediaType contentType(URI request) {
370        return HttpResponse.contentType(request);
371    }
372
373    /**
374     * Provides a resource bundle for localization.
375     * The default implementation looks up a bundle using the
376     * package name plus "l10n" as base name.
377     * 
378     * @return the resource bundle
379     */
380    protected ResourceBundle resourceBundle(Locale locale) {
381        return ResourceBundle.getBundle(
382            contentPath.replace('/', '.') + ".l10n", locale,
383            contentLoader, ResourceBundle.Control.getNoFallbackControl(
384                ResourceBundle.Control.FORMAT_DEFAULT));
385    }
