077/**
078 * A converter component that receives and sends byte buffers on a 
079 * network channel and web application layer messages on
080 * {@link IOSubchannel}s of its channel. 
081 * 
082 * Each {@link IOSubchannel} represents a connection established by 
083 * the browser. The {@link HttpServer} fires {@link Request} events 
084 * (and {@link Input} events, if there is associated data) on the 
085 * subchannels. Web application components (short "weblets") handle 
086 * these events and use 
087 * {@link LinkedIOSubchannel#respond(org.jgrapes.core.Event)}
088 * to send {@link Response} events and, if applicable, {@link Output}
089 * events with data belonging to the response.
090 * 
091 * Events must be fired by weblets while handling the {@link Request}
092 * or {@link Input} events only (to be precise: while handling events 
093 * processed by the associated {@link EventProcessor}) to ensure
094 * that responses and their associated data do not interleave. 
095 */
096@SuppressWarnings("PMD.ExcessiveImports")
097public class HttpServer extends Component {

337    /**
338     * Checks whether the request has been handled (value of {@link Request}
339     * event set to `true`) or the status code in the prepared response
340     * is no longer "Not Implemented". If not, but a fall back has been set, 
341     * send a "Not Found" response. If this isn't the case either, send 
342     * the default response ("Not implemented") to the client.
343     * 
344     * @param event
345     *            the request completed event
346     * @param appChannel the application channel 
347     * @throws InterruptedException if the execution was interrupted
348     */
349    @Handler
350    public void onRequestCompleted(
351            Request.In.Completed event, IOSubchannel appChannel)
352            throws InterruptedException {
353        final Request.In requestEvent = event.event();
354        // A check that also works with null.
355        if (Boolean.TRUE.equals(requestEvent.get())
356            || requestEvent.httpRequest().response().map(
357                response -> response.statusCode() != HttpStatus.NOT_IMPLEMENTED
358                    .statusCode())
359                .orElse(false)) {
360            // Some other component has taken care
361            return;
362        }
363
364        // Check if "Not Found" should be sent
365        if (providedFallbacks != null
366            && providedFallbacks.contains(requestEvent.getClass())) {
367            ResponseCreationSupport.sendResponse(
368                requestEvent.httpRequest(), appChannel, HttpStatus.NOT_FOUND);
369            return;
370        }
371
372        // Last resort
373        ResponseCreationSupport.sendResponse(requestEvent.httpRequest(),
374            appChannel, HttpStatus.NOT_IMPLEMENTED);
375    }

412    /**
413     * An application layer channel.
414     */
415    @SuppressWarnings("PMD.DataflowAnomalyAnalysis")
416    private class WebAppMsgChannel extends LinkedIOSubchannel {
417        // Starts as ServerEngine<HttpRequest,HttpResponse> but may change
418        private final ServerEngine<?, ?> engine;
419        private ManagedBuffer<ByteBuffer> outBuffer;
420        private final boolean secure;
421        private List<String> snis = Collections.emptyList();
422        private final ManagedBufferPool<ManagedBuffer<ByteBuffer>,
423                ByteBuffer> byteBufferPool;
424        private final ManagedBufferPool<ManagedBuffer<CharBuffer>,
425                CharBuffer> charBufferPool;
426        private ManagedBufferPool<?, ?> currentPool;
427        private final EventPipeline downPipeline;
428        private Upgraded pendingUpgraded;
429        private WsMessageHeader currentWsMessage;
430
431        /**
432         * Instantiates a new channel.
433         *
434         * @param event the event
435         * @param netChannel the net channel
436         */
437        @SuppressWarnings("PMD.AvoidLiteralsInIfCondition")
438        public WebAppMsgChannel(Accepted event, IOSubchannel netChannel) {
439            super(HttpServer.this, channel(), netChannel, newEventPipeline());
440            engine = new ServerEngine<>(
441                new HttpRequestDecoder(), new HttpResponseEncoder());
442            secure = event.isSecure();
443            if (secure) {
444                snis = new ArrayList<>();
445                for (SNIServerName sni : event.requestedServerNames()) {
446                    if (sni instanceof SNIHostName) {
447                        snis.add(((SNIHostName) sni).getAsciiName());
448                    }
449                }
450            }
451
452            // Calculate "good" application buffer size
453            int bufferSize = applicationBufferSize;
454            if (bufferSize <= 0) {
455                bufferSize = netChannel.byteBufferPool().bufferSize() - 512;
456                if (bufferSize < 4096) {
457                    bufferSize = 4096;
458                }
459            }
460
461            String channelName = Components.objectName(HttpServer.this)
462                + "." + Components.objectName(this);
463            byteBufferPool().setName(channelName + ".upstream.byteBuffers");
464            charBufferPool().setName(channelName + ".upstream.charBuffers");
465            // Allocate downstream buffer pools. Note that decoding WebSocket
466            // network packets may result in several WS frames that are each
467            // delivered in independent events. Therefore provide some
468            // additional buffers.
469            final int bufSize = bufferSize;
470            byteBufferPool = new ManagedBufferPool<>(ManagedBuffer::new,
471                () -> {
472                    return ByteBuffer.allocate(bufSize);
473                }, 2, 100)
474                    .setName(channelName + ".downstream.byteBuffers");
475            charBufferPool = new ManagedBufferPool<>(ManagedBuffer::new,
476                () -> {
477                    return CharBuffer.allocate(bufSize);
478                }, 2, 100)
479                    .setName(channelName + ".downstream.charBuffers");
480
481            // Downstream pipeline
482            downPipeline = newEventPipeline();
483        }
484
485        /**
486         * Handle {@link Input} events from the network.
487         *
488         * @param event the event
489         * @throws ProtocolException the protocol exception
490         * @throws InterruptedException the interrupted exception
491         */
492        @SuppressWarnings({ "PMD.DataflowAnomalyAnalysis",
493            "PMD.AvoidInstantiatingObjectsInLoops",
494            "PMD.AvoidDeeplyNestedIfStmts", "PMD.CollapsibleIfStatements" })
495        public void handleNetInput(Input<ByteBuffer> event)
496                throws ProtocolException, InterruptedException {
497            // Send the data from the event through the decoder.
498            ByteBuffer inData = event.data();
499            // Don't unnecessary allocate a buffer, may be header only message
500            ManagedBuffer<?> bodyData = null;
501            boolean wasOverflow = false;
502            while (inData.hasRemaining()) {
503                if (wasOverflow) {
504                    // Message has (more) body
505                    bodyData = currentPool.acquire();
506                }
507                Decoder.Result<?> result = engine.decode(inData,
508                    bodyData == null ? null : bodyData.backingBuffer(),
509                    event.isEndOfRecord());
510                if (result.response().isPresent()) {
511                    // Feedback required, send it
512                    responsePipeline().overrideRestriction().fire(
513                        new Response(result.response().get()), this);
514                    if (result.isResponseOnly()) {
515                        maybeCloseConnection(result);
516                        continue;
517                    }
518                }
519                if (result.isHeaderCompleted()) {
520                    if (!handleRequestHeader(engine.currentRequest().get())) {
521                        maybeCloseConnection(result);
522                        break;
523                    }
524                }
525                if (bodyData != null) {
526                    if (bodyData.position() > 0) {
527                        downPipeline.fire(Input.fromSink(
528                            bodyData, !result.isOverflow()
529                                && !result.isUnderflow()),
530                            this);
531                    } else {
532                        bodyData.unlockBuffer();
533                    }
534                    bodyData = null;
535                }
536                maybeCloseConnection(result);
537                wasOverflow = result.isOverflow();
538            }
539        }
540
541        private void maybeCloseConnection(Decoder.Result<?> result) {
542            if (result.closeConnection()) {
543                respond(new Close());
544            }
545        }

547        @SuppressWarnings("PMD.CollapsibleIfStatements")
548        private boolean handleRequestHeader(MessageHeader request) {
549            if (request instanceof HttpRequest) {
550                HttpRequest httpRequest = (HttpRequest) request;
551                if (httpRequest.hasPayload()) {
552                    if (httpRequest.findValue(
553                        HttpField.CONTENT_TYPE, Converters.MEDIA_TYPE)
554                        .map(type -> type.value().topLevelType()
555                            .equalsIgnoreCase("text"))
556                        .orElse(false)) {
557                        currentPool = charBufferPool;
558                    } else {
559                        currentPool = byteBufferPool;
560                    }
561                }
562                if (secure) {
563                    if (!snis.contains(httpRequest.host())) {
564                        if (acceptNoSni && snis.isEmpty()) {
565                            convertHostToNumerical(httpRequest);
566                        } else {
567                            ResponseCreationSupport.sendResponse(httpRequest,
568                                this, 421, "Misdirected Request");
569                            return false;
570                        }
571                    }
572                }
573                downPipeline.fire(Request.In.fromHttpRequest(httpRequest,
574                    secure, matchLevels), this);
575            } else if (request instanceof WsMessageHeader) {
576                WsMessageHeader wsMessage = (WsMessageHeader) request;
577                if (wsMessage.hasPayload()) {
578                    if (wsMessage.isTextMode()) {
579                        currentPool = charBufferPool;
580                    } else {
581                        currentPool = byteBufferPool;
582                    }
583                }
584            } else if (request instanceof WsCloseFrame) {
585                downPipeline.fire(
586                    new WebSocketClose((WsCloseFrame) request, this));
587            }
588            return true;
589        }
590
591        @SuppressWarnings({ "PMD.DataflowAnomalyAnalysis",
592            "PMD.UseStringBufferForStringAppends" })
593        private void convertHostToNumerical(HttpRequest request) {
594            int port = request.port();
595            String host;
596            try {
597                InetAddress addr = InetAddress.getByName(
598                    request.host());
599                host = addr.getHostAddress();
600                if (!(addr instanceof Inet4Address)) {
601                    host = "[" + host + "]";
602                }
603            } catch (UnknownHostException e) {
604                host = InetAddress.getLoopbackAddress().getHostAddress();
605            }
606            request.setHostAndPort(host, port);
607        }

609        /**
610         * Handle a response event from the application layer.
611         *
612         * @param event the event
613         * @throws InterruptedException the interrupted exception
614         */
615        @SuppressWarnings({ "PMD.AvoidInstantiatingObjectsInLoops",
616            "PMD.AvoidBranchingStatementAsLastInLoop" })
617        public void handleResponse(Response event) throws InterruptedException {
618            if (!engine.encoding()
619                .isAssignableFrom(event.response().getClass())) {
620                return;
621            }
622            final MessageHeader response = event.response();
623            // Start sending the response
624            @SuppressWarnings("unchecked")
625            ServerEngine<?, MessageHeader> msgEngine
626                = (ServerEngine<?, MessageHeader>) engine;
627            msgEngine.encode(response);
628            if (pendingUpgraded != null) {
629                if (response instanceof HttpResponse
630                    && ((HttpResponse) response).statusCode() % 100 == 1) {
631                    downPipeline.fire(pendingUpgraded, this);
632                }
633                pendingUpgraded = null;
634            }
635            boolean hasBody = response.hasPayload();
636            while (true) {
637                outBuffer = upstreamChannel().byteBufferPool().acquire();
638                final ManagedBuffer<ByteBuffer> buffer = outBuffer;
639                Codec.Result result = engine.encode(
640                    Codec.EMPTY_IN, buffer.backingBuffer(), !hasBody);
641                if (result.isOverflow()) {
642                    upstreamChannel().respond(Output.fromSink(buffer, false));
643                    continue;
644                }
645                if (hasBody) {
646                    // Keep buffer with incomplete response to be further
647                    // filled by subsequent Output events
648                    break;
649                }
650                // Response is complete
651                if (buffer.position() > 0) {
652                    upstreamChannel().respond(Output.fromSink(buffer, true));
653                } else {
654                    buffer.unlockBuffer();
655                }
656                outBuffer = null;
657                if (result.closeConnection()) {
658                    upstreamChannel().respond(new Close());
659                }
660                break;
661            }
662
663        }

665        /**
666         * Handle a {@link ProtocolSwitchAccepted} event from the 
667         * application layer.
668         *
669         * @param event the event
670         * @param appChannel the app channel
671         */
672        public void handleProtocolSwitchAccepted(
673                ProtocolSwitchAccepted event, WebAppMsgChannel appChannel) {
674            appChannel.setAssociated(URI.class,
675                event.requestEvent().requestUri());
676            final HttpResponse response = event.requestEvent()
677                .httpRequest().response().get()
678                .setStatus(HttpStatus.SWITCHING_PROTOCOLS)
679                .setField(HttpField.UPGRADE,
680                    new StringList(event.protocol()));
681            // We send the Upgraded event only after the response has
682            // successfully been encoded (and thus checked).
683            pendingUpgraded = new Upgraded(event.resourceName(),
684                event.protocol());
685            respond(new Response(response));
686        }

688        /**
689         * Handle output from the application layer.
690         *
691         * @param event the event
692         * @throws InterruptedException the interrupted exception
693         */
694        @SuppressWarnings({ "PMD.CyclomaticComplexity", "PMD.NcssCount",
695            "PMD.NPathComplexity", "PMD.AvoidInstantiatingObjectsInLoops" })
696        public void handleAppOutput(Output<?> event)
697                throws InterruptedException {
698            Buffer eventData = event.data();
699            Buffer input;
700            if (eventData instanceof ByteBuffer) {
701                input = ((ByteBuffer) eventData).duplicate();
702            } else if (eventData instanceof CharBuffer) {
703                input = ((CharBuffer) eventData).duplicate();
704            } else {
705                return;
706            }
707            if (engine.switchedTo().equals(Optional.of("websocket"))
708                && currentWsMessage == null) {
709                // When switched to WebSockets, we only have Input and Output
710                // events. Add header automatically.
711                @SuppressWarnings("unchecked")
712                ServerEngine<?, MessageHeader> wsEngine
713                    = (ServerEngine<?, MessageHeader>) engine;
714                currentWsMessage = new WsMessageHeader(
715                    event.buffer().backingBuffer() instanceof CharBuffer,
716                    true);
717                wsEngine.encode(currentWsMessage);
718            }
719            while (input.hasRemaining() || event.isEndOfRecord()) {
720                if (outBuffer == null) {
721                    outBuffer = upstreamChannel().byteBufferPool().acquire();
722                }
723                Codec.Result result = engine.encode(input,
724                    outBuffer.backingBuffer(), event.isEndOfRecord());
725                if (result.isOverflow()) {
726                    upstreamChannel()
727                        .respond(Output.fromSink(outBuffer, false));
728                    outBuffer = upstreamChannel().byteBufferPool().acquire();
729                    continue;
730                }
731                if (event.isEndOfRecord() || result.closeConnection()) {
732                    if (outBuffer.position() > 0) {
733                        upstreamChannel()
734                            .respond(Output.fromSink(outBuffer, true));
735                    } else {
736                        outBuffer.unlockBuffer();
737                    }
738                    outBuffer = null;
739                    if (result.closeConnection()) {
740                        upstreamChannel().respond(new Close());
741                    }
742                    break;
743                }
744            }
745            if (engine.switchedTo().equals(Optional.of("websocket"))
746                && event.isEndOfRecord()) {
747                currentWsMessage = null;
748            }
749        }
750
751        /**
752         * Handle a {@link Close} event from the application layer.
753         *
754         * @param event the event
755         * @throws InterruptedException the interrupted exception
756         */
757        public void handleClose(Close event) throws InterruptedException {
758            if (engine.switchedTo().equals(Optional.of("websocket"))) {
759                fire(new Response(new WsCloseFrame(null, null)), this);
760                return;
761            }
762            upstreamChannel().respond(new Close());
763        }

766         * Handle a {@link Closed} event from the network by forwarding
767         * it to the application layer.
768         *
769         * @param event the event
770         */
771        public void handleClosed(Closed event) {
772            downPipeline.fire(new Closed(), this);
773        }
