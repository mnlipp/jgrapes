080/**
081 * A converter component that receives and sends byte buffers on a 
082 * network channel and web application layer messages on
083 * {@link IOSubchannel}s of its channel. 
084 * 
085 * Each {@link IOSubchannel} represents a connection established by 
086 * the browser. The {@link HttpServer} fires {@link Request} events 
087 * (and {@link Input} events, if there is associated data) on the 
088 * subchannels. Web application components (short "weblets") handle 
089 * these events and use 
090 * {@link LinkedIOSubchannel#respond(org.jgrapes.core.Event)}
091 * to send {@link Response} events and, if applicable, {@link Output}
092 * events with data belonging to the response.
093 * 
094 * Events must be fired by weblets while handling the {@link Request}
095 * or {@link Input} events only (to be precise: while handling events 
096 * processed by the associated {@link EventProcessor}) to ensure
097 * that responses and their associated data do not interleave. 
098 */
099@SuppressWarnings("PMD.ExcessiveImports")
100public class HttpServer extends Component {

340    /**
341     * Checks whether the request has been handled (value of {@link Request}
342     * event set to `true`) or the status code in the prepared response
343     * is no longer "Not Implemented". If not, but a fall back has been set, 
344     * send a "Not Found" response. If this isn't the case either, send 
345     * the default response ("Not implemented") to the client.
346     * 
347     * @param event
348     *            the request completed event
349     * @param appChannel the application channel 
350     * @throws InterruptedException if the execution was interrupted
351     */
352    @Handler
353    public void onRequestCompleted(
354            Request.Completed event, IOSubchannel appChannel)
355            throws InterruptedException {
356        final Request requestEvent = event.event();
357        // A check that also works with null.
358        if (Boolean.TRUE.equals(requestEvent.get())
359            || requestEvent.httpRequest().response().map(
360                response -> response.statusCode() != HttpStatus.NOT_IMPLEMENTED
361                    .statusCode())
362                .orElse(false)) {
363            // Some other component has taken care
364            return;
365        }
366
367        // Check if "Not Found" should be sent
368        if (providedFallbacks != null
369            && providedFallbacks.contains(requestEvent.getClass())) {
370            ResponseCreationSupport.sendResponse(
371                requestEvent.httpRequest(), appChannel, HttpStatus.NOT_FOUND);
372            return;
373        }
374
375        // Last resort
376        ResponseCreationSupport.sendResponse(requestEvent.httpRequest(),
377            appChannel, HttpStatus.NOT_IMPLEMENTED);
378    }

437     * An application layer channel.
438     */
439    @SuppressWarnings("PMD.DataflowAnomalyAnalysis")
440    private class WebAppMsgChannel extends LinkedIOSubchannel {
441        // Starts as ServerEngine<HttpRequest,HttpResponse> but may change
442        private final ServerEngine<?, ?> engine;
443        private ManagedBuffer<ByteBuffer> outBuffer;
444        private final boolean secure;
445        private List<String> snis = Collections.emptyList();
446        private final ManagedBufferPool<ManagedBuffer<ByteBuffer>,
447                ByteBuffer> byteBufferPool;
448        private final ManagedBufferPool<ManagedBuffer<CharBuffer>,
449                CharBuffer> charBufferPool;
450        private ManagedBufferPool<?, ?> currentPool;
451        private final EventPipeline downPipeline;
452        private final Object syncUpgrading = new Object();
453        private UpgradedState upgradedTo = UpgradedState.NONE;
454        private WsMessageHeader currentWsMessage;
455
456        /**
457         * Instantiates a new channel.
458         *
459         * @param event the event
460         * @param netChannel the net channel
461         */
462        @SuppressWarnings("PMD.AvoidLiteralsInIfCondition")
463        public WebAppMsgChannel(Accepted event, IOSubchannel netChannel) {
464            super(HttpServer.this, channel(), netChannel, newEventPipeline());
465            engine = new ServerEngine<>(
466                new HttpRequestDecoder(), new HttpResponseEncoder());
467            secure = event.isSecure();
468            if (secure) {
469                snis = new ArrayList<>();
470                for (SNIServerName sni : event.requestedServerNames()) {
471                    if (sni instanceof SNIHostName) {
472                        snis.add(((SNIHostName) sni).getAsciiName());
473                    }
474                }
475            }
476
477            // Calculate "good" application buffer size
478            int bufferSize = applicationBufferSize;
479            if (bufferSize <= 0) {
480                bufferSize = netChannel.byteBufferPool().bufferSize() - 512;
481                if (bufferSize < 4096) {
482                    bufferSize = 4096;
483                }
484            }
485
486            String channelName = Components.objectName(HttpServer.this)
487                + "." + Components.objectName(this);
488            byteBufferPool().setName(channelName + ".upstream.byteBuffers");
489            charBufferPool().setName(channelName + ".upstream.charBuffers");
490            // Allocate downstream buffer pools. Note that decoding WebSocket
491            // network packets may result in several WS frames that are each
492            // delivered in independent events. Therefore provide some
493            // additional buffers.
494            final int bufSize = bufferSize;
495            byteBufferPool = new ManagedBufferPool<>(ManagedBuffer::new,
496                () -> {
497                    return ByteBuffer.allocate(bufSize);
498                }, 2, 100)
499                    .setName(channelName + ".downstream.byteBuffers");
500            charBufferPool = new ManagedBufferPool<>(ManagedBuffer::new,
501                () -> {
502                    return CharBuffer.allocate(bufSize);
503                }, 2, 100)
504                    .setName(channelName + ".downstream.charBuffers");
505
506            // Downstream pipeline
507            downPipeline = newEventPipeline();
508        }
509
510        /**
511         * Handle {@link Input} events from the network.
512         *
513         * @param event the event
514         * @throws ProtocolException the protocol exception
515         * @throws InterruptedException the interrupted exception
516         */
517        @SuppressWarnings({ "PMD.DataflowAnomalyAnalysis",
518            "PMD.AvoidInstantiatingObjectsInLoops",
519            "PMD.AvoidDeeplyNestedIfStmts", "PMD.CollapsibleIfStatements" })
520        public void handleNetInput(Input<ByteBuffer> event)
521                throws ProtocolException, InterruptedException {
522            // Send the data from the event through the decoder.
523            ByteBuffer inData = event.data();
524            // Don't unnecessary allocate a buffer, may be header only message
525            ManagedBuffer<?> bodyData = null;
526            boolean wasOverflow = false;
527            while (inData.hasRemaining()) {
528                if (wasOverflow) {
529                    // Message has (more) body
530                    bodyData = currentPool.acquire();
531                }
532                Decoder.Result<?> result = engine.decode(inData,
533                    bodyData == null ? null : bodyData.backingBuffer(),
534                    event.isEndOfRecord());
535                if (result.response().isPresent()) {
536                    // Feedback required, send it
537                    responsePipeline().overrideRestriction().fire(
538                        new Response(result.response().get()), this);
539                    if (result.response().get().isFinal()) {
540                        if (result.isHeaderCompleted()) {
541                            engine.currentRequest()
542                                .filter(WsCloseFrame.class::isInstance)
543                                .ifPresent(closeFrame -> {
544                                    downPipeline.fire(new WebSocketClosed(
545                                        (WsCloseFrame) closeFrame, this));
546                                });
547                        }
548                        break;
549                    }
550                    if (result.isResponseOnly()) {
551                        continue;
552                    }
553                }
554                if (result.isHeaderCompleted()) {
555                    if (!handleRequestHeader(engine.currentRequest().get())) {
556                        break;
557                    }
558                }
559                if (bodyData != null) {
560                    if (bodyData.position() > 0) {
561                        downPipeline.fire(Input.fromSink(
562                            bodyData, !result.isOverflow()
563                                && !result.isUnderflow()),
564                            this);
565                    } else {
566                        bodyData.unlockBuffer();
567                    }
568                    bodyData = null;
569                }
570                wasOverflow = result.isOverflow();
571            }
572        }
573
574        @SuppressWarnings("PMD.CollapsibleIfStatements")
575        private boolean handleRequestHeader(MessageHeader request) {
576            if (request instanceof HttpRequest) {
577                HttpRequest httpRequest = (HttpRequest) request;
578                if (httpRequest.hasPayload()) {
579                    if (httpRequest.findValue(
580                        HttpField.CONTENT_TYPE, Converters.MEDIA_TYPE)
581                        .map(type -> type.value().topLevelType()
582                            .equalsIgnoreCase("text"))
583                        .orElse(false)) {
584                        currentPool = charBufferPool;
585                    } else {
586                        currentPool = byteBufferPool;
587                    }
588                }
589                if (secure) {
590                    if (!snis.contains(httpRequest.host())) {
591                        if (acceptNoSni && snis.isEmpty()) {
592                            convertHostToNumerical(httpRequest);
593                        } else {
594                            ResponseCreationSupport.sendResponse(httpRequest,
595                                this, 421, "Misdirected Request");
596                            return false;
597                        }
598                    }
599                }
600                downPipeline.fire(Request.fromHttpRequest(httpRequest,
601                    secure, matchLevels), this);
602            } else if (request instanceof WsMessageHeader) {
603                WsMessageHeader wsMessage = (WsMessageHeader) request;
604                if (wsMessage.hasPayload()) {
605                    if (wsMessage.isTextMode()) {
606                        currentPool = charBufferPool;
607                    } else {
608                        currentPool = byteBufferPool;
609                    }
610                }
611            }
612            return true;
613        }
614
615        @SuppressWarnings({ "PMD.DataflowAnomalyAnalysis",
616            "PMD.UseStringBufferForStringAppends" })
617        private void convertHostToNumerical(HttpRequest request) {
618            int port = request.port();
619            String host;
620            try {
621                InetAddress addr = InetAddress.getByName(
622                    request.host());
623                host = addr.getHostAddress();
624                if (!(addr instanceof Inet4Address)) {
625                    host = "[" + host + "]";
626                }
627            } catch (UnknownHostException e) {
628                host = InetAddress.getLoopbackAddress().getHostAddress();
629            }
630            request.setHostAndPort(host, port);
631        }

633        /**
634         * Handle a response event from the appication layer.
635         *
636         * @param event the event
637         * @throws InterruptedException the interrupted exception
638         */
639        @SuppressWarnings({ "PMD.AvoidInstantiatingObjectsInLoops",
640            "PMD.AvoidBranchingStatementAsLastInLoop" })
641        public void handleResponse(Response event) throws InterruptedException {
642            if (!engine.encoding()
643                .isAssignableFrom(event.response().getClass())) {
644                return;
645            }
646            final MessageHeader response = event.response();
647            // Start sending the response
648            @SuppressWarnings("unchecked")
649            ServerEngine<?, MessageHeader> msgEngine
650                = (ServerEngine<?, MessageHeader>) engine;
651            msgEngine.encode(response);
652            boolean hasBody = response.hasPayload();
653            while (true) {
654                outBuffer = upstreamChannel().byteBufferPool().acquire();
655                final ManagedBuffer<ByteBuffer> buffer = outBuffer;
656                Codec.Result result = engine.encode(
657                    Codec.EMPTY_IN, buffer.backingBuffer(), !hasBody);
658                if (result.isOverflow()) {
659                    upstreamChannel().respond(Output.fromSink(buffer, false));
660                    continue;
661                }
662                if (hasBody) {
663                    // Keep buffer with incomplete response to be further
664                    // filled by Output events
665                    break;
666                }
667                // Response is complete
668                if (buffer.position() > 0) {
669                    upstreamChannel().respond(Output.fromSink(buffer, true));
670                } else {
671                    buffer.unlockBuffer();
672                }
673                outBuffer = null;
674                if (result.closeConnection()) {
675                    upstreamChannel().respond(new Close());
676                }
677                break;
678            }
679
680        }

682        /**
683         * Handle a {@link ProtocolSwitchAccepted} event from the 
684         * application layer.
685         *
686         * @param event the event
687         * @param appChannel the app channel
688         */
689        public void handleProtocolSwitchAccepted(
690                ProtocolSwitchAccepted event, WebAppMsgChannel appChannel) {
691            upgradedTo = UpgradedState.UPGRADING_TO_WS;
692            appChannel.setAssociated(URI.class,
693                event.requestEvent().requestUri());
694            final HttpResponse response = event.requestEvent()
695                .httpRequest().response().get()
696                .setStatus(HttpStatus.SWITCHING_PROTOCOLS)
697                .setField(HttpField.UPGRADE,
698                    new StringList(event.protocol()));
699            downPipeline.fire(new Upgraded(event.resourceName(),
700                event.protocol()), this);
701            respond(new Response(response));
702            // Response is sent, channel may be used by next level now.
703            synchronized (syncUpgrading) {
704                upgradedTo = UpgradedState.WEB_SOCKET;
705                syncUpgrading.notifyAll();
706            }
707        }
708
709        /**
710         * Handle an {@link Upgraded} event from the applictaion layer.
711         *
712         * @param event the event
713         * @param appChannel the app channel
714         * @throws InterruptedException the interrupted exception
715         */
716        public void handleUpgraded(Upgraded event,
717                WebAppMsgChannel appChannel) throws InterruptedException {
718            synchronized (syncUpgrading) {
719                while (upgradedTo == UpgradedState.UPGRADING_TO_WS) {
720                    syncUpgrading.wait();
721                }
722            }
723        }
724
725        /**
726         * Handle output from the application layer.
727         *
728         * @param event the event
729         * @throws InterruptedException the interrupted exception
730         */
731        @SuppressWarnings({ "PMD.CyclomaticComplexity", "PMD.NcssCount",
732            "PMD.NPathComplexity", "PMD.AvoidInstantiatingObjectsInLoops" })
733        public void handleAppOutput(Output<?> event)
734                throws InterruptedException {
735            Buffer eventData = event.data();
736            Buffer input;
737            if (eventData instanceof ByteBuffer) {
738                input = ((ByteBuffer) eventData).duplicate();
739            } else if (eventData instanceof CharBuffer) {
740                input = ((CharBuffer) eventData).duplicate();
741            } else {
742                return;
743            }
744            if (upgradedTo == UpgradedState.WEB_SOCKET
745                && currentWsMessage == null) {
746                // When switched to WebSockets, we only have Input and Output
747                // events. Add header automatically.
748                @SuppressWarnings("unchecked")
749                ServerEngine<?, MessageHeader> wsEngine
750                    = (ServerEngine<?, MessageHeader>) engine;
751                currentWsMessage = new WsMessageHeader(
752                    event.buffer().backingBuffer() instanceof CharBuffer,
753                    true);
754                wsEngine.encode(currentWsMessage);
755            }
756            while (input.hasRemaining() || event.isEndOfRecord()) {
757                if (outBuffer == null) {
758                    outBuffer = upstreamChannel().byteBufferPool().acquire();
759                }
760                Codec.Result result = engine.encode(input,
761                    outBuffer.backingBuffer(), event.isEndOfRecord());
762                if (result.isOverflow()) {
763                    upstreamChannel()
764                        .respond(Output.fromSink(outBuffer, false));
765                    outBuffer = upstreamChannel().byteBufferPool().acquire();
766                    continue;
767                }
768                if (event.isEndOfRecord() || result.closeConnection()) {
769                    if (outBuffer.position() > 0) {
770                        upstreamChannel()
771                            .respond(Output.fromSink(outBuffer, true));
772                    } else {
773                        outBuffer.unlockBuffer();
774                    }
775                    outBuffer = null;
776                    if (result.closeConnection()) {
777                        upstreamChannel().respond(new Close());
778                    }
779                    break;
780                }
781            }
782            if (upgradedTo == UpgradedState.WEB_SOCKET
783                && event.isEndOfRecord()) {
784                currentWsMessage = null;
785            }
786        }

788        /**
789         * Handle a {@link Close} event from the application layer.
790         *
791         * @param event the event
792         * @throws InterruptedException the interrupted exception
793         */
794        public void handleClose(Close event) throws InterruptedException {
795            if (upgradedTo == UpgradedState.WEB_SOCKET) {
796                respond(new Response(new WsCloseFrame(null, null)));
797                return;
798            }
799            upstreamChannel().respond(new Close());
800        }
