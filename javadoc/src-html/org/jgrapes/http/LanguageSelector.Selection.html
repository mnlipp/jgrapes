021import java.beans.ConstructorProperties;
022import java.io.Serializable;
023import java.lang.ref.WeakReference;
024import java.net.HttpCookie;
025import java.text.ParseException;
026import java.util.ArrayList;
027import java.util.Arrays;
028import java.util.Collections;
029import java.util.List;
030import java.util.Locale;

034import org.jdrupes.httpcodec.protocols.http.HttpRequest;

036import org.jdrupes.httpcodec.types.CookieList;

039import org.jgrapes.core.Associator;

048/**
049 * A component that attempts to derive information about language preferences
050 * from requests in the specified scope (usually "/") and make them 
051 * available as a {@link Selection} object associated with the request 
052 * event using `Selection.class` as association identifier.
053 * 
054 * The component first checks if the event has an associated {@link Session}
055 * and whether that session has a value with key `Selection.class`. If
056 * such an entry exists, its value is assumed to be a {@link Selection} object
057 * which is (re-)used for all subsequent operations. Else, a new
058 * {@link Selection} object is created (and associated with the session, if
059 * a session exists).
060 * 
061 * If the {@link Selection} represents explicitly set values, it is used as
062 * result (i.e. as object associated with the event by `Selection.class`).
063 * 
064 * Else, the selector tries to derive the language preferences from the
065 * request. It first checks for a cookie with the specified name
066 * (see {@link #cookieName()}). If a cookie is found, its value is
067 * used to set the preferred locales. If no cookie is found, 
068 * the values derived from the `Accept-Language header` are set
069 * as fall-backs.
070 * 
071 * Whenever the language preferences 
072 * change (see {@link Selection#prefer(Locale)}), a cookie with the
073 * specified name and a path value set to the scope is created or
074 * updated accordingly. 
075 */
076public class LanguageSelector extends Component {
077
078        private String path;
079        private static final DefaultMultiValueConverter<List<Locale>, Locale> 
080                LOCALE_LIST = new DefaultMultiValueConverter<>(
081                                ArrayList<Locale>::new, Converters.LANGUAGE);
082        private String cookieName = LanguageSelector.class.getName();
083        
084        /**
085         * Creates a new language selector component with its channel set to
086         * itself and the scope set to "/".
087         */
088        public LanguageSelector() {
089                this ("/");
090        }
091
092        /**
093         * Creates a new language selector component with its channel set to
094         * itself and the scope set to the given value.
095         * 
096         * @param scope the scope
097         */
098        public LanguageSelector(String scope) {
099                this (Channel.SELF, scope);
100        }
101
102        /**
103         * Creates a new language selector component with its channel set 
104         * to the given channel and the scope to "/".
105         * 
106         * @param componentChannel the component channel
107         */
108        public LanguageSelector(Channel componentChannel) {
109                this(componentChannel, "/");
110        }
111
112        /**
113         * Creates a new language selector component with its channel set 
114         * to the given channel and the scope to the given scope.
115         * 
116         * @param componentChannel the component channel
117         * @param scope the scope
118         */
119        public LanguageSelector(Channel componentChannel, String scope) {
120                super(componentChannel);
121                if ("/".equals(scope) || !scope.endsWith("/")) {
122                        path = scope;
123                } else {
124                        path = scope.substring(0, scope.length() - 1);
125                }
126                String pattern;
127                if (path.equals("/")) {
128                        pattern = "/**";
129                } else {
130                        pattern = path + "," + path + "/**";
131                }
132                RequestHandler.Evaluator.add(this, "onRequest", pattern);
133        }
134
135        /**
136         * Sets the name of the cookie used to store the locale.
137         * 
138         * @param cookieName the cookie name to use
139         * @return the locale selector for easy chaining
140         */
141        public LanguageSelector setCookieName(String cookieName) {
142                this.cookieName = cookieName;
143                return this;
144        }
145
146        /**
147         * Returns the cookie name. Defaults to the class name.
148         * 
149         * @return the cookie name
150         */
151        public String cookieName() {
152                return cookieName;
153        }
154
155        /**
156         * Associates the event with a {@link Selection} object
157         * using `Selection.class` as association identifier.
158         * 
159         * @param event the event
160         */
161        @RequestHandler(priority=990, dynamic=true)
162        public void onRequest(Request event) {
163                final HttpRequest request = event.httpRequest();
164                final Selection selection = event.associated(Session.class)
165                                .map(session -> (Selection)session.computeIfAbsent(
166                                        Selection.class, k -> new Selection(cookieName, path)))
167                                .orElseGet(() -> new Selection(cookieName, path));
168                selection.setCurrentEvent(event);
169                event.setAssociated(Selection.class, selection);
170                if (selection.isExplicitlySet()) {
171                        return;
172                }
173                
174                // Try to get locale from cookies
175                Optional<String> localeNames = request.findValue(
176                                HttpField.COOKIE, Converters.COOKIE_LIST)
177                                .flatMap(cl -> cl.valueForName(cookieName));
178                if (localeNames.isPresent()) {
179                        try {
180                                List<Locale> cookieLocales = LOCALE_LIST
181                                                .fromFieldValue(localeNames.get());
182                                if (cookieLocales.size() > 0) {
183                                        Collections.reverse(cookieLocales);
184                                        cookieLocales.stream().forEach(l -> selection.prefer(l));
185                                        return;
186                                }
187                        } catch (ParseException e) {
188                                // Unusable
189                        }
190                }
191
192                // Last resport: Accept-Language header field
193                Optional<List<ParameterizedValue<Locale>>> accepted = request.findValue(
194                                HttpField.ACCEPT_LANGUAGE, Converters.LANGUAGE_LIST);
195                if (accepted.isPresent()) {
196                        Locale[] locales = accepted.get().stream()
197                                .sorted(ParameterizedValue.WEIGHT_COMPARATOR)
198                                .map(pv -> pv.value()).toArray(Locale[]::new);
199                        selection.updateFallbacks(locales);
200                }
201        }
202        
203        @Handler(priority=1000)
204        public void onProtocolSwitchAccepted(
205                        ProtocolSwitchAccepted event, IOSubchannel channel) {
206                event.requestEvent().associated(Selection.class)
207                        .ifPresent(selection -> 
208                                channel.setAssociated(Selection.class, selection));
209        }
210
211        /**
212         * Convenience method to retrieve a locale from an associator.
213         * 
214         * @param assoc the associator
215         * @return the locale
216         */
217        public static Locale associatedLocale(Associator assoc) {
218                return assoc.associated(Selection.class)
219                                .map(s -> s.get()[0]).orElse(Locale.getDefault());
220        }
221        
222        @SuppressWarnings("serial")
223        public static class Selection implements Serializable {
224                private transient WeakReference<Request> currentEvent;
225                private String cookieName;
226                private String cookiePath;
227                private boolean explicitlySet;
228                private Locale[] locales;
229
230                @ConstructorProperties({"cookieName", "cookiePath"})
231                private Selection(String cookieName, String cookiePath) {
232                        this.cookieName = cookieName;
233                        this.cookiePath = cookiePath;
234                        this.currentEvent = new WeakReference<>(null);
235                        explicitlySet = false;
236                        locales = new Locale[] { Locale.getDefault() };
237                }
238
239                /**
240                 * Gets the cookie name.
241                 *
242                 * @return the cookieName
243                 */
244                public String getCookieName() {
245                        return cookieName;
246                }
247
248                /**
249                 * Gets the cookie path.
250                 *
251                 * @return the cookiePath
252                 */
253                public String getCookiePath() {
254                        return cookiePath;
255                }
256
257                /**
258                 * @return the explicitlySet
259                 */
260                public boolean isExplicitlySet() {
261                        return explicitlySet;
262                }
263
264                /**
265                 * 
266                 * @param locales
267                 */
268                private void updateFallbacks(Locale[] locales) {
269                        if (explicitlySet) {
270                                return;
271                        }
272                        this.locales = locales;
273                }
274                
275                /**
276                 * @param currentEvent the currentEvent to set
277                 */
278                private Selection setCurrentEvent(Request currentEvent) {
279                        this.currentEvent = new WeakReference<>(currentEvent);
280                        return this;
281                }
282
283                /**
284                 * Return the current locale.
285                 * 
286                 * @return the value;
287                 */
288                public Locale[] get() {
289                        return locales;
290                }
291
292                /**
293                 * Updates the current locale.
294                 * 
295                 * @param locale the locale
296                 * @return the selection for easy chaining
297                 */
298                public Selection prefer(Locale locale) {
299                        explicitlySet = true;
300                        List<Locale> list = new ArrayList<>(Arrays.asList(locales));
301                        list.remove(locale);
302                        list.add(0, locale);
303                        this.locales = list.toArray(new Locale[list.size()]);
304                        HttpCookie localesCookie = new HttpCookie(cookieName,
305                                        LOCALE_LIST.asFieldValue(list));
306                        localesCookie.setPath(cookiePath);
307                        Request req = currentEvent.get();
308                        if (req != null) {
309                                req.httpRequest().response().ifPresent(resp -> {
310                                        resp.computeIfAbsent(
311                                                HttpField.SET_COOKIE, CookieList::new)
312                                        .value().add(localesCookie);
313                                });
314                        }
315                        return this;
316                }
317
318                /* (non-Javadoc)
319                 * @see java.lang.Object#toString()
320                 */
321                @Override
322                public String toString() {
323                        StringBuilder builder = new StringBuilder();
324                        builder.append("Selection [");
325                        if (locales != null) {
326                                builder.append("locales=");
327                                builder.append(Arrays.toString(locales));
328                                builder.append(", ");
329                        }
330                        builder.append("explicitlySet=");
331                        builder.append(explicitlySet);
332                        builder.append("]");
333                        return builder.toString();
334                }
335                
336        }
