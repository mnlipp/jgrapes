019package org.jgrapes.http.demo.portlets.helloworld;

021import org.jdrupes.json.JsonBeanDecoder;
022import org.jdrupes.json.JsonBeanEncoder;
023import org.jdrupes.json.JsonDecodeException;
024import org.jgrapes.core.Channel;
025import org.jgrapes.core.CompletionLock;
026import org.jgrapes.core.Event;
027import org.jgrapes.core.Manager;
028import org.jgrapes.core.annotation.Handler;
029import org.jgrapes.http.LanguageSelector;
030import org.jgrapes.io.IOSubchannel;
031import org.jgrapes.portal.PortalView;
032import org.jgrapes.portal.events.AddPortletRequest;
033import org.jgrapes.portal.events.AddPortletType;
034import org.jgrapes.portal.events.DeletePortlet;
035import org.jgrapes.portal.events.DeletePortletRequest;

051import static org.jgrapes.portal.Portlet.*;
052import static org.jgrapes.portal.Portlet.RenderMode.*;
053
054import java.io.IOException;
055import java.util.Map;
056import java.util.Optional;
057import java.util.ResourceBundle;
058import java.util.Set;

060/**
061 * 
062 */
063public class HelloWorldPortlet extends FreeMarkerPortlet {
064
065        private final static Set<RenderMode> MODES = RenderMode.asSet(
066                        DeleteablePreview, View);
067        private final static String STORAGE_PATH = "/portlets/" 
068                        + HelloWorldPortlet.class.getName()     + "/";
069        
070        /**
071         * Creates a new component with its channel set to
072         * itself.
073         */
074        public HelloWorldPortlet() {
075                this(Channel.SELF);
076        }
077
078        /**
079         * Creates a new component with its channel set to the given 
080         * channel.
081         * 
082         * @param componentChannel the channel that the component's 
083         * handlers listen on by default and that 
084         * {@link Manager#fire(Event, Channel...)} sends the event to 
085         */
086        public HelloWorldPortlet(Channel componentChannel) {
087                super(componentChannel);
088        }
089
090        @Handler
091        public void onPortalReady(PortalReady event, IOSubchannel channel) 
092                        throws TemplateNotFoundException, MalformedTemplateNameException, 
093                        ParseException, IOException {
094                ResourceBundle resourceBundle = resourceSupplier().apply(
095                                LanguageSelector.associatedLocale(channel));
096                // Add HelloWorldPortlet resources to page
097                channel.respond(new AddPortletType(getClass().getName())
098                                .setDisplayName(resourceBundle.getString("portletName"))
099                                .addScript(PortalView.uriFromPath("HelloWorld-functions.js"))
100                                .addCss(PortalView.uriFromPath("HelloWorld-style.css"))
101                                .setInstantiable());
102                KeyValueStoreQuery query = new KeyValueStoreQuery(STORAGE_PATH, true);
103                channel.setAssociated(
104                                HelloWorldPortlet.class, new CompletionLock(event, 3000));
105                fire(query, channel);
106        }
107
108        @Handler
109        public void onKeyValueStoreData(
110                        KeyValueStoreData event, IOSubchannel channel) 
111                                        throws JsonDecodeException {
112                if (!event.event().query().equals(STORAGE_PATH)) {
113                        return;
114                }
115                channel.associated(HelloWorldPortlet.class, CompletionLock.class)
116                        .ifPresent(lock -> lock.remove());
117                for (String json: event.data().values()) {
118                        HelloWorldModel model = JsonBeanDecoder.create(json)
119                                        .readObject(HelloWorldModel.class);
120                        addToSession(channel, model);
121                }
122        }
123        
124        @Handler
125        public void onAddPortletRequest(AddPortletRequest event,
126                        IOSubchannel channel) throws TemplateNotFoundException, 
127                                MalformedTemplateNameException, ParseException, IOException {
128                if (!event.portletType().equals(getClass().getName())) {
129                        return;
130                }
131                
132                event.stop();
133                HelloWorldModel portletModel 
134                                = addToSession(channel, new HelloWorldModel());
135                String jsonState = JsonBeanEncoder.create()
136                                .writeObject(portletModel).toJson();
137                channel.respond(new KeyValueStoreUpdate().update(STORAGE_PATH 
138                                + portletModel.getPortletId(), jsonState));
139                Template tpl = freemarkerConfig().getTemplate("HelloWorld-preview.ftlh");
140                Map<String, Object> baseModel 
141                        = freemarkerBaseModel(event.renderSupport());
142                channel.respond(new RenderPortletFromProvider(
143                                HelloWorldPortlet.class, portletModel.getPortletId(),
144                                DeleteablePreview, MODES, newContentProvider(tpl, 
145                                                freemarkerModel(baseModel, portletModel, channel)),
146                                true));
147        }
148
149        @Handler
150        public void onDeletePortletRequest(DeletePortletRequest event, 
151                        IOSubchannel channel) {
152                Optional<PortletModelBean> optPortletModel 
153                        = modelFromSession(channel, event.portletId());
154                if (!optPortletModel.isPresent()) {
155                        return;
156                }
157        
158                event.stop();
159                removeFromSession(channel, optPortletModel.get());
160                channel.respond(new KeyValueStoreUpdate().delete(STORAGE_PATH 
161                                + optPortletModel.get().getPortletId()));
162                channel.respond(new DeletePortlet(
163                                optPortletModel.get().getPortletId()));
164        }
165        
166        @Handler
167        public void onRenderPortlet(RenderPortletRequest event,
168                        IOSubchannel channel) 
169                                        throws TemplateNotFoundException, 
170                                        MalformedTemplateNameException, ParseException, 
171                                        IOException {
172                Optional<PortletModelBean> optPortletModel 
173                        = modelFromSession(channel, event.portletId());
174                if (!optPortletModel.isPresent()) {
175                        return;
176                }
177                
178                event.stop();
179                Map<String, Object> baseModel 
180                        = freemarkerBaseModel(event.renderSupport());
181                HelloWorldModel portletModel = (HelloWorldModel)optPortletModel.get();
182                switch (event.renderMode()) {
183                case Preview:
184                case DeleteablePreview: {
185                        Template tpl = freemarkerConfig().getTemplate("HelloWorld-preview.ftlh");
186                        channel.respond(new RenderPortletFromProvider(
187                                        HelloWorldPortlet.class, portletModel.getPortletId(), 
188                                        DeleteablePreview, MODES,       newContentProvider(tpl, 
189                                                        freemarkerModel(baseModel, portletModel, channel)),
190                                        event.isForeground()));
191                        break;
192                }
193                case View: {
194                        Template tpl = freemarkerConfig().getTemplate("HelloWorld-view.ftlh");
195                        channel.respond(new RenderPortletFromProvider(
196                                        HelloWorldPortlet.class, portletModel.getPortletId(), 
197                                        View, MODES, newContentProvider(tpl, 
198                                                        freemarkerModel(baseModel, portletModel, channel)),
199                                        event.isForeground()));
200                        channel.respond(new NotifyPortletView(getClass().getName(),
201                                        portletModel.getPortletId(), "setWorldVisible",
202                                        portletModel.isWorldVisible()));
203                        break;
204                }
205                default:
206                        break;
207                }       
208        }
209        
210        @Handler
211        public void onChangePortletModel(NotifyPortletModel event,
212                        IOSubchannel channel) throws TemplateNotFoundException, 
213                        MalformedTemplateNameException, ParseException, IOException {
214                Optional<PortletModelBean> optPortletModel 
215                        = modelFromSession(channel, event.portletId());
216                if (!optPortletModel.isPresent()) {
217                        return;
218                }
219        
220                event.stop();
221                HelloWorldModel portletModel = (HelloWorldModel)optPortletModel.get();
222                portletModel.setWorldVisible(!portletModel.isWorldVisible());
223                
224                String jsonState = JsonBeanEncoder.create()
225                                .writeObject(portletModel).toJson();
226                channel.respond(new KeyValueStoreUpdate().update(STORAGE_PATH 
227                                + portletModel.getPortletId(), jsonState));
228                channel.respond(new NotifyPortletView(getClass().getName(),
229                                portletModel.getPortletId(), "setWorldVisible", 
230                                portletModel.isWorldVisible()));
231        }
232        
233        public static class HelloWorldModel extends PortletModelBean {
234
235                private boolean worldVisible = true;
236                
237                public HelloWorldModel() {
238                }
239                
240                /**
241                 * @param visible the visible to set
242                 */
243                public void setWorldVisible(boolean visible) {
244                        this.worldVisible = visible;
245                }
246
247                public boolean isWorldVisible() {
248                        return worldVisible;
249                }
250        }
