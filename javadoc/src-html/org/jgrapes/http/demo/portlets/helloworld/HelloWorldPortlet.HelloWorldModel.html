061/**
062 * 
063 */
064public class HelloWorldPortlet extends FreeMarkerPortlet {
065
066        private final static Set<RenderMode> MODES = RenderMode.asSet(
067                        DeleteablePreview, View);
068        
069        /**
070         * Creates a new component with its channel set to the given 
071         * channel.
072         * 
073         * @param componentChannel the channel that the component's 
074         * handlers listen on by default and that 
075         * {@link Manager#fire(Event, Channel...)} sends the event to 
076         */
077        public HelloWorldPortlet(Channel componentChannel) {
078                super(componentChannel, false);
079        }
080
081        private String storagePath(Session session) {
082                return "/" + session.getOrDefault(Principal.class, "").toString()
083                        + "/portlets/" + HelloWorldPortlet.class.getName()      + "/";
084        }
085        
086        @Handler
087        public void onPortalReady(PortalReady event, PortalSession portalSession) 
088                        throws TemplateNotFoundException, MalformedTemplateNameException, 
089                        ParseException, IOException {
090                ResourceBundle resourceBundle = resourceBundle(portalSession.locale());
091                // Add HelloWorldPortlet resources to page
092                portalSession.respond(new AddPortletType(type())
093                                .setDisplayName(resourceBundle.getString("portletName"))
094                                .addScript(PortalView.uriFromPath("HelloWorld-functions.js"))
095                                .addCss(PortalView.uriFromPath("HelloWorld-style.css"))
096                                .setInstantiable());
097                KeyValueStoreQuery query = new KeyValueStoreQuery(
098                                storagePath(portalSession.browserSession()), true);
099                portalSession.setAssociated(
100                                HelloWorldPortlet.class, new CompletionLock(event, 3000));
101                fire(query, portalSession);
102        }
103
104        @Handler
105        public void onKeyValueStoreData(
106                        KeyValueStoreData event, PortalSession channel) 
107                                        throws JsonDecodeException {
108                if (!event.event().query().equals(storagePath(channel.browserSession()))) {
109                        return;
110                }
111                channel.associated(HelloWorldPortlet.class, CompletionLock.class)
112                        .ifPresent(lock -> lock.remove());
113                for (String json: event.data().values()) {
114                        HelloWorldModel model = JsonBeanDecoder.create(json)
115                                        .readObject(HelloWorldModel.class);
116                        putInSession(channel.browserSession(), model);
117                }
118        }
119        
120        @Override
121        public String doAddPortlet(AddPortletRequest event,
122                        PortalSession channel) throws Exception {
123                String portletId = generatePortletId();
124                HelloWorldModel portletModel = putInSession
125                                (channel.browserSession(), new HelloWorldModel(portletId));
126                String jsonState = JsonBeanEncoder.create()
127                                .writeObject(portletModel).toJson();
128                channel.respond(new KeyValueStoreUpdate().update(
129                                storagePath(channel.browserSession()) + portletModel.getPortletId(),
130                                jsonState));
131                Template tpl = freemarkerConfig().getTemplate("HelloWorld-preview.ftlh");
132                channel.respond(new RenderPortlet(
133                                HelloWorldPortlet.class, portletModel.getPortletId(),
134                                DeleteablePreview, MODES, true, templateProcessor(tpl, 
135                                                fmModel(event, channel, portletModel))));
136                return portletId;
137        }
138
139        /* (non-Javadoc)
140         * @see org.jgrapes.portal.AbstractPortlet#doDeletePortlet(org.jgrapes.portal.events.DeletePortletRequest, org.jgrapes.io.IOSubchannel, org.jgrapes.portal.AbstractPortlet.PortletModelBean)
141         */
142        @Override
143        protected void doDeletePortlet(DeletePortletRequest event,
144                PortalSession channel, String portletId, 
145                Serializable portletState) throws Exception {
146                channel.respond(new KeyValueStoreUpdate().delete(
147                                storagePath(channel.browserSession()) + portletId));
148                channel.respond(new DeletePortlet(portletId));
149        }
150        
151        
152        /* (non-Javadoc)
153         * @see org.jgrapes.portal.AbstractPortlet#doRenderPortlet(org.jgrapes.portal.events.RenderPortletRequest, org.jgrapes.io.IOSubchannel, org.jgrapes.portal.AbstractPortlet.PortletModelBean)
154         */
155        @Override
156        protected void doRenderPortlet(RenderPortletRequest event,
157                PortalSession channel, String portletId, 
158                Serializable retrievedState) throws Exception {
159                HelloWorldModel portletModel = (HelloWorldModel)retrievedState;
160                switch (event.renderMode()) {
161                case Preview:
162                case DeleteablePreview: {
163                        Template tpl = freemarkerConfig().getTemplate("HelloWorld-preview.ftlh");
164                        channel.respond(new RenderPortlet(
165                                        HelloWorldPortlet.class, portletModel.getPortletId(), 
166                                        DeleteablePreview, MODES,       event.isForeground(),
167                                        templateProcessor(
168                                                        tpl, fmModel(event, channel, portletModel))));
169                        break;
170                }
171                case View: {
172                        Template tpl = freemarkerConfig().getTemplate("HelloWorld-view.ftlh");
173                        channel.respond(new RenderPortlet(
174                                        HelloWorldPortlet.class, portletModel.getPortletId(), 
175                                        View, MODES, event.isForeground(), templateProcessor(
176                                                        tpl, fmModel(event, channel, portletModel))));
177                        channel.respond(new NotifyPortletView(type(),
178                                        portletModel.getPortletId(), "setWorldVisible",
179                                        portletModel.isWorldVisible()));
180                        break;
181                }
182                default:
183                        break;
184                }
185        }
186        
187        /* (non-Javadoc)
188         * @see org.jgrapes.portal.AbstractPortlet#doNotifyPortletModel(org.jgrapes.portal.events.NotifyPortletModel, org.jgrapes.io.IOSubchannel, org.jgrapes.http.Session, java.io.Serializable)
189         */
190        @Override
191        protected void doNotifyPortletModel(NotifyPortletModel event,
192                PortalSession channel, Serializable portletState)
193                throws Exception {
194                event.stop();
195                HelloWorldModel portletModel = (HelloWorldModel)portletState;
196                portletModel.setWorldVisible(!portletModel.isWorldVisible());
197                
198                String jsonState = JsonBeanEncoder.create()
199                                .writeObject(portletModel).toJson();
200                channel.respond(new KeyValueStoreUpdate().update(
201                                storagePath(channel.browserSession()) + portletModel.getPortletId(),
202                                jsonState));
203                channel.respond(new NotifyPortletView(type(),
204                                portletModel.getPortletId(), "setWorldVisible", 
205                                portletModel.isWorldVisible()));
206        }
207
208        @SuppressWarnings("serial")
209        public static class HelloWorldModel extends PortletBaseModel {
210
211                private boolean worldVisible = true;
212                
213                /**
214                 * Creates a new model with the given type and id.
215                 * 
216                 * @param portletId the portlet id
217                 */
218                @ConstructorProperties({"portletId"})
219                public HelloWorldModel(String portletId) {
220                        super(portletId);
221                }
222
223                /**
224                 * @param visible the visible to set
225                 */
226                public void setWorldVisible(boolean visible) {
227                        this.worldVisible = visible;
228                }
229
230                public boolean isWorldVisible() {
231                        return worldVisible;
232                }
233        }
