019package org.jgrapes.http.demo.portlets.helloworld;
020
021import org.jgrapes.core.Channel;
022import org.jgrapes.core.Event;
023import org.jgrapes.core.Manager;
024import org.jgrapes.core.annotation.Handler;
025import org.jgrapes.http.LanguageSelector;
026import org.jgrapes.io.IOSubchannel;
027import org.jgrapes.portal.PortalView;
028import org.jgrapes.portal.events.AddPortletRequest;
029import org.jgrapes.portal.events.AddPortletType;
030import org.jgrapes.portal.events.DeletePortlet;
031import org.jgrapes.portal.events.DeletePortletRequest;
032import org.jgrapes.portal.events.NotifyPortletModel;
033import org.jgrapes.portal.events.NotifyPortletView;
034import org.jgrapes.portal.events.PortalReady;
035import org.jgrapes.portal.events.RenderPortletFromProvider;
036import org.jgrapes.portal.events.RenderPortletRequest;
037import org.jgrapes.portal.freemarker.FreeMarkerPortlet;
038
039import freemarker.core.ParseException;
040import freemarker.template.MalformedTemplateNameException;
041import freemarker.template.Template;
042import freemarker.template.TemplateNotFoundException;
043
044import static org.jgrapes.portal.Portlet.*;
045import static org.jgrapes.portal.Portlet.RenderMode.*;
046
047import java.io.IOException;
048import java.util.Map;
049import java.util.Optional;
050import java.util.ResourceBundle;
051import java.util.Set;
052
053/**
054 * 
055 */
056public class HelloWorldPortlet extends FreeMarkerPortlet {
057
058        private final static Set<RenderMode> MODES = RenderMode.asSet(
059                        DeleteablePreview, View);
060        
061        /**
062         * Creates a new component with its channel set to
063         * itself.
064         */
065        public HelloWorldPortlet() {
066                this(Channel.SELF);
067        }
068
069        /**
070         * Creates a new component with its channel set to the given 
071         * channel.
072         * 
073         * @param componentChannel the channel that the component's 
074         * handlers listen on by default and that 
075         * {@link Manager#fire(Event, Channel...)} sends the event to 
076         */
077        public HelloWorldPortlet(Channel componentChannel) {
078                super(componentChannel);
079        }
080
081        @Handler
082        public void onPortalReady(PortalReady event, IOSubchannel channel) 
083                        throws TemplateNotFoundException, MalformedTemplateNameException, 
084                        ParseException, IOException {
085                ResourceBundle resourceBundle = resourceSupplier().apply(
086                                LanguageSelector.associatedLocale(channel));
087                // Add HelloWorldPortlet resources to page
088                channel.respond(new AddPortletType(getClass().getName())
089                                .setDisplayName(resourceBundle.getString("portletName"))
090                                .addScript(PortalView.uriFromPath("HelloWorld-functions.js"))
091                                .addCss(PortalView.uriFromPath("HelloWorld-style.css"))
092                                .setInstantiable());
093        }
094        
095        @Handler
096        public void onAddPortletRequest(AddPortletRequest event,
097                        IOSubchannel channel) throws TemplateNotFoundException, 
098                                MalformedTemplateNameException, ParseException, IOException {
099                if (!event.portletType().equals(getClass().getName())) {
100                        return;
101                }
102                
103                event.stop();
104                HelloWorldModel portletModel 
105                        = addToSession(channel, new HelloWorldModel());
106                Template tpl = freemarkerConfig().getTemplate("HelloWorld-preview.ftlh");
107                Map<String, Object> baseModel 
108                        = freemarkerBaseModel(event.renderSupport());
109                channel.respond(new RenderPortletFromProvider(
110                                portletModel.getPortletId(), DeleteablePreview, 
111                                MODES, newContentProvider(tpl, 
112                                                freemarkerModel(baseModel, portletModel, channel))));
113        }
114
115        @Handler
116        public void onDeletePortletRequest(DeletePortletRequest event, 
117                        IOSubchannel channel) {
118                Optional<PortletModelBean> optPortletModel 
119                        = modelFromSession(channel, event.portletId());
120                if (!optPortletModel.isPresent()) {
121                        return;
122                }
123        
124                event.stop();
125                removeFromSession(channel, optPortletModel.get());
126                channel.respond(new DeletePortlet(
127                                optPortletModel.get().getPortletId()));
128        }
129        
130        @Handler
131        public void onRenderPortlet(RenderPortletRequest event,
132                        IOSubchannel channel) 
133                                        throws TemplateNotFoundException, 
134                                        MalformedTemplateNameException, ParseException, 
135                                        IOException {
136                Optional<PortletModelBean> optPortletModel 
137                        = modelFromSession(channel, event.portletId());
138                if (!optPortletModel.isPresent()) {
139                        return;
140                }
141                
142                event.stop();
143                Map<String, Object> baseModel 
144                        = freemarkerBaseModel(event.renderSupport());
145                HelloWorldModel portletModel = (HelloWorldModel)optPortletModel.get();
146                switch (event.renderMode()) {
147                case Preview:
148                case DeleteablePreview: {
149                        Template tpl = freemarkerConfig().getTemplate("HelloWorld-preview.ftlh");
150                        channel.respond(new RenderPortletFromProvider(
151                                        portletModel.getPortletId(), DeleteablePreview, MODES,
152                                        newContentProvider(tpl, 
153                                                        freemarkerModel(baseModel, portletModel, channel))));
154                }
155                case View: {
156                        Template tpl = freemarkerConfig().getTemplate("HelloWorld-view.ftlh");
157                        channel.respond(new RenderPortletFromProvider(
158                                        portletModel.getPortletId(), View, MODES,
159                                        newContentProvider(tpl, 
160                                                        freemarkerModel(baseModel, portletModel, channel))));
161                }
162                default:
163                        break;
164                }       
165                channel.respond(new NotifyPortletView(getClass().getName(),
166                                portletModel.getPortletId(), "setWorldVisible",
167                                portletModel.isWorldVisible()));
168        }
169        
170        @Handler
171        public void onChangePortletModel(NotifyPortletModel event,
172                        IOSubchannel channel) throws TemplateNotFoundException, 
173                        MalformedTemplateNameException, ParseException, IOException {
174                Optional<PortletModelBean> optPortletModel 
175                        = modelFromSession(channel, event.portletId());
176                if (!optPortletModel.isPresent()) {
177                        return;
178                }
179        
180                event.stop();
181                HelloWorldModel portletModel = (HelloWorldModel)optPortletModel.get();
182                portletModel.setVisible(!portletModel.isWorldVisible());
183                
184                channel.respond(new NotifyPortletView(getClass().getName(),
185                                portletModel.getPortletId(), "setWorldVisible", 
186                                portletModel.isWorldVisible()));
187        }
188        
189        public class HelloWorldModel extends PortletModelBean {
190
191                private boolean visible = true;
192                
193                public HelloWorldModel() {
194                }
195                
196                /**
197                 * @param visible the visible to set
198                 */
199                public void setVisible(boolean visible) {
200                        this.visible = visible;
201                }
202
203                public boolean isWorldVisible() {
204                        return visible;
205                }
206        }
207        
208}
