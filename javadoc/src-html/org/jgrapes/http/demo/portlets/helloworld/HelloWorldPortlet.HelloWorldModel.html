019package org.jgrapes.http.demo.portlets.helloworld;
020
021import org.jgrapes.core.Channel;
022import org.jgrapes.core.Event;
023import org.jgrapes.core.Manager;
024import org.jgrapes.core.annotation.Handler;
025import org.jgrapes.http.LanguageSelector;
026import org.jgrapes.io.IOSubchannel;
027import org.jgrapes.portal.PortalView;
028import org.jgrapes.portal.events.AddPortletRequest;
029import org.jgrapes.portal.events.AddPortletType;
030import org.jgrapes.portal.events.DeletePortlet;
031import org.jgrapes.portal.events.DeletePortletRequest;
032import org.jgrapes.portal.events.NotifyPortletModel;
033import org.jgrapes.portal.events.NotifyPortletView;
034import org.jgrapes.portal.events.PortalReady;
035import org.jgrapes.portal.events.RenderPortletFromProvider;
036import org.jgrapes.portal.events.RenderPortletRequest;
037import org.jgrapes.portal.freemarker.FreeMarkerPortlet;
038
039import freemarker.core.ParseException;
040import freemarker.template.MalformedTemplateNameException;
041import freemarker.template.Template;
042import freemarker.template.TemplateNotFoundException;
043
044import static org.jgrapes.portal.Portlet.*;
045import static org.jgrapes.portal.Portlet.RenderMode.*;
046
047import java.io.IOException;
048import java.util.ArrayList;
049import java.util.Collection;
050import java.util.Map;
051import java.util.Optional;
052import java.util.ResourceBundle;
053import java.util.Set;
054
055/**
056 * 
057 */
058public class HelloWorldPortlet extends FreeMarkerPortlet {
059
060        private final static Set<RenderMode> MODES = RenderMode.asSet(
061                        DeleteablePreview, View);
062        
063        /**
064         * Creates a new component with its channel set to
065         * itself.
066         */
067        public HelloWorldPortlet() {
068                this(Channel.SELF);
069        }
070
071        /**
072         * Creates a new component with its channel set to the given 
073         * channel.
074         * 
075         * @param componentChannel the channel that the component's 
076         * handlers listen on by default and that 
077         * {@link Manager#fire(Event, Channel...)} sends the event to 
078         */
079        public HelloWorldPortlet(Channel componentChannel) {
080                super(componentChannel);
081        }
082
083        @Handler
084        public void onPortalReady(PortalReady event, IOSubchannel channel) 
085                        throws TemplateNotFoundException, MalformedTemplateNameException, 
086                        ParseException, IOException {
087                ResourceBundle resourceBundle = resourceSupplier().apply(
088                                LanguageSelector.associatedLocale(channel));
089                // Add HelloWorldPortlet resources to page
090                channel.respond(new AddPortletType(getClass().getName())
091                                .setDisplayName(resourceBundle.getString("portletName"))
092                                .addScript(PortalView.uriFromPath("HelloWorld-functions.js"))
093                                .addCss(PortalView.uriFromPath("HelloWorld-style.css"))
094                                .setInstantiable());
095                Collection<PortletModelBean> portletModels
096                        = new ArrayList<>(modelsFromSession(channel));
097                if (portletModels.size() == 0) {
098                        portletModels.add(addToSession(channel, new HelloWorldModel()));
099                }
100                Map<String, Object> baseModel 
101                        = freemarkerBaseModel(event.renderSupport());
102                for (PortletModelBean portletModel: portletModels) {
103                        Template tpl = freemarkerConfig().getTemplate("HelloWorld-preview.ftlh");
104                        channel.respond(new RenderPortletFromProvider(
105                                        portletModel.getPortletId(), DeleteablePreview, 
106                                        MODES, newContentProvider(tpl, 
107                                                        freemarkerModel(baseModel, portletModel, channel))));
108                }
109        }
110        
111        @Handler
112        public void onAddPortletRequest(AddPortletRequest event,
113                        IOSubchannel channel) throws TemplateNotFoundException, 
114                                MalformedTemplateNameException, ParseException, IOException {
115                if (!event.portletType().equals(getClass().getName())) {
116                        return;
117                }
118                
119                event.stop();
120                HelloWorldModel portletModel 
121                        = addToSession(channel, new HelloWorldModel());
122                Template tpl = freemarkerConfig().getTemplate("HelloWorld-preview.ftlh");
123                Map<String, Object> baseModel 
124                        = freemarkerBaseModel(event.renderSupport());
125                channel.respond(new RenderPortletFromProvider(
126                                portletModel.getPortletId(), DeleteablePreview, 
127                                MODES, newContentProvider(tpl, 
128                                                freemarkerModel(baseModel, portletModel, channel))));
129        }
130
131        @Handler
132        public void onDeletePortletRequest(DeletePortletRequest event, 
133                        IOSubchannel channel) {
134                Optional<PortletModelBean> optPortletModel 
135                        = modelFromSession(channel, event.portletId());
136                if (!optPortletModel.isPresent()) {
137                        return;
138                }
139        
140                event.stop();
141                removeFromSession(channel, optPortletModel.get());
142                channel.respond(new DeletePortlet(
143                                optPortletModel.get().getPortletId()));
144        }
145        
146        @Handler
147        public void onRenderPortlet(RenderPortletRequest event,
148                        IOSubchannel channel) 
149                                        throws TemplateNotFoundException, 
150                                        MalformedTemplateNameException, ParseException, 
151                                        IOException {
152                Optional<PortletModelBean> optPortletModel 
153                        = modelFromSession(channel, event.portletId());
154                if (!optPortletModel.isPresent()) {
155                        return;
156                }
157                
158                event.stop();
159                HelloWorldModel portletModel = (HelloWorldModel)optPortletModel.get();
160                Template tpl = freemarkerConfig().getTemplate("HelloWorld-view.ftlh");
161                channel.respond(new RenderPortletFromProvider(
162                                portletModel.getPortletId(), View, MODES,
163                                newContentProvider(tpl, 
164                                                freemarkerModel(freemarkerBaseModel(
165                                                                event.renderSupport()), portletModel, channel))));
166                channel.respond(new NotifyPortletView(getClass().getName(),
167                                portletModel.getPortletId(), "setWorldVisible",
168                                portletModel.isWorldVisible()));
169        }
170        
171        @Handler
172        public void onChangePortletModel(NotifyPortletModel event,
173                        IOSubchannel channel) throws TemplateNotFoundException, 
174                        MalformedTemplateNameException, ParseException, IOException {
175                Optional<PortletModelBean> optPortletModel 
176                        = modelFromSession(channel, event.portletId());
177                if (!optPortletModel.isPresent()) {
178                        return;
179                }
180        
181                event.stop();
182                HelloWorldModel portletModel = (HelloWorldModel)optPortletModel.get();
183                portletModel.setVisible(!portletModel.isWorldVisible());
184                
185                channel.respond(new NotifyPortletView(getClass().getName(),
186                                portletModel.getPortletId(), "setWorldVisible", 
187                                portletModel.isWorldVisible()));
188        }
189        
190        public class HelloWorldModel extends PortletModelBean {
191
192                private boolean visible = true;
193                
194                public HelloWorldModel() {
195                }
196                
197                /**
198                 * @param visible the visible to set
199                 */
200                public void setVisible(boolean visible) {
201                        this.visible = visible;
202                }
203
204                public boolean isWorldVisible() {
205                        return visible;
206                }
207        }
208        
209}
