032import java.security.cert.CertificateException;

038import org.jgrapes.core.Channel;

062/**
063 *
064 */
065public class HttpServerDemo extends Component implements BundleActivator {
066
067        HttpServerDemo app;
068        
069        /* (non-Javadoc)
070         * @see org.osgi.framework.BundleActivator#start(org.osgi.framework.BundleContext)
071         */
072        @Override
073        public void start(BundleContext context) throws Exception {
074                // The demo component is the application
075                app = new HttpServerDemo();
076                // Attach a general nio dispatcher
077                app.attach(new NioDispatcher());
078
079                // Network level unencrypted channel.
080                Channel httpTransport = new NamedChannel("httpTransport");
081                // Create a TCP server listening on port 8888
082                app.attach(new TcpServer(httpTransport)
083                                .setServerAddress(new InetSocketAddress(8888)));
084
085                // Create TLS "converter"
086                KeyStore serverStore = KeyStore.getInstance("JKS");
087                try (FileInputStream kf 
088                                = new FileInputStream("demo-resources/localhost.jks")) {
089                        serverStore.load(kf, "nopass".toCharArray());
090                }
091                KeyManagerFactory kmf = KeyManagerFactory.getInstance(
092                                KeyManagerFactory.getDefaultAlgorithm());
093        kmf.init(serverStore, "nopass".toCharArray());
094                SSLContext sslContext = SSLContext.getInstance("TLS");
095                sslContext.init(kmf.getKeyManagers(), null, new SecureRandom());
096                // Create a TCP server for SSL
097                Channel securedNetwork = app.attach(
098                                new TcpServer().setServerAddress(new InetSocketAddress(4443))
099                                .setBacklog(3000).setConnectionLimiter(new PermitsPool(50)));
100                app.attach(new SslServer(httpTransport, securedNetwork, sslContext));
101
102                // Create an HTTP server as converter between transport and application
103                // layer.
104                app.attach(new HttpServer(app, 
105                        httpTransport, GetRequest.class, PostRequest.class));
106                
107                // Build application layer
108                app.attach(new PreferencesStore(app.channel(), this.getClass()));
109                app.attach(new InMemorySessionManager(app.channel()));
110                app.attach(new LanguageSelector(app.channel()));
111                app.attach(new FileStorage(app.channel(), 65536));
112                app.attach(new StaticContentDispatcher(app.channel(),
113                        "/**", Paths.get("demo-resources/static-content").toUri()));
114                app.attach(new StaticContentDispatcher(app.channel(),
115                        "/doc|**", Paths.get("../../jgrapes.gh-pages/javadoc").toUri()));
116                app.attach(new PostProcessor(app.channel()));
117                app.attach(new WsEchoServer(app.channel()));
118                Portal portal = app.attach(new Portal(Channel.SELF, app.channel(), 
119                                new URI("/portal/"))).setResourceBundleSupplier(l -> 
120                                ResourceBundle.getBundle(
121                                        getClass().getPackage().getName() + ".portal-l10n", l,
122                                        ResourceBundle.Control.getNoFallbackControl(
123                                                        ResourceBundle.Control.FORMAT_DEFAULT)));
124                portal.attach(new PortalLocalBackedKVStore(
125                                portal, portal.prefix().getPath()));
126                portal.attach(new KVStoreBasedPortalPolicy(portal));
127                portal.attach(new NewPortalSessionPolicy(portal));
128                portal.attach(new HelloWorldPortlet(portal));
129                Components.start(app);
130        }
131
132        /* (non-Javadoc)
133         * @see org.osgi.framework.BundleActivator#stop(org.osgi.framework.BundleContext)
134         */
135        @Override
136        public void stop(BundleContext context) throws Exception {
137                app.fire(new Stop(), Channel.BROADCAST);
138                Components.awaitExhaustion();
139        }
140
141        /**
142         * @param args
143         * @throws IOException 
144         * @throws InterruptedException 
145         * @throws NoSuchAlgorithmException 
146         * @throws KeyStoreException 
147         * @throws UnrecoverableKeyException 
148         * @throws CertificateException 
149         * @throws KeyManagementException 
150         */
151        public static void main(String[] args) throws Exception {
152                new HttpServerDemo().start(null);
153        }
