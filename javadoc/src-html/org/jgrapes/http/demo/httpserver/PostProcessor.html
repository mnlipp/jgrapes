019package org.jgrapes.http.demo.httpserver;
020
021import java.io.UnsupportedEncodingException;
022import java.util.Optional;
023
024import org.jdrupes.httpcodec.protocols.http.HttpConstants.HttpStatus;
025import org.jdrupes.httpcodec.protocols.http.HttpField;
026import org.jdrupes.httpcodec.protocols.http.HttpRequest;
027import org.jdrupes.httpcodec.protocols.http.HttpResponse;
028import org.jdrupes.httpcodec.types.MediaType;
029import org.jdrupes.httpcodec.util.FormUrlDecoder;
030import org.jgrapes.core.Channel;
031import org.jgrapes.core.Component;
032import org.jgrapes.core.annotation.Handler;
033import org.jgrapes.http.Session;
034import org.jgrapes.http.annotation.RequestHandler;
035import org.jgrapes.http.events.PostRequest;
036import org.jgrapes.http.events.Response;
037import org.jgrapes.io.IOSubchannel;
038import org.jgrapes.io.events.Input;
039import org.jgrapes.io.events.Output;
040import org.jgrapes.io.util.ManagedByteBuffer;
041
042/**
043 * 
044 */
045public class PostProcessor extends Component {
046
047        protected static class FormContext {
048                public HttpRequest request;
049                public Session session;
050                public FormUrlDecoder fieldDecoder = new FormUrlDecoder();
051        }
052
053        /**
054         * @param componentChannel
055         */
056        public PostProcessor(Channel componentChannel) {
057                super(componentChannel);
058        }
059
060        @RequestHandler(patterns="/form")
061        public void onPost(PostRequest event, IOSubchannel channel) {
062                FormContext ctx = channel
063                                .associated(this, FormContext::new);
064                ctx.request = event.httpRequest();
065                ctx.session = event.associated(Session.class).get();
066                event.stop();
067        }
068        
069        @Handler
070        public void onInput(Input<ManagedByteBuffer> event, IOSubchannel channel) 
071                        throws InterruptedException, UnsupportedEncodingException {
072                Optional<FormContext> ctx = channel.associated(this, FormContext.class);
073                if (!ctx.isPresent()) {
074                        return;
075                }
076                ctx.get().fieldDecoder.addData(event.buffer().backingBuffer());
077                if (!event.isEndOfRecord()) {
078                        return;
079                }
080                
081                long invocations = (Long)ctx.get().session.computeIfAbsent(
082                                "invocations", key -> { return 0L; });
083                ctx.get().session.put("invocations", invocations + 1);
084                
085                HttpResponse response = ctx.get().request.response().get();
086                response.setStatus(HttpStatus.OK);
087                response.setHasPayload(true);
088                response.setField(HttpField.CONTENT_TYPE,
089                        MediaType.builder().setType("text", "plain")
090                                .setParameter("charset", "utf-8").build());
091                String data = "First name: "
092                        + ctx.get().fieldDecoder.fields().get("firstname")
093                        + "\r\n" + "Last name: "
094                        + ctx.get().fieldDecoder.fields().get("lastname")
095                        + "\r\n" + "Previous invocations: " + invocations;
096                channel.respond(new Response(response));
097                ManagedByteBuffer out = channel.byteBufferPool().acquire();
098                out.put(data.getBytes("utf-8"));
099                channel.respond(new Output<>(out, true));
100        }
101
102}
