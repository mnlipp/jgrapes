019package org.jgrapes.http;

021import java.net.URI;
022import java.text.ParseException;
023import java.util.ArrayList;
024import java.util.Arrays;
025import java.util.List;
026import java.util.Spliterator;
027import java.util.Spliterators.AbstractSpliterator;
028import java.util.StringTokenizer;
029import java.util.function.Consumer;
030import java.util.regex.Matcher;
031import java.util.regex.Pattern;
032import java.util.stream.StreamSupport;
033
034/**
035 * A resource pattern can be used to filter URIs. A pattern looks  
036 * similar to a URI (<code>scheme://host:port/path</code>) with the 
037 * following differences:
038 * 
039 *  * The <em>scheme</em> may be a single protocol name or a list
040 *    of protocol names separated by commas, or an asterisk, which is matched
041 *    by URIs with any scheme. The scheme part ({@code scheme://}) is optional 
042 *    in a pattern. Omitting it is equivalent to specifying an asterisk.
043 *    
044 *  * The <em>host</em> may be a host name or an IP address or an asterisk.
045 *    Unless the value is an asterisk, filtered URIs must match the value
046 *    literally.
047 *    
048 *  * The <em>port</em> may be a number or an asterisk.
049 *    Unless the value is an asterisk, filtered URIs must match it.
050 *   
051 *  * Specifying a port ({@code :port}) is optional. If omitted, 
052 *    it is equivalent to specifying an asterisk.
053 *    
054 *  * If the scheme part is omitted, the {@code host:port} part may 
055 *    completely be left out as well, which is
056 *    equivalent to specifying an asterisk for both host and port.
057 *    
058 *  * The optional path part consist of one or more path separated by commas.
059 *  
060 *    Each path consists of a sequence of names and asterisks separated
061 *    by slashes or a vertical bar ("|"). A name must be matched by the 
062 *    corresponding path element of filtered URIs, an asterisk is matched by 
063 *    any corresponding path element (which, however, must exist in the 
064 *    filtered URI). The final element in 
065 *    the path of a pattern may be two asterisks ({@code **}), which matches 
066 *    any remaining path elements in the filtered URI.
067 *    
068 *    Using a vertical bar instead of a slash separates the path in a
069 *    prefix part and the rest. The number of prefix segments is the
070 *    value returned by the match methods.
071 *
072 */
073public class ResourcePattern {
074
075        private static Pattern resourcePattern = Pattern.compile(
076                        "^((?<proto>[^:]+|\\*)://)?" // Optional protocol (2)
077                        + "(" // Start of optional host/port part
078                                + "(?<host>([^\\[/][^:/]*)|(\\[[^\\]]*\\]))" // Host (4)
079                                + "(:(?<port>([0-9]+)|(\\*)))?" // Optional port (8)
080                        + ")?" // End of optional host/port
081                        + "(?<path>/.*)?"); // Finally path (11)
082        
083        private String pattern;
084        private String protocol;
085        private String host;
086        private String port;
087        private String path;
088        private String[][] pathPatternElements;
089        private int[] prefixSegs;
090
091        /**
092         * Creates a new resource pattern.
093         * 
094         * @param pattern the pattern to be used for matching
095         * @throws ParseException if an invalid pattern is specified
096         */
097        public ResourcePattern(String pattern) throws ParseException {
098                this.pattern = pattern;
099                Matcher rpm = resourcePattern.matcher(pattern);
100                if (!rpm.matches()) {
101                        throw new ParseException("Invalid pattern: " + pattern, 0);
102                }
103                protocol = rpm.group("proto");
104                host = rpm.group("host");
105                port = rpm.group("port");
106                path = rpm.group("path");
107                if (path == null) {
108                        pathPatternElements = new String[0][];
109                } else {
110                        String[] paths = path.split(",");
111                        pathPatternElements = new String[paths.length][];
112                        prefixSegs = new int[paths.length];
113                        for (int i = 0; i < paths.length; i++) {
114                                List<String> segs = new ArrayList<>();
115                                prefixSegs[i] = 0;
116                                StringTokenizer st = new StringTokenizer(paths[i], "/|", true);
117                                while(st.hasMoreTokens()) {
118                                        String token = st.nextToken();
119                                        switch (token) {
120                                        case "/":
121                                                continue;
122                                        case "|":
123                                                prefixSegs[i] = segs.size();
124                                                continue;
125                                        default:
126                                                segs.add(token);
127                                                continue;
128                                        }
129                                }
130                                if (paths[i].endsWith("/") || paths[i].endsWith("|")) {
131                                        segs.add("");
132                                }
133                                pathPatternElements[i] = segs.toArray(new String[segs.size()]);
134                        }
135                }
136        }
137        
138        /**
139         * @return the pattern (string) that was used to create this 
140         * resource pattern.
141         */
142        public String pattern() {
143                return pattern;
144        }
145        
146        /**
147         * @return the protocol value specified in the pattern or {@code null}
148         */
149        public String protocol() {
150                return protocol;
151        }
152
153        /**
154         * @return the host value specified in the pattern or {@code null}
155         */
156        public String host() {
157                return host;
158        }
159
160        /**
161         * @return the port value specified in the pattern or {@code null}
162         */
163        public String port() {
164                return port;
165        }
166
167        /**
168         * @return the path value specified in the pattern or {@code null}
169         */
170        public String path() {
171                return path;
172        }
173
174        /**
175         * Matches the given resource URI against the pattern.
176         * 
177         * @param resource the URI specifying the resource to match
178         * @return -1 if the resource does not match, else the number
179         * of prefix segments (which may be 0)
180         */
181        public int matches(URI resource) {
182                if (protocol != null && !protocol.equals("*")) {
183                        if (resource.getScheme() == null) {
184                                return -1;
185                        }
186                        if (Arrays.stream(protocol.split(","))
187                                .noneMatch(p -> p.equals(resource.getScheme()))) {
188                                return -1;
189                        }
190                }
191                if (host != null && !host.equals("*")) {
192                        if (resource.getHost() == null
193                                || !resource.getHost().equals(host)) {
194                                return -1;
195                        }
196                }
197                if (port != null && !port.equals("*")) {
198                        if (Integer.parseInt(port) != resource.getPort()) {
199                                return -1;
200                        }
201                }
202
203                String[] reqElements = StreamSupport.stream(
204                                new PathSpliterator(resource.getPath()), false)
205                                .toArray(size -> new String[size]);
206                for (int pathIdx = 0; pathIdx < pathPatternElements.length; pathIdx++) {
207                        if (matchPath(pathPatternElements[pathIdx], reqElements)) {
208                                return prefixSegs[pathIdx];
209                        }
210                }
211                return -1;
212        }
213
214        /**
215         * Matches the given pattern against the given resource URI.
216         * 
217         * @param pattern the pattern to match
218         * @param resource the URI specifying the resource to match
219         * @return {@code true} if the resource URI matches
220         * @throws ParseException if an invalid pattern is specified
221         */
222        public static boolean matches(String pattern, URI resource)
223                throws ParseException {
224                return (new ResourcePattern(pattern)).matches(resource) >= 0;
225        }
226
227        private boolean matchPath(String[] patternElements, String[] reqElements) {
228                int pathIdx = 0;
229                int reqIdx = 0;
230                while (true) {
231                        if (pathIdx == patternElements.length) {
232                                return reqIdx == reqElements.length;
233                        }
234                        String matchElement = patternElements[pathIdx++];
235                        if (reqIdx == reqElements.length) {
236                                return false;
237                        }
238                        if (matchElement.equals("**")) {
239                                return true;
240                        }
241                        String reqElement = reqElements[reqIdx++];
242                        if (!matchElement.equals("*") && !matchElement.equals(reqElement)) {
243                                return false;
244                        }
245                }
246                
247        }

249        /* (non-Javadoc)
250         * @see java.lang.Object#toString()
251         */
252        @Override
253        public String toString() {
254                StringBuilder builder = new StringBuilder();
255                builder.append("ResourcePattern [");
256                builder.append(pattern);
257                builder.append("]");
258                return builder.toString();
259        }
260
261        /* (non-Javadoc)
262         * @see java.lang.Object#hashCode()
263         */
264        @Override
265        public int hashCode() {
266                final int prime = 31;
267                int result = 1;
268                result = prime * result + ((pattern == null) ? 0 : pattern.hashCode());
269                return result;
270        }
271
272        /* (non-Javadoc)
273         * @see java.lang.Object#equals(java.lang.Object)
274         */
275        @Override
276        public boolean equals(Object obj) {
277                if (this == obj) {
278                        return true;
279                }
280                if (obj == null) {
281                        return false;
282                }
283                if (getClass() != obj.getClass()) {
284                        return false;
285                }
286                ResourcePattern other = (ResourcePattern) obj;
287                if (pattern == null) {
288                        if (other.pattern != null) {
289                                return false;
290                        }
291                } else if (!pattern.equals(other.pattern)) {
292                        return false;
293                }
294                return true;
295        }
296        
297        /**
298         * Returns the segments of the path. If the path ends with a slash,
299         * an empty string is returned as final segment.
300         */
301        public static class PathSpliterator extends AbstractSpliterator<String> {
302                private StringTokenizer tokenizer;
303                private boolean endsWithSlash;
304
305                /**
306                 * Creates a new spliterator for the given path, using "/"
307                 * as path separator.
308                 * 
309                 * @param path the path
310                 */
311                public PathSpliterator(String path) {
312                        this(path, "/");
313                }
314                
315                /**
316                 * Creates a new spliterator for the given path, using 
317                 * the character from delimiters as seperators.
318                 * 
319                 * @param path the path
320                 * @param delimiters the delimiters
321                 */
322                public PathSpliterator(String path, String delimiters) {
323                        super(Long.MAX_VALUE, Spliterator.ORDERED 
324                                        | Spliterator.IMMUTABLE);
325                        tokenizer = new StringTokenizer(path, delimiters);
326                        endsWithSlash = path.endsWith("/");
327                }
328
329                /* (non-Javadoc)
330                 * @see java.util.Spliterator#tryAdvance(java.util.function.Consumer)
331                 */
332                @Override
333                public boolean tryAdvance(Consumer<? super String> consumer) {
334                        if (tokenizer == null) {
335                                return false;
336                        }
337                        if (tokenizer.hasMoreTokens()) {
338                                consumer.accept(tokenizer.nextToken());
339                                return true;
340                        }
341                        tokenizer = null;
342                        if (endsWithSlash) {
343                                consumer.accept("");
344                                return true;
345                        }
346                        return false;
347                }
348                
349        }
