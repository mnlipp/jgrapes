022import java.lang.annotation.Documented;
023import java.lang.annotation.ElementType;
024import java.lang.annotation.Retention;

028import java.text.ParseException;
029import java.util.Arrays;
030import java.util.HashSet;
031import java.util.Set;
032import java.util.stream.Collectors;

034import org.jgrapes.core.Channel;
035import org.jgrapes.core.ComponentType;
036import org.jgrapes.core.Components;
037import org.jgrapes.core.DefaultChannel;
038import org.jgrapes.core.Eligible;
039import org.jgrapes.core.Event;
040import org.jgrapes.core.HandlerScope;
041import org.jgrapes.core.Self;
042import org.jgrapes.core.annotation.Handler.NoChannel;
043import org.jgrapes.core.annotation.Handler.NoEvent;
044import org.jgrapes.core.annotation.HandlerDefinition;
045import org.jgrapes.core.internal.Common;

047import org.jgrapes.http.events.Request;

049/**
050 * This annotation marks a method as handler for events. The method is 
051 * invoked for events that have a type derived from {@link Request} and
052 * are matched by one of the specified {@link ResourcePattern}s.
053 * 
054 * @author Michael N. Lipp
055 */
056@Documented
057@Retention(value=RetentionPolicy.RUNTIME)
058@Target(value=ElementType.METHOD)
059@HandlerDefinition(evaluator=RequestHandler.Evaluator.class)
060public @interface RequestHandler {
061        
062        /**
063         * Specifies classes of events that the handler is to receive.
064         * 
065         * @return the event classes
066         */
067        @SuppressWarnings("rawtypes")
068        Class<? extends Event>[] events() default NoEvent.class;
069        
070        /**
071         * Specifies classes of channels that the handler listens on.
072         * 
073         * @return the channel classes
074         */
075        Class<? extends Channel>[] channels() default NoChannel.class;
076
077        /**
078         * Specifies the patterns that the handler is supposed to handle
079         * (see {@link ResourcePattern}).
080         * 
081         * @return the patterns
082         */
083        String[] patterns() default "";
084
085        /**
086         * Specifies a priority. The value is used to sort handlers.
087         * Handlers with higher priority are invoked first.
088         * 
089         * @return the priority
090         */
091        int priority() default 0;
092
093        /**
094         * Returns {@code true} if the annotated annotation defines a
095         * dynamic handler. A dynamic handler must be added to the set of
096         * handlers of a component explicitly.
097         * 
098         * @return the result
099         */
100        boolean dynamic() default false;
101        
102        /**
103         * This class provides the {@link Evaluator} for the {@link RequestHandler}
104         * annotation. It implements the behavior as described for the annotation.
105         * 
106         * @author Michael N. Lipp
107         */
108        public static class Evaluator implements HandlerDefinition.Evaluator {
109
110                /* (non-Javadoc)
111                 * @see org.jgrapes.core.annotation.HandlerDefinition.Evaluator#getPriority()
112                 */
113                @Override
114                public int priority(Annotation annotation) {
115                        return ((RequestHandler)annotation).priority();
116                }
117
118                @Override
119                public HandlerScope scope(
120                                ComponentType component, Method method, 
121                                Object[] eventValues, Object[] channelValues) {
122                        RequestHandler annotation 
123                                = method.getAnnotation(RequestHandler.class);
124                        if (annotation.dynamic()) {
125                                return null;
126                        }
127                        return new Scope(component, method, (RequestHandler)annotation,
128                                        null);
129                }
130
131                /**
132                 * Adds the given method of the given component as a 
133                 * dynamic handler for a specific pattern. Other informations
134                 * are taken from the annotation.
135                 * 
136                 * @param component the component
137                 * @param method the name of the method that implements the handler
138                 * @param pattern the pattern
139                 */
140                public static void add(ComponentType component, String method,
141                                String pattern) {
142                        try {
143                                for (Method m: component.getClass().getMethods()) {
144                                        if (!m.getName().equals(method)) {
145                                                continue;
146                                        }
147                                        if (m.getParameterTypes().length != 0
148                                                        && !(m.getParameterTypes().length == 1
149                                                                 && Event.class.isAssignableFrom(
150                                                                                 m.getParameterTypes()[0]))) {
151                                                continue;
152                                        }
153                                        for (Annotation annotation: m.getDeclaredAnnotations()) {
154                                                Class<?> annoType = annotation.annotationType();
155                                                HandlerDefinition hda 
156                                                        = annoType.getAnnotation(HandlerDefinition.class);
157                                                if (hda == null
158                                                        || !RequestHandler.class.isAssignableFrom(annoType)
159                                                        || !((RequestHandler)annotation).dynamic()) {
160                                                        continue;
161                                                }
162                                                Scope scope = new Scope(component, m, 
163                                                                (RequestHandler)annotation, pattern);
164                                                Components.manager(component)
165                                                        .addHandler(m, scope, 
166                                                                        ((RequestHandler)annotation).priority());
167                                                return;
168                                        }
169                                }
170                                throw new IllegalArgumentException(
171                                                "No method named \"" + method + "\" with DynamicHandler"
172                                                + " annotation and correct parameter list.");
173                        } catch (SecurityException e) {
174                                throw (RuntimeException)
175                                        (new IllegalArgumentException().initCause(e));
176                        }
177                }
178                
179                public static class Scope implements HandlerScope {
180
181                        private Set<Object> handledEventTypes = new HashSet<>();
182                        private Set<Object> handledChannels = new HashSet<>();
183                        private Set<ResourcePattern> handledPatterns = new HashSet<>();
184
185                        public Scope(ComponentType component, 
186                                        Method method, RequestHandler annotation, String pattern) {
187                                // Get all event keys from the handler annotation.
188                                if (annotation.events()[0] != NoEvent.class) {
189                                        handledEventTypes.addAll(Arrays.asList(annotation.events()));
190                                }
191                                // If no event types are given, try first parameter.
192                                if (handledEventTypes.isEmpty()) {
193                                        Class<?>[] paramTypes = method.getParameterTypes();
194                                        if (paramTypes.length > 0) {
195                                                if (Event.class.isAssignableFrom(paramTypes[0])) {
196                                                        handledEventTypes.add(paramTypes[0]);
197                                                }
198                                        }
199                                }
200                                
201                                // Get channel keys from the annotation.
202                                boolean addDefaultChannel = false;
203                                if (annotation.channels()[0] != NoChannel.class) {
204                                        for (Class<?> c: annotation.channels()) {
205                                                if (c == Self.class) {
206                                                        if (component instanceof Channel) {
207                                                                handledChannels
208                                                                        .add(((Channel)component).defaultCriterion());
209                                                        } else {
210                                                                throw new IllegalArgumentException(
211                                                                        "Canot use channel This.class in annotation"
212                                                                        + " of " + method + " because " 
213                                                                        + getClass().getName() 
214                                                                        + " does not implement Channel.");
215                                                        }
216                                                } else if (c == DefaultChannel.class) {
217                                                        addDefaultChannel = true;
218                                                } else {
219                                                        handledChannels.add(c);
220                                                }
221                                        }
222                                }
223                                if (handledChannels.size() == 0 || addDefaultChannel) {
224                                        handledChannels.add(Components.manager(component)
225                                                .channel().defaultCriterion());
226                                }
227                                
228                                try {
229                                // Get all paths from the annotation.
230                                        if (!annotation.patterns()[0].equals("")) {
231                                                for (String p: annotation.patterns()) {
232                                                        handledPatterns.add(new ResourcePattern(p));
233                                                }
234                                        }
235                                
236                                        // Add additionally provided path
237                                        if (pattern != null) {
238                                                handledPatterns.add(new ResourcePattern(pattern));
239                                        }
240                                } catch (ParseException e) {
241                                        throw new IllegalArgumentException(e.getMessage());
242                                }
243                        }
244                        
245                        @Override
246                        public boolean includes(Eligible event, Eligible[] channels) {
247                                boolean match = false;
248                                for (Object eventType: handledEventTypes) {
249                                        if (Class.class.isInstance(eventType)
250                                                && ((Class<?>) eventType)
251                                                        .isAssignableFrom(event.getClass())) {
252                                                match = true;
253                                                break;
254                                        }
255                                        
256                                }
257                                if (!match) {
258                                        return false;
259                                }
260                                match = false;
261                                // Try channels
262                                for (Eligible channel: channels) {
263                                        for (Object channelValue: handledChannels) {
264                                                if (channel.isEligibleFor(channelValue)) {
265                                                        match = true;
266                                                        break;
267                                                }
268                                        }
269                                }
270                                if (!match) {
271                                        return false;
272                                }
273                                if (!(event instanceof Request)) {
274                                        return false;
275                                }
276                                for (ResourcePattern rp: handledPatterns) {
277                                        if (((Request)event).isEligibleFor(
278                                                        Request.createMatchValue(Request.class, rp))) {
279                                                return true;
280                                        }
281                                }
282                                return false;
283                        }
284
285                        /* (non-Javadoc)
286                         * @see java.lang.Object#toString()
287                         */
288                        @Override
289                        public String toString() {
290                                StringBuilder builder = new StringBuilder();
291                                builder.append("Scope [");
292                                if (handledEventTypes != null) {
293                                        builder.append("handledEventTypes=");
294                                        builder.append(handledEventTypes.stream().map(v -> {
295                                                if (v instanceof Class) {
296                                                        return Common.classToString((Class<?>) v);
297                                                }
298                                                return v.toString();
299                                        }).collect(Collectors.toSet()));
300                                        builder.append(", ");
301                                }
302                                if (handledChannels != null) {
303                                        builder.append("handledChannels=");
304                                        builder.append(handledChannels);
305                                        builder.append(", ");
306                                }
307                                if (handledPatterns != null) {
308                                        builder.append("handledPatterns=");
309                                        builder.append(handledPatterns);
310                                }
311                                builder.append("]");
312                                return builder.toString();
313                        }
314                        
315                }
316        }
