061/**
062 * A dispatcher for requests for static content, usually files.
063 */
064public class StaticContentDispatcher extends Component {
065
066        private static MimetypesFileTypeMap typesMap = new MimetypesFileTypeMap();
067        
068        private ResourcePattern resourcePattern;
069        private URI contentRoot = null;
070        private Path contentDirectory = null;
071        private List<ValidityInfo> validityInfos = new ArrayList<>();
072        
073        /**
074         * Creates new dispatcher that tries to fulfill requests matching 
075         * the given resource pattern from the given content root.
076         * 
077         * An attempt is made to convert the content root to a {@link Path}
078         * in a {@link FileSystem}. If this fails, the content root is
079         * used as a URL against which requests are resolved and data
080         * is obtained by open an input stream from the resulting URL.
081         * In the latter case information such as directory listings and
082         * modification times aren't available. 
083         * 
084         * @param componentChannel this component's channel
085         * @param resourcePattern the pattern that requests must match 
086         * in order to be handled by this component 
087         * (see {@link ResourcePattern})
088         * @param contentRoot the location with content to serve 
089         * @see Component#Component(Channel)
090         */
091        public StaticContentDispatcher(Channel componentChannel, 
092                        String resourcePattern, URI contentRoot) {
093                super(componentChannel);
094                try {
095                        this.resourcePattern = new ResourcePattern(resourcePattern);
096                } catch (ParseException e) {
097                        throw new IllegalArgumentException(e);
098                }
099                try {
100                        this.contentDirectory = Paths.get(contentRoot);
101                } catch (FileSystemNotFoundException e) {
102                        this.contentRoot = contentRoot;
103                }
104                RequestHandler.Evaluator.add(this, "onGet", resourcePattern);
105        }
106
107        /**
108         * Creates a new component base with its channel set to
109         * itself.
110         * 
111         * @param resourcePattern the pattern that requests must match with to 
112         * be handled by this component 
113         * (see {@link ResourcePattern#matches(String, java.net.URI)})
114         * @param contentRoot the location with content to serve 
115         * @see Component#Component()
116         */
117        public StaticContentDispatcher(String resourcePattern, 
118                        URI contentRoot) {
119                this(Channel.SELF, resourcePattern, contentRoot);
120        }
121
122        /**
123         * Returns the validity infos as unmodifiable list.
124         * 
125         * @return the validityInfos
126         * @see #validityInfos()
127         */
128        public List<ValidityInfo> validityInfos() {
129                return Collections.unmodifiableList(validityInfos);
130        }
131
132        /**
133         * Sets validity infos for generating the `Cache-Control` (`max-age`)
134         * header of the response. If the type of the resource served 
135         * matches an entry in the list, the entry's `maxAge` value is used 
136         * for generating the header. If no info matches or no infos are set, 
137         * a default of 600 seconds is used as time span.
138         * 
139         * @param validityInfos the validityInfos to set
140         */
141        public void setValidityInfos(List<ValidityInfo> validityInfos) {
142                this.validityInfos = validityInfos;
143        }
144
145        /**
146         * Add a validity info to the list.
147         * 
148         * @return this object for easy chaining
149         * @see #validityInfos()
150         */
151        public StaticContentDispatcher addValidityInfo(ValidityInfo info) {
152                validityInfos.add(info);
153                return this;
154        }
155
156        @RequestHandler(dynamic=true)
157        public void onGet(GetRequest event, IOSubchannel channel)
158                        throws ParseException, IOException {
159                int prefixSegs = resourcePattern.matches(event.requestUri());
160                if (prefixSegs < 0) {
161                        return;
162                }
163                if (!(contentDirectory != null 
164                                ? getFromFileSystem(event, channel, prefixSegs)
165                                : getFromUri(event, channel, prefixSegs))) {
166                                return;
167                }
168                event.stop();
169        }
170
171        private boolean getFromFileSystem(GetRequest event, IOSubchannel channel,
172                int prefixSegs) throws IOException, ParseException {
173                // Final wrapper for usage in closure
174                final Path[] assembly = new Path[] { contentDirectory };
175                Arrays.stream(event.requestUri().getPath().split("/"))
176                        .skip(prefixSegs + 1)
177                        .forEach(e -> assembly[0] = assembly[0].resolve(e));
178                Path resourcePath = assembly[0];
179                if (Files.isDirectory(resourcePath)) {
180                        Path indexPath = resourcePath.resolve("index.html");
181                        if (Files.isReadable(indexPath)) {
182                                resourcePath = indexPath;
183                        } else {
184                                return false;
185                        }
186                }
187                if (!Files.isReadable(resourcePath)) {
188                        return false;
189                }
190                
191                // Get content type and derive max-age
192                String mimeTypeName;
193                try {
194                        mimeTypeName = Files.probeContentType(resourcePath);
195                } catch(IOException e) {
196                        mimeTypeName = null;
197                }
198                if (mimeTypeName == null) {
199                        // probeContentType has been reported to fail on some platforms.
200                        // Use old approach as fall back.
201                        mimeTypeName = typesMap.getContentType(resourcePath.toFile());
202                }
203                MediaType mediaType = Converters.MEDIA_TYPE.fromFieldValue(mimeTypeName);
204                long maxAge = 600;
205                for (ValidityInfo info: validityInfos) {
206                        if (info.mediaRange().matches(mediaType)) {
207                                break;
208                        }
209                }
210
211                // Set max age in cache-control header
212                List<Directive> directives = new ArrayList<>();
213                directives.add(new Directive("max-age", maxAge));
214                HttpResponse response = event.httpRequest().response().get();
215                response.setField(HttpField.CACHE_CONTROL, directives);
216
217                // Check if sending is really required.
218                Instant lastModified = Files.getLastModifiedTime(resourcePath)
219                                .toInstant().with(ChronoField.NANO_OF_SECOND, 0);
220                Optional<Instant> modifiedSince = event.httpRequest()
221                                .findValue(HttpField.IF_MODIFIED_SINCE, Converters.DATE_TIME);
222                if (modifiedSince.isPresent() 
223                                && !lastModified.isAfter(modifiedSince.get())) {
224                        response.setStatus(HttpStatus.NOT_MODIFIED);
225                        channel.respond(new Response(response));
226                } else {
227                        if ("text".equals(mediaType.topLevelType())) {
228                                mediaType = MediaType.builder().from(mediaType)
229                                                .setParameter("charset", System.getProperty(
230                                                                "file.encoding", "UTF-8")).build();
231                        }
232                        response.setField(HttpField.CONTENT_TYPE, mediaType);
233                        response.setStatus(HttpStatus.OK);
234                        response.setHasPayload(true);
235                        response.setField(HttpField.LAST_MODIFIED, lastModified);
236                        channel.respond(new Response(response));
237                        fire(new StreamFile(resourcePath, StandardOpenOption.READ), channel);
238                }
239                return true;
240        }
241
242        private boolean getFromUri(GetRequest event, IOSubchannel channel,
243                int prefixSegs) throws ParseException {
244                // Final wrapper for usage in closure
245                final URI[] assembly = new URI[] { contentRoot };
246                Arrays.stream(event.requestUri().getPath().split("/"))
247                        .skip(prefixSegs + 1)
248                        .forEach(e -> assembly[0] = assembly[0].resolve(e));
249                URL resourceUrl;
250                try {
251                        resourceUrl = assembly[0].toURL();
252                } catch (MalformedURLException e1) {
253                        return false;
254                }
255                URLConnection resConn;
256                InputStream resIn;
257                try {
258                        resConn = resourceUrl.openConnection();
259                        resIn = resConn.getInputStream();
260                } catch (IOException e1) {
261                        try {
262                                resourceUrl = resourceUrl.toURI().resolve("index.html").toURL();
263                                resConn = resourceUrl.openConnection();
264                                resIn = resConn.getInputStream();
265                        } catch (URISyntaxException | IOException e2) {
266                                return false;
267                        }
268                }
269                
270                // Get content type and derive max-age
271                String mimeTypeName;
272                try {
273                        // probeContentType is most advanced, but may fail if it tries
274                        // to look at the file's content (which doesn't exist).
275                        mimeTypeName = Files.probeContentType(
276                                        Paths.get(resourceUrl.getPath()));
277                } catch (IOException e) {
278                        mimeTypeName = null;
279                }
280                if (mimeTypeName == null) {
281                        mimeTypeName = typesMap.getContentType(resourceUrl.getPath());
282                }
283                MediaType mediaType = Converters.MEDIA_TYPE.fromFieldValue(mimeTypeName);
284                long maxAge = 600;
285                for (ValidityInfo info: validityInfos) {
286                        if (info.mediaRange().matches(mediaType)) {
287                                break;
288                        }
289                }
290
291                // Set max age in cache-control header
292                List<Directive> directives = new ArrayList<>();
293                directives.add(new Directive("max-age", maxAge));
294                HttpResponse response = event.httpRequest().response().get();
295                response.setField(HttpField.CACHE_CONTROL, directives);
296
297                // Send response 
298                if ("text".equals(mediaType.topLevelType())) {
299                        mediaType = MediaType.builder().from(mediaType)
300                                        .setParameter("charset", System.getProperty(
301                                                        "file.encoding", "UTF-8")).build();
302                }
303                response.setField(HttpField.CONTENT_TYPE, mediaType);
304                response.setStatus(HttpStatus.OK);
305                response.setHasPayload(true);
306                response.setField(HttpField.LAST_MODIFIED, Instant.now());
307                channel.respond(new Response(response));
308                
309                // Start sending content
310                channel.responsePipeline().executorService()
311                        .submit(new InputStreamPipeline(resIn, channel));
312                return true;
313        }

315        /**
316         * Describes an association between a media range and a  
317         * maximum age in seconds. 
318         */
319        public static class ValidityInfo {
320                private MediaRange mediaRange;
321                private long maxAge;
322
323                /**
324                 * @param mediaRange
325                 * @param maxAge
326                 */
327                public ValidityInfo(MediaRange mediaRange, long maxAge) {
328                        super();
329                        this.mediaRange = mediaRange;
330                        this.maxAge = maxAge;
331                }
332
333                /**
334                 * The media range matches by this info.
335                 * 
336                 * @return the mediaRange
337                 */
338                public MediaRange mediaRange() {
339                        return mediaRange;
340                }
341                
342                /**
343                 * The time span in seconds until the response expires.
344                 * 
345                 * @return the time span
346                 */
347                public long maxAge() {
348                        return maxAge;
349                }
350        }
