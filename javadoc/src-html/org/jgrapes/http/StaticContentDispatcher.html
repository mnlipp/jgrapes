036import java.util.List;
037import java.util.Optional;
038
039import org.jdrupes.httpcodec.protocols.http.HttpConstants.HttpStatus;
040import org.jdrupes.httpcodec.protocols.http.HttpField;

053/**
054 * A dispatcher for requests for static content, usually files.
055 */
056public class StaticContentDispatcher extends Component {
057
058        private ResourcePattern resourcePattern;
059        private URI contentRoot = null;
060        private Path contentDirectory = null;
061        private List<ValidityInfo> validityInfos = new ArrayList<>();
062        
063        /**
064         * Creates new dispatcher that tries to fulfill requests matching 
065         * the given resource pattern from the given content root.
066         * 
067         * An attempt is made to convert the content root to a {@link Path}
068         * in a {@link FileSystem}. If this fails, the content root is
069         * used as a URL against which requests are resolved and data
070         * is obtained by open an input stream from the resulting URL.
071         * In the latter case modification times aren't available. 
072         * 
073         * @param componentChannel this component's channel
074         * @param resourcePattern the pattern that requests must match 
075         * in order to be handled by this component 
076         * (see {@link ResourcePattern})
077         * @param contentRoot the location with content to serve 
078         * @see Component#Component(Channel)
079         */
080        public StaticContentDispatcher(Channel componentChannel, 
081                        String resourcePattern, URI contentRoot) {
082                super(componentChannel);
083                try {
084                        this.resourcePattern = new ResourcePattern(resourcePattern);
085                } catch (ParseException e) {
086                        throw new IllegalArgumentException(e);
087                }
088                try {
089                        this.contentDirectory = Paths.get(contentRoot);
090                } catch (FileSystemNotFoundException e) {
091                        this.contentRoot = contentRoot;
092                }
093                RequestHandler.Evaluator.add(this, "onGet", resourcePattern);
094        }
095
096        /**
097         * Creates a new component base with its channel set to
098         * itself.
099         * 
100         * @param resourcePattern the pattern that requests must match with to 
101         * be handled by this component 
102         * (see {@link ResourcePattern#matches(String, java.net.URI)})
103         * @param contentRoot the location with content to serve 
104         * @see Component#Component()
105         */
106        public StaticContentDispatcher(String resourcePattern, URI contentRoot) {
107                this(Channel.SELF, resourcePattern, contentRoot);
108        }
109
110        /**
111         * Returns the validity infos as unmodifiable list.
112         * 
113         * @return the validityInfos
114         * @see #validityInfos()
115         */
116        public List<ValidityInfo> validityInfos() {
117                return Collections.unmodifiableList(validityInfos);
118        }
119
120        /**
121         * Sets validity infos for generating the `Cache-Control` (`max-age`)
122         * header of the response. If the type of the resource served 
123         * matches an entry in the list, the entry's `maxAge` value is used 
124         * for generating the header. If no info matches or no infos are set, 
125         * a default of 600 seconds is used as time span.
126         * 
127         * @param validityInfos the validityInfos to set
128         */
129        public void setValidityInfos(List<ValidityInfo> validityInfos) {
130                this.validityInfos = validityInfos;
131        }
132
133        /**
134         * Add a validity info to the list.
135         * 
136         * @return this object for easy chaining
137         * @see #validityInfos()
138         */
139        public StaticContentDispatcher addValidityInfo(ValidityInfo info) {
140                validityInfos.add(info);
141                return this;
142        }
143
144        @RequestHandler(dynamic=true)
145        public void onGet(GetRequest event, IOSubchannel channel)
146                        throws ParseException, IOException {
147                int prefixSegs = resourcePattern.matches(event.requestUri());
148                if (prefixSegs < 0) {
149                        return;
150                }
151                if (contentDirectory != null) {
152                        getFromFileSystem(event, channel, prefixSegs);
153                } else {
154                        getFromUri(event, channel, prefixSegs);
155                }
156        }
157
158        private boolean getFromFileSystem(GetRequest event, IOSubchannel channel,
159                int prefixSegs) throws IOException, ParseException {
160                // Final wrapper for usage in closure
161                final Path[] assembly = new Path[] { contentDirectory };
162                Arrays.stream(event.requestUri().getPath().split("/"))
163                        .skip(prefixSegs + 1)
164                        .forEach(e -> assembly[0] = assembly[0].resolve(e));
165                Path resourcePath = assembly[0];
166                if (Files.isDirectory(resourcePath)) {
167                        Path indexPath = resourcePath.resolve("index.html");
168                        if (Files.isReadable(indexPath)) {
169                                resourcePath = indexPath;
170                        } else {
171                                return false;
172                        }
173                }
174                if (!Files.isReadable(resourcePath)) {
175                        return false;
176                }
177                
178                // Get content type
179                HttpResponse response = event.httpRequest().response().get();
180                MediaType mediaType = HttpResponse.contentType(resourcePath.toUri());
181                
182                // Derive max-age
183                ResponseCreationSupport.setMaxAge(response, validityInfos, mediaType, 600);
184
185                // Check if sending is really required.
186                Instant lastModified = Files.getLastModifiedTime(resourcePath)
187                                .toInstant().with(ChronoField.NANO_OF_SECOND, 0);
188                Optional<Instant> modifiedSince = event.httpRequest()
189                                .findValue(HttpField.IF_MODIFIED_SINCE, Converters.DATE_TIME);
190                event.stop();
191                if (modifiedSince.isPresent() 
192                                && !lastModified.isAfter(modifiedSince.get())) {
193                        response.setStatus(HttpStatus.NOT_MODIFIED);
194                        response.setField(HttpField.LAST_MODIFIED, lastModified);
195                        channel.respond(new Response(response));
196                } else {
197                        response.setContentType(mediaType);
198                        response.setStatus(HttpStatus.OK);
199                        response.setField(HttpField.LAST_MODIFIED, lastModified);
200                        channel.respond(new Response(response));
201                        fire(new StreamFile(resourcePath, StandardOpenOption.READ), channel);
202                }
203                return true;
204        }
205
206        private boolean getFromUri(GetRequest event, IOSubchannel channel,
207                int prefixSegs) throws ParseException {
208                return ResponseCreationSupport.sendStaticContent(
209                                event, channel, path -> {
210                                        try {
211                                                return contentRoot.resolve(
212                                                                ResponseCreationSupport.removeSegments(
213                                                                                path, prefixSegs + 1)).toURL();
214                                        } catch (MalformedURLException e) {
215                                                throw new IllegalArgumentException(e);
216                                        }
217                                }, validityInfos);
218        }
219        
220}
