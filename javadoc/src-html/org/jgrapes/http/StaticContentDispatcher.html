021import java.io.IOException;
022import java.nio.file.Files;
023import java.nio.file.Path;
024import java.nio.file.StandardOpenOption;
025import java.text.ParseException;
026import java.time.Instant;
027import java.time.temporal.ChronoField;
028import java.util.ArrayList;
029import java.util.Arrays;
030import java.util.Collections;
031import java.util.List;
032import java.util.Optional;
033
034import org.jdrupes.httpcodec.protocols.http.HttpConstants.HttpStatus;
035import org.jdrupes.httpcodec.protocols.http.HttpField;
036import org.jdrupes.httpcodec.protocols.http.HttpResponse;
037import org.jdrupes.httpcodec.types.Converters;
038import org.jdrupes.httpcodec.types.Directive;
039import org.jdrupes.httpcodec.types.MediaRange;
040import org.jdrupes.httpcodec.types.MediaType;

049/**
050 */
051public class StaticContentDispatcher extends Component {
052
053        private ResourcePattern resourcePattern;
054        private Path contentDirectory;
055        private List<ValidityInfo> validityInfos = new ArrayList<>();
056        
057        /**
058         * @param resourcePattern the pattern that requests must match with to 
059         * be handled by this component 
060         * (see {@link ResourcePattern#matches(String, java.net.URI)})
061         * @param contentDirectory the directory with content to serve 
062         * @see Component#Component()
063         */
064        public StaticContentDispatcher(String resourcePattern, 
065                        Path contentDirectory) {
066                this(Channel.SELF, resourcePattern, contentDirectory);
067        }
068
069        /**
070         * @param componentChannel
071         *            this component's channel
072         * @param resourcePattern the pattern that requests must match 
073         * in order to be handled by this component 
074         * (see {@link ResourcePattern#matches(String, java.net.URI)})
075         * @param contentDirectory the directory with content to serve 
076         * @see Component#Component(Channel)
077         */
078        public StaticContentDispatcher(Channel componentChannel, 
079                        String resourcePattern, Path contentDirectory) {
080                super(componentChannel);
081                try {
082                        this.resourcePattern = new ResourcePattern(resourcePattern);
083                } catch (ParseException e) {
084                        throw new IllegalArgumentException(e);
085                }
086                this.contentDirectory = contentDirectory;
087                RequestHandler.Evaluator.add(this, "onGet", resourcePattern);
088        }
089
090        /**
091         * Returns the validity infos as unmodifiable list.
092         * 
093         * @return the validityInfos
094         * @see #validityInfos()
095         */
096        public List<ValidityInfo> validityInfos() {
097                return Collections.unmodifiableList(validityInfos);
098        }
099
100        /**
101         * Sets validity infos for generating the `Cache-Control` (`max-age`)
102         * header of the response. If the type of the resource served 
103         * matches an entry in the list, the entry's `maxAge` value is used 
104         * for generating the header. If no info matches or no infos are set, 
105         * a default of 600 seconds is used as time span.
106         * 
107         * @param validityInfos the validityInfos to set
108         */
109        public void setValidityInfos(List<ValidityInfo> validityInfos) {
110                this.validityInfos = validityInfos;
111        }
112
113        /**
114         * Add a validity info to the list.
115         * 
116         * @return this object for easy chaining
117         * @see #validityInfos()
118         */
119        public StaticContentDispatcher addValidityInfo(ValidityInfo info) {
120                validityInfos.add(info);
121                return this;
122        }
123
124        @RequestHandler(dynamic=true)
125        public void onGet(GetRequest event, IOSubchannel channel)
126                        throws ParseException, IOException {
127                int prefixSegs = resourcePattern.matches(event.requestUri());
128                if (prefixSegs < 0) {
129                        return;
130                }
131                
132                // Final wrapper for usage in closure
133                final Path[] assembly = new Path[] { contentDirectory };
134                Arrays.stream(event.requestUri().getPath().split("/"))
135                        .skip(prefixSegs + 1)
136                        .forEach(e -> assembly[0] = assembly[0].resolve(e));
137                Path resourcePath = assembly[0];
138                if (Files.isDirectory(resourcePath)) {
139                        Path indexPath = resourcePath.resolve("index.html");
140                        if (Files.isReadable(indexPath)) {
141                                resourcePath = indexPath;
142                        } else {
143                                return;
144                        }
145                }
146                if (!Files.isReadable(resourcePath)) {
147                        return;
148                }
149                
150                // Get content type and derive max-age
151                String mimeTypeName = Files.probeContentType(resourcePath);
152                if (mimeTypeName == null) {
153                        mimeTypeName = "application/octet-stream";
154                }
155                MediaType mediaType = Converters.MEDIA_TYPE.fromFieldValue(mimeTypeName);
156                long maxAge = 600;
157                for (ValidityInfo info: validityInfos) {
158                        if (info.mediaRange().matches(mediaType)) {
159                                break;
160                        }
161                }
162
163                // Set max age in cache-control header
164                List<Directive> directives = new ArrayList<>();
165                directives.add(new Directive("max-age", maxAge));
166                HttpResponse response = event.request().response().get();
167                response.setField(HttpField.CACHE_CONTROL, directives);
168
169                // Check if sending is really required.
170                Instant lastModified = Files.getLastModifiedTime(resourcePath)
171                                .toInstant().with(ChronoField.NANO_OF_SECOND, 0);
172                Optional<Instant> modifiedSince = event.request()
173                                .findValue(HttpField.IF_MODIFIED_SINCE, Converters.DATE_TIME);
174                if (modifiedSince.isPresent() 
175                                && !lastModified.isAfter(modifiedSince.get())) {
176                        response.setStatus(HttpStatus.NOT_MODIFIED);
177                        channel.respond(new Response(response));
178                } else {
179                        if ("text".equals(mediaType.topLevelType())) {
180                                mediaType = MediaType.builder().from(mediaType)
181                                                .setParameter("charset", System.getProperty(
182                                                                "file.encoding", "UTF-8")).build();
183                        }
184                        response.setField(HttpField.CONTENT_TYPE, mediaType);
185                        response.setStatus(HttpStatus.OK);
186                        response.setMessageHasBody(true);
187                        response.setField(HttpField.LAST_MODIFIED, lastModified);
188                        channel.respond(new Response(response));
189                        fire(new StreamFile(resourcePath, StandardOpenOption.READ), channel);
190                }
191                event.stop();
192        }
193
194        /**
195         * Describes an association between a media range and a  
196         * maximum age in seconds. 
197         */
198        public static class ValidityInfo {
199                private MediaRange mediaRange;
200                private long maxAge;
201
202                /**
203                 * @param mediaRange
204                 * @param maxAge
205                 */
206                public ValidityInfo(MediaRange mediaRange, long maxAge) {
207                        super();
208                        this.mediaRange = mediaRange;
209                        this.maxAge = maxAge;
210                }
211
212                /**
213                 * The media range matches by this info.
214                 * 
215                 * @return the mediaRange
216                 */
217                public MediaRange mediaRange() {
218                        return mediaRange;
219                }
220                
221                /**
222                 * The time span in seconds until the response expires.
223                 * 
224                 * @return the time span
225                 */
226                public long maxAge() {
227                        return maxAge;
228                }
229        }
