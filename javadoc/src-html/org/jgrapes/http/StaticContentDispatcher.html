019package org.jgrapes.http;
020
021import java.io.IOException;
022import java.nio.file.Files;
023import java.nio.file.Path;
024import java.nio.file.StandardOpenOption;
025import java.text.ParseException;
026import java.time.Instant;
027import java.time.temporal.ChronoField;
028import java.util.Arrays;
029import java.util.Optional;
030
031import org.jdrupes.httpcodec.protocols.http.HttpConstants.HttpStatus;
032import org.jdrupes.httpcodec.protocols.http.HttpField;
033import org.jdrupes.httpcodec.protocols.http.HttpResponse;
034import org.jdrupes.httpcodec.types.Converters;
035import org.jdrupes.httpcodec.types.MediaType;
036import org.jgrapes.core.Channel;
037import org.jgrapes.core.Component;
038import org.jgrapes.http.annotation.RequestHandler;
039import org.jgrapes.http.events.GetRequest;
040import org.jgrapes.http.events.Response;
041import org.jgrapes.io.IOSubchannel;
042import org.jgrapes.io.events.StreamFile;
043
044/**
045 * @author Michael N. Lipp
046 */
047public class StaticContentDispatcher extends Component {
048
049        private ResourcePattern resourcePattern;
050        private Path contentDirectory;
051        
052        /**
053         * @param resourcePattern the pattern that requests must match with to 
054         * be handled by this component 
055         * (see {@link ResourcePattern#matches(String, java.net.URI)})
056         * @param contentDirectory the directory with content to serve 
057         * @see Component#Component()
058         */
059        public StaticContentDispatcher(String resourcePattern, 
060                        Path contentDirectory) {
061                this(Channel.SELF, resourcePattern, contentDirectory);
062        }
063
064        /**
065         * @param componentChannel
066         *            this component's channel
067         * @param resourcePattern the pattern that requests must match 
068         * in order to be handled by this component 
069         * (see {@link ResourcePattern#matches(String, java.net.URI)})
070         * @param contentDirectory the directory with content to serve 
071         * @see Component#Component(Channel)
072         */
073        public StaticContentDispatcher(Channel componentChannel, 
074                        String resourcePattern, Path contentDirectory) {
075                super(componentChannel);
076                try {
077                        this.resourcePattern = new ResourcePattern(resourcePattern);
078                } catch (ParseException e) {
079                        throw new IllegalArgumentException(e);
080                }
081                this.contentDirectory = contentDirectory;
082                RequestHandler.Evaluator.add(this, "onGet", resourcePattern);
083        }
084
085        @RequestHandler(dynamic=true)
086        public void onGet(GetRequest event) throws ParseException, IOException {
087                int prefixSegs = resourcePattern.matches(event.getRequestUri());
088                if (prefixSegs < 0) {
089                        return;
090                }
091                // Final wrapper for usage in closure
092                final Path[] assembly = new Path[] { contentDirectory };
093                Arrays.stream(event.getRequestUri().getPath().split("/"))
094                        .skip(prefixSegs + 1)
095                        .forEach(e -> assembly[0] = assembly[0].resolve(e));
096                Path resourcePath = assembly[0];
097                if (Files.isDirectory(resourcePath)) {
098                        Path indexPath = resourcePath.resolve("index.html");
099                        if (Files.isReadable(indexPath)) {
100                                resourcePath = indexPath;
101                        } else {
102                                return;
103                        }
104                }
105                if (!Files.isReadable(resourcePath)) {
106                        return;
107                }
108                // Check if sending is really required.
109                Instant lastModified = Files.getLastModifiedTime(resourcePath)
110                                .toInstant().with(ChronoField.NANO_OF_SECOND, 0);
111                Optional<Instant> modifiedSince = event.getRequest()
112                                .getValue(HttpField.IF_MODIFIED_SINCE, Converters.DATE_TIME);
113                HttpResponse response = event.getRequest().getResponse().get();
114                IOSubchannel channel = event.firstChannel(IOSubchannel.class);
115                if (modifiedSince.isPresent() 
116                                && !lastModified.isAfter(modifiedSince.get())) {
117                        response.setStatus(HttpStatus.NOT_MODIFIED);
118                        channel.fire(new Response(response));
119                } else {
120                        String mimeTypeName = Files.probeContentType(resourcePath);
121                        if (mimeTypeName == null) {
122                                mimeTypeName = "application/octet-stream";
123                        }
124                        MediaType mediaType = MediaType.fromString(mimeTypeName);
125                        if ("text".equals(mediaType.getTopLevelType())) {
126                                mediaType = MediaType.builder().from(mediaType)
127                                                .setParameter("charset", System.getProperty(
128                                                                "file.encoding", "UTF-8")).build();
129                        }
130                        response.setField(HttpField.CONTENT_TYPE, mediaType);
131                        response.setStatus(HttpStatus.OK);
132                        response.setMessageHasBody(true);
133                        response.setField(HttpField.LAST_MODIFIED, lastModified);
134                        channel.fire(new Response(response));
135                        channel.fire(new StreamFile(resourcePath, StandardOpenOption.READ));
136                }
137                event.stop();
138        }
139
140}
