058/**
059 * A base class for session managers. A session manager associates 
060 * {@link Request} events with a {@link Session} object using 
061 * `Session.class` as association identifier. The {@link Request}
062 * handler has a default priority of 1000.
063 * 
064 * Managers track requests using a cookie with a given name and path. The 
065 * path is a prefix that has to be matched by the request, often "/".
066 * If no cookie with the given name (see {@link #idName()}) is found,
067 * a new cookie with that name and the specified path is created.
068 * The cookie's value is the unique session id that is used to lookup
069 * the session object.
070 * 
071 * Session managers provide additional support for web sockets. If a
072 * web socket is accepted, the session associated with the request
073 * is automatically associated with the {@link IOSubchannel} that
074 * is subsequently used for the web socket events. This allows
075 * handlers for web socket messages to access the session like
076 * {@link Request} handlers.
077 * 
078 * @see EventBase#setAssociated(Object, Object)
079 * @see "[OWASP Session Management Cheat Sheet](https://www.owasp.org/index.php/Session_Management_Cheat_Sheet)"
080 */
081public abstract class SessionManager extends Component {
082        
083        private static SecureRandom secureRandom = new SecureRandom();
084
085        private String idName = "id";
086        private String path = "/";
087        private long absoluteTimeout = 9*60*60*1000;
088        private long idleTimeout = 30*60*1000;
089        private int maxSessions = 1000;
090        
091        /**
092         * Creates a new session manager with its channel set to
093         * itself and the path set to "/". The manager handles
094         * all {@link Request} events.
095         */
096        public SessionManager() {
097                this("/");
098        }
099        
100        /**
101         * Creates a new session manager with its channel set to
102         * itself and the path set to the given path. The manager
103         * handles all requests that match the given path, using the
104         * same rules as browsers do for selecting the cookies that
105         * are to be sent.
106         * 
107         * @param path the path
108         */
109        public SessionManager(String path) {
110                this(Channel.SELF, path);
111        }
112
113        /**
114         * Creates a new session manager with its channel set to
115         * the given channel and the path to "/". The manager handles
116         * all {@link Request} events.
117         * 
118         * @param componentChannel the component channel
119         */
120        public SessionManager(Channel componentChannel) {
121                this(componentChannel, "/");
122        }
123
124        /**
125         * Creates a new session manager with the given channel and path.
126         * The manager handles all requests that match the given path, using
127         * the same rules as browsers do for selecting the cookies that
128         * are to be sent.
129         *  
130         * @param componentChannel the component channel
131         * @param path the path
132         */
133        public SessionManager(Channel componentChannel, String path) {
134                this(componentChannel, derivePattern(path), 1000, path);
135        }
136
137        /**
138         * Derives the resource pattern from the path.
139         *
140         * @param path the path
141         * @return the pattern
142         */
143        protected static String derivePattern(String path) {
144                String pattern;
145                if (path.equals("/")) {
146                        pattern = "/**";
147                } else {
148                        String patternBase = path;
149                        if (patternBase.endsWith("/")) {
150                                patternBase = path.substring(0, path.length() - 1);
151                        }
152                        pattern = path + "," + path + "/**";
153                }
154                return pattern;
155        }
156        
157        /**
158         * Creates a new session manager using the given channel and path.
159         * The manager handles only requests that match the given pattern.
160         * The handler is registered with the given priority.
161         * 
162         * This constructor can be used if special handling of top level
163         * requests is needed.
164         *
165         * @param componentChannel the component channel
166         * @param pattern the path part of a {@link ResourcePattern}
167         * @param priority the priority
168         * @param path the path
169         */
170        public SessionManager(Channel componentChannel, String pattern, 
171                        int priority, String path) {
172                super(componentChannel);
173                this.path = path;
174                RequestHandler.Evaluator.add(this, "onRequest", pattern, priority);
175                MBeanView.addManager(this);
176        }
177
178        /**
179         * The name used for the session id cookie. Defaults to "`id`".
180         * 
181         * @return the id name
182         */
183        public String idName() {
184                return idName;
185        }
186        
187        /**
188         * @param idName the id name to set
189         * 
190         * @return the session manager for easy chaining
191         */
192        public SessionManager setIdName(String idName) {
193                this.idName = idName;
194                return this;
195        }
196
197        /**
198         * Set the maximum number of sessions. If the value is zero or less,
199         * an unlimited number of sessions is supported. The default value
200         * is 1000.
201         * 
202         * If adding a new session would exceed the limit, first all
203         * sessions older than {@link #absoluteTimeout()} are removed.
204         * If this doesn't free a slot, the least recently used session
205         * is removed.
206         * 
207         * @param maxSessions the maxSessions to set
208         * @return the session manager for easy chaining
209         */
210        public SessionManager setMaxSessions(int maxSessions) {
211                this.maxSessions = maxSessions;
212                return this;
213        }
214
215        /**
216         * @return the maxSessions
217         */
218        public int maxSessions() {
219                return maxSessions;
220        }
221
222        /**
223         * Sets the absolute timeout for a session in seconds. The absolute
224         * timeout is the time after which a session is invalidated (relative
225         * to its creation time). Defaults to 9 hours. Zero or less disables
226         * the timeout.
227         * 
228         * @param absoluteTimeout the absolute timeout
229         * @return the session manager for easy chaining
230         */
231        public SessionManager setAbsoluteTimeout(int absoluteTimeout) {
232                this.absoluteTimeout = absoluteTimeout * 1000;
233                return this;
234        }
235
236        /**
237         * @return the absolute session timeout (in seconds)
238         */
239        public int absoluteTimeout() {
240                return (int)(absoluteTimeout / 1000);
241        }
242
243        /**
244         * Sets the idle timeout for a session in seconds. Defaults to 30 minutes.
245         * Zero or less disables the timeout. 
246         * 
247         * @param idleTimeout the absolute timeout
248         * @return the session manager for easy chaining
249         */
250        public SessionManager setIdleTimeout(int idleTimeout) {
251                this.idleTimeout = idleTimeout * 1000;
252                return this;
253        }
254
255        /**
256         * @return the idle timeout (in seconds)
257         */
258        public int idleTimeout() {
259                return (int)(idleTimeout / 1000);
260        }
261
262        /**
263         * Associates the event with a {@link Session} object
264         * using `Session.class` as association identifier.
265         * 
266         * @param event the event
267         */
268        @RequestHandler(dynamic=true)
269        public void onRequest(Request event) {
270                if (event.associated(Session.class).isPresent()) {
271                        return;
272                }
273                final HttpRequest request = event.httpRequest();
274                Optional<String> requestedSessionId = request.findValue(
275                        HttpField.COOKIE, Converters.COOKIE_LIST)
276                        .flatMap(cookies -> cookies.stream().filter(
277                                cookie -> cookie.getName().equals(idName()))
278                                .findFirst().map(HttpCookie::getValue));
279                if (requestedSessionId.isPresent()) {
280                        String sessionId = requestedSessionId.get();
281                        synchronized(SessionManager.this) {
282                                Optional<Session> session = lookupSession(sessionId);
283                                if (session.isPresent()) {
284                                        Instant now = Instant.now();
285                                        if ((absoluteTimeout <= 0
286                                                        || Duration.between(session.get().createdAt(), 
287                                                                        now).toMillis() < absoluteTimeout)
288                                                && (idleTimeout <= 0
289                                                        || Duration.between(session.get().lastUsedAt(),
290                                                                        now).toMillis() < idleTimeout)) {
291                                                event.setAssociated(Session.class, session.get());
292                                                session.get().updateLastUsedAt();
293                                                return;
294                                        }
295                                        // Invalidate, too old 
296                                        removeSession(sessionId);
297                                }
298                        }
299                }
300                String sessionId = createSessionId(request.response().get());
301                Session session = createSession(sessionId);
302                event.setAssociated(Session.class, session);
303        }
304
305        /**
306         * Creates a new session with the given id.
307         * 
308         * @param sessionId
309         * @return the session
310         */
311        protected abstract Session createSession(String sessionId);
312
313        /**
314         * Lookup the session with the given id.
315         * 
316         * @param sessionId
317         * @return the session
318         */
319        protected abstract Optional<Session> lookupSession(String sessionId);
320
321        /**
322         * Removed the given session.
323         * 
324         * @param sessionId the session id
325         */
326        protected abstract void removeSession(String sessionId);
327
328        /**
329         * Return the number of established sessions.
330         * 
331         * @return the result
332         */
333        protected abstract int sessionCount();
334        
335        /**
336         * Creates a session id and adds the corresponding cookie to the
337         * response.
338         * 
339         * @param response the response
340         * @return the session id
341         */
342        protected String createSessionId(HttpResponse response) {
343                StringBuilder sessionIdBuilder = new StringBuilder();
344                byte[] bytes = new byte[16];
345                secureRandom.nextBytes(bytes);
346                for (byte b: bytes) {
347                        sessionIdBuilder.append(Integer.toHexString(b & 0xff));
348                }
349                String sessionId = sessionIdBuilder.toString();
350                HttpCookie sessionCookie = new HttpCookie(idName(), sessionId);
351                sessionCookie.setPath(path);
352                sessionCookie.setHttpOnly(true);
353                response.computeIfAbsent(HttpField.SET_COOKIE, CookieList::new)
354                        .value().add(sessionCookie);
355                response.computeIfAbsent(
356                                HttpField.CACHE_CONTROL, CacheControlDirectives::new)
357                        .value().add(new Directive("no-cache", "SetCookie, Set-Cookie2"));
358                return sessionId;
359        }
360        
361        /**
362         * Discards the given session.
363         * 
364         * @param event the event
365         */
366        @Handler(channels=Channel.class)
367        public void discard(DiscardSession event) {
368                removeSession(event.session().id());
369        }
370
371        /**
372         * Associates the channel with the session from the upgrade request.
373         * 
374         * @param event the event
375         * @param channel the channel
376         */
377        @Handler(priority=1000)
378        public void onWebSocketAccepted(
379                        WebSocketAccepted event, IOSubchannel channel) {
380                event.requestEvent().associated(Session.class)
381                        .ifPresent(session -> {
382                                channel.setAssociated(Session.class, session);
383                        });
384        }

386        /**
387         * An MBean interface for getting information about the 
388         * established sessions.
389         */
390        public static interface SessionManagerMXBean {
391                
392                public static class SessionManagerInfo {
393
394                        private SessionManager sessionManager;
395                        
396                        public SessionManagerInfo(SessionManager sessionManager) {
397                                this.sessionManager = sessionManager;
398                        }
399                        
400                        public int getMaxSessions() {
401                                return sessionManager.maxSessions();
402                        }
403                        
404                        public int getAbsoluteTimeout() {
405                                return sessionManager.absoluteTimeout();
406                        }
407                        
408                        public int getIdleTimeout() {
409                                return sessionManager.idleTimeout();
410                        }
411                        
412                        public int getSessionCount() {
413                                return sessionManager.sessionCount();
414                        }
415                }
416                
417                SortedMap<String,SessionManagerInfo> getManagers();
418        }
419        
420        private static class MBeanView implements SessionManagerMXBean {
421                private static Set<SessionManager> allManagers
422                        = Collections.synchronizedSet(
423                                        Collections.newSetFromMap(
424                                                        new WeakHashMap<SessionManager, Boolean>()));
425                
426                public static void addManager(SessionManager manager) {
427                        allManagers.add(manager);
428                }
429                
430                @Override
431                public SortedMap<String,SessionManagerInfo> getManagers() {
432                        SortedMap<String,SessionManagerInfo> result = new TreeMap<>();
433                        for (SessionManager manager: allManagers) {
434                                result.put(Components.objectName(manager) 
435                                                + " (" + manager.idName() + "," + manager.path + ")",
436                                                new SessionManagerInfo(manager));
437                        }
438                        return result;
439                }
440        }

443                try {
444                        MBeanServer mbs = ManagementFactory.getPlatformMBeanServer(); 
445                        ObjectName mxbeanName = new ObjectName("org.jgrapes.http:type="
446                                        + SessionManager.class.getSimpleName());
447                        mbs.registerMBean(new MBeanView(), mxbeanName);
448                } catch (MalformedObjectNameException | InstanceAlreadyExistsException
449                                | MBeanRegistrationException | NotCompliantMBeanException e) {
450                        // Does not happen
451                }               
452        }
