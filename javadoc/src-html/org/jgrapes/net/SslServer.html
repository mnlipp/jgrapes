019package org.jgrapes.net;

021import java.net.InetSocketAddress;
022import java.net.SocketAddress;
023import java.nio.ByteBuffer;
024import java.util.Collections;
025import java.util.List;
026import java.util.Optional;
027import java.util.logging.Level;
028import java.util.logging.Logger;
029
030import javax.net.ssl.ExtendedSSLSession;
031import javax.net.ssl.SNIServerName;
032import javax.net.ssl.SSLContext;

039import org.jgrapes.core.Channel;
040import org.jgrapes.core.Component;
041import org.jgrapes.core.annotation.Handler;

043import org.jgrapes.io.events.Close;
044import org.jgrapes.io.events.Closed;
045import org.jgrapes.io.events.Input;
046import org.jgrapes.io.events.Output;
047import org.jgrapes.io.util.LinkedIOSubchannel;

050import org.jgrapes.io.util.ManagedByteBuffer;

052
053/**
054 * A component that receives and send byte buffers on an
055 * encrypted channel and sends and receives the corresponding
056 * decrypted data on its own channel.
057 */
058public class SslServer extends Component {
059
060        private static final Logger logger 
061                = Logger.getLogger(SslServer.class.getName());
062        
063        private SSLContext sslContext;
064        
065        /**
066         * @param plainChannel the component's channel
067         * @param encryptedChannel the channel with the encrypted data
068         * @param sslContext the SSL context to use
069         */
070        public SslServer(Channel plainChannel, Channel encryptedChannel,
071                        SSLContext sslContext) {
072                super(plainChannel);
073                this.sslContext = sslContext;
074                Handler.Evaluator.add(
075                                this, "onAccepted", encryptedChannel.defaultCriterion());
076                Handler.Evaluator.add(
077                                this, "onInput", encryptedChannel.defaultCriterion());
078                Handler.Evaluator.add(
079                                this, "onClosed", encryptedChannel.defaultCriterion());
080        }
081
082        /**
083         * Creates a new downstream connection as {@link LinkedIOSubchannel} 
084         * of the network connection together with an {@link SSLEngine}.
085         * 
086         * @param event
087         *            the accepted event
088         */
089        @Handler(dynamic=true)
090        public void onAccepted(Accepted event, IOSubchannel encryptedChannel) {
091                new PlainChannel(event, encryptedChannel);
092        }
093
094        /**
095         * Handles encrypted data from upstream (the network). The data is 
096         * send through the {@link SSLEngine} and events are sent downstream 
097         * (and in the initial phases upstream) according to the conversion 
098         * results.
099         * 
100         * @param event the event
101         * @param encryptedChannel the channel for exchanging the encrypted data
102         * @throws InterruptedException 
103         * @throws SSLException 
104         */
105        @Handler(dynamic = true)
106        public void onInput(
107                        Input<ManagedByteBuffer> event, IOSubchannel encryptedChannel)
108                throws InterruptedException, SSLException {
109                @SuppressWarnings("unchecked")
110                final Optional<PlainChannel> plainChannel = (Optional<PlainChannel>)
111                                LinkedIOSubchannel.downstreamChannel(this, encryptedChannel);
112                if (plainChannel.isPresent()) {
113                        plainChannel.get().sendDownstream(event);
114                }
115        }
116
117        /**
118         * Handles a close event from the encrypted channel (client).
119         * 
120         * @param event the event
121         * @param encryptedChannel the channel for exchanging the encrypted data
122         * @throws InterruptedException 
123         * @throws SSLException 
124         */
125        @Handler(dynamic = true)
126        public void onClosed(Closed event, IOSubchannel encryptedChannel)
127                throws SSLException, InterruptedException {
128                @SuppressWarnings("unchecked")
129                final Optional<PlainChannel> plainChannel = (Optional<PlainChannel>)
130                                LinkedIOSubchannel.downstreamChannel(this, encryptedChannel);
131                if (plainChannel.isPresent()) {
132                        plainChannel.get().upstreamClosed();
133                }
134        }
135        
136        /**
137         * Sends decrypted data through the engine and then upstream.
138         * 
139         * @param event
140         *            the event with the data
141         * @throws InterruptedException if the execution was interrupted
142         * @throws SSLException if some SSL related problem occurs
143         */
144        @Handler
145        public void onOutput(Output<ManagedBuffer<ByteBuffer>> event,
146                PlainChannel plainChannel)
147                throws InterruptedException, SSLException {
148                if (plainChannel.converterComponent() != this) {
149                        return;
150                }
151                plainChannel.sendUpstream(event);
152        }
153
154        /**
155         * Handles a close event from downstream.
156         * 
157         * @param event
158         *            the close event
159         * @throws SSLException if an SSL related problem occurs
160         * @throws InterruptedException if the execution was interrupted
161         */
162        @Handler
163        public void onClose(Close event, PlainChannel plainChannel)
164                throws InterruptedException, SSLException {
165                if (plainChannel.converterComponent() != this) {
166                        return;
167                }
168                plainChannel.close(event);
169        }
170        
171        private class PlainChannel extends LinkedIOSubchannel {
172                public SocketAddress localAddress;
173                public SocketAddress remoteAddress;
174                public SSLEngine sslEngine;
175                private ManagedBufferQueue<ManagedByteBuffer, ByteBuffer> downstreamPool;
176                private boolean isInputClosed = false;
177
178                public PlainChannel(Accepted event, IOSubchannel upstreamChannel) {
179                        super(SslServer.this, upstreamChannel);
180                        localAddress = event.localAddress();
181                        remoteAddress = event.remoteAddress();
182                        if (remoteAddress instanceof InetSocketAddress) {
183                                sslEngine = sslContext.createSSLEngine(
184                                                ((InetSocketAddress)remoteAddress).getAddress().getHostAddress(),
185                                                ((InetSocketAddress)remoteAddress).getPort());
186                        } else {
187                                sslEngine = sslContext.createSSLEngine();
188                        }
189                        sslEngine.setUseClientMode(false);
190                        
191                        // Create buffer pools, adding 50 to application buffer size, see 
192                        // https://docs.oracle.com/javase/8/docs/technotes/guides/security/jsse/samples/sslengine/SSLEngineSimpleDemo.java
193                        int bufSize = sslEngine.getSession().getApplicationBufferSize() + 50;
194                        downstreamPool = new ManagedBufferQueue<>(ManagedByteBuffer::new,
195                                        () -> { return ByteBuffer.allocate(bufSize); }, 2); 
196                        setByteBufferPool(new ManagedBufferQueue<>(ManagedByteBuffer::new,
197                                        () -> { return ByteBuffer.allocate(bufSize); }, 2));
198                }
199                
200                public void sendDownstream(Input<ManagedByteBuffer> event)
201                                throws SSLException, InterruptedException {
202                        ManagedByteBuffer unwrapped = downstreamPool.acquire();
203                        ByteBuffer input = event.buffer().duplicate();
204                        SSLEngineResult unwrapResult;
205                        while (true) {
206                                unwrapResult = sslEngine.unwrap(
207                                                input, unwrapped.backingBuffer());
208                                // Handle any handshaking procedures
209                                switch (unwrapResult.getHandshakeStatus()) {
210                                case NEED_TASK:
211                                        while (true) {
212                                                Runnable runnable = sslEngine.getDelegatedTask();
213                                                if (runnable == null) {
214                                                        break;
215                                                }
216                                                runnable.run();
217                                        }
218                                        continue;
219                                        
220                                case NEED_WRAP:
221                                        ManagedByteBuffer feedback 
222                                                = upstreamChannel().byteBufferPool().acquire();
223                                        SSLEngineResult wrapResult = sslEngine.wrap(
224                                                        ManagedByteBuffer.EMPTY_BUFFER
225                                                        .backingBuffer(), feedback.backingBuffer());
226                                        upstreamChannel().respond(new Output<>(feedback, false));
227                                        if (wrapResult.getHandshakeStatus() == HandshakeStatus.FINISHED) {
228                                                fireAccepted();
229                                        }
230                                        continue;
231                                        
232                                case FINISHED:
233                                        fireAccepted();
234                                        // fall through
235                                case NEED_UNWRAP:
236                                        if (input.hasRemaining()) {
237                                                continue;
238                                        }
239                                        break;
240                                        
241                                default:
242                                        break;
243                                }
244                                // Handshake done
245                                break;
246                        }
247                        if (unwrapped.position() == 0) {
248                                // Was only handshake
249                                unwrapped.unlockBuffer();
250                        } else {
251                                // forward data received
252                                unwrapped.flip();
253                                fire(new Input<>(unwrapped, sslEngine.isInboundDone()), this);
254                        }
255                        
256                        // final message?
257                        if (unwrapResult.getStatus() == Status.CLOSED
258                                        && ! isInputClosed) {
259                                Closed evt = new Closed();
260                                newEventPipeline().fire(evt, this);
261                                evt.get();
262                                isInputClosed = true;
263                        }
264                }
265
266                public void sendUpstream(Output<ManagedBuffer<ByteBuffer>> event)
267                                throws SSLException, InterruptedException {
268                        ByteBuffer output = event.buffer().backingBuffer().duplicate();
269                        while (output.hasRemaining() && !sslEngine.isInboundDone()) {
270                                ManagedByteBuffer out 
271                                        = upstreamChannel().byteBufferPool().acquire();
272                                sslEngine.wrap(output, out.backingBuffer());
273                                upstreamChannel().respond(
274                                                new Output<>(out, event.isEndOfRecord()));
275                        }
276                }
277
278                public void close(Close event) 
279                                throws InterruptedException, SSLException {
280                        sslEngine.closeOutbound();
281                        while (!sslEngine.isOutboundDone()) {
282                                ManagedByteBuffer feedback
283                                        = upstreamChannel().byteBufferPool().acquire();
284                                sslEngine.wrap(ManagedByteBuffer.EMPTY_BUFFER
285                                        .backingBuffer(), feedback.backingBuffer());
286                                upstreamChannel().respond(new Output<>(feedback, false));
287                        }
288                        upstreamChannel().respond(new Close());
289                }
290
291                private void fireAccepted() {
292                        List<SNIServerName> snis = Collections.emptyList();
293                        if (sslEngine.getSession() instanceof ExtendedSSLSession) {
294                                snis = ((ExtendedSSLSession)sslEngine.getSession())
295                                        .getRequestedServerNames();
296                        }
297                        fire(new Accepted(
298                                        localAddress, remoteAddress, true, snis), this);
299                }
300                
301                public void upstreamClosed()
302                                throws SSLException, InterruptedException {
303                        if (!isInputClosed) {
304                                // was not properly closed on SSL layer
305                                Closed evt = new Closed();
306                                newEventPipeline().fire(evt, this);
307                                evt.get();
308                        }
309                        try {
310                                sslEngine.closeInbound();
311                                while (!sslEngine.isOutboundDone()) {
312                                        ManagedByteBuffer feedback 
313                                                = upstreamChannel().byteBufferPool().acquire();
314                                        sslEngine.wrap(ManagedByteBuffer.EMPTY_BUFFER
315                                                        .backingBuffer(),feedback.backingBuffer());
316                                        upstreamChannel().respond(new Output<>(feedback, false));
317                                }
318                        } catch (SSLException e) {
319                                // Several clients (notably chromium, see
320                                // https://bugs.chromium.org/p/chromium/issues/detail?id=118366
321                                // don't close the connection properly. So nobody is really
322                                // interested in this message
323                                logger.log(Level.FINEST, e.getMessage(), e);
324                        }
325                }
326
327        }
