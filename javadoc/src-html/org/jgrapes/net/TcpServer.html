061/**
062 * Provides a TCP server. The server binds to the given address. If the
063 * address is {@code null}, address and port are automatically assigned.
064 * <P>
065 * The end of record flag is not used by the server.
066 */
067public class TcpServer extends Component implements NioHandler {
068
069        private SocketAddress serverAddress = null;
070        private ServerSocketChannel serverSocketChannel = null;
071        private int bufferSize = 0;
072        private Set<TcpChannel> channels = new HashSet<>();
073        private boolean closing = false;
074        private ExecutorService executorService = null;
075        private int backlog = 0;
076        private PermitsPool connLimiter = null;
077        private Registration registration = null;
078        private AvailabilityListener permitsListener = new AvailabilityListener() {
079                
080                @Override
081                public void availabilityChanged(PermitsPool pool, boolean available) {
082                        if (registration == null) {
083                                return;
084                        }
085                        registration.updateInterested(
086                                        (Boolean)available ? SelectionKey.OP_ACCEPT : 0);
087                }
088        };

171        /**

178        public TcpServer setConnectionLimiter(PermitsPool connectionLimiter) {
179                if (connLimiter != null) {
180                        connLimiter.removeListener(permitsListener);

183                if (connLimiter != null) {

218        /**
219         * Starts the server.
220         * 
221         * @param event the start event
222         * @throws IOException if an I/O exception occurred
223         */
224        @Handler
225        public void onStart(Start event) throws IOException {
226                closing = false;
227                serverSocketChannel = ServerSocketChannel.open();
228                serverSocketChannel.bind(serverAddress, backlog);
229                fire(new NioRegistration(this, serverSocketChannel, 
230                                SelectionKey.OP_ACCEPT, this), Channel.BROADCAST);
231        }
232
233        @Handler(channels=Self.class)
234        public void onRegistered(NioRegistration.Completed event) 
235                        throws InterruptedException, IOException {
236                NioHandler handler = event.event().handler(); 
237                if (handler == this) {
238                        if (event.event().get() == null) {
239                                fire(new Error(event, 
240                                                "Registration failed, no NioDispatcher?"));
241                                return;
242                        }
243                        registration = event.event().get();
244                        fire(new Ready(serverSocketChannel.getLocalAddress()));
245                        return;
246                }
247                if (handler instanceof TcpChannel) {
248                        ((TcpChannel)handler)
249                                .registrationComplete(event.event());
250                }
251        }
252
253        /* (non-Javadoc)
254         * @see org.jgrapes.io.NioSelectable#handleOps(int)
255         */
256        @Override
257        public void handleOps(int ops) {
258                synchronized (channels) {
259                        if ((ops & SelectionKey.OP_ACCEPT) != 0 && !closing) {
260                                try {
261                                        if (connLimiter == null || connLimiter.tryAcquire()) {
262                                                SocketChannel socketChannel = serverSocketChannel.accept();
263                                                channels.add(new TcpChannel(socketChannel));
264                                        }
265                                } catch (IOException e) {
266                                        fire(new IOError(null, e));
267                                }
268                        }
269                }
270        }
271
272        private boolean removeChannel(TcpChannel channel) {
273                synchronized (channels) {
274                        if(!channels.remove(channel)) {
275                                // Closed already
276                                return false;
277                        }
278                        if (connLimiter != null) {
279                                connLimiter.release();
280                        }
281                        // In case the server is shutting down
282                        channels.notifyAll();
283                }                       
284                return true;
285        }

287        /**
288         * Writes the data passed in the event to the client. The end of record
289         * flag is ignored.
290         * 
291         * @param event the event
292         * @throws IOException if an error occurs
293         */
294        @Handler
295        public void onOutput(Output<ManagedByteBuffer> event,
296                        TcpChannel channel) throws IOException {
297                if (channels.contains(channel)) {
298                        channel.write(event);
299                }
300        }

302        /**
303         * Shuts down the server or one of the connections to the server
304         * 
305         * @param event the event
306         * @throws IOException if an I/O exception occurred
307         * @throws InterruptedException if the execution was interrupted 
308         */
309        @Handler
310        public void onClose(Close event) throws IOException, InterruptedException {
311                boolean subOnly = true;
312                for (Channel channel: event.channels()) {
313                        if (channel instanceof TcpChannel) {
314                                if (channels.contains(channel)) {
315                                        ((TcpChannel)channel).close();
316                                }
317                        } else {
318                                subOnly = false;
319                        }
320                }
321                if (subOnly || !serverSocketChannel.isOpen()) {
322                        return;
323                }
324                synchronized (channels) {
325                        closing = true;
326                        // Copy to avoid concurrent modification exception
327                        Set<TcpChannel> conns = new HashSet<>(channels);
328                        for (TcpChannel conn : conns) {
329                                conn.close();
330                        }
331                        while (channels.size() > 0) {
332                                channels.wait();
333                        }
334                }
335                serverSocketChannel.close();
336                closing = false;
337                fire(new Closed());
338        }
339
340        /**
341         * Shuts down the server.
342         * 
343         * @param event the event
344         */
345        @Handler
346        public void onStop(Stop event) {
347                if (closing || !serverSocketChannel.isOpen()) {
348                        return;
349                }
350                newSyncEventPipeline().fire(new Close(), channel());
351        }

361        /**
362         * The internal representation of a connected client. 
363         */
364        private class TcpChannel 
365                extends DefaultSubchannel implements NioHandler {
366
367                private SocketChannel nioChannel;
368                private EventPipeline downPipeline;
369                private ManagedBufferQueue<ManagedByteBuffer, ByteBuffer> readBuffers;
370                private Registration registration = null;
371                private Queue<ManagedByteBuffer.Reader> pendingWrites = new ArrayDeque<>();
372                private boolean pendingClose = false;
373                
374                /**
375                 * @param nioChannel the channel
376                 * @throws IOException if an I/O error occured
377                 */
378                public TcpChannel(SocketChannel nioChannel)     throws IOException {
379                        super(TcpServer.this);
380                        this.nioChannel = nioChannel;
381                        if (executorService != null) {
382                                downPipeline = newEventPipeline(executorService);
383                        } else {
384                                downPipeline = newEventPipeline();
385                        }
386
387                        String channelName = Components.objectName(TcpServer.this)
388                                        + "." + Components.objectName(this);
389                        int writeBufferSize = bufferSize == 0 
390                                        ? nioChannel.socket().getSendBufferSize() : bufferSize;
391                        setByteBufferPool(new ManagedBufferQueue<>(ManagedByteBuffer::new,
392                                        () -> { return ByteBuffer.allocate(writeBufferSize); }, 2)
393                                        .setName(channelName + ".upstream.buffers"));
394                        
395                        int readBufferSize = bufferSize == 0 
396                                        ? nioChannel.socket().getReceiveBufferSize() : bufferSize;
397                        readBuffers = new ManagedBufferQueue<>(ManagedByteBuffer::new,
398                                        () -> { return ByteBuffer.allocate(readBufferSize); }, 2)
399                                        .setName(channelName + ".downstream.buffers");
400                        
401                        // Register with dispatcher
402                        nioChannel.configureBlocking(false);
403                        TcpServer.this.fire(
404                                new NioRegistration(this, nioChannel, 0, TcpServer.this),
405                                Channel.BROADCAST);
406                }
407
408                /**
409                 * Invoked when registration has completed.
410                 * 
411                 * @param event the completed event
412                 * @throws InterruptedException if the execution was interrupted
413                 * @throws IOException if an I/O error occurred
414                 */
415                public void registrationComplete(NioRegistration event)
416                        throws InterruptedException, IOException {
417                        registration = event.get();
418                        downPipeline.fire(new Accepted(nioChannel.getLocalAddress(),
419                                        nioChannel.getRemoteAddress(), false,
420                                        Collections.emptyList()), this);
421                        registration.updateInterested(SelectionKey.OP_READ);
422
423                }
424
425                /**
426                 * Write the data on this channel.
427                 * 
428                 * @param event the event
429                 * @throws IOException if an error occurred
430                 */
431                public void write(Output<ManagedByteBuffer> event) throws IOException {
432                        synchronized(pendingWrites) {
433                                if (!nioChannel.isOpen()) {
434                                        return;
435                                }
436                                ManagedByteBuffer.Reader reader = event.buffer().newReader();
437                                if (!pendingWrites.isEmpty()) {
438                                        reader.managedBuffer().lockBuffer();
439                                        pendingWrites.add(reader);
440                                        return;
441                                }
442                                nioChannel.write(reader.get());
443                                if (!reader.get().hasRemaining()) {
444                                        return;
445                                }
446                                reader.managedBuffer().lockBuffer();
447                                pendingWrites.add(reader);
448                                if (pendingWrites.size() == 1) {
449                                        registration.updateInterested(
450                                                        SelectionKey.OP_READ | SelectionKey.OP_WRITE);
451                                }
452                        }
453                }
454
455                @Override
456                public void handleOps(int ops) {
457                        try {
458                                if ((ops & SelectionKey.OP_READ) != 0) {
459                                        handleReadOp();
460                                }
461                                if ((ops & SelectionKey.OP_WRITE) != 0) {
462                                        handleWriteOp();
463                                }
464                        } catch (InterruptedException | IOException e) {
465                                downPipeline.fire(new IOError(null, e));
466                        }
467                }
468
469                /**
470                 * Gets a buffer from the pool and reads available data into it.
471                 * Sends the result as event. 
472                 * 
473                 * @throws InterruptedException
474                 * @throws IOException
475                 */
476                private void handleReadOp() throws InterruptedException, IOException {
477                        ManagedByteBuffer buffer;
478                        buffer = readBuffers.acquire();
479                        int bytes = nioChannel.read(buffer.backingBuffer());
480                        if (bytes == 0) {
481                                buffer.unlockBuffer();
482                                return;
483                        }
484                        if (bytes > 0) {
485                                buffer.flip();
486                                downPipeline.fire(
487                                                new Input<ManagedByteBuffer>(buffer, false), this);
488                                return;
489                        }
490                        // EOF (-1) from client
491                        buffer.unlockBuffer();
492                        synchronized (nioChannel) {
493                                if (nioChannel.socket().isOutputShutdown()) {
494                                        // Client confirms our close, complete close
495                                        nioChannel.close();
496                                        return;
497                                }
498                                
499                        }
500                        // Client initiates close
501                        removeChannel();
502                        synchronized (pendingWrites) {
503                                synchronized (nioChannel) {
504                                        if (!pendingWrites.isEmpty()) {
505                                                // Pending writes, delay close
506                                                pendingClose = true;
507                                                // Mark as client initiated close
508                                                nioChannel.shutdownInput();
509                                                return;
510                                        }
511                                        // Nothing left to do, close
512                                        nioChannel.close();
513                                }
514                        }
515                }
516
517                /**
518                 * Checks if there is still data to be written. This may be
519                 * a left over in an incompletely written buffer or a complete
520                 * pending buffer. 
521                 * 
522                 * @throws IOException
523                 * @throws InterruptedException 
524                 */
525                private void handleWriteOp() 
526                                throws IOException, InterruptedException {
527                        while (true) {
528                                ManagedByteBuffer.Reader head = null;
529                                synchronized (pendingWrites) {
530                                        if (pendingWrites.isEmpty()) {
531                                                // Nothing left to write, stop getting ops
532                                                registration.updateInterested(SelectionKey.OP_READ);
533                                                // Was the connection closed while we were writing?
534                                                if (pendingClose) {
535                                                        synchronized (nioChannel) {
536                                                                if (nioChannel.socket().isInputShutdown()) {
537                                                                        // Delayed close from client, complete
538                                                                        nioChannel.close();
539                                                                } else {
540                                                                        // Delayed close from server, initiate
541                                                                        nioChannel.shutdownOutput();
542                                                                }
543                                                        }
544                                                        pendingClose = false;
545                                                }
546                                                break; // Nothing left to do
547                                        }
548                                        head = pendingWrites.peek();
549                                        if (!head.get().hasRemaining()) {
550                                                // Nothing left in head buffer, try next
551                                                head.managedBuffer().unlockBuffer();
552                                                pendingWrites.remove();
553                                                continue;
554                                        }
555                                }
556                                nioChannel.write(head.get()); // write...
557                                break; // ... and wait for next op
558                        }
559                }
560
561                /**
562                 * Closes this channel.
563                 * 
564                 * @throws IOException if an error occurs
565                 * @throws InterruptedException if the execution was interrupted 
566                 */
567                public void close() throws IOException, InterruptedException {
568                        removeChannel();
569                        synchronized (pendingWrites) {
570                                if (!pendingWrites.isEmpty()) {
571                                        // Pending writes, delay close until done
572                                        pendingClose = true;
573                                        return;
574                                }
575                                // Nothing left to do, proceed
576                                synchronized (nioChannel) {
577                                        if (nioChannel.isOpen()) {
578                                                // Initiate close, must be confirmed by client
579                                                nioChannel.shutdownOutput();
580                                        }
581                                }
582                        }
583                }
584
585                private void removeChannel() throws InterruptedException {
586                        if (TcpServer.this.removeChannel(this)) {
587                                Closed evt = new Closed();
588                                downPipeline.fire(evt, this);
589                                evt.get();
590                        }
591                }
592

594                 * @see java.lang.Object#toString()
595                 */
596                public String toString() {
597                        return IOSubchannel.toString(this);
598                }
599        }
