019package org.jgrapes.net;
020
021import java.io.IOException;
022import java.net.ConnectException;
023import java.nio.channels.SocketChannel;
024import org.jgrapes.core.Channel;
025import org.jgrapes.core.Self;
026import org.jgrapes.core.annotation.Handler;
027import org.jgrapes.core.events.Error;
028import org.jgrapes.io.NioHandler;
029import org.jgrapes.io.events.Close;
030import org.jgrapes.io.events.ConnectError;
031import org.jgrapes.io.events.NioRegistration;
032import org.jgrapes.io.events.OpenTcpConnection;
033import org.jgrapes.net.events.Connected;
034
035/**
036 * A component that reads from or write to a TCP connection.
037 */
038public class TcpConnector extends TcpConnectionManager {
039
040    private int bufferSize = 1536;
041
042    /**
043     * Create a new instance using the given channel.
044     * 
045     * @param channel the component's channel 
046     */
047    public TcpConnector(Channel channel) {
048        super(channel);
049    }
050
051    /**
052     * Creates a new connector, using itself as component channel. 
053     */
054    public TcpConnector() {
055        this(Channel.SELF);
056    }
057
058    /**
059     * Sets the buffer size for the send an receive buffers.
060     * If no size is set, the system defaults will be used.
061     * 
062     * @param size the size to use for the send and receive buffers
063     * @return the TCP connector for easy chaining
064     */
065    public TcpConnector setBufferSize(int size) {
066        this.bufferSize = size;
067        return this;
068    }
069
070    /**
071     * Return the configured buffer size.
072     *
073     * @return the bufferSize
074     */
075    public int bufferSize() {
076        return bufferSize;
077    }
078
079    /**
080     * Opens a connection to the end point specified in the event.
081     *
082     * @param event the event
083     */
084    @Handler
085    public void onOpenConnection(OpenTcpConnection event) {
086        try {
087            SocketChannel socketChannel = SocketChannel.open(event.address());
088            channels.add(new TcpChannelImpl(socketChannel));
089        } catch (ConnectException e) {
090            fire(new ConnectError(event, "Connection refused.", e));
091        } catch (IOException e) {
092            fire(new ConnectError(event, "Failed to open TCP connection.", e));
093        }
094    }
095
096    /**
097     * Called when the new socket channel has successfully been registered
098     * with the nio dispatcher.
099     *
100     * @param event the event
101     * @throws InterruptedException the interrupted exception
102     * @throws IOException Signals that an I/O exception has occurred.
103     */
104    @Handler(channels = Self.class)
105    public void onRegistered(NioRegistration.Completed event)
106            throws InterruptedException, IOException {
107        NioHandler handler = event.event().handler();
108        if (!(handler instanceof TcpChannelImpl)) {
109            return;
110        }
111        if (event.event().get() == null) {
112            fire(new Error(event, "Registration failed, no NioDispatcher?",
113                new Throwable()));
114            return;
115        }
116        TcpChannelImpl channel = (TcpChannelImpl) handler;
117        channel.registrationComplete(event.event());
118        channel.downPipeline()
119            .fire(new Connected(channel.nioChannel().getLocalAddress(),
120                channel.nioChannel().getRemoteAddress()), channel);
121    }
122
123    /**
124     * Shuts down the one of the connections.
125     *
126     * @param event the event
127     * @throws IOException if an I/O exception occurred
128     * @throws InterruptedException if the execution was interrupted
129     */
130    @Handler
131    @SuppressWarnings("PMD.DataflowAnomalyAnalysis")
132    public void onClose(Close event) throws IOException, InterruptedException {
133        for (Channel channel : event.channels()) {
134            if (channel instanceof TcpChannelImpl
135                && channels.contains(channel)) {
136                ((TcpChannelImpl) channel).close();
137            }
138        }
139    }
140}
