019package org.jgrapes.core;
020
021import java.util.Map;
022import java.util.WeakHashMap;
023import java.util.concurrent.atomic.AtomicLong;
024import java.util.logging.Level;
025
026import org.jgrapes.core.annotation.ComponentManager;
027import org.jgrapes.core.events.Start;
028import org.jgrapes.core.events.Started;
029import org.jgrapes.core.internal.Common;
030import org.jgrapes.core.internal.ComponentVertex;
031import org.jgrapes.core.internal.GeneratorRegistry;
032
033/**
034 * This class provides some utility functions.
035 */
036public class Components {
037
038        private Components() {
039        }
040        
041        /**
042         * Returns a component's manager. For a component that inherits
043         * from {@link org.jgrapes.core.Component} this method simply returns
044         * the component as it is its own manager.
045         * 
046         * For components that implement {@link ComponentType} but don't inherit from 
047         * {@link org.jgrapes.core.Component} the method returns the value of 
048         * the attribute annotated as manager slot. If this attribute is still
049         * empty, this method makes the component the root
050         * of a new tree and returns its manager.
051         * 
052         * @param component the component
053         * @return the component (with its manager attribute set)
054         */
055        public static Manager manager(ComponentType component) {
056                return ComponentVertex.componentVertex(component, null);
057        }
058
059        /**
060         * Returns a component's manager like {@link #manager(ComponentType)}.
061         * If the manager slot attribute is empty, the component is initialized
062         * with its component channel set to the given parameter. Invoking
063         * this method overrides any channel set in the
064         * {@link ComponentManager} annotation.
065         * 
066         * This method is usually invoked by the constructor of a class
067         * that implements {@link ComponentType}.
068         * 
069         * @param component the component
070         * @param componentChannel the channel that the component's 
071         * handlers listen on by default and that 
072         * {@link Manager#fire(Event, Channel...)} sends the event to 
073         * @return the component (with its manager attribute set)
074         * @see Component#Component(Channel)
075         */
076        public static Manager manager(
077                        ComponentType component, Channel componentChannel) {
078                return ComponentVertex.componentVertex(component, componentChannel);
079        }
080
081        /**
082         * Fires a {@link Start} event with an associated
083         * {@link Started} completion event on the broadcast channel
084         * of the given application and wait for the completion of the
085         * <code>Start</code> event.
086         * 
087         * @param application the application to start
088         * @throws InterruptedException if the execution was interrupted
089         */
090        public static void start(ComponentType application) 
091                        throws InterruptedException {
092                manager(application).fire(new Start() , Channel.BROADCAST).get();
093        }
094
095        /**
096         * Wait until all generators and event queues are exhausted. When this
097         * stage is reached, nothing can happen anymore unless a new event is
098         * sent from an external thread.
099         * 
100         * @throws InterruptedException if the current thread was interrupted
101         * while waiting
102         */
103        public static void awaitExhaustion() throws InterruptedException {
104                GeneratorRegistry.instance().awaitExhaustion();
105        }
106
107        /**
108         * Wait until all generators and event queues are exhausted or
109         * the maximum wait time has expired.
110         * 
111         * @param timeout the wait time in milliseconds
112         * @return {@code true} if exhaustion state was reached
113         * @throws InterruptedException if the execution was interrupted 
114         * @see #awaitExhaustion()
115         */
116        public static boolean awaitExhaustion(long timeout) 
117                        throws InterruptedException {
118                return GeneratorRegistry.instance().awaitExhaustion(timeout);
119        }
120
121        /**
122         * Utility method that checks if an assertion error has occurred
123         * while executing handlers. If so, the error is thrown and
124         * the assertion error store is reset.
125         * <P>
126         * This method is intended for junit tests. It enables easy propagation
127         * of assertion failures to the main thread.
128         * 
129         * @throws AssertionError if an assertion error occurred while
130         * executing the application
131         */
132        public static void checkAssertions() {
133                Common.checkAssertions();
134        }
135        
136        /**
137         * Returns the full name of the object's class together with an id (see 
138         * {@link #objectId(Object)}). The result can be used as a unique
139         * human readable identifier for arbitrary objects.
140         * 
141         * @param object
142         *            the object
143         * @return the object's name
144         */
145        public static String objectFullName(Object object) {
146                StringBuilder builder = new StringBuilder();
147                builder.append(object.getClass().getName());
148                builder.append('#');
149                builder.append(objectId(object));
150                return builder.toString();
151        }
152
153        /**
154         * Returns the name of the object's class together with an id (see 
155         * {@link #objectId(Object)}). May be used to implement {@code toString()}
156         * with identifiable objects. If the log level is "finer", the full
157         * class name will be used for the returned value, else the simple name.
158         * 
159         * @param object
160         *            the object
161         * @return the object's name
162         */
163        public static String objectName(Object object) {
164                StringBuilder builder = new StringBuilder();
165                builder.append(Components.classToString(object.getClass()));
166                builder.append('#');
167                builder.append(objectId(object));
168                return builder.toString();
169        }
170
171        private static Map<Object, String> objectIds = new WeakHashMap<>();
172        private static Map<Class<?>, AtomicLong> idCounters = new WeakHashMap<>();
173
174        private static String getId(Class<?> scope, Object object) {
175                if (object == null) {
176                        return "?";
177                }
178                synchronized (objectIds) {
179                        return objectIds.computeIfAbsent(object, k -> 
180                                Long.toString(idCounters
181                                                .computeIfAbsent(scope, l -> new AtomicLong())
182                                                        .incrementAndGet()));
183                        
184                }
185        }
186
187        /**
188         * Returns the full name or simple name of the class depending
189         * on the log level.
190         * 
191         * @param clazz the class
192         * @return the name
193         */
194        public static String classToString(Class<?> clazz) {
195                if (Common.classNames.isLoggable(Level.FINER)) {
196                        return clazz.getName();
197                } else {
198                        return clazz.getSimpleName();
199                }
200        }
201        
202        /**
203         * Returns an id of the object that is unique within a specific scope. Ids
204         * are generated and looked up in the scope of the object's class unless the
205         * class implements {@link IdInfoProvider}.
206         * 
207         * @param object
208         *            the object
209         * @return the object's name
210         */
211        public static String objectId(Object object) {
212                if (object instanceof IdInfoProvider) {
213                        return getId(((IdInfoProvider) object).idScope(),
214                                ((IdInfoProvider) object).idObject());
215                } else {
216                        return getId(object.getClass(), object);
217                }
218        }
219
220        /**
221         * Implemented by classes that want a special class (scope) to be used
222         * for looking up their id or want to map to another object for getting the
223         * id (see {@link Components#objectId(Object)}).
224         */
225        public static interface IdInfoProvider {
226                
227                /**
228                 * Returns the scope.
229                 * 
230                 * @return the scope
231                 */
232                Class<?> idScope();
233                
234                /**
235                 * Returns the object to be used for generating the id.
236                 * 
237                 * @return the object (defaults to {@code this})
238                 */
239                default Object idObject() {
240                        return this;
241                }
242        }
243
