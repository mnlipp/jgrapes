039/**
040 * This class is the base class for all events.
041 *  
042 * By default (i.e. as implemented by this class), the event's kind is
043 * represented by its Java class and the eligibility is based on
044 * the "is a" relationship between classes. An event is eligible if its class
045 * is equal to or a super class of the class used as criterion. 
046 * This default behavior can be changed by overriding the
047 * methods from {@link Eligible}. See {@link NamedEvent} as an example.
048 * 
049 * @param <T>
050 *            the result type of the event. Use {@link Void} if handling the
051 *            event does not produce a result
052 */
053public class Event<T> extends EventBase<T> {
054
055    /** The channels that this event is to be fired on if no
056     * channels are specified explicitly when firing. */
057    private Channel[] channels;
058    /** Indicates that the event should not processed further. */
059    private boolean stopped;
060    /** The results of handling the event (if any). */
061    private List<T> results;
062    /** Context data. */
063    private Map<Object, Object> contextData;
064    private boolean cancelled;
065
066    /**
067     * Creates a new event. Passing channels is equivalent to first
068     * creating the event and then calling {@link #setChannels(Channel...)}
069     * with the given channels.
070     * 
071     * @param channels the channels to set
072     */
073    public Event(Channel... channels) {
074        super();
075        this.channels = Arrays.copyOf(channels, channels.length);
076    }
077
078    /**
079     * Returns the class of this event as representation of its kind.
080     * 
081     * @return the class of this event
082     * 
083     * @see org.jgrapes.core.Eligible#defaultCriterion()
084     */
085    @Override
086    public Object defaultCriterion() {
087        return getClass();
088    }
089
090    /**
091     * Returns <code>true</code> if the `criterion` 
092     * is of the same class or a base class of this event's class.
093     * 
094     * @see org.jgrapes.core.Eligible#isEligibleFor(java.lang.Object)
095     */
096    @Override
097    public boolean isEligibleFor(Object criterion) {
098        return Class.class.isInstance(criterion)
099            && ((Class<?>) criterion).isAssignableFrom(getClass());
100    }
101
102    /**
103     * Return the event pipeline that currently processes the event
104     * (if any).
105     * 
106     * @return the event pipeline if the event is being processed
107     */
108    @SuppressWarnings("PMD.UselessOverridingMethod")
109    public Optional<EventPipeline> processedBy() {
110        return super.processedBy();
111    }
112
113    /**
114     * Implements the default behavior for handling events thrown
115     * by a handler. Fires a {@link HandlingError handling error} event
116     * for this event and the given throwable.
117     * 
118     * @see HandlingError
119     */
120    @Override
121    protected void handlingError(
122            EventPipeline eventProcessor, Throwable throwable) {
123        eventProcessor.fire(
124            new HandlingError(this, throwable), channels());
125    }
126
127    /**
128     * Sets the channels that the event is fired on if no channels
129     * are specified explicitly when firing the event
130     * (see {@link org.jgrapes.core.Manager#fire(Event, Channel...)}).
131     * 
132     * @param channels the channels to set
133     * @return the object for easy chaining
134     * 
135     * @throws IllegalStateException if the method is called after
136     * this event has been fired
137     */
138    public Event<T> setChannels(Channel... channels) {
139        if (enqueued()) {
140            throw new IllegalStateException(
141                "Channels cannot be changed after fire");
142        }
143        this.channels = Arrays.copyOf(channels, channels.length);
144        return this;
145    }
146
147    /**
148     * Returns the channels associated with the event. Before an
149     * event has been fired, this returns the channels set with
150     * {@link #setChannels(Channel[])}. After an event has been
151     * fired, this returns the channels that the event has
152     * effectively been fired on 
153     * (see {@link Manager#fire(Event, Channel...)}).
154     * 
155     * @return the channels (never `null`, but may be empty)
156     */
157    @Override
158    public Channel[] channels() {
159        return Arrays.copyOf(channels, channels.length);
160    }
161
162    /**
163     * Returns the subset of channels that are assignable to the given type.
164     * 
165     * @param <C> the given type's class
166     * @param type the class to look for
167     * @return the filtered channels
168     * @see #channels()
169     */
170    @SuppressWarnings({ "unchecked", "PMD.ShortVariable",
171        "PMD.AvoidDuplicateLiterals" })
172    public <C> C[] channels(Class<C> type) {
173        return Arrays.stream(channels)
174            .filter(c -> type.isAssignableFrom(c.getClass())).toArray(
175                size -> (C[]) Array.newInstance(type, size));
176    }
177
178    /**
179     * Execute the given handler for all channels of the given type.
180     * 
181     * @param <E> the type of the event
182     * @param <C> the type of the channel
183     * @param type the channel type
184     * @param handler the handler
185     */
186    @SuppressWarnings({ "unchecked", "PMD.ShortVariable" })
187    public <E extends EventBase<?>, C extends Channel> void forChannels(
188            Class<C> type, BiConsumer<E, C> handler) {
189        Arrays.stream(channels)
190            .filter(c -> type.isAssignableFrom(c.getClass()))
191            .forEach(c -> handler.accept((E) this, (C) c));
192    }
193
194    /**
195     * Returns the events to be thrown when this event has completed
196     * (see {@link #isDone()}).
197     * 
198     * @return the completed events
199     */
200    public Set<Event<?>> completionEvents() {
201        return completionEvents == null ? Collections.emptySet()
202            : Collections.unmodifiableSet(completionEvents);
203    }
204
205    /**
206     * Adds the given event to the events to be thrown when this event 
207     * has completed (see {@link #isDone()}). Such an event is called 
208     * a "completion event".
209     * 
210     * Completion events are considered to be caused by the event that 
211     * caused the completed event. If an event *e1* caused an event
212     * *e2* which has a completion event *e2c*, *e1* is only put in 
213     * state completed when *e2c* has been handled.
214     * 
215     * Completion events are handled by the same {@link EventProcessor}
216     * as the event that has been completed.
217     * 
218     * @param completionEvent the completion event to add
219     * @return the object for easy chaining
220     */
221    public Event<T> addCompletionEvent(Event<?> completionEvent) {
222        if (completionEvents == null) {
223            completionEvents = new HashSet<>();
224        }
225        completionEvents.add(completionEvent);
226        return this;
227    }

229    /*
230     * (non-Javadoc)
231     * 
232     * @see org.jgrapes.core.internal.EventBase#setRequiresResult(boolean)
233     */
234    @Override
235    public Event<T> setRequiresResult(boolean value) {
236        return (Event<T>) super.setRequiresResult(value);
237    }
238
239    /**
240     * Check if this event has completed. An event is completed
241     * if 
242     *  * all its handlers have been invoked (or the event has
243     *    been stopped or cancelled), 
244     *  * all events caused by it have completed,
245     *  * no {@link CompletionLock}s remain, and  
246     *  * a result has been set (only required if 
247     *    {@link #setRequiresResult(boolean)} has been called with `true`).
248     * 
249     * @return the completed state
250     */
251    @Override
252    public boolean isDone() {
253        return completed;
254    }
255
256    /**
257     * Invoked after all handlers for the event have been executed. 
258     * May be overridden by derived classes to cause some immediate effect
259     * (instead of e.g. waiting for the completion event). The default 
260     * implementation does nothing. This method is invoked by the event 
261     * handler thread and must not block.
262     */
263    protected void handled() {
264        // Default is to do nothing.
265    }
266
267    /**
268     * Can be called during the execution of an event handler to indicate
269     * that the event should not be processed further. All remaining 
270     * handlers for this event will be skipped.
271     * 
272     * @return the object for easy chaining
273     */
274    public Event<T> stop() {
275        stopped = true;
276        return this;
277    }
278
279    /**
280     * Returns <code>true</code> if {@link #stop} has been called.
281     * 
282     * @return the stopped state
283     */
284    public boolean isStopped() {
285        return stopped;
286    }
287
288    /**
289     * Prevents the invocation of further handlers (like {@link #stop()} 
290     * and (in addition) the invocation of any added completed events.
291     * 
292     * @param mayInterruptIfRunning ignored
293     * @return `false` if the event has already been completed
294     * @see java.util.concurrent.Future#cancel(boolean)
295     */
296    @Override
297    public boolean cancel(boolean mayInterruptIfRunning) {
298        if (!completed && !cancelled) {
299            stop();
300            cancelled = true;
301            return true;
302        }
303        return false;
304    }
305
306    @Override
307    public boolean isCancelled() {
308        return cancelled;
309    }
310
311    /**
312     * Sets the result of handling this event. If this method is invoked 
313     * more then once, the various results are collected in a list. This
314     * can happen if the event is handled by several components. 
315     * 
316     * @param result the result to set
317     * @return the object for easy chaining
318     */
319    public Event<T> setResult(T result) {
320        synchronized (this) {
321            if (results == null) {
322                // Make sure that we have a valid result before
323                // calling decrementOpen
324                results = new ArrayList<>();
325                results.add(result);
326                firstResultAssigned();
327                return this;
328            }
329            results.add(result);
330            return this;
331        }
332    }
333
334    /**
335     * Allows access to the intermediate result before the 
336     * completion of the event. 
337     * 
338     * @return the intermediate results (which may be an empty list)
339     */
340    protected List<T> currentResults() {
341        return results == null ? Collections.emptyList()
342            : Collections.unmodifiableList(results);
343    }
344
345    /**
346     * Tie the result of this event to the result of the other event.
347     * Changes of either event's results will subsequently be applied
348     * to both events.
349     * <P>
350     * This is useful when an event is replaced by another event during
351     * handling like:
352     * {@code fire((new Event()).tieTo(oldEvent.stop()))}  
353     * 
354     * @param other the event to tie to
355     * @return the object for easy chaining
356     */
357    public Event<T> tieTo(Event<T> other) {
358        synchronized (this) {
359            if (other.results == null) {
360                other.results = new ArrayList<>();
361            }
362            results = other.results;
363            return this;
364        }
365    }

367    /**
368     * Waits for the event to be completed (see {@link #isDone()})
369     * and returns the first (or only) result.
370     * 
371     * @see Future#get()
372     */
373    @Override
374    public T get() throws InterruptedException {
375        while (true) {
376            synchronized (this) {
377                if (completed) {
378                    return results == null || results.isEmpty()
379                        ? null
380                        : results.get(0);
381                }
382                wait();
383            }
384        }
385    }
386
387    /**
388     * Causes the invoking thread to wait until the processing of the 
389     * event has been completed (see {@link #isDone()}) or the given 
390     * timeout has expired and returns the first (or only) result. 
391     * 
392     * @return the result
393     * @see java.util.concurrent.Future#get(long, java.util.concurrent.TimeUnit)
394     */
395    @Override
396    public T get(long timeout, TimeUnit unit)
397            throws InterruptedException, TimeoutException {
398        synchronized (this) {
399            if (completed) {
400                return results == null || results.isEmpty()
401                    ? null
402                    : results.get(0);
403            }
404            wait(unit.toMillis(timeout));
405        }
406        if (completed) {
407            return results == null || results.isEmpty()
408                ? null
409                : results.get(0);
410        }
411        throw new TimeoutException();
412    }
413
414    /**
415     * Waits for the event to be completed (see {@link #isDone()})
416     * and returns the list of results (which may be empty if the
417     * event's result type is {@link Void}).
418     * 
419     * @return the results
420     * @see Future#get()
421     */
422    public List<T> results() throws InterruptedException {
423        while (true) {
424            synchronized (this) {
425                if (completed) {
426                    return results == null ? Collections.emptyList()
427                        : Collections.unmodifiableList(results);
428                }
429                wait();
430            }
431        }
432    }
433
434    /**
435     * Causes the invoking thread to wait until the processing of the 
436     * event has been completed (see {@link #isDone()}) or given timeout 
437     * has expired and returns the list of results (which may be empty
438     * if the event's result type is {@link Void}). 
439     * 
440     * @return the results
441     * @see java.util.concurrent.Future#get(long, java.util.concurrent.TimeUnit)
442     */
443    public List<T> results(long timeout, TimeUnit unit)
444            throws InterruptedException, TimeoutException {
445        synchronized (this) {
446            if (completed) {
447                return results == null ? Collections.emptyList()
448                    : Collections.unmodifiableList(results);
449            }
450            wait(unit.toMillis(timeout));
451        }
452        if (completed) {
453            return results == null ? Collections.emptyList()
454                : Collections.unmodifiableList(results);
455        }
456        throw new TimeoutException();
457    }

459    @Override
460    @SuppressWarnings("PMD.ShortVariable")
461    public Event<T> setAssociated(Object by, Object with) {
462        if (contextData == null) {
463            contextData = new ConcurrentHashMap<>();
464        }
465        if (with == null) {
466            contextData.remove(by);
467        } else {
468            contextData.put(by, with);
469        }
470        return this;
471    }
472
473    @Override
474    @SuppressWarnings("PMD.ShortVariable")
475    public <V> Optional<V> associated(Object by, Class<V> type) {
476        if (contextData == null) {
477            return Optional.empty();
478        }
479        return Optional.ofNullable(contextData.get(by))
480            .filter(found -> type.isAssignableFrom(found.getClass()))
481            .map(match -> type.cast(match));
482    }

484    /*
485     * (non-Javadoc)
486     * 
487     * @see java.lang.Object#toString()
488     */
489    @Override
490    public String toString() {
491        StringBuilder builder = new StringBuilder();
492        builder.append(Components.objectName(this))
493            .append(" [");
494        if (channels != null) {
495            builder.append("channels=");
496            builder.append(Channel.toString(channels));
497        }
498        builder.append(']');
499        return builder.toString();
500    }
