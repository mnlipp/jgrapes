019package org.jgrapes.core;
020
021import org.jgrapes.core.events.HandlingError;
022import org.jgrapes.core.internal.Common;
023import org.jgrapes.core.internal.EventBase;
024
025/**
026 * This class is the base class for all events.
027 *  
028 * By default (i.e. as implemented by this class), the event's kind is
029 * represented by its Java class and the eligibility is based on
030 * the "is a" relationship between classes. An event is eligible if its class
031 * is equal to or a super class of the class used as criterion. 
032 * This default behavior can be changed by overriding the
033 * methods from {@link Eligible}. See {@link NamedEvent} as an example.
034 * 
035 * @param <T>
036 *            the result type of the event. Use {@link Void} if handling the
037 *            event does not produce a result
038 */
039public class Event<T> extends EventBase<T> {
040
041        /**
042         * Creates a new event. Passing channels is equivalent to first
043         * creating the event and then calling {@link #setChannels(Channel...)}
044         * with the given channels.
045         * 
046         * @param channels the channels to set
047         */
048        public Event(Channel... channels) {
049                super();
050                if (channels.length > 0) {
051                        setChannels(channels);
052                }
053        }
054
055        /**
056         * Creates a new event. Sets the associated completed event to the
057         * event passed as parameter and sets this event as result of
058         * the completed event.
059         * <P>
060         * The channels are handled as in {@link Event#Event(Channel...)}.
061         * 
062         * @param completedEvent the event to to set
063         * @param channels the channels to set
064         */
065        public Event(CompletedEvent<? extends Event<T>> completedEvent,
066                        Channel... channels) {
067                super();
068                @SuppressWarnings("unchecked")
069                Event<Event<T>> ce = (Event<Event<T>>)completedEvent;
070                ce.setResult(this);
071                addCompletedEvent(ce);
072                if (channels.length > 0) {
073                        setChannels(channels);
074                }
075        }
076
077        /**
078         * Returns the class of this event as representation of its kind.
079         * 
080         * @return the class of this event
081         * 
082         * @see org.jgrapes.core.Eligible#defaultCriterion()
083         */
084        @Override
085        public Object defaultCriterion() {
086                return getClass();
087        }
088
089        /**
090         * Returns <code>true</code> if the `criterion` 
091         * is of the same class or a base class of this event's class.
092         * 
093         * @see org.jgrapes.core.Eligible#isEligibleFor(java.lang.Object)
094         */
095        @Override
096        public boolean isEligibleFor(Object criterion) {
097                return Class.class.isInstance(criterion)
098                                && ((Class<?>)criterion).isAssignableFrom(getClass());
099        }
100
101        /**
102         * Implements the default behavior for handling events thrown
103         * by a handler. Fires a {@link HandlingError handling error} event
104         * for this event and the given throwable.
105         * 
106         * @see HandlingError
107         */
108        @Override
109        protected void handlingError(
110                        EventPipeline eventProcessor, Throwable throwable) {
111                eventProcessor.fire(
112                                new HandlingError(this, throwable), channels());
113        }
114
115        /* (non-Javadoc)
116         * @see java.lang.Object#toString()
117         */
118        @Override
119        public String toString() {
120                StringBuilder builder = new StringBuilder();
121                builder.append(Components.objectName(this));
122                builder.append(" [");
123                if (channels != null) {
124                        builder.append("channels=");
125                        builder.append(Common.channelsToString(channels));
126                }
127                builder.append("]");
128                return builder.toString();
129        }
130
131}
