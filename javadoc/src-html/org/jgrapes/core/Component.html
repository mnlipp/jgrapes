019package org.jgrapes.core;
020
021import org.jgrapes.core.internal.ComponentVertex;
022
023/**
024 * This class can be used as base class for implementing a component. 
025 * <P>
026 * This class implements the {@link Manager} interface. Contrary 
027 * to classes that only implement {@link ComponentType}, derived 
028 * classes therefore don't need a manager attribute to get access to the 
029 * component management methods provided by this interface.
030 * <P>
031 * This class also implements the {@code Channel} interface in such a way
032 * that each instance of this class can be used as an independant
033 * communication bus.
034 * 
035 * @see ComponentType
036 */
037public abstract class Component extends ComponentVertex 
038        implements ComponentType, Channel {
039
040        private Channel componentChannel;
041        
042        /**
043         * Creates a new component base with its channel set to
044         * itself.
045         */
046        public Component() {
047                super();
048                componentChannel = this;
049                initComponentsHandlers();
050        }
051
052        /**
053         * Creates a new component base with its channel set to the given 
054         * channel. As a special case {@link Channel#SELF} can be
055         * passed to the constructor to make the component use itself
056         * as channel. The special value is necessary as you 
057         * obviously cannot pass an object to be constructed to its 
058         * constructor.
059         * 
060         * @param componentChannel the channel that the component's 
061         * handlers listen on by default and that 
062         * {@link Manager#fire(Event, Channel...)} sends the event to 
063         */
064        public Component(Channel componentChannel) {
065                super();
066                if (componentChannel == SELF) {
067                        this.componentChannel = this;
068                } else {
069                        this.componentChannel = componentChannel;
070                }
071                initComponentsHandlers();
072        }
073
074        /* (non-Javadoc)
075         * @see org.jgrapes.core.internal.ComponentVertex#getComponent()
076         */
077        @Override
078        protected ComponentType component() {
079                return this;
080        }
081
082        /**
083         * Returns the channel associated with the component.
084         * 
085         * @return the channel as assigned by the constructor.
086         * 
087         * @see org.jgrapes.core.Manager#channel()
088         */
089        @Override
090        public Channel channel() {
091                return componentChannel;
092        }
093
094        /**
095         * Return the object itself as value.
096         */
097        @Override
098        public Object defaultCriterion() {
099                return this;
100        }
101
102        /**
103         * Matches the object itself (using identity comparison) or the
104         * {@link Channel} class.
105         * 
106         * @see Channel#isEligibleFor(Object)
107         */
108        @Override
109        public boolean isEligibleFor(Object value) {
110                return value.equals(Channel.class) 
111                                || value == defaultCriterion();
112        }
113
114}
