019package org.jgrapes.core.events;
020
021import org.jgrapes.core.Channel;
022import org.jgrapes.core.Components;
023import org.jgrapes.core.Event;
024
025/**
026 * This event signals that an error occurred while processing an event.
027 */
028public class Error extends Event<Void> {
029
030    private final Event<?> event;
031    private final String message;
032    private Throwable throwable;
033
034    /**
035     * Creates a new event as a copy of an existing event. Useful
036     * for forwarding an event.
037     *
038     * @param event the event to copy
039     */
040    public Error(Error event) {
041        this.event = event.event;
042        this.message = event.message;
043        this.throwable = event.throwable;
044    }
045
046    /**
047     * Creates a new event.
048     * 
049     * @param event the event being processed when the problem occurred
050     * @param message the message
051     */
052    public Error(Event<?> event, String message) {
053        this.event = event;
054        this.message = message;
055    }
056
057    /**
058     * Creates a new event caused by the given throwable.
059     * 
060     * @param event the event being processed when the problem occurred
061     * @param message the message
062     * @param throwable the throwable
063     */
064    public Error(Event<?> event, String message, Throwable throwable) {
065        this.event = event;
066        this.message = message;
067        this.throwable = throwable;
068    }
069
070    /**
071     * Creates a new event caused by the given throwable. The message
072     * is initialized from the throwable.
073     * 
074     * @param event the event being processed when the problem occurred
075     * @param throwable the throwable
076     */
077    public Error(Event<?> event, Throwable throwable) {
078        this.event = event;
079        this.message = throwable.getMessage() == null
080            ? throwable.getClass().getName()
081            : throwable.getMessage();
082        this.throwable = throwable;
083    }
084
085    /**
086     * Returns the event that was handled when the problem occurred.
087     * 
088     * @return the event
089     */
090    public Event<?> event() {
091        return event;
092    }
093
094    /**
095     * Returns the message passed to the constructor.
096     * 
097     * @return the message
098     */
099    public String message() {
100        return message;
101    }
102
103    /**
104     * Returns the throwable that caused the problem.
105     * 
106     * @return the throwable or {@code null} if the problem wasn't caused
107     * by a throwable.
108     */
109    public Throwable throwable() {
110        return throwable;
111    }
112
113    /*
114     * (non-Javadoc)
115     * 
116     * @see java.lang.Object#toString()
117     */
118    @Override
119    public String toString() {
120        StringBuilder builder = new StringBuilder();
121        builder.append(Components.objectName(this))
122            .append(" [");
123        if (channels().length > 0) {
124            builder.append("channels=");
125            builder.append(Channel.toString(channels()));
126        }
127        if (message != null) {
128            builder.append(", message=\"");
129            builder.append(message);
130            builder.append('"');
131        }
132        builder.append(']');
133        return builder.toString();
134    }
135}
