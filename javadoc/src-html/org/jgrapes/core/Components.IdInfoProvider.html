226        private static Map<Object, String> objectIds = new WeakHashMap<>();
227        private static Map<Class<?>, AtomicLong> idCounters = new WeakHashMap<>();
228
229        private static String getId(Class<?> scope, Object object) {
230                if (object == null) {
231                        return "?";
232                }
233                synchronized (objectIds) {
234                        return objectIds.computeIfAbsent(object, k -> 
235                                Long.toString(idCounters
236                                                .computeIfAbsent(scope, l -> new AtomicLong())
237                                                        .incrementAndGet()));
238                        
239                }
240        }

242        /**
243         * Returns the full name or simple name of the class depending
244         * on the log level.
245         * 
246         * @param clazz the class
247         * @return the name
248         */
249        public static String className(Class<?> clazz) {
250                if (Common.classNames.isLoggable(Level.FINER)) {
251                        return clazz.getName();
252                } else {
253                        return simpleClassName(clazz);
254                }
255        }
256        
257        /**
258         * Returns the simple name of a class. Contrary to 
259         * {@link Class#getSimpleName()}, this method returns
260         * the last segement of the full name for anonymous
261         * classes (instead of an empty string).
262         * 
263         * @param clazz the class
264         * @return the name
265         */
266        public static String simpleClassName(Class<?> clazz) {
267                if (!clazz.isAnonymousClass()) {
268                        return clazz.getSimpleName();
269                }
270                // Simple name of anonymous class is empty
271                String name = clazz.getName();
272                int lastDot = name.lastIndexOf('.');
273                if (lastDot <= 0) {
274                        return name;
275                }
276                return name.substring(lastDot + 1);
277        }

279        /**
280         * Returns an id of the object that is unique within a specific scope. Ids
281         * are generated and looked up in the scope of the object's class unless the
282         * class implements {@link IdInfoProvider}.
283         * 
284         * @param object
285         *            the object
286         * @return the object's name
287         */
288        public static String objectId(Object object) {
289                if (object instanceof IdInfoProvider) {
290                        return getId(((IdInfoProvider) object).idScope(),
291                                ((IdInfoProvider) object).idObject());
292                } else {
293                        return getId(object.getClass(), object);
294                }
295        }

336        /**
337         * Represents a timer as created by 
338         * {@link Components#schedule(TimeoutHandler, Instant)}.
339         */
340        public static class Timer {
341                private Scheduler scheduler;
342                private TimeoutHandler timeoutHandler;
343                private Instant scheduledFor;
344
345                private Timer(Scheduler scheduler,
346                                TimeoutHandler timeoutHandler, Instant scheduledFor) {
347                        this.scheduler = scheduler;
348                        this.timeoutHandler = timeoutHandler;
349                        this.scheduledFor = scheduledFor;
350                }
351
352                /**
353                 * Reschedules the timer for the given instant.
354                 * 
355                 * @param scheduledFor the instant
356                 */
357                public void reschedule(Instant scheduledFor) {
358                        scheduler.reschedule(this, scheduledFor);
359                }
360
361                /**
362                 * Reschedules the timer for the given duration after now.
363                 * 
364                 * @param scheduledFor the timeout
365                 */
366                public void reschedule(Duration scheduledFor) {
367                        reschedule(Instant.now().plus(scheduledFor));                   
368                }
369                
370                /**
371                 * Returns the timeout handler of this timer.
372                 * 
373                 * @return the handler
374                 */
375                public TimeoutHandler timeoutHandler() {
376                        return timeoutHandler;
377                }
378
379                /**
380                 * Returns the instant that this handler is scheduled for.
381                 * 
382                 * @return the instant
383                 */
384                public Instant scheduledFor() {
385                        return scheduledFor;
386                }
387
388                /**
389                 * Cancels this timer.
390                 */
391                public void cancel() {
392                        scheduler.cancel(this);
393                }
394        }

416        /**
417         * A general purpose scheduler.
418         */
419        private static class Scheduler extends Thread {
420
421                private PriorityQueue<Timer> timers 
422                        = new PriorityQueue<>(10,
423                                        Comparator.comparing(Timer::scheduledFor)); 
424
425                public Scheduler() {
426                        setName("Components.Scheduler");
427                        setDaemon(true);
428                        start();
429                }
430                
431                public Timer schedule(
432                                TimeoutHandler timeoutHandler, Instant scheduledFor) {
433                        Timer timer = new Timer(this, timeoutHandler, scheduledFor);
434                        synchronized (timers) {
435                                timers.add(timer);
436                                timers.notify();
437                        }
438                        return timer;
439                }
440
441                private void reschedule(Timer timer, Instant scheduledFor) {
442                        synchronized (timers) {
443                                timers.remove(timer);
444                                timer.scheduledFor = scheduledFor;
445                                timers.add(timer);
446                                timers.notify();
447                        }
448                }
449                
450                private void cancel(Timer timer) {
451                        synchronized (timers) {
452                                timers.remove(timer);
453                                timers.notify();
454                        }
455                }
456                
457                @Override
458                public void run() {
459                        while (true) {
460                                while (true) {
461                                        final Timer first;
462                                        synchronized (timers) {
463                                                first = timers.peek();
464                                                if (first == null 
465                                                                || first.scheduledFor().isAfter(Instant.now())) {
466                                                        break;
467                                                }
468                                                timers.poll();
469                                        }
470                                        timerExecutorService.submit( 
471                                                        () -> first.timeoutHandler().timeout(first));
472                                }
473                                try {
474                                        synchronized (timers) {
475                                                if (timers.size() == 0) {
476                                                        timers.wait();
477                                                } else {
478                                                        timers.wait(Math.max(1, Duration.between(Instant.now(), 
479                                                                        timers.peek().scheduledFor()).toMillis()));
480                                                }
481                                        }
482                                } catch (Throwable e) {
483                                        // Keep running.
484                                }
485                        }
486                }
487        }
488        
489        private static Scheduler scheduler = new Scheduler();
