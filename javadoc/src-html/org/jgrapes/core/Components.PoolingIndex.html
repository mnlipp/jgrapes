255    private static Map<Class<?>, AtomicLong> idCounters // NOPMD

258    private static String getId(Class<?> scope, Object object) {
259        if (object == null) {
260            return "?";
261        }
262        synchronized (objectIds) {
263            return objectIds.computeIfAbsent(object,
264                key -> Long.toString(idCounters
265                    .computeIfAbsent(scope, newKey -> new AtomicLong())
266                    .incrementAndGet()));
267
268        }
269    }

286    /**
287     * Returns the simple name of a class. Contrary to 
288     * {@link Class#getSimpleName()}, this method returns
289     * the last segement of the full name for anonymous
290     * classes (instead of an empty string).
291     * 
292     * @param clazz the class
293     * @return the name
294     */
295    public static String simpleClassName(Class<?> clazz) {
296        if (!clazz.isAnonymousClass()) {
297            return clazz.getSimpleName();
298        }
299        // Simple name of anonymous class is empty
300        String name = clazz.getName();
301        int lastDot = name.lastIndexOf('.');
302        if (lastDot <= 0) {
303            return name;
304        }
305        return name.substring(lastDot + 1);
306    }

309     * Returns an id of the object that is unique within a specific scope. Ids
310     * are generated and looked up in the scope of the object's class unless the
311     * class implements {@link IdInfoProvider}.
312     * 
313     * @param object
314     *            the object
315     * @return the object's name
316     */
317    public static String objectId(Object object) {
318        if (object == null) {
319            return "?";
320        }
321        if (object instanceof IdInfoProvider) {
322            return getId(((IdInfoProvider) object).idScope(),
323                ((IdInfoProvider) object).idObject());
324        } else {
325            return getId(object.getClass(), object);
326        }

448    /**
449     * A general purpose scheduler.
450     */
451    private static class Scheduler extends Thread {
452
453        private final PriorityQueue<Timer> timers
454            = new PriorityQueue<>(10,
455                Comparator.comparing(Timer::scheduledFor));
456
457        /**
458         * Instantiates a new scheduler.
459         */
460        public Scheduler() {
461            setName("Components.Scheduler");
462            setDaemon(true);
463            start();
464        }
465
466        /**
467         * Schedule the handler and return the resulting timer.
468         *
469         * @param timeoutHandler the timeout handler
470         * @param scheduledFor the scheduled for
471         * @return the timer
472         */
473        public Timer schedule(
474                TimeoutHandler timeoutHandler, Instant scheduledFor) {
475            @SuppressWarnings("PMD.AccessorClassGeneration")
476            Timer timer = new Timer(this, timeoutHandler, scheduledFor);
477            synchronized (timers) {
478                timers.add(timer);
479                timers.notifyAll();
480            }
481            return timer;
482        }
483
484        private void reschedule(Timer timer, Instant scheduledFor) {
485            synchronized (timers) {
486                timers.remove(timer);
487                timer.scheduledFor = scheduledFor;
488                timers.add(timer);
489                timers.notifyAll();
490            }
491        }
492
493        private void cancel(Timer timer) {
494            synchronized (timers) {
495                timers.remove(timer);
496                timers.notifyAll();
497                timer.scheduledFor = null;
498            }
499        }
500
501        @Override
502        public void run() {
503            while (true) {
504                while (true) {
505                    @SuppressWarnings("PMD.AvoidFinalLocalVariable")
506                    final Timer first;
507                    synchronized (timers) {
508                        first = timers.peek();
509                        if (first == null
510                            || first.scheduledFor().isAfter(Instant.now())) {
511                            break;
512                        }
513                        timers.poll();
514                    }
515                    timerExecutorService.submit(
516                        () -> first.timeoutHandler().timeout(first));
517                }
518                try {
519                    synchronized (timers) {
520                        if (timers.size() == 0) {
521                            timers.wait();
522                        } else {
523                            timers
524                                .wait(Math.max(1,
525                                    Duration.between(Instant.now(),
526                                        timers.peek().scheduledFor())
527                                        .toMillis()));
528                        }
529                    }
530                } catch (Exception e) { // NOPMD
531                    // Keep running.
532                }
533            }
534        }
535    }

777    /**
778     * An index of pooled items. Each key is associated with a set
779     * of values. Values can be added or retrieved from the set.
780     *
781     * @param <K> the key type
782     * @param <V> the value type
783     */
784    public static class PoolingIndex<K, V> {
785
786        @SuppressWarnings("PMD.UseConcurrentHashMap")
787        private final Map<K, Set<ValueReference>> backing = new HashMap<>();
788        private final ReferenceQueue<V> orphanedEntries
789            = new ReferenceQueue<>();
790
791        @SuppressWarnings("PMD.CommentRequired")
792        private class ValueReference extends WeakReference<V> {
793
794            private final K key;
795
796            public ValueReference(K key, V referent) {
797                super(referent, orphanedEntries);
798                this.key = key;
799            }
800
801            /*
802             * (non-Javadoc)
803             * 
804             * @see java.lang.Object#hashCode()
805             */
806            @Override
807            public int hashCode() {
808                V value = get();
809                if (value == null) {
810                    return 0;
811                }
812                return value.hashCode();
813            }
814
815            /*
816             * (non-Javadoc)
817             * 
818             * @see java.lang.Object#equals(java.lang.Object)
819             */
820            @Override
821            public boolean equals(Object obj) {
822                if (obj == null) {
823                    return false;
824                }
825                if (!obj.getClass().equals(ValueReference.class)) {
826                    return false;
827                }
828                V value1 = get();
829                @SuppressWarnings("unchecked")
830                V value2 = ((ValueReference) obj).get();
831                if (value1 == null || value2 == null) {
832                    return false;
833                }
834                return value1.equals(value2);
835            }
836        }
837
838        @SuppressWarnings("PMD.CompareObjectsWithEquals")
839        private void cleanOrphaned() {
840            while (true) {
841                @SuppressWarnings("unchecked")
842                ValueReference orphaned
843                    = (ValueReference) orphanedEntries.poll();
844                if (orphaned == null) {
845                    return;
846                }
847                synchronized (this) {
848                    Set<ValueReference> set = backing.get(orphaned.key);
849                    if (set == null) {
850                        continue;
851                    }
852                    Iterator<ValueReference> iter = set.iterator();
853                    while (iter.hasNext()) {
854                        ValueReference ref = iter.next();
855                        if (ref == orphaned) {
856                            iter.remove();
857                            if (set.isEmpty()) {
858                                backing.remove(orphaned.key);
859                            }
860                            break;
861                        }
862                    }
863                }
864            }
865        }
866
867        /**
868         * Remove all entries.
869         */
870        public void clear() {
871            backing.clear();
872            cleanOrphaned();
873        }
874
875        /**
876         * Checks if the key is in the index.
877         *
878         * @param key the key
879         * @return true, if successful
880         */
881        public boolean containsKey(Object key) {
882            return backing.containsKey(key);
883        }
884
885        /**
886         * Checks if the index is empty.
887         *
888         * @return true, if is empty
889         */
890        public boolean isEmpty() {
891            return backing.isEmpty();
892        }
893
894        /**
895         * Returns all keys.
896         *
897         * @return the sets the
898         */
899        public Set<K> keySet() {
900            synchronized (this) {
901                return backing.keySet();
902            }
903        }
904
905        /**
906         * Adds the value to the pool of values associated with the key.
907         *
908         * @param key the key
909         * @param value the value
910         * @return the v
911         */
912        public V add(K key, V value) {
913            synchronized (this) {
914                cleanOrphaned();
915                backing.computeIfAbsent(key, k -> new HashSet<>())
916                    .add(new ValueReference(key, value));
917                return value;
918            }
919        }
920
921        /**
922         * Retrives and removes an item from the pool associated with the key.
923         *
924         * @param key the key
925         * @return the removed item or `null` if the pool is empty
926         */
927        @SuppressWarnings("PMD.AvoidBranchingStatementAsLastInLoop")
928        public V poll(K key) {
929            synchronized (this) {
930                cleanOrphaned();
931                Set<ValueReference> set = backing.get(key);
932                if (set == null) {
933                    return null;
934                }
935                Iterator<ValueReference> iter = set.iterator();
936                while (iter.hasNext()) {
937                    ValueReference ref = iter.next();
938                    V value = ref.get();
939                    iter.remove();
940                    if (value == null) {
941                        continue;
942                    }
943                    if (set.isEmpty()) {
944                        backing.remove(key);
945                    }
946                    return value;
947                }
948            }
949            return null;
950        }
951
952        /**
953         * Removes all values associated with the key.
954         *
955         * @param key the key
956         */
957        public void removeAll(K key) {
958            synchronized (this) {
959                cleanOrphaned();
960                backing.remove(key);
961            }
962        }
963
964        /**
965         * Removes the given value from the pool associated with the given key.
966         *
967         * @param key the key
968         * @param value the value
969         * @return the v
970         */
971        @SuppressWarnings("PMD.DataflowAnomalyAnalysis")
972        public V remove(K key, V value) {
973            synchronized (this) {
974                cleanOrphaned();
975                Set<ValueReference> set = backing.get(key);
976                if (set == null) {
977                    return null;
978                }
979                Iterator<ValueReference> iter = set.iterator();
980                while (iter.hasNext()) {
981                    ValueReference ref = iter.next();
982                    V stored = ref.get();
983                    boolean found = false;
984                    if (stored == null) {
985                        iter.remove();
986                    }
987                    if (stored.equals(value)) {
988                        iter.remove();
989                        found = true;
990                    }
991                    if (set.isEmpty()) {
992                        backing.remove(key);
993                    }
994                    if (found) {
995                        return value;
996                    }
997                }
998            }
999            return null;
1000        }
1001
1002        /**
1003         * Removes the value from the first pool in which it is found.
1004         *
1005         * @param value the value
1006         * @return the value or `null` if the value is not found
1007         */
1008        public V remove(V value) {
1009            synchronized (this) {
1010                for (Set<ValueReference> set : backing.values()) {
1011                    Iterator<ValueReference> iter = set.iterator();
1012                    while (iter.hasNext()) {
1013                        ValueReference ref = iter.next();
1014                        V stored = ref.get();
1015                        if (stored == null) {
1016                            iter.remove();
1017                        }
1018                        if (stored.equals(value)) {
1019                            iter.remove();
1020                            return value;
1021                        }
1022                    }
1023                }
1024            }
1025            return null;
1026        }
1027
1028        /**
1029         * Returns the number of keys in the index.
1030         *
1031         * @return the numer of keys
1032         */
1033        public int keysSize() {
1034            return backing.size();
1035        }
