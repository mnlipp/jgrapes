045        /**
046         * Returns a component's manager. For a component that inherits
047         * from {@link org.jgrapes.core.Component} this method simply returns
048         * the component as it is its own manager.
049         * 
050         * For components that implement {@link ComponentType} but don't inherit from 
051         * {@link org.jgrapes.core.Component} the method returns the value of 
052         * the attribute annotated as manager slot. If this attribute is still
053         * empty, this method makes the component the root
054         * of a new tree and returns its manager.
055         * 
056         * @param component the component
057         * @return the component (with its manager attribute set)
058         */
059        public static Manager manager(ComponentType component) {
060                return ComponentVertex.componentVertex(component, null);
061        }
062
063        /**
064         * Returns a component's manager like {@link #manager(ComponentType)}.
065         * If the manager slot attribute is empty, the component is initialized
066         * with its component channel set to the given parameter. Invoking
067         * this method overrides any channel set in the
068         * {@link ComponentManager} annotation.
069         * 
070         * This method is usually invoked by the constructor of a class
071         * that implements {@link ComponentType}.
072         * 
073         * @param component the component
074         * @param componentChannel the channel that the component's 
075         * handlers listen on by default and that 
076         * {@link Manager#fire(Event, Channel...)} sends the event to 
077         * @return the component (with its manager attribute set)
078         * @see Component#Component(Channel)
079         */
080        public static Manager manager(
081                        ComponentType component, Channel componentChannel) {
082                return ComponentVertex.componentVertex(component, componentChannel);
083        }

085        /**
086         * Fires a {@link Start} event with an associated
087         * {@link Started} completion event on the broadcast channel
088         * of the given application and wait for the completion of the
089         * <code>Start</code> event.
090         * 
091         * @param application the application to start
092         * @throws InterruptedException if the execution was interrupted
093         */
094        public static void start(ComponentType application) 
095                        throws InterruptedException {
096                manager(application).fire(new Start() , Channel.BROADCAST).get();
097        }

104         * @throws InterruptedException if the current thread was interrupted
105         * while waiting

117         * @throws InterruptedException if the execution was interrupted

120        public static boolean awaitExhaustion(long timeout)

140        /**
141         * Returns the full name of the object's class together with an id (see 
142         * {@link #objectId(Object)}). The result can be used as a unique
143         * human readable identifier for arbitrary objects.
144         * 
145         * @param object
146         *            the object
147         * @return the object's name
148         */
149        public static String objectFullName(Object object) {
150                StringBuilder builder = new StringBuilder();
151                builder.append(object.getClass().getName());
152                builder.append('#');
153                builder.append(objectId(object));
154                return builder.toString();
155        }
156
157        /**
158         * Returns the name of the object's class together with an id (see 
159         * {@link #objectId(Object)}). May be used to implement {@code toString()}
160         * with identifiable objects. If the log level is "finer", the full
161         * class name will be used for the returned value, else the simple name.
162         * 
163         * @param object
164         *            the object
165         * @return the object's name
166         */
167        public static String objectName(Object object) {
168                StringBuilder builder = new StringBuilder();
169                builder.append(Components.classToString(object.getClass()));
170                builder.append('#');
171                builder.append(objectId(object));
172                return builder.toString();
173        }
174
175        private static Map<Object, String> objectIds = new WeakHashMap<>();
176        private static Map<Class<?>, AtomicLong> idCounters = new WeakHashMap<>();
177
178        private static String getId(Class<?> scope, Object object) {
179                if (object == null) {
180                        return "?";
181                }
182                synchronized (objectIds) {
183                        return objectIds.computeIfAbsent(object, k -> 
184                                Long.toString(idCounters
185                                                .computeIfAbsent(scope, l -> new AtomicLong())
186                                                        .incrementAndGet()));
187                        
188                }
189        }
190
191        /**
192         * Returns the full name or simple name of the class depending
193         * on the log level.
194         * 
195         * @param clazz the class
196         * @return the name
197         */
198        public static String classToString(Class<?> clazz) {
199                if (Common.classNames.isLoggable(Level.FINER)) {
200                        return clazz.getName();
201                } else {
202                        return clazz.getSimpleName();
203                }
204        }
205        
206        /**
207         * Returns an id of the object that is unique within a specific scope. Ids
208         * are generated and looked up in the scope of the object's class unless the
209         * class implements {@link IdInfoProvider}.
210         * 
211         * @param object
212         *            the object
213         * @return the object's name
214         */
215        public static String objectId(Object object) {
216                if (object instanceof IdInfoProvider) {
217                        return getId(((IdInfoProvider) object).idScope(),
218                                ((IdInfoProvider) object).idObject());
219                } else {
220                        return getId(object.getClass(), object);
221                }
222        }
223
224        /**
225         * Implemented by classes that want a special class (scope) to be used
226         * for looking up their id or want to map to another object for getting the
227         * id (see {@link Components#objectId(Object)}).
228         */
229        public static interface IdInfoProvider {
230                
231                /**
232                 * Returns the scope.
233                 * 
234                 * @return the scope
235                 */
236                Class<?> idScope();
237                
238                /**
239                 * Returns the object to be used for generating the id.
240                 * 
241                 * @return the object (defaults to {@code this})
242                 */
243                default Object idObject() {
244                        return this;
245                }
246        }

248        /**
249         * Instances are added to the scheduler in order to be invoked
250         * at a given time.
251         */
252        @FunctionalInterface
253        public static interface TimeoutHandler {
254
255                /**
256                 * Invoked when the timeout occurs.
257                 * 
258                 * @param scheduledFor the time that the handler was scheduled for
259                 */
260                void timeout(Timer timer);
261        }
262
263        /**
264         * Represents a timer as created by 
265         * {@link Components#schedule(TimeoutHandler, Instant)}.
266         */
267        public static class Timer {
268                private Scheduler scheduler;
269                private TimeoutHandler timeoutHandler;
270                private Instant scheduledFor;
271
272                private Timer(Scheduler scheduler,
273                                TimeoutHandler timeoutHandler, Instant scheduledFor) {
274                        this.scheduler = scheduler;
275                        this.timeoutHandler = timeoutHandler;
276                        this.scheduledFor = scheduledFor;
277                }
278
279                /**
280                 * Returns the timeout handler of this timer.
281                 * 
282                 * @return the handler
283                 */
284                public TimeoutHandler timeoutHandler() {
285                        return timeoutHandler;
286                }
287
288                /**
289                 * Returns the instant that this handler is scheduled for.
290                 * 
291                 * @return the instant
292                 */
293                public Instant scheduledFor() {
294                        return scheduledFor;
295                }
296
297                /**
298                 * Cancels this timer.
299                 */
300                public void cancel() {
301                        scheduler.cancel(this);
302                }
303        }
304
305        /**
306         * A general purpose scheduler.
307         */
308        private static class Scheduler extends Thread {
309
310                private PriorityQueue<Timer> timers 
311                        = new PriorityQueue<>(10,
312                                        Comparator.comparing(Timer::scheduledFor)); 
313
314                public Scheduler() {
315                        setName("Components.Scheduler");
316                        setDaemon(true);
317                        start();
318                }
319                
320                public Timer schedule(
321                                TimeoutHandler timeoutHandler, Instant scheduledFor) {
322                        Timer timer = new Timer(this, timeoutHandler, scheduledFor);
323                        synchronized (this) {
324                                timers.add(timer);
325                                notify();
326                        }
327                        return timer;
328                }
329                
330                private void cancel(Timer timer) {
331                        synchronized (this) {
332                                timers.remove(timer);
333                                notify();
334                        }
335                }
336                
337                @Override
338                public void run() {
339                        while (true) {
340                                Instant now = Instant.now();
341                                synchronized (this) {
342                                        while (true) {
343                                                final Timer first = timers.peek();
344                                                if (first == null || first.scheduledFor().isAfter(now)) {
345                                                        break;
346                                                }
347                                                timers.poll();
348                                                new Thread(() -> first.timeoutHandler()
349                                                                .timeout(first)).start();
350                                        }
351                                        try {
352                                                if (timers.size() == 0) {
353                                                        wait();
354                                                } else {
355                                                        wait(Math.max(1, Duration.between(now, 
356                                                                        timers.peek().scheduledFor()).toMillis()));
357                                                }
358                                        } catch (Throwable e) {
359                                        }
360                                }
361                        }
362                }
363        }
364        
365        private static Scheduler scheduler = new Scheduler();
366
367        /**
368         * Schedules the given timeout handler for the given instance. 
369         * 
370         * @param timeoutHandler the handler
371         * @param scheduledFor the instance in time
372         * @return the timer
373         */
374        public static Timer schedule(
375                        TimeoutHandler timeoutHandler, Instant scheduledFor) {
376                return scheduler.schedule(timeoutHandler, scheduledFor);
377        }
378
379        /**
380         * Schedules the given timeout handler for the given 
381         * offset from now. 
382         * 
383         * @param timeoutHandler the handler
384         * @param scheduledFor the time to wait
385         * @return the timer
386         */
387        public static Timer schedule(
388                        TimeoutHandler timeoutHandler, Duration scheduledFor) {
389                return scheduler.schedule(
390                                timeoutHandler, Instant.now().plus(scheduledFor));
391        }
