019package org.jgrapes.core;
020
021import java.lang.reflect.Method;
022import java.util.Iterator;
023import java.util.List;
024import java.util.concurrent.ExecutorService;
025import java.util.stream.Stream;
026import org.jgrapes.core.events.Start;
027
028/**
029 * The interface that provides the methods for manipulating the
030 * component and the component's hierarchy and for firing events. 
031 * Every component has access to a manager implementation that 
032 * manages the component.
033 * 
034 * The `Manager` for a component that extends from 
035 * {@link Component} is provided by the base class itself.
036 * Components that only implement the {@link ComponentType} interface
037 * get their `Manager` assigned to their annotated 
038 * attribute when they are attached to the component tree.
039 * 
040 * @see ComponentType
041 */
042public interface Manager extends Iterable<ComponentType> {
043
044    /**
045     * Sets the (optional) name of the component.
046     * 
047     * @param name the name to set
048     * @return the component (for comfortable chaining)
049     */
050    ComponentType setName(String name);
051
052    /**
053     * Returns the (optional) name of the component.
054     */
055    String name();
056
057    /**
058     * Returns the path of the component. The path is the concatenation
059     * of a slash ('/') and the component's name for all components
060     * from the root component to this component. If a component
061     * doesn't have a name set, the simple name of its class
062     * (see {@link Class#getSimpleName()}) is used instead.
063     */
064    String componentPath();
065
066    /**
067     * Detaches the component managed by this manager (with its children,
068     * if any) from the component tree that it currently belongs to.
069     * <P>
070     * This method results in a <code>IllegalStateException</code> if
071     * called on a tree before a {@link org.jgrapes.core.events.Start} 
072     * event has been fired on 
073     * it. The Reason for this restriction is that distributing
074     * buffered events between the two separated trees cannot easily be
075     * defined in an intuitive way.
076     * 
077     * @return the component (for comfortable chaining)
078     * @throws IllegalStateException if invoked before a <code>Start</code>
079     * event
080     */
081    ComponentType detach();
082
083    /**
084     * Attaches the given component node (or complete tree) as a child 
085     * to the component managed by this manager. The node or tree may not
086     * have been started.
087     * 
088     * If a component (or component tree) is attached to an already
089     * started tree, a {@link Start} event is automatically fired 
090     * with the list of components from the attached subtree as 
091     * channels. This guarantees that every component gets a 
092     * {@link Start} event once. For efficiency, it is therefore 
093     * preferable to build a subtree first and attach it, instead
094     * of attaching the nodes to the existing tree one by one.
095     * 
096     * @param <T> the component node's type
097     * @param child the component to add
098     * @return the added component (for comfortable chaining)
099     */
100    <T extends ComponentType> T attach(T child);
101
102    /**
103     * Returns the child components of the component managed by 
104     * this manager as unmodifiable list.
105     * 
106     * @return the child components
107     */
108    List<ComponentType> children();
109
110    /**
111     * Returns the parent of the component managed by this manager.
112     * 
113     * @return the parent component or <code>null</code> if the
114     * component is not registered with another component
115     */
116    ComponentType parent();
117
118    /**
119     * Returns the root of the tree the component 
120     * managed by this manager belongs to.
121     * 
122     * @return the root
123     */
124    ComponentType root();
125
126    /**
127     * Returns the channel of the component managed by this manager.
128     * 
129     * @return the channel that the component's 
130     * handlers listen on by default and that 
131     * {@link Manager#fire(Event, Channel...)} sends the event to 
132     */
133    Channel channel();
134
135    /**
136     * Fires the given event on the given channel. If no channels are
137     * specified as parameters, the event is fired on the event's 
138     * channel (see {@link Event#channels()}). If the event doesn't
139     * specify channels either, the event is fired on the 
140     * channel of the component managed by this manager 
141     * (see {@link #channel()}).
142     * <P>
143     * If an event is fired inside an event handler, it is added to the
144     * {@link EventPipeline} that has invoked the handler. If an event is fired
145     * by some other thread (not associated with a pipeline), a new pipeline
146     * is created for handling the event (and any events triggered by it). 
147     * 
148     * @param <T> the result type of the event
149     * @param event the event to fire
150     * @param channels the channels to fire the event on
151     * @return the event (for easy chaining)
152     */
153    <T> Event<T> fire(Event<T> event, Channel... channels);
154
155    /**
156     * Adds a handler for the given method with the given scope
157     * and priority.
158     * 
159     * This method is usually not invoked directly. Rather, a {@link
160     * org.jgrapes.core.annotation.HandlerDefinition.Evaluator} provides
161     * `add(...)` methods that evaluate the required `method` and
162     * `scope` arguments from easier to provide arguments and then
163     * call this method.
164     * 
165     * @param method the method to invoke 
166     * @param scope the handler scope to be used for matching events
167     * @param priority the priority of the handler
168     * @see org.jgrapes.core.annotation.Handler.Evaluator#add 
169     */
170    void addHandler(Method method, HandlerScope scope, int priority);
171
172    /**
173     * Returns the pipeline used when firing an event.
174     * 
175     * @return the event pipeline
176     * @see #fire(Event, Channel...) 
177     */
178    EventPipeline activeEventPipeline();
179
180    /**
181     * Return a new {@link EventPipeline} that processes the added events
182     * using a thread from a thread pool.
183     * 
184     * @return the pipeline
185     */
186    EventPipeline newEventPipeline();
187
188    /**
189     * Return a new {@link EventPipeline} that processes the added events
190     * using threads from the given executor service.
191     * 
192     * @param executorService the executor service
193     * @return the pipeline
194     */
195    EventPipeline newEventPipeline(ExecutorService executorService);
196
197    /**
198     * Return a new {@link EventPipeline} that processes an added event
199     * (and all events caused by it) before returning from the
200     * {@link EventPipeline#fire} method.
201     * <P>
202     * The returned event pipeline is not thread-safe, i.e. no other thread
203     * may call <code>add</code> while an invocation of <code>add</code>
204     * is being processed.
205     * 
206     * @return the pipeline
207     */
208    EventPipeline newSyncEventPipeline();
209
210    /**
211     * Register the managed component as a running generator. 
212     */
213    void registerAsGenerator();
214
215    /**
216     * Unregister the managed component as a running generator.
217     */
218    void unregisterAsGenerator();
219
220    /**
221     * Returns an iterator that visits the components of the
222     * component subtree that has this node as root.
223     * 
224     * @return the iterator
225     */
226    @Override
227    Iterator<ComponentType> iterator();
228
229    /**
230     * Returns the components visited when traversing the
231     * tree that starts with this component.
232     * 
233     * @return the stream
234     */
235    @SuppressWarnings("PMD.DataflowAnomalyAnalysis")
236    default Stream<ComponentType> stream() {
237        Stream.Builder<ComponentType> res = Stream.builder();
238        for (ComponentType c : this) {
239            res.accept(c);
240        }
241        return res.build();
242    }
243}
