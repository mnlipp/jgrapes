019package org.jgrapes.core.internal;
020
021import java.util.Collections;
022import java.util.HashSet;
023import java.util.List;
024import java.util.Set;
025import java.util.concurrent.Future;
026import java.util.concurrent.atomic.AtomicInteger;
027
028import org.jgrapes.core.Associator;
029import org.jgrapes.core.Channel;
030import org.jgrapes.core.CompletionLock;
031import org.jgrapes.core.Eligible;
032import org.jgrapes.core.Event;
033import org.jgrapes.core.EventPipeline;
034
035/**
036 * Provides the implementations of methods to class {@link Event} that
037 * need access to classes or methods that are visible in the implementation
038 * package only. The class is not intended to be used as base class
039 * for any other class.
040 * 
041 * @param <T> the result type of the event. Use {@link Void} if handling
042 * the event does not produce a result
043 */
044public abstract class EventBase<T> 
045        implements Eligible, Future<T>, Associator {
046
047        /** The event that caused this event. */
048        private EventBase<?> generatedBy = null;
049        /** Number of events that have to be dispatched until completion.
050         * This is one for the event itself and one more for each event
051         * that has this event as its cause. */
052        private AtomicInteger openCount = new AtomicInteger(0);
053        /** Completion locks. */
054        private Set<CompletionLockBase> completionLocks = null;
055        /** Set when the event is enqueued, reset when it has been completed. */
056        private EventProcessor processedBy = null;
057        /** Indicates that the event should not processed further. */
058        private boolean stopped = false;
059        /** The events to be fired upon completion. Using this attribute
060         * provides a slightly faster access than invoking
061         * {@link Event#completionEvents()}, which wraps the result in
062         * an unmodifiable set. */
063        protected Set<Event<?>> completionEvents = null;
064        /** Set when the event has been completed. */
065        protected boolean completed = false;
066        
067        public abstract Channel[] channels();
068
069        protected abstract void handled();
070
071        protected abstract List<T> currentResults();
072        
073        /**
074         * Returns <code>true</code> if the event has been enqueued in a pipeline.
075         * 
076         * @return the result
077         */
078        protected boolean enqueued() {
079                return processedBy != null || completed || isCancelled();
080        }
081
082        /**
083         * Invoked when an exception occurs while invoking a handler for an event.
084         * 
085         * @param eventProcessor the manager that has invoked the handler
086         * @param throwable the exception that has been thrown by the handler
087         */
088        protected abstract void handlingError(
089                        EventPipeline eventProcessor, Throwable throwable);
090
091        /**
092         * Can be called during the execution of an event handler to indicate
093         * that the event should not be processed further. All remaining 
094         * handlers for this event will be skipped.
095         * 
096         * @return the object for easy chaining
097         */
098        public Event<T> stop() {
099                stopped = true;
100                return (Event<T>)this;
101        }
102
103        /**
104         * Returns <code>true</code> if {@link #stop} has been called.
105         * 
106         * @return the stopped state
107         */
108        public boolean isStopped() {
109                return stopped;
110        }
111
112        /**
113         * If an event is fired while processing another event, note
114         * the event being processed. This allows us to track the cause
115         * of events to the "initial" (externally) generated event that
116         * triggered everything.
117         * 
118         * @param causingEvent the causing event to set
119         */
120        void generatedBy(EventBase<?> causingEvent) {
121                openCount.incrementAndGet();
122                generatedBy = causingEvent;
123                if (causingEvent != null) {
124                        causingEvent.openCount.incrementAndGet();
125                }
126        }
127
128        /**
129         * Set the processor that will (eventually) process the event. 
130         * 
131         * @param processor the processor
132         */
133        void processedBy(EventProcessor processor) {
134                this.processedBy = processor;
135        }
136        
137        /**
138         * @param pipeline
139         */
140        void decrementOpen() {
141                if (openCount.decrementAndGet() == 0 && !completed) {
142                        synchronized (this) {
143                                completed = true;
144                                notifyAll();
145                        }
146                        if (completionEvents != null && !isCancelled()) {
147                                processedBy.updateNewEventsParent(generatedBy);
148                                for (Event<?> e: completionEvents) {
149                                        Channel[] completeChannels = e.channels();
150                                        if (completeChannels == null) {
151                                                // Note that channels cannot be null, as it is set
152                                                // when firing the event and an event is never fired
153                                                // on no channels.
154                                                completeChannels = channels();
155                                                e.setChannels(completeChannels);
156                                        }
157                                        processedBy.add(e, completeChannels);
158                                }
159                        }
160                        if (generatedBy != null) {
161                                generatedBy.decrementOpen();
162                        }
163                        processedBy = null; // No longer needed
164                }
165        }
166
167        /**
168         * Adds the given completion lock. 
169         * 
170         * @param lock the lock
171         * @see CompletionLock
172         */
173        Event<T> addCompletionLock(CompletionLockBase lock) {
174                synchronized (this) {
175                        if (completionLocks == null) {
176                                completionLocks = Collections.synchronizedSet(new HashSet<>());
177                        }
178                }
179                if (completionLocks.add(lock)) {
180                        openCount.incrementAndGet();
181                        lock.startTimer();
182                }
183                return (Event<T>)this;
184        }
185
186        /**
187         * Removes the given completion lock. 
188         * 
189         * @param lock the lock
190         * @see CompletionLock
191         */
192        void removeCompletionLock(CompletionLockBase lock) {
193                if (completionLocks == null) {
194                        return;
195                }
196                if (completionLocks.remove(lock)) {
197                        decrementOpen();
198                }
199                lock.cancelTimer();
200        }
201        
202}
