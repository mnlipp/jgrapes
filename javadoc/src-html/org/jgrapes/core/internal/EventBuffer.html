019package org.jgrapes.core.internal;
020
021import java.util.concurrent.ExecutorService;
022
023import org.jgrapes.core.Channel;
024import org.jgrapes.core.Event;
025import org.jgrapes.core.events.Start;
026
027/**
028 * The event buffer is used before a tree has been started. It simply
029 * buffers all events until a {@link Start} event is added.
030 */
031public class EventBuffer implements InternalEventPipeline {
032        
033        private ComponentTree componentTree;
034        /** Buffered events. */
035        private EventQueue buffered = new EventQueue();
036        /** The event pipeline that we delegate to after the start
037         * event has been detected. */
038        private InternalEventPipeline activePipeline = null;
039        
040        EventBuffer(ComponentTree componentTree) {
041                super();
042                this.componentTree = componentTree;
043        }
044
045        @Override
046        public synchronized void merge(InternalEventPipeline other) {
047                if (!(other instanceof EventBuffer)) {
048                        throw new IllegalArgumentException(
049                                        "Can only merge events from an EventBuffer.");
050                }
051                buffered.addAll(((EventBuffer) other).retrieveEvents());
052        }
053
054        @Override
055        public synchronized <T extends Event<?>> T add(T event,
056                Channel... channels) {
057                // If thread1 adds the start event and thread2 gets here before we
058                // have changed the event processor for the tree, forward the
059                // event to the event processor that should already have been used.
060                if (activePipeline != null) {
061                        activePipeline.add(event, channels);
062                        return event;
063                }
064                // Event gets enqueued (increments reference count).
065                ((EventBase<?>)event).generatedBy(null);
066                buffered.add(event, channels);
067                if (event instanceof Start) {
068                        // Merge all events into an "standard" event processor
069                        // and set it as default processor for the tree (with
070                        // any thread specific pipelines taking precedence).
071                        EventProcessor processor = new EventProcessor(componentTree);
072                        activePipeline = new FeedBackPipelineFilter(processor);
073                        componentTree.setEventPipeline(activePipeline);
074                        processor.add(buffered);
075                }
076                return event;
077        }
078
079        synchronized EventQueue retrieveEvents() {
080                EventQueue old = buffered;
081                buffered = new EventQueue();
082                return old;
083        }
084
085        /* (non-Javadoc)
086         * @see org.jgrapes.core.internal.InternalEventPipeline#executorService()
087         */
088        @Override
089        public ExecutorService executorService() {
090                return EventProcessor.defaultExecutorService;
091        }
092
093        /* (non-Javadoc)
094         * @see java.lang.Object#toString()
095         */
096        @Override
097        public String toString() {
098                StringBuilder builder = new StringBuilder();
099                builder.append("EventBuffer [");
100                if (buffered != null) {
101                        builder.append("buffered=");
102                        builder.append(buffered);
103                }
104                builder.append("]");
105                return builder.toString();
106        }
107}
