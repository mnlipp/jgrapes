019package org.jgrapes.core.internal;
020
021import java.util.concurrent.ExecutorService;
022
023import org.jgrapes.core.Channel;
024import org.jgrapes.core.Components;
025import org.jgrapes.core.Event;
026import org.jgrapes.core.EventPipeline;
027
028/**
029 * This class provides the default implementation of an {@link EventPipeline}.
030 */
031public class EventProcessor implements InternalEventPipeline, Runnable {
032
033    @SuppressWarnings("PMD.VariableNamingConventions")
034    protected static final ThreadLocal<EventBase<?>> newEventsParent
035        = new ThreadLocal<>();
036
037    private final ExecutorService executorService;
038    private final ComponentTree componentTree;
039    private final EventPipeline asEventPipeline;
040    protected final EventQueue queue = new EventQueue();
041    private boolean isExecuting;
042
043    /**
044     * Instantiates a new event processor.
045     *
046     * @param tree the tree
047     */
048    /* default */ EventProcessor(ComponentTree tree) {
049        this(tree, Components.defaultExecutorService());
050    }
051
052    /* default */ EventProcessor(ComponentTree tree,
053            ExecutorService executorService) {
054        this.componentTree = tree;
055        this.executorService = executorService;
056        asEventPipeline = new CheckingPipelineFilter(this);
057    }
058
059    /* default */ EventPipeline asEventPipeline() {
060        return asEventPipeline;
061    }
062
063    /**
064     * Called before adding completion events. The parent of
065     * a completion event is not the event that has completed but
066     * the event that generated the original event.
067     *
068     * @param parent the new parent
069     */
070    /* default */ void updateNewEventsParent(EventBase<?> parent) {
071        newEventsParent.set(parent);
072    }
073
074    @Override
075    public <T extends Event<?>> T add(T event, Channel... channels) {
076        ((EventBase<?>) event).generatedBy(newEventsParent.get());
077        ((EventBase<?>) event).processedBy(this);
078        queue.add(event, channels);
079        synchronized (this) {
080            if (!isExecuting) {
081                GeneratorRegistry.instance().add(this);
082                isExecuting = true;
083                executorService.execute(this);
084            }
085        }
086        return event;
087    }
088
089    /* default */ void add(EventQueue source) {
090        while (true) {
091            EventChannelsTuple entry = source.poll();
092            if (entry == null) {
093                break;
094            }
095            entry.event.processedBy(this);
096            queue.add(entry);
097        }
098        synchronized (this) {
099            if (!isExecuting) {
100                GeneratorRegistry.instance().add(this);
101                isExecuting = true;
102                executorService.execute(this);
103            }
104        }
105    }
106
107    @Override
108    public void merge(InternalEventPipeline other) {
109        if (!(other instanceof BufferingEventPipeline)) {
110            throw new IllegalArgumentException(
111                "Can only merge events from an BufferingEventPipeline.");
112        }
113        add(((BufferingEventPipeline) other).retrieveEvents());
114    }
115
116    @Override
117    public void run() {
118        String origName = Thread.currentThread().getName();
119        try {
120            Thread.currentThread().setName(
121                origName + " (P" + Components.objectId(this) + ")");
122            FeedBackPipelineFilter.setAssociatedPipeline(this);
123            while (true) {
124                // No lock needed if queue is filled
125                EventChannelsTuple next = queue.peek();
126                if (next == null) {
127                    synchronized (this) {
128                        // Retry with lock for proper synchronization
129                        next = queue.peek();
130                        if (next == null) {
131                            GeneratorRegistry.instance().remove(this);
132                            isExecuting = false;
133                            break;
134                        }
135                    }
136                }
137                newEventsParent.set(next.event);
138                componentTree.dispatch(
139                    asEventPipeline, next.event, next.channels);
140                newEventsParent.get().decrementOpen();
141                queue.remove();
142            }
143        } finally {
144            Thread.currentThread().setName(origName);
145            newEventsParent.set(null);
146            FeedBackPipelineFilter.setAssociatedPipeline(null);
147        }
148    }
149
150    /*
151     * (non-Javadoc)
152     * 
153     * @see org.jgrapes.core.internal.InternalEventPipeline#executorService()
154     */
155    @Override
156    public ExecutorService executorService() {
157        return executorService;
158    }
159
160    /*
161     * (non-Javadoc)
162     * 
163     * @see java.lang.Object#toString()
164     */
165    @Override
166    public String toString() {
167        StringBuilder builder = new StringBuilder();
168        builder.append(Components.objectName(this))
169            .append(" [");
170        if (queue != null) {
171            builder.append("queue=")
172                .append(queue);
173        }
174        builder.append(']');
175        return builder.toString();
176    }
177
178}
