019package org.jgrapes.core.internal;
020
021import java.util.WeakHashMap;
022import java.util.concurrent.ExecutorService;
023import java.util.concurrent.Executors;
024
025import org.jgrapes.core.Channel;
026import org.jgrapes.core.ComponentType;
027import org.jgrapes.core.Components;
028import org.jgrapes.core.Event;
029import org.jgrapes.core.EventPipeline;
030
031/**
032 * This class provides the default implementation of an {@link EventPipeline}.
033 */
034public class EventProcessor implements ExecutingEventPipeline, Runnable {
035
036        private static ExecutorService defaultExecutorService 
037                = Executors.newCachedThreadPool();
038        
039        protected static ThreadLocal<EventBase<?>> 
040                currentlyHandling = new ThreadLocal<>();
041        
042        private ExecutorService executorService;
043        private ComponentTree componentTree;
044        private EventPipeline asEventPipeline;
045        protected EventQueue queue = new EventQueue();
046        private WeakHashMap<ComponentType, Object> componentContext 
047                = new WeakHashMap<>();
048        
049        EventProcessor(ComponentTree tree) {
050                this(tree, defaultExecutorService);
051        }
052
053        EventProcessor(ComponentTree tree, ExecutorService executorService) {
054                this.componentTree = tree;
055                this.executorService = executorService;
056                asEventPipeline = new CheckingPipelineFilter(this);
057        }
058
059        @Override
060        public <T extends Event<?>> T add(T event, Channel... channels) {
061                ((EventBase<?>)event).generatedBy(currentlyHandling.get());
062                synchronized (queue) {
063                        boolean wasEmpty = queue.isEmpty();
064                        queue.add(event, channels);
065                        if (wasEmpty) {
066                                GeneratorRegistry.instance().add(this);
067                                executorService.submit(this);
068                        }
069                }
070                return event;
071        }
072
073        void add(EventQueue source) {
074                synchronized (queue) {
075                        boolean wasEmpty = queue.isEmpty();
076                        queue.addAll(source);
077                        source.clear();
078                        if (wasEmpty) {
079                                GeneratorRegistry.instance().add(this);
080                                executorService.submit(this);
081                        }
082                }
083        }
084        
085        @Override
086        public void merge(InternalEventPipeline other) {
087                if (!(other instanceof EventBuffer)) {
088                        throw new IllegalArgumentException(
089                                        "Can only merge events from an EventBuffer.");
090                }
091                add(((EventBuffer) other).retrieveEvents());
092        }
093
094        @Override
095        public synchronized void run() {
096                try {
097                        if (queue.isEmpty()) {
098                                return;
099                        }
100                        FeedBackPipelineFilter.setAssociatedPipeline(this);
101                        while (true) {
102                                EventChannelsTuple next = queue.peek();
103                                currentlyHandling.set(next.event);
104                                componentTree.dispatch(
105                                                asEventPipeline, next.event, next.channels);
106                                currentlyHandling.get().decrementOpen(this);
107                                synchronized (queue) {
108                                        queue.remove();
109                                        if (queue.isEmpty()) {
110                                                break;
111                                        }
112                                }
113                        }
114                } finally {
115                        GeneratorRegistry.instance().remove(this);
116                        currentlyHandling.set(null);;
117                        FeedBackPipelineFilter.setAssociatedPipeline(null);
118                }
119        }
120
121        /* (non-Javadoc)
122         * @see java.lang.Object#toString()
123         */
124        @Override
125        public String toString() {
126                StringBuilder builder = new StringBuilder();
127                builder.append(Components.objectName(this));
128                builder.append(" [");
129                if (queue != null) {
130                        builder.append("queue=");
131                        builder.append(queue);
132                }
133                builder.append("]");
134                return builder.toString();
135        }
136
137        /* (non-Javadoc)
138         * @see org.jgrapes.core.EventPipeline#setContext(org.jgrapes.core.Component, java.lang.Object)
139         */
140        @Override
141        public void setComponentContext(ComponentType component, Object data) {
142                componentContext.put(component, data);
143        }
144
145        /* (non-Javadoc)
146         * @see org.jgrapes.core.EventPipeline#getContext(org.jgrapes.core.Component)
147         */
148        @Override
149        public Object componentContext(ComponentType component) {
150                return componentContext.get(component);
151        }
152}
