019package org.jgrapes.core.internal;
020
021import java.util.concurrent.ExecutorService;
022import org.jgrapes.core.Channel;
023import org.jgrapes.core.Components;
024import org.jgrapes.core.Event;
025import org.jgrapes.core.EventPipeline;
026
027/**
028 * This class provides the default implementation of an {@link EventPipeline}.
029 */
030public class EventProcessor implements InternalEventPipeline, Runnable {
031
032    @SuppressWarnings("PMD.FieldNamingConventions")
033    protected static final ThreadLocal<EventBase<?>> newEventsParent
034        = new ThreadLocal<>();
035
036    private final ExecutorService executorService;
037    private final ComponentTree componentTree;
038    private final EventPipeline asEventPipeline;
039    protected final EventQueue queue = new EventQueue();
040    private boolean isExecuting;
041
042    /**
043     * Instantiates a new event processor.
044     *
045     * @param tree the tree
046     */
047    /* default */ EventProcessor(ComponentTree tree) {
048        this(tree, Components.defaultExecutorService());
049    }
050
051    /* default */ EventProcessor(ComponentTree tree,
052            ExecutorService executorService) {
053        this.componentTree = tree;
054        this.executorService = executorService;
055        asEventPipeline = new CheckingPipelineFilter(tree, this);
056    }
057
058    /**
059     * Gets the component tree.
060     *
061     * @return the component tree
062     */
063    protected ComponentTree tree() {
064        return componentTree;
065    }
066
067    /* default */ EventPipeline asEventPipeline() {
068        return asEventPipeline;
069    }
070
071    /**
072     * Called before adding completion events. The parent of
073     * a completion event is not the event that has completed but
074     * the event that generated the original event.
075     *
076     * @param parent the new parent
077     */
078    /* default */ void updateNewEventsParent(EventBase<?> parent) {
079        newEventsParent.set(parent);
080    }
081
082    @Override
083    public <T extends Event<?>> T add(T event, Channel... channels) {
084        ((EventBase<?>) event).generatedBy(newEventsParent.get());
085        ((EventBase<?>) event).processedBy(this);
086        queue.add(event, channels);
087        synchronized (this) {
088            if (!isExecuting) {
089                GeneratorRegistry.instance().add(this);
090                isExecuting = true;
091                executorService.execute(this);
092            }
093        }
094        return event;
095    }
096
097    /* default */ void add(EventQueue source) {
098        while (true) {
099            EventChannelsTuple entry = source.poll();
100            if (entry == null) {
101                break;
102            }
103            entry.event.processedBy(this);
104            queue.add(entry);
105        }
106        synchronized (this) {
107            if (!isExecuting) {
108                GeneratorRegistry.instance().add(this);
109                isExecuting = true;
110                executorService.execute(this);
111            }
112        }
113    }
114
115    @Override
116    public void merge(InternalEventPipeline other) {
117        if (!(other instanceof BufferingEventPipeline)) {
118            throw new IllegalArgumentException(
119                "Can only merge events from an BufferingEventPipeline.");
120        }
121        add(((BufferingEventPipeline) other).retrieveEvents());
122    }
123
124    @Override
125    public void run() {
126        String origName = Thread.currentThread().getName();
127        try {
128            Thread.currentThread().setName(
129                origName + " (P" + Components.objectId(this) + ")");
130            componentTree.setDispatchingPipeline(this);
131            while (true) {
132                // No lock needed if queue is filled
133                EventChannelsTuple next = queue.peek();
134                if (next == null) {
135                    synchronized (this) {
136                        // Retry with lock for proper synchronization
137                        next = queue.peek();
138                        if (next == null) {
139                            GeneratorRegistry.instance().remove(this);
140                            isExecuting = false;
141                            break;
142                        }
143                    }
144                }
145                newEventsParent.set(next.event);
146                componentTree.dispatch(
147                    asEventPipeline, next.event, next.channels);
148                newEventsParent.get().decrementOpen();
149                queue.remove();
150            }
151        } finally {
152            newEventsParent.set(null);
153            componentTree.setDispatchingPipeline(null);
154            Thread.currentThread().setName(origName);
155        }
156    }
157
158    /*
159     * (non-Javadoc)
160     * 
161     * @see org.jgrapes.core.internal.InternalEventPipeline#executorService()
162     */
163    @Override
164    public ExecutorService executorService() {
165        return executorService;
166    }
167
168    /*
169     * (non-Javadoc)
170     * 
171     * @see java.lang.Object#toString()
172     */
173    @Override
174    public String toString() {
175        StringBuilder builder = new StringBuilder();
176        builder.append(Components.objectName(this))
177            .append(" [");
178        if (queue != null) {
179            builder.append("queue=")
180                .append(queue);
181        }
182        builder.append(']');
183        return builder.toString();
184    }
185
186}
