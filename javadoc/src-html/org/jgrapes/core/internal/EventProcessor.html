019package org.jgrapes.core.internal;
020
021import java.util.concurrent.ExecutorService;
022
023import org.jgrapes.core.Channel;
024import org.jgrapes.core.Components;
025import org.jgrapes.core.Event;
026import org.jgrapes.core.EventPipeline;
027
028/**
029 * This class provides the default implementation of an {@link EventPipeline}.
030 */
031public class EventProcessor implements InternalEventPipeline, Runnable {
032
033        protected static final ThreadLocal<EventBase<?>> 
034                newEventsParent = new ThreadLocal<>();
035        
036        private final ExecutorService executorService;
037        private final ComponentTree componentTree;
038        private final EventPipeline asEventPipeline;
039        protected final EventQueue queue = new EventQueue();
040        private boolean isExecuting = false;
041        
042        EventProcessor(ComponentTree tree) {
043                this(tree, Components.defaultExecutorService());
044        }
045
046        EventProcessor(ComponentTree tree, ExecutorService executorService) {
047                this.componentTree = tree;
048                this.executorService = executorService;
049                asEventPipeline = new CheckingPipelineFilter(this);
050        }
051
052        EventPipeline asEventPipeline() {
053                return asEventPipeline;
054        }
055        
056        /**
057         * Called before adding completion events. The parent of
058         * a completion event is not the event that has completed but
059         * the event that generated the original event.
060         *
061         * @param parent the new parent
062         */
063        void updateNewEventsParent(EventBase<?> parent) {
064                newEventsParent.set(parent);
065        }
066        
067        @Override
068        public <T extends Event<?>> T add(T event, Channel... channels) {
069                ((EventBase<?>)event).generatedBy(newEventsParent.get());
070                ((EventBase<?>)event).processedBy(this);
071                queue.add(event, channels);
072                synchronized (this) {
073                        if (!isExecuting) {
074                                GeneratorRegistry.instance().add(this);
075                                isExecuting = true;
076                                executorService.execute(this);
077                        }
078                }
079                return event;
080        }
081
082        void add(EventQueue source) {
083                while (true) {
084                        EventChannelsTuple entry = source.poll();
085                        if (entry == null) {
086                                break;
087                        }
088                        entry.event.processedBy(this);
089                        queue.add(entry);
090                }
091                synchronized (this) {
092                        if (!isExecuting) {
093                                GeneratorRegistry.instance().add(this);
094                                isExecuting = true;
095                                executorService.execute(this);
096                        }
097                }
098        }
099        
100        @Override
101        public void merge(InternalEventPipeline other) {
102                if (!(other instanceof BufferingEventPipeline)) {
103                        throw new IllegalArgumentException(
104                                        "Can only merge events from an BufferingEventPipeline.");
105                }
106                add(((BufferingEventPipeline) other).retrieveEvents());
107        }
108
109        @Override
110        public void run() {
111                String origName = Thread.currentThread().getName();
112                try {
113                        Thread.currentThread().setName(
114                                        origName + " (P" + Components.objectId(this) + ")");
115                        FeedBackPipelineFilter.setAssociatedPipeline(this);
116                        while (true) {
117                                // No lock needed if queue is filled
118                                EventChannelsTuple next = queue.peek();
119                                if (next == null) {
120                                        synchronized (this) {
121                                                // Retry with lock for proper synchronization
122                                                next = queue.peek();
123                                                if (next == null) {
124                                                        GeneratorRegistry.instance().remove(this);
125                                                        isExecuting = false;
126                                                        break;
127                                                }
128                                        }
129                                }
130                                newEventsParent.set(next.event);
131                                componentTree.dispatch(
132                                                asEventPipeline, next.event, next.channels);
133                                newEventsParent.get().decrementOpen();
134                                queue.remove();
135                        }
136                } finally {
137                        Thread.currentThread().setName(origName);
138                        newEventsParent.set(null);;
139                        FeedBackPipelineFilter.setAssociatedPipeline(null);
140                }
141        }
142
143        /* (non-Javadoc)
144         * @see org.jgrapes.core.internal.InternalEventPipeline#executorService()
145         */
146        @Override
147        public ExecutorService executorService() {
148                return executorService;
149        }
150
151        /* (non-Javadoc)
152         * @see java.lang.Object#toString()
153         */
154        @Override
155        public String toString() {
156                StringBuilder builder = new StringBuilder();
157                builder.append(Components.objectName(this));
158                builder.append(" [");
159                if (queue != null) {
160                        builder.append("queue=");
161                        builder.append(queue);
162                }
163                builder.append("]");
164                return builder.toString();
165        }
166
167}
