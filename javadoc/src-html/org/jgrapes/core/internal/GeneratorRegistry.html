019package org.jgrapes.core.internal;
020
021import java.util.IdentityHashMap;
022import java.util.Map;
023import java.util.logging.Level;
024import java.util.logging.Logger;
025import org.jgrapes.core.ComponentType;
026
027/**
028 * A registry for generators. Used to track generators and determine
029 * whether the application has stopped.
030 */
031public class GeneratorRegistry {
032
033    @SuppressWarnings("PMD.FieldNamingConventions")
034    private static final Logger generatorTracking
035        = Logger.getLogger(ComponentType.class.getPackage().getName()
036            + ".generatorTracking");
037
038    private long running;
039    private Thread keepAlive;
040    private Map<Object, Object> generators;
041
042    /**
043     * Holds a generator instance.
044     */
045    private static final class InstanceHolder {
046        @SuppressWarnings("PMD.AccessorClassGeneration")
047        private static final GeneratorRegistry INSTANCE
048            = new GeneratorRegistry();
049    }
050
051    private GeneratorRegistry() {
052        if (generatorTracking.isLoggable(Level.FINE)) {
053            generators = new IdentityHashMap<>();
054        }
055    }
056
057    /**
058     * Returns the singleton instance of the registry.
059     *
060     * @return the generator registry
061     */
062    public static GeneratorRegistry instance() {
063        return InstanceHolder.INSTANCE;
064    }
065
066    /**
067     * Adds a generator.
068     *
069     * @param obj the obj
070     */
071    @SuppressWarnings({ "PMD.GuardLogStatement", "PMD.AvoidDuplicateLiterals" })
072    public void add(Object obj) {
073        synchronized (this) {
074            running += 1;
075            if (generators != null) {
076                generators.put(obj, null);
077                generatorTracking.finest(() -> "Added generator " + obj
078                    + ", " + generators.size() + " generators registered: "
079                    + generators.keySet());
080            }
081            if (running == 1) { // NOPMD, no, not using a constant for this.
082                keepAlive = new Thread("GeneratorRegistry") {
083                    @Override
084                    public void run() {
085                        try {
086                            while (true) {
087                                Thread.sleep(Long.MAX_VALUE);
088                            }
089                        } catch (InterruptedException e) {
090                            // Okay, then stop
091                        }
092                    }
093                };
094                keepAlive.start();
095            }
096        }
097    }
098
099    /**
100     * Removes the generator.
101     *
102     * @param obj the generator
103     */
104    @SuppressWarnings("PMD.GuardLogStatement")
105    public void remove(Object obj) {
106        synchronized (this) {
107            running -= 1;
108            if (generators != null) {
109                generators.remove(obj);
110                generatorTracking.finest(() -> "Removed generator " + obj
111                    + ", " + generators.size() + " generators registered: "
112                    + generators.keySet());
113            }
114            if (running == 0) {
115                generatorTracking
116                    .finest(() -> "Zero generators, notifying all.");
117                keepAlive.interrupt();
118                notifyAll();
119            }
120        }
121    }
122
123    /**
124     * Checks if is exhausted (no generators left)
125     *
126     * @return true, if is exhausted
127     */
128    public boolean isExhausted() {
129        return running == 0;
130    }
131
132    /**
133     * Await exhaustion.
134     *
135     * @throws InterruptedException the interrupted exception
136     */
137    @SuppressWarnings({ "PMD.CollapsibleIfStatements",
138        "PMD.GuardLogStatement" })
139    public void awaitExhaustion() throws InterruptedException {
140        synchronized (this) {
141            if (generators != null) {
142                if (running != generators.size()) {
143                    generatorTracking
144                        .severe(() -> "Generator count doesn't match tracked.");
145                }
146            }
147            while (running > 0) {
148                if (generators != null) {
149                    generatorTracking
150                        .fine(() -> "Thread " + Thread.currentThread().getName()
151                            + " is waiting, " + generators.size()
152                            + " generators registered: "
153                            + generators.keySet());
154                }
155                wait();
156            }
157            generatorTracking
158                .finest("Thread " + Thread.currentThread().getName()
159                    + " continues.");
160        }
161    }
162
163    /**
164     * Await exhaustion with a timeout.
165     *
166     * @param timeout the timeout
167     * @return true, if successful
168     * @throws InterruptedException the interrupted exception
169     */
170    @SuppressWarnings({ "PMD.CollapsibleIfStatements",
171        "PMD.GuardLogStatement" })
172    public boolean awaitExhaustion(long timeout)
173            throws InterruptedException {
174        synchronized (this) {
175            if (generators != null) {
176                if (running != generators.size()) {
177                    generatorTracking.severe(
178                        "Generator count doesn't match tracked.");
179                }
180            }
181            if (isExhausted()) {
182                return true;
183            }
184            if (generators != null) {
185                generatorTracking
186                    .fine(() -> "Waiting, generators: " + generators.keySet());
187            }
188            wait(timeout);
189            if (generators != null) {
190                generatorTracking
191                    .fine(() -> "Waited, generators: " + generators.keySet());
192            }
193            return isExhausted();
194        }
195    }
196}
