019package org.jgrapes.core.internal;
020
021import java.util.IdentityHashMap;
022import java.util.Map;
023import java.util.logging.Level;
024import java.util.logging.Logger;
025
026import org.jgrapes.core.ComponentType;
027
028/**
029 * A registry for generators. Used to track generators and determine
030 * whether the application has stopped.
031 */
032public class GeneratorRegistry {
033
034    @SuppressWarnings("PMD.VariableNamingConventions")
035    private static final Logger generatorTracking
036        = Logger.getLogger(ComponentType.class.getPackage().getName()
037            + ".generatorTracking");
038
039    private long running;
040    private Thread keepAlive;
041    private Map<Object, Object> generators;
042
043    /**
044     * Holds a generator instance.
045     */
046    private static final class InstanceHolder {
047        @SuppressWarnings("PMD.AccessorClassGeneration")
048        private static final GeneratorRegistry INSTANCE
049            = new GeneratorRegistry();
050    }
051
052    private GeneratorRegistry() {
053        if (generatorTracking.isLoggable(Level.FINE)) {
054            generators = new IdentityHashMap<>();
055        }
056    }
057
058    /**
059     * Returns the singleton instance of the registry.
060     *
061     * @return the generator registry
062     */
063    public static GeneratorRegistry instance() {
064        return InstanceHolder.INSTANCE;
065    }
066
067    /**
068     * Adds a generator.
069     *
070     * @param obj the obj
071     */
072    @SuppressWarnings({ "PMD.GuardLogStatement", "PMD.AvoidDuplicateLiterals" })
073    public void add(Object obj) {
074        synchronized (this) {
075            running += 1;
076            if (generators != null) {
077                generators.put(obj, null);
078                generatorTracking.finest(() -> "Added generator " + obj
079                    + ", " + generators.size() + " generators registered: "
080                    + generators.keySet());
081            }
082            if (running == 1) { // NOPMD, no, not using a constant for this.
083                keepAlive = new Thread("GeneratorRegistry") {
084                    @Override
085                    public void run() {
086                        try {
087                            while (true) {
088                                Thread.sleep(Long.MAX_VALUE);
089                            }
090                        } catch (InterruptedException e) {
091                            // Okay, then stop
092                        }
093                    }
094                };
095                keepAlive.start();
096            }
097        }
098    }
099
100    /**
101     * Removes the generator.
102     *
103     * @param obj the generator
104     */
105    @SuppressWarnings("PMD.GuardLogStatement")
106    public void remove(Object obj) {
107        synchronized (this) {
108            running -= 1;
109            if (generators != null) {
110                generators.remove(obj);
111                generatorTracking.finest(() -> "Removed generator " + obj
112                    + ", " + generators.size() + " generators registered: "
113                    + generators.keySet());
114            }
115            if (running == 0) {
116                keepAlive.interrupt();
117                notifyAll();
118            }
119        }
120    }
121
122    /**
123     * Checks if is exhausted (no generators left)
124     *
125     * @return true, if is exhausted
126     */
127    public boolean isExhausted() {
128        return running == 0;
129    }
130
131    /**
132     * Await exhaustion.
133     *
134     * @throws InterruptedException the interrupted exception
135     */
136    @SuppressWarnings({ "PMD.CollapsibleIfStatements",
137        "PMD.GuardLogStatement" })
138    public void awaitExhaustion() throws InterruptedException {
139        synchronized (this) {
140            if (generators != null) {
141                if (running != generators.size()) {
142                    generatorTracking
143                        .severe(() -> "Generator count doesn't match tracked.");
144                }
145            }
146            while (running > 0) {
147                if (generators != null) {
148                    generatorTracking
149                        .fine(() -> "Thread " + Thread.currentThread().getName()
150                            + " is waiting, " + generators.size()
151                            + " generators registered: "
152                            + generators.keySet());
153                }
154                wait();
155            }
156            generatorTracking
157                .finest("Thread " + Thread.currentThread().getName()
158                    + " continues.");
159        }
160    }
161
162    /**
163     * Await exhaustion with a timeout.
164     *
165     * @param timeout the timeout
166     * @return true, if successful
167     * @throws InterruptedException the interrupted exception
168     */
169    @SuppressWarnings({ "PMD.CollapsibleIfStatements",
170        "PMD.GuardLogStatement" })
171    public boolean awaitExhaustion(long timeout)
172            throws InterruptedException {
173        synchronized (this) {
174            if (generators != null) {
175                if (running != generators.size()) {
176                    generatorTracking.severe(
177                        "Generator count doesn't match tracked.");
178                }
179            }
180            if (isExhausted()) {
181                return true;
182            }
183            if (generators != null) {
184                generatorTracking
185                    .fine(() -> "Waiting, generators: " + generators.keySet());
186            }
187            wait(timeout);
188            if (generators != null) {
189                generatorTracking
190                    .fine(() -> "Waited, generators: " + generators.keySet());
191            }
192            return isExhausted();
193        }
194    }
195}
