019package org.jgrapes.core.internal;
020
021import org.jgrapes.core.ComponentType;
022import org.jgrapes.core.events.Error;
023
024/**
025 * A fallback error printer.
026 */
027public class ErrorPrinter implements ComponentType {
028
029    /**
030     * Prints the error.
031     *
032     * @param event the event
033     */
034    @SuppressWarnings({ "PMD.DataflowAnomalyAnalysis",
035        "PMD.AvoidCatchingGenericException",
036        "PMD.UseStringBufferForStringAppends", "PMD.SystemPrintln" })
037    public void printError(Error event) {
038        String msg = "Unhandled " + event;
039        System.err.println(msg);
040        if (event.throwable() == null) {
041            System.err.println("No stack trace available.");
042        } else {
043            event.throwable().printStackTrace();
044        }
045    }
046
047}
