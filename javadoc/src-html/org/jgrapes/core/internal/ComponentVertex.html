047/**
048 * ComponentVertex is the base class for all nodes in the component tree.
049 * ComponentVertex is extended by {@link org.jgrapes.core.Component}
050 * for the use as base class for component implementations. As an 
051 * alternative for implementing components with an independent base class,
052 * the derived class {@link org.jgrapes.core.internal.ComponentProxy} can be
053 * used. 
054 */
055public abstract class ComponentVertex implements Manager, Channel {
056
057        /** Handler factory cache. */
058        private static Map<Class<? extends HandlerDefinition.Evaluator>,
059                        HandlerDefinition.Evaluator> definitionEvaluators 
060                        = Collections.synchronizedMap(new HashMap<>());
061        /** The component's (optional) name. */
062        private String name = null;
063        /** Reference to the common properties of the tree nodes. */
064        private ComponentTree tree = null;
065        /** Reference to the parent node. */
066        private ComponentVertex parent = null;
067        /** All the node's children */
068        private List<ComponentVertex> children = new ArrayList<>();
069        /** The handlers provided by this component. */
070        private List<HandlerReference> handlers;
071        
072        /** 
073         * Initialize the ComponentVertex. By default it forms a stand-alone
074         * tree, i.e. the root is set to the component itself.
075         */
076        protected ComponentVertex() {
077        }
078
079        /**
080         * Initialize the handler list of this component. May only be called
081         * when {@link #component()} can be relied on to return the
082         * correct value.
083         */
084        protected void initComponentsHandlers() {
085                handlers = new ArrayList<HandlerReference>();
086                // Have a look at all methods.
087                for (Method m : component().getClass().getMethods()) {
088                        maybeAddHandler(m);
089                }
090                handlers = Collections.synchronizedList(handlers);
091        }
092
093        private void maybeAddHandler(Method method) {
094                for (Annotation annotation: method.getDeclaredAnnotations()) {
095                        Class<?> annoType = annotation.annotationType();
096                        HandlerDefinition hda 
097                                = annoType.getAnnotation(HandlerDefinition.class);
098                        if (hda == null) {
099                                continue;
100                        }
101                        HandlerDefinition.Evaluator evaluator = defintionEvaluator(hda);
102                        HandlerScope scope = evaluator.scope(component(), method, null,
103                                null);
104                        if (scope == null) {
105                                continue;
106                        }
107                        handlers.add(HandlerReference.newRef(
108                                        component(), method, evaluator.priority(annotation),
109                                                        scope));
110                }
111        }
112
113        private HandlerDefinition.Evaluator defintionEvaluator(
114                HandlerDefinition hda) {
115                HandlerDefinition.Evaluator evaluator = definitionEvaluators
116                                .computeIfAbsent(hda.evaluator(), key -> {
117                        try {
118                                HandlerDefinition.Evaluator hde 
119                                        = hda.evaluator().newInstance();
120                                definitionEvaluators.put(key, hde);
121                                return hde;
122                        } catch (InstantiationException
123                                | IllegalAccessException e) {
124                                throw new RuntimeException(e);
125                        }
126                });
127                return evaluator;
128        }

157        private void buildPath(StringBuilder sb) {

218        /**
219         * Return the tree that this node belongs to. If the node does not
220         * belong to a tree yet, a tree is automatically created.
221         * 
222         * @return the tree
223         */
224        ComponentTree tree() {
225                if (tree != null) {
226                        return tree;
227                }
228                // Build complete tree before assigning it.
229                ComponentTree newTree = new ComponentTree(this);
230                newTree.setEventPipeline(new BufferingEventPipeline(newTree));
231                tree = newTree;
232                fire(new Attached(component(), null), channel());
233                return tree;
234        }

236        /**
237         * Set the reference to the common properties of this component 
238         * and all its children to the given value.
239         * 
240         * @param comp the new root
241         */
242        private synchronized void setTree(ComponentTree tree) {
243                this.tree = tree;
244                for (ComponentVertex child: children) {
245                        child.setTree(tree);
246                }
247        }

249        /* (non-Javadoc)
250         * @see org.jgrapes.core.Manager#attach(Component)
251         */
252        @Override
253        public synchronized <T extends ComponentType> T attach(T child) {
254                ComponentVertex childNode = componentVertex(child, null);
255                List<Channel> attachedAsChannels = new ArrayList<>();
256                synchronized (childNode) {
257                        if (tree != null && tree.isStarted()) {
258                                for (TreeIterator itr = new TreeIterator(childNode);
259                                                itr.hasNext();) {
260                                        attachedAsChannels.add(itr.next());
261                                }
262                        }
263                        synchronized (tree()) {
264                                if (childNode.tree == null) { 
265                                        // Newly created, stand-alone child node
266                                        childNode.parent = ComponentVertex.this;
267                                        childNode.setTree(tree);
268                                        children.add(childNode);
269                                } else {
270                                        // Attaching a tree...
271                                        synchronized (childNode.tree) {
272                                                if (childNode.parent != null) {
273                                                        throw new IllegalStateException(
274                                                                        "Cannot attach a node with a parent.");
275                                                }
276                                                if (childNode.tree.isStarted()) {
277                                                        throw new IllegalStateException(
278                                                                        "Cannot attach a started subtree.");
279                                                }
280                                                childNode.parent = ComponentVertex.this;
281                                                ComponentTree childTree = childNode.tree;
282                                                childNode.setTree(tree);
283                                                children.add(childNode);
284                                                tree.mergeEvents(childTree);
285                                        }
286                                }
287                                tree.clearHandlerCache();
288                        }
289                }
290                Channel parentChan = channel();
291                if (parentChan == null) {
292                        parentChan = Channel.BROADCAST;
293                }
294                Channel childChan = childNode.channel();
295                if (childChan == null) {
296                        parentChan = Channel.BROADCAST;
297                }
298                Attached evt = new Attached(childNode.component(), component());
299                if (parentChan.equals(Channel.BROADCAST) 
300                        || childChan.equals(Channel.BROADCAST)) {
301                        fire(evt, Channel.BROADCAST);
302                } else if (parentChan.equals(childChan)) {
303                        fire(evt, parentChan);
304                } else {
305                        fire(evt, parentChan, childChan);
306                }
307                if (!attachedAsChannels.isEmpty()) {
308                        fire(new Start(), attachedAsChannels.toArray(new Channel[0]));
309                }
310                return child;
311        }

313        /**
314         * Remove the component from the tree, making it a stand-alone tree.
315         */
316        public synchronized ComponentType detach() {
317                if (parent != null) {
318                        ComponentVertex oldParent = parent;
319                        synchronized (tree) {
320                                if (!tree.isStarted()) {
321                                        throw new IllegalStateException(
322                                                        "Components may not be detached from a tree before"
323                                                        + " a Start event has been fired on it.");
324                                }
325                                synchronized (oldParent) {
326                                        parent.children.remove(ComponentVertex.this);
327                                        parent.tree.clearHandlerCache();
328                                        parent = null;
329                                }
330                                ComponentTree newTree 
331                                        = new ComponentTree(ComponentVertex.this);
332                                newTree.setEventPipeline(new FeedBackPipelineFilter(
333                                                new EventProcessor(newTree)));
334                                setTree(newTree);
335                        }
336                        Detached evt = new Detached(component(), oldParent.component());
337                        oldParent.fire(evt);
338                        evt = new Detached(component(), oldParent.component());
339                        fire(evt);
340                }
341                return component();
342        }

353         * A simple wrapper that converts a component vertex iterator
354         * to a component (type) iterator.
355         */
356        private static class ComponentIterator implements Iterator<ComponentType> {
357
358                private TreeIterator baseIterator;
359                
360                /**
361                 * @param baseIterator
362                 */
363                public ComponentIterator(TreeIterator baseIterator) {
364                        this.baseIterator = baseIterator;
365                }
366
367                @Override
368                public boolean hasNext() {
369                        return baseIterator.hasNext();
370                }
371
372                @Override
373                public ComponentType next() {
374                        return baseIterator.next().component();
375                }
376                
377        }

379        /**
380         * An iterator for getting all nodes of the tree.
381         */
382        private static class TreeIterator implements Iterator<ComponentVertex> {
383
384                private class Pos {
385                        public ComponentVertex current;
386                        public Iterator<ComponentVertex> childIter;
387                        
388                        public Pos(ComponentVertex cm) {
389                                current = cm;
390                                childIter = current.children.iterator();
391                        }
392                }
393                
394                private Stack<Pos> stack = new Stack<Pos>();
395                private ComponentTree tree;
396                
397                public TreeIterator(ComponentVertex root) {
398                        tree = root.tree();
399                        stack.push(new Pos(root));
400                }
401
402                /* (non-Javadoc)
403                 * @see java.util.Iterator#hasNext()
404                 */
405                @Override
406                public boolean hasNext() {
407                        return !stack.empty();
408                }
409
410                /* (non-Javadoc)
411                 * @see java.util.Iterator#next()
412                 */
413                @Override
414                public ComponentVertex next() {
415                        if (stack.empty()) {
416                                throw new NoSuchElementException();
417                        }
418                        Pos pos = stack.peek();
419                        ComponentVertex res = pos.current;
420                        while (true) {
421                                synchronized (pos.current) {
422                                        if (pos.current.tree != tree) {
423                                                throw new ConcurrentModificationException();
424                                        }
425                                        if (pos.childIter.hasNext()) {
426                                                stack.push(new Pos(pos.childIter.next()));
427                                                break;
428                                        }
429                                }
430                                stack.pop();
431                                if (stack.empty()) {
432                                        break;
433                                }
434                                pos = stack.peek();
435                        }
436                        return res;
437                }
438
439                /* (non-Javadoc)
440                 * @see java.util.Iterator#remove()
441                 */
442                @Override
443                public void remove() {
444                        throw new UnsupportedOperationException();
445                }
446        }
447
448        @Override
449        public void addHandler(Method method, HandlerScope scope, int priority) {
450                handlers.add(HandlerReference.newRef(component(),
451                        method, priority, scope));
452        }

454        /* (non-Javadoc)
455         * @see org.jgrapes.core.core.Manager#fire
456         * (org.jgrapes.core.Event, org.jgrapes.core.Channel)
457         */
458        @Override
459        public <T> Event<T> fire(Event<T> event, Channel... channels) {
460                if (channels.length == 0) {
461                        channels = event.channels();
462                        if (channels == null || channels.length == 0) {
463                                channels = new Channel[] { channel() };
464                        }
465                }
466                event.setChannels(channels);
467                tree().fire(event, channels);
468                return event;
469        }

471        /**
472         * Collects all handlers. Iterates over the tree with this object
473         * as root and for all child components adds the matching handlers to
474         * the result set recursively.
475         * 
476         * @param hdlrs the result set
477         * @param event the event to match
478         * @param channels the channels to match
479         */
480        void collectHandlers(Collection<HandlerReference> hdlrs, 
481                        EventBase<?> event, Channel[] channels) {
482                for (HandlerReference hdlr: handlers) {
483                        if (hdlr.handles(event, channels)) {
484                                hdlrs.add(hdlr);
485                        }
486                }
487                for (ComponentVertex child: children) {
488                        child.collectHandlers(hdlrs, event, channels);
489                }
490        }
