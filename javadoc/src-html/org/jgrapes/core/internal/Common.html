019package org.jgrapes.core.internal;
020
021import java.lang.reflect.InvocationTargetException;
022import java.util.Collections;
023import java.util.HashMap;
024import java.util.Map;
025import java.util.concurrent.atomic.AtomicReference;
026import java.util.logging.Logger;
027
028import org.jgrapes.core.ComponentType;
029import org.jgrapes.core.annotation.HandlerDefinition;
030import org.jgrapes.core.annotation.HandlerDefinition.Evaluator;
031
032/**
033 * Common utility methods.
034 */
035@SuppressWarnings("PMD.MoreThanOneLogger")
036public final class Common {
037
038    @SuppressWarnings("PMD.VariableNamingConventions")
039    public static final Logger classNames
040        = Logger.getLogger(ComponentType.class.getPackage().getName()
041            + ".classNames");
042
043    @SuppressWarnings("PMD.VariableNamingConventions")
044    /* default */ static final Logger fireRestrictionLogger
045        = Logger.getLogger(Common.class.getPackage().getName()
046            + ".fireRestriction");
047
048    /** Handler factory cache. */
049    @SuppressWarnings("PMD.UseConcurrentHashMap")
050    private static Map<Class<? extends HandlerDefinition.Evaluator>,
051            HandlerDefinition.Evaluator> definitionEvaluators
052                = Collections.synchronizedMap(new HashMap<>());
053    private static AtomicReference<AssertionError> assertionError
054        = new AtomicReference<>();
055
056    private Common() {
057    }
058
059    /**
060     * Create a new definition evaluator.
061     *
062     * @param hda the handler definition annotation
063     * @return the evaluator
064     */
065    public static Evaluator definitionEvaluator(
066            HandlerDefinition hda) {
067        return definitionEvaluators.computeIfAbsent(hda.evaluator(), key -> {
068            try {
069                return hda.evaluator().getConstructor().newInstance();
070            } catch (InstantiationException | IllegalAccessException
071                    | IllegalArgumentException | InvocationTargetException
072                    | NoSuchMethodException | SecurityException e) {
073                throw new IllegalStateException(e);
074            }
075        });
076    }
077
078    /* default */ static void setAssertionError(AssertionError error) {
079        assertionError.compareAndSet(null, error);
080    }
081
082    /**
083     * Check if an assertion has been set and if, throw it.
084     */
085    public static void checkAssertions() {
086        AssertionError error = assertionError.getAndSet(null);
087        if (error != null) {
088            throw error;
089        }
090    }
091
092}
