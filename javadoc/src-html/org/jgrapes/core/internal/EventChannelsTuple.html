019package org.jgrapes.core.internal;
020
021import java.util.Arrays;
022import org.jgrapes.core.Channel;
023
024/**
025 * This class provides a container for an event and an arbitrary 
026 * number of channels. Instances represent a particular event being fired
027 * on several channels. They are used e.g. to queue the information
028 * about an event being fired on some channels.
029 */
030public class EventChannelsTuple {
031    public EventBase<?> event;
032    public Channel[] channels;
033
034    /**
035     * Create a new instance.
036     * 
037     * @param event the event
038     * @param channels the channels
039     */
040    @SuppressWarnings("PMD.UseVarargs")
041    public EventChannelsTuple(EventBase<?> event, Channel[] channels) {
042        super();
043        this.event = event;
044        this.channels = Arrays.copyOf(channels, channels.length);
045    }
046
047    /*
048     * (non-Javadoc)
049     * 
050     * @see java.lang.Object#hashCode()
051     */
052    @Override
053    @SuppressWarnings("PMD.DataflowAnomalyAnalysis")
054    public int hashCode() {
055        @SuppressWarnings("PMD.AvoidFinalLocalVariable")
056        final int prime = 31;
057        int result = 1;
058        result = prime * result + Arrays.hashCode(channels);
059        result = prime * result + ((event == null) ? 0 : event.hashCode());
060        return result;
061    }
062
063    /*
064     * (non-Javadoc)
065     * 
066     * @see java.lang.Object#equals(java.lang.Object)
067     */
068    @Override
069    public boolean equals(Object obj) {
070        if (this == obj) {
071            return true;
072        }
073        if (obj == null) {
074            return false;
075        }
076        if (getClass() != obj.getClass()) {
077            return false;
078        }
079        EventChannelsTuple other = (EventChannelsTuple) obj;
080        if (!Arrays.equals(channels, other.channels)) {
081            return false;
082        }
083        if (event == null) {
084            if (other.event != null) {
085                return false;
086            }
087        } else if (!event.equals(other.event)) {
088            return false;
089        }
090        return true;
091    }
092
093    /*
094     * (non-Javadoc)
095     * 
096     * @see java.lang.Object#toString()
097     */
098    @Override
099    public String toString() {
100        StringBuilder builder = new StringBuilder(50);
101        builder.append("EventChannelsTuple [");
102        if (event != null) {
103            builder.append("event=")
104                .append(event)
105                .append(", ");
106        }
107        if (channels != null) {
108            builder.append("channels=")
109                .append(Arrays.toString(channels));
110        }
111        builder.append(']');
112        return builder.toString();
113    }
114
115}
