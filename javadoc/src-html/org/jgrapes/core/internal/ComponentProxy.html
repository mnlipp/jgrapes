019package org.jgrapes.core.internal;
020
021import java.lang.reflect.Field;
022
023import org.jgrapes.core.Channel;
024import org.jgrapes.core.ComponentType;
025import org.jgrapes.core.NamedChannel;
026import org.jgrapes.core.annotation.ComponentManager;
027import org.jgrapes.core.annotation.Handler;
028
029/**
030 * The ComponentProxy is a special ComponentVertex that references the
031 * object implementing the Component interface (instead of being
032 * its base class).
033 * 
034 * @author Michael N. Lipp
035 */
036public class ComponentProxy extends ComponentVertex {
037
038        /** The reference to the actual component. */
039        private ComponentType component = null;
040        /** The referenced component's channel. */
041        private Channel componentChannel = Channel.BROADCAST;
042        
043        private static Field getManagerField(Class<?> clazz) {
044                try {
045                        while (true) {
046                                for (Field field: clazz.getDeclaredFields()) {
047                                        if (field.getAnnotation(ComponentManager.class) != null) {
048                                                return field;
049                                        }
050                                }
051                                clazz = clazz.getSuperclass();
052                                if (clazz == null) {
053                                        throw new IllegalArgumentException(
054                                                        "Components must have a manager attribute");
055                                }
056                        }
057                } catch (SecurityException e) {
058                        throw (RuntimeException)(new IllegalArgumentException(
059                                        "Cannot access component's manager attribute")).initCause(e);
060                }
061        }
062
063        private static Channel getComponentChannel(Field field) {
064                ComponentManager cma = field.getAnnotation(ComponentManager.class);
065                if (cma.channel() != Handler.NoChannel.class) {
066                        if (cma.channel() != Channel.BROADCAST.getDefaultCriterion()) {
067                                try {
068                                        return cma.channel().newInstance();
069                                } catch (InstantiationException | IllegalAccessException e) {
070                                        // Ignored
071                                }
072                        }
073                        return Channel.BROADCAST;
074                }
075                if (!cma.namedChannel().equals("")) {
076                        return new NamedChannel(cma.namedChannel());
077                }
078                return Channel.BROADCAST;
079        }
080        
081        /**
082         * Create a new component proxy for the component and assign it to 
083         * the specified field which must be of type {@link Manager}.
084         * 
085         * @param field the field that gets the proxy assigned
086         * @param component the component
087         */
088        private ComponentProxy(Field field, ComponentType component) {
089                this.component = component;
090                try {
091                        field.set(component, this);
092                        componentChannel = getComponentChannel(field);
093                        initComponentsHandlers();
094                } catch (SecurityException | IllegalAccessException e) {
095                        throw (RuntimeException)(new IllegalArgumentException(
096                                        "Cannot access component's manager attribute")).initCause(e);
097                }
098        }
099
100        /**
101         * Return the component node for a component that is represented
102         * by a proxy in the tree.
103         * 
104         * @param component the component
105         * @return the node representing the component in the tree
106         */
107        static ComponentVertex getComponentProxy(ComponentType component) {
108                ComponentProxy componentProxy = null;
109                try {
110                        Field field = getManagerField(component.getClass());
111                        synchronized (component) {
112                                if (!field.isAccessible()) {
113                                        field.setAccessible(true);
114                                        componentProxy = (ComponentProxy)field.get(component);
115                                        if (componentProxy == null) {
116                                                componentProxy = new ComponentProxy(field, component);
117                                        }
118                                        field.setAccessible(false);
119                                } else {
120                                        componentProxy = (ComponentProxy)field.get(component);
121                                        if (componentProxy == null) {
122                                                componentProxy = new ComponentProxy(field, component);
123                                        }
124                                }
125                        }
126                } catch (SecurityException | IllegalAccessException e) {
127                        throw (RuntimeException)(new IllegalArgumentException(
128                                        "Cannot access component's manager attribute")).initCause(e);
129                }
130                return componentProxy;
131        }
132
133        public ComponentType getComponent() {
134                return component;
135        }
136
137        /* (non-Javadoc)
138         * @see org.jgrapes.core.Manager#getChannel()
139         */
140        @Override
141        public Channel getChannel() {
142                return componentChannel;
143        }
144}
