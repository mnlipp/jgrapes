019package org.jgrapes.core;
020
021import java.util.Arrays;
022
023/**
024 * A class for events identified by name. Instances of this class 
025 * represent events that use their name for matching 
026 * events with handlers.
027 * 
028 * @author Michael N. Lipp
029 */
030public final class NamedEvent<T> extends Event<T> {
031
032        private String name;
033        
034        /**
035         * Creates a new named event with the given name.
036         * 
037         * @param name the event's name
038         */
039        public NamedEvent(String name) {
040                super();
041                this.name = name;
042        }
043
044        /**
045         * Returns the name of the Event as value.
046         * 
047         * @return the name
048         * 
049         * @see org.jgrapes.core.Channel#getDefaultCriterion()
050         */
051        @Override
052        public Object getDefaultCriterion() {
053                return name;
054        }
055
056        /**
057         * Returns <code>true</code> if the <code>value</code>
058         * is the event class or is equal to the event's name.
059         * 
060         * @see org.jgrapes.core.Eligible#isEligibleFor(java.lang.Object)
061         */
062        @Override
063        public boolean isEligibleFor(Object value) {
064                return value.equals(Event.class) || value.equals(name);
065        }
066
067        /* (non-Javadoc)
068         * @see java.lang.Object#toString()
069         */
070        @Override
071        public String toString() {
072                StringBuilder result = new StringBuilder();
073                result.append("NamedEvent [name=");
074                result.append(name);
075                if (channels() != null) {
076                        result.append(", " + "channels=" + Arrays.toString(channels())); 
077                }
078                result.append("]");
079                return result.toString();
080        }
081}
