019package org.jgrapes.core;
020
021import java.io.Serializable;
022import java.util.Map;
023import java.util.Optional;
024
025import org.jgrapes.core.Associator;
026
027/**
028 * Combines a type and an id value to a key for an {@link Associator}
029 * or {@link Map}. This kind of key can be used if using an instance of 
030 * the type as key is too specific or not desirable due to garbage collection. 
031 */
032public class TypedIdKey<V> {
033
034    /**
035     * Associates the given value's type and the id with the given value
036     * using the given associator. 
037     *
038     * @param <V> the value type
039     * @param associator the associator
040     * @param id the id
041     * @param value the value
042     * @return the value for easy chaining
043     */
044    public static <V> V associate(Associator associator, Object id, V value) {
045        associator.setAssociated(new TypeIdPair<>(value.getClass(), id),
046            value);
047        return value;
048    }
049
050    /**
051     * Associates the given value's type and the id with the given value
052     * using the given map. 
053     *
054     * @param <V> the value type
055     * @param map the map
056     * @param id the id
057     * @param value the value
058     * @return the value for easy chaining
059     */
060    public static <V> V put(Map<Object, ? super V> map, Object id, V value) {
061        map.put(new TypeIdPair<>(value.getClass(), id), value);
062        return value;
063    }
064
065    /**
066     * Associates the given value's type and the id with the given value
067     * using the given map. Special version for maps that must be serializable.
068     *
069     * @param <V> the value type
070     * @param map the map
071     * @param id the id
072     * @param value the value
073     * @return the value for easy chaining
074     */
075    public static <V extends Serializable> V
076            put(Map<Serializable, ? super V> map, Serializable id, V value) {
077        map.put(new TypeIdPair<>(value.getClass(), id), value);
078        return value;
079    }
080
081    /**
082     * Retrieves a value with the given type and id from the given associator.
083     *
084     * @param <V> the value type
085     * @param associator the associator
086     * @param type the type
087     * @param id the id
088     * @return the associated value, if any
089     */
090    public static <V> Optional<V> associated(Associator associator,
091            Class<V> type, Object id) {
092        return associator.associated(new TypeIdPair<>(type, id), type);
093    }
094
095    /**
096     * Retrieves a value with the given type and id from the given map.
097     *
098     * @param <V> the value type
099     * @param map the map
100     * @param type the type
101     * @param id the id
102     * @return the associated value, if any
103     */
104    @SuppressWarnings("unchecked")
105    public static <V> Optional<V> get(Map<Object, Object> map,
106            Class<V> type, Object id) {
107        return Optional
108            .ofNullable((V) map.get(new TypeIdPair<>(type, id)));
109    }
110
111    /**
112     * Retrieves a value with the given type and id from the given map.
113     * special version for maps that must be serializable.
114     *
115     * @param <V> the value type
116     * @param map the map
117     * @param type the type
118     * @param id the id
119     * @return the associated value, if any
120     */
121    @SuppressWarnings("unchecked")
122    public static <V extends Serializable> Optional<V> get(
123            Map<Serializable, Serializable> map,
124            Class<V> type, Serializable id) {
125        return Optional
126            .ofNullable((V) map.get(new TypeIdPair<>(type, id)));
127    }
128
129    @SuppressWarnings("serial")
130    private static class TypeIdPair<V> implements Serializable {
131
132        protected Class<V> type;
133        protected Object id;
134
135        public TypeIdPair(Class<V> type, Object id) {
136            super();
137            this.type = type;
138            this.id = id;
139        }
140
141        @Override
142        public int hashCode() {
143            final int prime = 31;
144            int result = 1;
145            result = prime * result + ((id == null) ? 0 : id.hashCode());
146            result = prime * result + ((type == null) ? 0 : type.hashCode());
147            return result;
148        }
149
150        @Override
151        public boolean equals(Object obj) {
152            if (this == obj) {
153                return true;
154            }
155            if (obj == null) {
156                return false;
157            }
158            if (getClass() != obj.getClass()) {
159                return false;
160            }
161            TypeIdPair<?> other = (TypeIdPair<?>) obj;
162            if (id == null) {
163                if (other.id != null) {
164                    return false;
165                }
166            } else if (!id.equals(other.id)) {
167                return false;
168            }
169            if (type == null) {
170                if (other.type != null) {
171                    return false;
172                }
173            } else if (!type.equals(other.type)) {
174                return false;
175            }
176            return true;
177        }
178
179        @Override
180        public String toString() {
181            return "TypedIdKey [" + type + ":" + id + "]";
182        }
183
184    }
185
186}
