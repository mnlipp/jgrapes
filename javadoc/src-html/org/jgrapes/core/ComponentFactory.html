019package org.jgrapes.core;
020
021import java.util.Collections;
022import java.util.Map;
023import java.util.Optional;
024import java.util.ServiceLoader;
025
026import org.jgrapes.core.Channel;
027import org.jgrapes.core.ComponentType;
028import org.jgrapes.core.Manager;
029
030/**
031 * A general purpose factory interface for components.
032 * 
033 * In some use cases, components may not be known at compile-time, but
034 * only made available at run-time. The usual pattern for this is to lookup
035 * a factory service using the {@link ServiceLoader} and create the
036 * components using the factory (or factories) found.
037 * 
038 * Because JGrapes components interact with their environment only
039 * through events, they do not provide special APIs and it is possible
040 * to define this generic factory service interface.
041 * Of course, lookup by the {@link ServiceLoader} will usually be done
042 * using a derived interface (or base class) that represents the special
043 * kind of components required and allows an invocation
044 * of {@link ServiceLoader#load(Class)} that returns a "filtered"
045 * set of factories.
046 */
047public interface ComponentFactory {
048
049        /**
050         * Returns the type of the components created by this factory.
051         * 
052         * @return the component type
053         */
054        Class<? extends ComponentType> componentType();
055        
056        /**
057         * Creates a new component with its channel set to
058         * itself.
059         * 
060         * @return the component
061         */
062        default ComponentType create() {
063                return create(Channel.SELF);
064        }
065        
066        /**
067         * Creates a new component with its channel set to the given 
068         * channel.
069         * 
070         * @param componentChannel the channel that the component's 
071         * handlers listen on by default and that 
072         * {@link Manager#fire(Event, Channel...)} sends the event to
073         * @return the component
074         */
075        default ComponentType create(Channel componentChannel) {
076                return create(componentChannel, Collections.emptyMap()).get();
077        }
078        
079        /**
080         * Creates a new component with its channel set to the given 
081         * channel using the given additional properties. If the requested
082         * properties or combination of properties cannot be provided by
083         * the component, the factory may return an empty {@link Optional}. 
084         * 
085         * @param componentChannel the channel that the component's 
086         * handlers listen on by default and that 
087         * {@link Manager#fire(Event, Channel...)} sends the event to 
088         * @param properties additional properties for the creation of the
089         * component
090         * @return the component
091         */
092        Optional<ComponentType> create(
093                        Channel componentChannel, Map<Object, Object> properties);
094}
