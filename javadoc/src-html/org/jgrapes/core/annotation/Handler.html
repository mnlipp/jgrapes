049/**
050 * This is the basic, general purpose handler annotation provided as part of the
051 * core package.
052 * 
053 * The annotated method is invoked for events that have a type (or
054 * name) matching the given `events` (or `namedEvents`) element 
055 * of the annotation and that are fired on one of
056 * the `channels` (or `namedChannels`) specified in the annotation.
057 * 
058 * If neither event classes nor named events are specified in the 
059 * annotation, the class of the annotated method's first parameter (which 
060 * must be of type {@link Event} or a derived type) is used as (single)
061 * event class (see the examples in {@link #events()} and 
062 * {@link #namedEvents()}).
063 * 
064 * Channel matching is performed by matching the event's channels
065 * (see {@link Event#channels()}) with the channels specified in the
066 * handler. The matching algorithm invokes
067 * {@link Eligible#isEligibleFor(Object) isEligibleFor} for each of the 
068 * event's channels with the class (or name, see {@link #channels()} and 
069 * {@link Handler#namedChannels()}) of each of the channels specified
070 * in the handler.
071 * 
072 * If neither channel classes not named channels are specified in the 
073 * handler, or `{@link Default Channel.Default}.class` is specified as one 
074 * of the channel classes, the matching algorithm invokes
075 * {@link Eligible#isEligibleFor(Object) isEligibleFor} for each of
076 * the event's channels with the default criterion of the component's 
077 * channel (see {@link Manager#channel()} and 
078 * {@link Eligible#defaultCriterion()}) as argument.
079 *
080 * Finally, independent of any specified channels, the matching algorithm 
081 * invokes {@link Eligible#isEligibleFor(Object) isEligibleFor} 
082 * for each of the event's channels with the component's default criterion
083 * as argument. This results in a match if
084 * the component itself is used as one of the event's channels
085 * (see the description of {@link Eligible}).
086 * 
087 * If a match is found for a given event's properties and a handler's
088 * specified attributes, the handler method is invoked. 
089 * The method can have an additional optional parameter of type
090 * {@link Channel} (or a derived type). This parameter does not 
091 * influence the eligibility of the method regarding a given event,
092 * it determines how the method is invoked. If the method does not
093 * have a second parameter, it is invoked once if an event 
094 * matches. If the parameter exists, the method is invoked once for
095 * each of the event's channels, provided that the optional parameter's
096 * type is assignable from the event's channel.
097 * 
098 * Because annotation elements accept only literals as values, they
099 * cannot be used to register handlers with properties that are only
100 * known at runtime. It is therefore possible to specify a 
101 * {@link Handler} annotation with element `dynamic=true`. Such a 
102 * handler must be added explicitly by invoking 
103 * {@link Evaluator#add(ComponentType, String, Object)} or
104 * {@link Evaluator#add(ComponentType, String, Object, Object, int)},
105 * thus specifying some of the handler's properties dynamically (i.e.
106 * at runtime).
107 * 
108 * A special case is the usage of a channel that is only known at
109 * runtime. If there are several handlers for events on such a
110 * channel, a lot of methods will become dynamic. To avoid this,
111 * {@link Component}s support a {@link ChannelReplacements}
112 * parameter in their constructor. Using this, it is possible
113 * to specify a specially defined {@link Channel} class in the
114 * annotation that is replaced by a channel that is only known
115 * at runtime.
116 * 
117 * @see Component#channel()
118 */
119@Documented
120@Retention(RetentionPolicy.RUNTIME)
121@Target(ElementType.METHOD)
122@HandlerDefinition(evaluator = Handler.Evaluator.class)
123public @interface Handler {
124
125    /** The default value for the <code>events</code> parameter of
126     * the annotation. Indicates that the parameter is not used. */
127    final class NoEvent extends Event<Void> {
128    }
129
130    /** The default value for the <code>channels</code> parameter of
131     * the annotation. Indicates that the parameter is not used. */
132    final class NoChannel extends ClassChannel {
133    }
134
135    /**
136     * Specifies classes of events that the handler is to receive.
137     * 
138     * ```java
139     * class SampleComponent extends Component {
140     * 
141     *    {@literal @}Handler
142     *     public void onStart(Start event) {
143     *         // Invoked for Start events on the component's channel,
144     *         // event object made available
145     *     }
146     * 
147     *    {@literal @}Handler(events=Start.class)
148     *     public void onStart() {
149     *         // Invoked for Start events on the component's channel,
150     *         // not interested in the event object
151     *     }
152     * 
153     *    {@literal @}Handler(events={Start.class, Stop.class})
154     *     public void onStart(Event<?> event) {
155     *         // Invoked for Start and Stop events on the component's
156     *         // channel, event made available (may need casting to 
157     *         // access specific properties) 
158     *     }
159     * }
160     * ```
161     * 
162     * @return the event classes
163     */
164    @SuppressWarnings("rawtypes")
165    Class<? extends Event>[] events() default NoEvent.class;
166
167    /**
168     * Specifies names of {@link NamedEvent}s that the handler is to receive.
169     * 
170     * ```java
171     * class SampleComponent extends Component {
172     * 
173     *    {@literal @}Handler(namedEvents="Test")
174     *     public void onTest(Event<?> event) {
175     *         // Invoked for (named) "Test" events (new NamedEvent("Test")) 
176     *         // on the component's channel, event object made available
177     *     }
178     * }
179     * ```
180     * 
181     * @return the event names
182     */
183    String[] namedEvents() default "";
184
185    /**
186     * Specifies classes of channels that the handler listens on. If none
187     * are specified, the component's channel is used.
188     * 
189     * ```java
190     * class SampleComponent extends Component {
191     * 
192     *    {@literal @}Handler(channels=Feedback.class)
193     *     public void onStart(Start event) {
194     *         // Invoked for Start events on the "Feedback" channel
195     *         // (class Feedback implements Channel {...}),
196     *         // event object made available
197     *     }
198     * }
199     * ```
200     * 
201     * Specifying `channels=Channel.class` make the handler listen
202     * for all events, independent of the channel that they are fired on.
203     * 
204     * Specifying `channels=Self.class` make the handler listen
205     * for events that are fired on the conponent.
206     * 
207     * @return the channel classes
208     */
209    Class<? extends Channel>[] channels() default NoChannel.class;
210
211    /**
212     * Specifies names of {@link NamedChannel}s that the handler listens on.
213     * 
214     * ```java
215     * class SampleComponent extends Component {
216     * 
217     *    {@literal @}Handler(namedChannels="Feedback")
218     *     public void onStart(Start event) {
219     *         // Invoked for Start events on the (named) channel "Feedback"
220     *         // (new NamedChannel("Feedback")), event object made available
221     *     }
222     * }
223     * ```
224     * 
225     * @return the channel names
226     */
227    String[] namedChannels() default "";
228
229    /**
230     * Specifies a priority. The value is used to sort handlers.
231     * Handlers with higher priority are invoked first.
232     * 
233     * @return the priority
234     */
235    int priority() default 0;
236
237    /**
238     * Returns {@code true} if the annotated method defines a
239     * dynamic handler. A dynamic handler must be added to the set of
240     * handlers of a component explicitly at run time using
241     * {@link Evaluator#add(ComponentType, String, Object)}
242     * or {@link Evaluator#add(ComponentType, String, Object, Object, int)}.
243     * 
244     * ```java
245     * class SampleComponent extends Component {
246     * 
247     *     SampleComponent() {
248     *         Handler.Evaluator.add(this, "onStartDynamic", someChannel);
249     *     }
250     * 
251     *    {@literal @}Handler(dynamic=true)
252     *     public void onStartDynamic(Start event) {
253     *         // Only invoked if added as handler at runtime
254     *     }
255     * }
256     * ```
257     * 
258     * @return the result
259     */
260    boolean dynamic() default false;
261
262    /**
263     * This class provides the {@link Evaluator} for the 
264     * {@link Handler} annotation provided by the core package. It 
265     * implements the behavior as described for the annotation. 
266     */
267    class Evaluator implements HandlerDefinition.Evaluator {
268
269        @Override
270        public HandlerScope scope(
271                ComponentType component, Method method,
272                ChannelReplacements channelReplacements) {
273            Handler annotation = method.getAnnotation(Handler.class);
274            if (annotation == null || annotation.dynamic()) {
275                return null;
276            }
277            return new Scope(component, method, annotation,
278                channelReplacements, null, null);
279        }
280
281        /*
282         * (non-Javadoc)
283         * 
284         * @see
285         * org.jgrapes.core.annotation.HandlerDefinition.Evaluator#getPriority()
286         */
287        @Override
288        public int priority(Annotation annotation) {
289            return ((Handler) annotation).priority();
290        }
291
292        /**
293         * Adds the given method of the given component as a dynamic handler for
294         * a specific event and channel. The method with the given name must be
295         * annotated as dynamic handler and must have a single parameter of type
296         * {@link Event} (or a derived type as appropriate for the event type to
297         * be handled). It can have an optional parameter of type 
298         * {@link Channel}.
299         * 
300         * @param component
301         *            the component
302         * @param method
303         *            the name of the method that implements the handler
304         * @param eventValue
305         *            the event key that should be used for matching this
306         *            handler with an event. This is equivalent to an
307         *            <code>events</code>/<code>namedEvents</code> parameter
308         *            used with a single value in the handler annotation, but
309         *            here all kinds of Objects are allowed as key values.
310         * @param channelValue
311         *            the channel value that should be used for matching 
312         *            an event's channel with this handler. This is equivalent 
313         *            to a `channels`/`namedChannels` parameter with a single
314         *            value in the handler annotation, but
315         *            here all kinds of Objects are allowed as values. As a
316         *            convenience, if the actual object provided is a
317         *            {@link Channel}, its default criterion is used for 
318         *            matching.
319         * @param priority
320         *            the priority of the handler
321         */
322        public static void add(ComponentType component, String method,
323                Object eventValue, Object channelValue, int priority) {
324            addInternal(component, method, eventValue, channelValue, priority);
325        }
326
327        /**
328         * Add a handler like 
329         * {@link #add(ComponentType, String, Object, Object, int)}
330         * but take the values for event and priority from the annotation.
331         * 
332         * @param component the component
333         * @param method the name of the method that implements the handler
334         * @param channelValue the channel value that should be used 
335         * for matching an event's channel with this handler
336         */
337        public static void add(ComponentType component, String method,
338                Object channelValue) {
339            addInternal(component, method, null, channelValue, null);
340        }
341
342        @SuppressWarnings({ "PMD.CyclomaticComplexity",
343            "PMD.AvoidBranchingStatementAsLastInLoop" })
344        private static void addInternal(ComponentType component, String method,
345                Object eventValue, Object channelValue, Integer priority) {
346            try {
347                if (channelValue instanceof Channel) {
348                    channelValue = ((Eligible) channelValue).defaultCriterion();
349                }
350                for (Method m : component.getClass().getMethods()) {
351                    if (!m.getName().equals(method)) {
352                        continue;
353                    }
354                    for (Annotation annotation : m.getDeclaredAnnotations()) {
355                        Class<?> annoType = annotation.annotationType();
356                        if (!(annoType.equals(Handler.class))) {
357                            continue;
358                        }
359                        HandlerDefinition hda
360                            = annoType.getAnnotation(HandlerDefinition.class);
361                        if (hda == null
362                            || !Handler.class.isAssignableFrom(annoType)
363                            || !((Handler) annotation).dynamic()) {
364                            continue;
365                        }
366                        @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
367                        Scope scope = new Scope(component, m,
368                            (Handler) annotation, Collections.emptyMap(),
369                            eventValue == null ? null
370                                : new Object[] { eventValue },
371                            new Object[] { channelValue });
372                        Components.manager(component).addHandler(m, scope,
373                            priority == null
374                                ? ((Handler) annotation).priority()
375                                : priority);
376                        return;
377                    }
378                }
379                throw new IllegalArgumentException(
380                    "No method named \"" + method + "\" with DynamicHandler"
381                        + " annotation and correct parameter list.");
382            } catch (SecurityException e) {
383                throw (RuntimeException) (new IllegalArgumentException()
384                    .initCause(e));
385            }
386        }
387
388        /**
389         * The handler scope implementation used by the evaluator.
390         */
391        private static class Scope implements HandlerScope {
392
393            private final Set<Object> eventCriteria = new HashSet<>();
394            private final Set<Object> channelCriteria = new HashSet<>();
395
396            /**
397             * Instantiates a new scope.
398             *
399             * @param component the component
400             * @param method the method
401             * @param annotation the annotation
402             * @param channelReplacements the channel replacements
403             * @param eventValues the event values
404             * @param channelValues the channel values
405             */
406            @SuppressWarnings({ "PMD.CyclomaticComplexity", "PMD.NcssCount",
407                "PMD.NPathComplexity", "PMD.UseVarargs",
408                "PMD.AvoidDeeplyNestedIfStmts",
409                "PMD.CollapsibleIfStatements" })
410            public Scope(ComponentType component, Method method,
411                    Handler annotation,
412                    Map<Class<? extends Channel>, Object> channelReplacements,
413                    Object[] eventValues, Object[] channelValues) {
414                if (!HandlerDefinition.Evaluator.checkMethodSignature(method)) {
415                    throw new IllegalArgumentException("Method \""
416                        + method.toString() + "\" cannot be used as"
417                        + " handler (wrong signature).");
418                }
419                if (eventValues != null) { // NOPMD, != is easier to read
420                    eventCriteria.addAll(Arrays.asList(eventValues));
421                } else {
422                    // Get all event values from the handler annotation.
423                    if (annotation.events()[0] != Handler.NoEvent.class) {
424                        eventCriteria
425                            .addAll(Arrays.asList(annotation.events()));
426                    }
427                    // Get all named events from the annotation and add to event
428                    // keys.
429                    if (!annotation.namedEvents()[0].equals("")) {
430                        eventCriteria.addAll(
431                            Arrays.asList(annotation.namedEvents()));
432                    }
433                    // If no event types are given, try first parameter.
434                    if (eventCriteria.isEmpty()) {
435                        Class<?>[] paramTypes = method.getParameterTypes();
436                        if (paramTypes.length > 0) {
437                            if (Event.class.isAssignableFrom(paramTypes[0])) {
438                                eventCriteria.add(paramTypes[0]);
439                            }
440                        }
441                    }
442                }
443
444                if (channelValues != null) { // NOPMD, != is easier to read
445                    channelCriteria.addAll(Arrays.asList(channelValues));
446                } else {
447                    // Get channel values from the annotation.
448                    boolean addDefaultChannel = false;
449                    if (annotation.channels()[0] != Handler.NoChannel.class) {
450                        for (Class<?> c : annotation.channels()) {
451                            if (c == Self.class) {
452                                if (!(component instanceof Channel)) {
453                                    throw new IllegalArgumentException(
454                                        "Canot use channel This.class in "
455                                            + "annotation of " + method
456                                            + " because " + getClass().getName()
457                                            + " does not implement Channel.");
458                                }
459                                // Will be added anyway, see below, but
460                                // channelCriteria must not remain empty,
461                                // else the default channel is added.
462                                channelCriteria.add(
463                                    ((Channel) component).defaultCriterion());
464                            } else if (c == Channel.Default.class) {
465                                addDefaultChannel = true;
466                            } else {
467                                channelCriteria.add(channelReplacements == null
468                                    ? c
469                                    : channelReplacements.getOrDefault(c, c));
470                            }
471                        }
472                    }
473                    // Get named channels from annotation and add to channel
474                    // keys.
475                    if (!annotation.namedChannels()[0].equals("")) {
476                        channelCriteria.addAll(
477                            Arrays.asList(annotation.namedChannels()));
478                    }
479                    if (channelCriteria.isEmpty() || addDefaultChannel) {
480                        channelCriteria.add(Components.manager(component)
481                            .channel().defaultCriterion());
482                    }
483                }
484                // Finally, a component always handles events
485                // directed at it directly.
486                if (component instanceof Channel) {
487                    channelCriteria.add(
488                        ((Channel) component).defaultCriterion());
489                }
490
491            }
492
493            @Override
494            public boolean includes(Eligible event, Eligible[] channels) {
495                for (Object eventValue : eventCriteria) {
496                    if (event.isEligibleFor(eventValue)) {
497                        // Found match regarding event, now try channels
498                        for (Eligible channel : channels) {
499                            for (Object channelValue : channelCriteria) {
500                                if (channel.isEligibleFor(channelValue)) {
501                                    return true;
502                                }
503                            }
504                        }
505                        return false;
506                    }
507                }
508                return false;
509            }
510
511            /*
512             * (non-Javadoc)
513             * 
514             * @see java.lang.Object#toString()
515             */
516            @Override
517            public String toString() {
518                StringBuilder builder = new StringBuilder();
519                builder.append("Scope [");
520                if (eventCriteria != null) {
521                    builder.append("handledEvents=")
522                        .append(eventCriteria.stream().map(crit -> {
523                            if (crit instanceof Class) {
524                                return Components.className((Class<?>) crit);
525                            }
526                            return crit.toString();
527                        }).collect(Collectors.toSet()));
528                    builder.append(", ");
529                }
530                if (channelCriteria != null) {
531                    builder.append("handledChannels=");
532                    builder.append(channelCriteria);
533                }
534                builder.append(']');
535                return builder.toString();
536            }
537
538        }
539    }
