050/**
051 * This is the basic, general purpose handler annotation provided as part of the
052 * core package.
053 * 
054 * The annotated method is invoked for events that have a type (or
055 * name) matching the given `events` (or `namedEvents`) element 
056 * of the annotation and that are fired on one of
057 * the `channels` (or `namedChannels`) specified in the annotation.
058 * 
059 * If neither event classes nor named events are specified in the 
060 * annotation, the class of the annotated method's first parameter (which 
061 * must be of type {@link Event} or a derived type) is used as (single)
062 * event class (see the examples in {@link #events()} and 
063 * {@link #namedEvents()}).
064 * 
065 * Channel matching is performed by matching the event's channels
066 * (see {@link Event#channels()}) with the channels specified in the
067 * handler. The matching algorithm invokes  
068 * {@link Eligible#isEligibleFor(Object) isEligibleFor} for each of the 
069 * event's channels with the class (or name, see {@link #channels()} and 
070 * {@link Handler#namedChannels()}) of each of the channels specified
071 * in the handler.
072 * 
073 * If neither channel classes not named channels are specified in the 
074 * handler, or `{@link Default Channel.Default}.class` is specified as one 
075 * of the channel classes, the matching algorithm invokes  
076 * {@link Eligible#isEligibleFor(Object) isEligibleFor} for each of
077 * the event's channels with the default criterion of the component's 
078 * channel (see {@link Manager#channel()} and 
079 * {@link Eligible#defaultCriterion()}) as argument.
080 *
081 * Finally, independent of any specified channels, the matching algorithm 
082 * invokes {@link Eligible#isEligibleFor(Object) isEligibleFor} 
083 * for each of the event's channels with the component's default criterion
084 * as argument. This results in a match if
085 * the component itself is used as one of the event's channels
086 * (see the description of {@link Eligible}).
087 * 
088 * If a match is found for a given event's properties and a handler's
089 * specified attributes, the handler method is invoked. 
090 * The method can have an additional optional parameter of type
091 * {@link Channel} (or a derived type). This parameter does not 
092 * influence the eligibility of the method regarding a given event,
093 * it determines how the method is invoked. If the method does not
094 * have a second parameter, it is invoked once if an event 
095 * matches. If the parameter exists, the method is invoked once for
096 * each of the event's channels, provided that the optional parameter's
097 * type is assignable from the event's channel.
098 * 
099 * Because annotation elements accept only literals as values, they
100 * cannot be used to register handlers with properties that are only
101 * known at runtime. It is therefore possible to specify a 
102 * {@link Handler} annotation with element `dynamic=true`. Such a 
103 * handler must be added explicitly by invoking 
104 * {@link Evaluator#add(ComponentType, String, Object)} or
105 * {@link Evaluator#add(ComponentType, String, Object, Object, int)},
106 * thus specifying some of the handler's properties dynamically (i.e.
107 * at runtime).
108 * 
109 * A special case is the usage of a channel that is only known at
110 * runtime. If there are several handlers for events on such a
111 * channel, a lot of methods will become dynamic. To avoid this,
112 * {@link Component}s support a {@link ChannelReplacements}
113 * parameter in their constructor. Using this, it is possible
114 * to specify a specially defined {@link Channel} class in the
115 * annotation that is replaced by a channel that is only known
116 * at runtime.
117 * 
118 * @see Component#channel()
119 */
120@Documented
121@Retention(RetentionPolicy.RUNTIME)
122@Target(ElementType.METHOD)
123@HandlerDefinition(evaluator = Handler.Evaluator.class)
124public @interface Handler {
125
126    /** The default value for the <code>events</code> parameter of
127     * the annotation. Indicates that the parameter is not used. */
128    final class NoEvent extends Event<Void> {
129    }
130
131    /** The default value for the <code>channels</code> parameter of
132     * the annotation. Indicates that the parameter is not used. */
133    final class NoChannel extends ClassChannel {
134    }
135
136    /**
137     * Specifies classes of events that the handler is to receive.
138     * 
139     * ```java
140     * class SampleComponent extends Component {
141     * 
142     *    {@literal @}Handler
143     *     public void onStart(Start event) {
144     *         // Invoked for Start events on the component's channel,
145     *         // event object made available
146     *     }
147     * 
148     *    {@literal @}Handler(events=Start.class)
149     *     public void onStart() {
150     *         // Invoked for Start events on the component's channel,
151     *         // not interested in the event object
152     *     }
153     * 
154     *    {@literal @}Handler(events={Start.class, Stop.class})
155     *     public void onStart(Event<?> event) {
156     *         // Invoked for Start and Stop events on the component's
157     *         // channel, event made available (may need casting to 
158     *         // access specific properties) 
159     *     }
160     * }
161     * ```
162     * 
163     * @return the event classes
164     */
165    @SuppressWarnings("rawtypes")
166    Class<? extends Event>[] events() default NoEvent.class;
167
168    /**
169     * Specifies names of {@link NamedEvent}s that the handler is to receive.
170     * 
171     * ```java
172     * class SampleComponent extends Component {
173     * 
174     *    {@literal @}Handler(namedEvents="Test")
175     *     public void onTest(Event<?> event) {
176     *         // Invoked for (named) "Test" events (new NamedEvent("Test")) 
177     *         // on the component's channel, event object made available
178     *     }
179     * }
180     * ```
181     * 
182     * @return the event names
183     */
184    String[] namedEvents() default "";
185
186    /**
187     * Specifies classes of channels that the handler listens on. If none
188     * are specified, the component's channel is used.
189     * 
190     * ```java
191     * class SampleComponent extends Component {
192     * 
193     *    {@literal @}Handler(channels=Feedback.class)
194     *     public void onStart(Start event) {
195     *         // Invoked for Start events on the "Feedback" channel
196     *         // (class Feedback implements Channel {...}),
197     *         // event object made available
198     *     }
199     * }
200     * ```
201     * 
202     * Specifying `channels=Channel.class` make the handler listen
203     * for all events, independent of the channel that they are fired on.
204     * 
205     * @return the channel classes
206     */
207    Class<? extends Channel>[] channels() default NoChannel.class;
208
209    /**
210     * Specifies names of {@link NamedChannel}s that the handler listens on.
211     * 
212     * ```java
213     * class SampleComponent extends Component {
214     * 
215     *    {@literal @}Handler(namedChannels="Feedback")
216     *     public void onStart(Start event) {
217     *         // Invoked for Start events on the (named) channel "Feedback"
218     *         // (new NamedChannel("Feedback")), event object made available
219     *     }
220     * }
221     * ```
222     * 
223     * @return the channel names
224     */
225    String[] namedChannels() default "";
226
227    /**
228     * Specifies a priority. The value is used to sort handlers.
229     * Handlers with higher priority are invoked first.
230     * 
231     * @return the priority
232     */
233    int priority() default 0;
234
235    /**
236     * Returns {@code true} if the annotated method defines a
237     * dynamic handler. A dynamic handler must be added to the set of
238     * handlers of a component explicitly at run time using
239     * {@link Evaluator#add(ComponentType, String, Object)}
240     * or {@link Evaluator#add(ComponentType, String, Object, Object, int)}.
241     * 
242     * ```java
243     * class SampleComponent extends Component {
244     * 
245     *     SampleComponent() {
246     *         Handler.Evaluator.add(this, "onStartDynamic", someChannel);
247     *     }
248     * 
249     *    {@literal @}Handler(dynamic=true)
250     *     public void onStartDynamic(Start event) {
251     *         // Only invoked if added as handler at runtime
252     *     }
253     * }
254     * ```
255     * 
256     * @return the result
257     */
258    boolean dynamic() default false;

260    /**
261     * This class provides the {@link Evaluator} for the 
262     * {@link Handler} annotation provided by the core package. It 
263     * implements the behavior as described for the annotation. 
264     */
265    class Evaluator implements HandlerDefinition.Evaluator {
266
267        @Override
268        public HandlerScope scope(
269                ComponentType component, Method method,
270                ChannelReplacements channelReplacements) {
271            Handler annotation = method.getAnnotation(Handler.class);
272            if (annotation == null || annotation.dynamic()) {
273                return null;
274            }
275            return new Scope(component, method, annotation,
276                channelReplacements, null, null);
277        }

279        /*
280         * (non-Javadoc)
281         * 
282         * @see
283         * org.jgrapes.core.annotation.HandlerDefinition.Evaluator#getPriority()
284         */
285        @Override
286        public int priority(Annotation annotation) {
287            return ((Handler) annotation).priority();
288        }

290        /**
291         * Adds the given method of the given component as a dynamic handler for
292         * a specific event and channel. The method with the given name must be
293         * annotated as dynamic handler and must have a single parameter of type
294         * {@link Event} (or a derived type as appropriate for the event type to
295         * be handled). It can have an optional parameter of type 
296         * {@link Channel}.
297         * 
298         * @param component
299         *            the component
300         * @param method
301         *            the name of the method that implements the handler
302         * @param eventValue
303         *            the event key that should be used for matching this
304         *            handler with an event. This is equivalent to an
305         *            <code>events</code>/<code>namedEvents</code> parameter
306         *            used with a single value in the handler annotation, but
307         *            here all kinds of Objects are allowed as key values.
308         * @param channelValue
309         *            the channel value that should be used for matching 
310         *            an event's channel with this handler. This is equivalent 
311         *            to a `channels`/`namedChannels` parameter with a single
312         *            value in the handler annotation, but
313         *            here all kinds of Objects are allowed as values. As a
314         *            convenience, if the actual object provided is a
315         *            {@link Channel}, its default criterion is used for 
316         *            matching.
317         * @param priority
318         *            the priority of the handler
319         */
320        public static void add(ComponentType component, String method,
321                Object eventValue, Object channelValue, int priority) {
322            addInternal(component, method, eventValue, channelValue, priority);
323        }
324
325        /**
326         * Add a handler like 
327         * {@link #add(ComponentType, String, Object, Object, int)}
328         * but take the values for event and priority from the annotation.
329         * 
330         * @param component the component
331         * @param method the name of the method that implements the handler
332         * @param channelValue the channel value that should be used 
333         * for matching an event's channel with this handler
334         */
335        public static void add(ComponentType component, String method,
336                Object channelValue) {
337            addInternal(component, method, null, channelValue, null);
338        }
339
340        @SuppressWarnings({ "PMD.CyclomaticComplexity",
341            "PMD.AvoidBranchingStatementAsLastInLoop" })
342        private static void addInternal(ComponentType component, String method,
343                Object eventValue, Object channelValue, Integer priority) {
344            try {
345                if (channelValue instanceof Channel) {
346                    channelValue = ((Eligible) channelValue).defaultCriterion();
347                }
348                for (Method m : component.getClass().getMethods()) {
349                    if (!m.getName().equals(method)) {
350                        continue;
351                    }
352                    for (Annotation annotation : m.getDeclaredAnnotations()) {
353                        Class<?> annoType = annotation.annotationType();
354                        if (!(annoType.equals(Handler.class))) {
355                            continue;
356                        }
357                        HandlerDefinition hda
358                            = annoType.getAnnotation(HandlerDefinition.class);
359                        if (hda == null
360                            || !Handler.class.isAssignableFrom(annoType)
361                            || !((Handler) annotation).dynamic()) {
362                            continue;
363                        }
364                        @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
365                        Scope scope = new Scope(component, m,
366                            (Handler) annotation, Collections.emptyMap(),
367                            eventValue == null ? null
368                                : new Object[] { eventValue },
369                            new Object[] { channelValue });
370                        Components.manager(component).addHandler(m, scope,
371                            priority == null
372                                ? ((Handler) annotation).priority()
373                                : priority);
374                        return;
375                    }
376                }
377                throw new IllegalArgumentException(
378                    "No method named \"" + method + "\" with DynamicHandler"
379                        + " annotation and correct parameter list.");
380            } catch (SecurityException e) {
381                throw (RuntimeException) (new IllegalArgumentException()
382                    .initCause(e));
383            }
384        }
385
386        /**
387         * The handler scope implementation used by the evaluator.
388         */
389        private static class Scope implements HandlerScope {
390
391            private final Set<Object> eventCriteria = new HashSet<Object>();
392            private final Set<Object> channelCriteria = new HashSet<Object>();
393
394            /**
395             * Instantiates a new scope.
396             *
397             * @param component the component
398             * @param method the method
399             * @param annotation the annotation
400             * @param channelReplacements the channel replacements
401             * @param eventValues the event values
402             * @param channelValues the channel values
403             */
404            @SuppressWarnings({ "PMD.CyclomaticComplexity", "PMD.NcssCount",
405                "PMD.NPathComplexity", "PMD.UseVarargs",
406                "PMD.AvoidDeeplyNestedIfStmts",
407                "PMD.CollapsibleIfStatements" })
408            public Scope(ComponentType component, Method method,
409                    Handler annotation,
410                    Map<Class<? extends Channel>, Object> channelReplacements,
411                    Object[] eventValues, Object[] channelValues) {
412                if (!HandlerDefinition.Evaluator.checkMethodSignature(method)) {
413                    throw new IllegalArgumentException("Method \""
414                        + method.toString() + "\" cannot be used as"
415                        + " handler (wrong signature).");
416                }
417                if (eventValues != null) { // NOPMD, != is easier to read
418                    eventCriteria.addAll(Arrays.asList(eventValues));
419                } else {
420                    // Get all event values from the handler annotation.
421                    if (annotation.events()[0] != Handler.NoEvent.class) {
422                        eventCriteria
423                            .addAll(Arrays.asList(annotation.events()));
424                    }
425                    // Get all named events from the annotation and add to event
426                    // keys.
427                    if (!annotation.namedEvents()[0].equals("")) {
428                        eventCriteria.addAll(
429                            Arrays.asList(annotation.namedEvents()));
430                    }
431                    // If no event types are given, try first parameter.
432                    if (eventCriteria.isEmpty()) {
433                        Class<?>[] paramTypes = method.getParameterTypes();
434                        if (paramTypes.length > 0) {
435                            if (Event.class.isAssignableFrom(paramTypes[0])) {
436                                eventCriteria.add(paramTypes[0]);
437                            }
438                        }
439                    }
440                }
441
442                if (channelValues != null) { // NOPMD, != is easier to read
443                    channelCriteria.addAll(Arrays.asList(channelValues));
444                } else {
445                    // Get channel values from the annotation.
446                    boolean addDefaultChannel = false;
447                    if (annotation.channels()[0] != Handler.NoChannel.class) {
448                        for (Class<?> c : annotation.channels()) {
449                            if (c == Self.class) {
450                                if (!(component instanceof Channel)) {
451                                    throw new IllegalArgumentException(
452                                        "Canot use channel This.class in "
453                                            + "annotation of " + method
454                                            + " because " + getClass().getName()
455                                            + " does not implement Channel.");
456                                }
457                                // Will be added anyway, see below
458                            } else if (c == Channel.Default.class) {
459                                addDefaultChannel = true;
460                            } else {
461                                channelCriteria.add(channelReplacements == null
462                                    ? c
463                                    : channelReplacements.getOrDefault(c, c));
464                            }
465                        }
466                    }
467                    // Get named channels from annotation and add to channel
468                    // keys.
469                    if (!annotation.namedChannels()[0].equals("")) {
470                        channelCriteria.addAll(
471                            Arrays.asList(annotation.namedChannels()));
472                    }
473                    if (channelCriteria.isEmpty() || addDefaultChannel) {
474                        channelCriteria.add(Components.manager(component)
475                            .channel().defaultCriterion());
476                    }
477                }
478                // Finally, a component always handles events
479                // directed at it directly.
480                if (component instanceof Channel) {
481                    channelCriteria.add(
482                        ((Channel) component).defaultCriterion());
483                }
484
485            }
486
487            @Override
488            public boolean includes(Eligible event, Eligible[] channels) {
489                for (Object eventValue : eventCriteria) {
490                    if (event.isEligibleFor(eventValue)) {
491                        // Found match regarding event, now try channels
492                        for (Eligible channel : channels) {
493                            for (Object channelValue : channelCriteria) {
494                                if (channel.isEligibleFor(channelValue)) {
495                                    return true;
496                                }
497                            }
498                        }
499                        return false;
500                    }
501                }
502                return false;
503            }
504
505            /*
506             * (non-Javadoc)
507             * 
508             * @see java.lang.Object#toString()
509             */
510            @Override
511            public String toString() {
512                StringBuilder builder = new StringBuilder();
513                builder.append("Scope [");
514                if (eventCriteria != null) {
515                    builder.append("handledEvents=")
516                        .append(eventCriteria.stream().map(crit -> {
517                            if (crit instanceof Class) {
518                                return Components.className((Class<?>) crit);
519                            }
520                            return crit.toString();
521                        }).collect(Collectors.toSet()));
522                    builder.append(", ");
523                }
524                if (channelCriteria != null) {
525                    builder.append("handledChannels=");
526                    builder.append(channelCriteria);
527                }
528                builder.append(']');
529                return builder.toString();
530            }

532        }
