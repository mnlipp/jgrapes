019package org.jgrapes.core.annotation;
020
021import java.lang.annotation.Annotation;
022import java.lang.annotation.Documented;
023import java.lang.annotation.ElementType;
024import java.lang.annotation.Retention;
025import java.lang.annotation.RetentionPolicy;
026import java.lang.annotation.Target;
027import java.lang.reflect.Method;
028
029import org.jgrapes.core.Channel;
030import org.jgrapes.core.ComponentType;
031import org.jgrapes.core.Event;
032import org.jgrapes.core.HandlerScope;
033
034/**
035 * This annotation tags some other annotation as a handler annotation. 
036 * The tagged annotation can then be used to mark a method as a handler.
037 *  
038 * Every handler definition annotation must provide an {@link Evaluator} 
039 * to allow access to the properties of the handler annotation in a 
040 * uniform way. 
041 */
042@Documented
043@Retention(value=RetentionPolicy.RUNTIME)
044@Target(value=ElementType.ANNOTATION_TYPE)
045public @interface HandlerDefinition {
046        
047        /**
048         * Returns the evaluator for the annotated handler annotation.
049         * 
050         * @return the evaluator
051         */
052        Class<? extends Evaluator> evaluator();
053
054        /**
055         * This interface allows access to the properties defined by arbitrary
056         * handler annotations in a uniform way. Handler annotations
057         * must specify the scope of a handler, i.e. for which events and
058         * channels the handler should be invoked, and the priority of
059         * the handler.  
060         */
061        public interface Evaluator {
062
063                /**
064                 * Returns the information about the events and channels handled
065                 * by the handler as a {@link HandlerScope} object. This method
066                 * is invoked during object initialization. It may return
067                 * {@code null} if a handler is not supposed to be added for
068                 * this method during initialization (dynamic handler,
069                 * see {@link Handler#dynamic()}). 
070                 * 
071                 * @param component the component
072                 * @param method the annotated method
073                 * @param eventValues event values that can be used in addition
074                 * to or as replacements for the values specified in the
075                 * annotation
076                 * @param channelValues channel values that can be used in addition
077                 * to or as replacements for the values specified in the
078                 * annotation
079                 * @return the scope or {@code null} if a handler for the method
080                 * should not be created
081                 */
082                HandlerScope scope(ComponentType component, Method method, 
083                                        Object[] eventValues, Object[] channelValues);
084                
085                /**
086                 * Returns the priority defined by the annotation
087                 * 
088                 * @param annotation the annotation
089                 * @return the priority
090                 */
091                int priority(Annotation annotation);
092                
093                /**
094                 * Utility method for checking if the method can be used as handler.
095                 * 
096                 * @param method the method
097                 * @return the result
098                 */
099                public static boolean checkMethodSignature(Method method) {
100                        return method.getParameterTypes().length == 0
101                                || (method.getParameterTypes().length == 1
102                                        && Event.class.isAssignableFrom(
103                                                method.getParameterTypes()[0]))
104                                || (method.getParameterTypes().length == 2
105                                        && Event.class.isAssignableFrom(
106                                                method.getParameterTypes()[0]))
107                                        && Channel.class.isAssignableFrom(
108                                                method.getParameterTypes()[1]);
109                }
110        }
111
112}
