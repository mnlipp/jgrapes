058/**
059 * 
060 */
061public class SysInfoPortlet extends FreeMarkerPortlet {
062
063        private final static Set<RenderMode> MODES = RenderMode.asSet(
064                        DeleteablePreview, View);
065        
066        /**
067         * Creates a new component with its channel set to the given 
068         * channel.
069         * 
070         * @param componentChannel the channel that the component's 
071         * handlers listen on by default and that 
072         * {@link Manager#fire(Event, Channel...)} sends the event to 
073         */
074        public SysInfoPortlet(Channel componentChannel) {
075                super(componentChannel, true);
076                setPeriodicRefresh(Duration.ofSeconds(1));
077        }
078
079        @Handler
080        public void onPortalReady(PortalReady event, PortalSession portalSession) 
081                        throws TemplateNotFoundException, MalformedTemplateNameException, 
082                        ParseException, IOException {
083                ResourceBundle resourceBundle = resourceBundle(portalSession.locale());
084                // Add SysInfoPortlet resources to page
085                portalSession.respond(new AddPortletType(type())
086                                .setDisplayName(resourceBundle.getString("portletName"))
087                                .addScript(PortalView.uriFromPath("SysInfo-functions.ftl.js"))
088                                .addCss(PortalView.uriFromPath("SysInfo-style.css"))
089                                .setInstantiable());
090        }
091
092        /* (non-Javadoc)
093         * @see org.jgrapes.portal.AbstractPortlet#generatePortletId()
094         */
095        @Override
096        protected String generatePortletId() {
097                return type() + "-" + super.generatePortletId();
098        }
099
100        /* (non-Javadoc)
101         * @see org.jgrapes.portal.AbstractPortlet#modelFromSession(org.jgrapes.io.IOSubchannel, java.lang.String)
102         */
103        @SuppressWarnings("unchecked")
104        @Override
105        protected <T extends Serializable> Optional<T> stateFromSession(
106                        Session session, String portletId, Class<T> type) {
107                if (portletId.startsWith(type() + "-")) {
108                        return Optional.of((T)new SysInfoModel(portletId));
109                }
110                return Optional.empty();
111        }
112
113        @Override
114        public String doAddPortlet(AddPortletRequest event,
115                        PortalSession portalSession) throws Exception {
116                String portletId = generatePortletId();
117                SysInfoModel portletModel = putInSession(
118                                portalSession.browserSession(), new SysInfoModel(portletId));
119                Template tpl = freemarkerConfig().getTemplate("SysInfo-preview.ftl.html");
120                portalSession.respond(new RenderPortlet(
121                                SysInfoPortlet.class, portletModel.getPortletId(),
122                                templateProcessor(tpl, fmModel(event, portalSession, portletModel)))
123                                .setRenderMode(DeleteablePreview).setSupportedModes(MODES)
124                                .setForeground(true));
125                updateView(portalSession, portletId, portalSession.locale());
126                return portletId;
127        }
128        
129        /* (non-Javadoc)
130         * @see org.jgrapes.portal.AbstractPortlet#doRenderPortlet(org.jgrapes.portal.events.RenderPortletRequest, org.jgrapes.io.IOSubchannel, org.jgrapes.portal.AbstractPortlet.PortletModelBean)
131         */
132        @Override
133        protected void doRenderPortlet(RenderPortletRequest event,
134                PortalSession portalSession, String portletId, 
135                Serializable retrievedState) throws Exception {
136                SysInfoModel portletModel = (SysInfoModel)retrievedState;
137                Locale locale = portalSession.locale();
138                switch (event.renderMode()) {
139                case Preview:
140                case DeleteablePreview: {
141                        Template tpl = freemarkerConfig().getTemplate("SysInfo-preview.ftl.html");
142                        portalSession.respond(new RenderPortlet(
143                                        SysInfoPortlet.class, portletModel.getPortletId(), 
144                                        templateProcessor(tpl, fmModel(event, portalSession, portletModel)))
145                                        .setRenderMode(DeleteablePreview).setSupportedModes(MODES)
146                                        .setForeground(event.isForeground()));
147                        updateView(portalSession, portletModel.getPortletId(), locale);
148                        break;
149                }
150                case View: {
151                        Template tpl = freemarkerConfig().getTemplate("SysInfo-view.ftl.html");
152                        portalSession.respond(new RenderPortlet(
153                                        SysInfoPortlet.class, portletModel.getPortletId(), 
154                                        templateProcessor(tpl, fmModel(event, portalSession, portletModel)))
155                                        .setRenderMode(View).setSupportedModes(MODES)
156                                        .setForeground(event.isForeground()));
157                        break;
158                }
159                default:
160                        break;
161                }
162        }
163
164        private void updateView(IOSubchannel channel, String portletId,
165                Locale locale) {
166                Runtime runtime = Runtime.getRuntime();
167                channel.respond(new NotifyPortletView(type(),
168                                portletId, "updateMemorySizes", 
169                                System.currentTimeMillis(), runtime.maxMemory(),
170                                runtime.totalMemory(), runtime.freeMemory()));
171        }
172        
173        /* (non-Javadoc)
174         * @see org.jgrapes.portal.AbstractPortlet#doDeletePortlet(org.jgrapes.portal.events.DeletePortletRequest, org.jgrapes.io.IOSubchannel, org.jgrapes.portal.AbstractPortlet.PortletModelBean)
175         */
176        @Override
177        protected void doDeletePortlet(DeletePortletRequest event,
178                PortalSession channel, String portletId, 
179                Serializable retrievedState) throws Exception {
180                channel.respond(new DeletePortlet(portletId));
181        }
182
183        @Override
184        public void doRefreshPortletViews() {
185                for (Map.Entry<PortalSession, Set<String>> e : portletIdsByPortalSession()
186                        .entrySet()) {
187                        PortalSession portalSession = e.getKey();
188                        Locale locale = portalSession.locale();
189                        for (String portletId : e.getValue()) {
190                                updateView(portalSession, portletId, locale);
191                        }
192                }
193        }
194        
195        @SuppressWarnings("serial")
196        public static class SysInfoModel extends PortletBaseModel {
197
198                /**
199                 * Creates a new model with the given type and id.
200                 * 
201                 * @param portletId the portlet id
202                 */
203                @ConstructorProperties({"portletId"})
204                public SysInfoModel(String portletId) {
205                        super(portletId);
206                }
207
208                public Properties systemProperties() {
209                        return System.getProperties();
210                }
211                
212                public Runtime runtime() {
213                        return Runtime.getRuntime();
214                }
215        }
