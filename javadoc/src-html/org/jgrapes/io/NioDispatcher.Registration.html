019package org.jgrapes.io;
020
021import java.io.IOException;
022import java.io.InterruptedIOException;
023import java.nio.channels.SelectableChannel;
024import java.nio.channels.SelectionKey;
025import java.nio.channels.Selector;
026import java.util.Set;
027import org.jgrapes.core.Component;
028import org.jgrapes.core.annotation.Handler;
029import org.jgrapes.core.events.Start;
030import org.jgrapes.core.events.Stop;
031import org.jgrapes.io.events.NioRegistration;
032
033/**
034 * A helper component that provides the central hub for non blocking
035 * I/O components. Exactly one {@code NioDispatcher} must exist in
036 * any tree with {@link NioHandler} components. 
037 * 
038 * @author Michael N. Lipp
039 */
040public class NioDispatcher extends Component implements Runnable {
041
042        private Selector selector = null;
043        private Thread runner = null;
044        private boolean running = false;
045        private Object selectorGate = new Object();
046        
047        /**
048         * Creates a new Dispatcher.
049         * 
050         * @throws IOException if an I/O exception occurred
051         */
052        public NioDispatcher() throws IOException {
053                selector = Selector.open();
054        }
055
056        /**
057         * Starts this dispatcher. A dispatcher has an associated thread that
058         * keeps it running.
059         * 
060         * @param event the event
061         */
062        @Handler
063        public synchronized void onStart(Start event) {
064                if (running) {
065                        return;
066                }
067                running = true;
068                runner = new Thread(this, "NioDispatcher");
069                runner.start();
070        }
071
072        /**
073         * Stops the thread that is associated with this dispatcher.
074         * 
075         * @param event the event
076         * @throws InterruptedException if the execution is interrupted
077         */
078        @Handler(priority=-10000)
079        public synchronized void onStop(Stop event) throws InterruptedException {
080                if (runner == null) {
081                        return;
082                }
083                running = false;
084                // It just might happen that the wakeup() occurs between the
085                // check for running and the select() in the thread's run loop,
086                // but we -- obviously -- cannot put the select() in a 
087                // synchronized(this).
088                while (runner.isAlive()) {
089                        selector.wakeup();
090                        runner.join(10);
091                }
092                runner = null;
093        }
094
095        /**
096         * Invoked once by the thread associated with the dispatcher. Handles
097         * all events from the underlying {@link Selector}.  
098         */
099        @Override
100        public void run() {
101                try {
102                        registerAsGenerator();
103                        while (running) {
104                                try {
105                                        selector.select();
106                                        Set<SelectionKey> selected = selector.selectedKeys();
107                                        for (SelectionKey key: selected) {
108                                                ((NioHandler)key.attachment())
109                                                        .handleOps(key.readyOps());
110                                        }
111                                        selected.clear();
112                                        synchronized (selectorGate) {
113                                                // Delay next iteration if another thread has the lock.
114                                                // "Find bugs" complains, but this is really okay.
115                                        }
116                                } catch (InterruptedIOException e) {
117                                        break;
118                                } catch (IOException e) {
119                                        // Can be ignored
120                                }
121                        }
122                } finally {
123                        unregisterAsGenerator();
124                }
125        }
126
127        @Handler
128        public void onNioRegistration(NioRegistration event)
129                        throws IOException {
130                SelectableChannel channel = event.ioChannel();
131                channel.configureBlocking(false);
132                SelectionKey key;
133                synchronized (selectorGate) {
134                        selector.wakeup(); // make sure selector isn't blocking
135                        key = channel.register(
136                                        selector, event.ops(), event.handler());
137                }
138                event.setResult(new Registration(key));
139        }
140        
141        public class Registration extends NioRegistration.Registration {
142
143                private SelectionKey key;
144                
145                public Registration(SelectionKey key) {
146                        super();
147                        this.key = key;
148                }
149
150                @Override
151                public void updateInterested(int ops) {
152                        synchronized (selectorGate) {
153                                selector.wakeup(); // make sure selector isn't blocking
154                                key.interestOps(ops);
155                        }
156                }
157        }
158        
159}
