019package org.jgrapes.io.events;
020
021import java.nio.Buffer;
022import java.nio.ByteBuffer;
023import java.nio.CharBuffer;
024
025import org.jgrapes.core.Channel;
026import org.jgrapes.core.Components;
027import org.jgrapes.core.Event;
028import org.jgrapes.io.util.BufferCollector;
029import org.jgrapes.io.util.ManagedBuffer;
030
031/**
032 * This event signals that a new chunk of internally generated data is to be
033 * forwarded to some destination. The data is kept in a buffer. The buffer is
034 * returned to the pool upon successful processing of the event. This type of
035 * event is commonly used for data flowing out of the application.
036 * 
037 * As a convenience, the class provides the methods known
038 * from {@link Buffer} as short-cuts for invoking
039 * `buffer().`*method()*.
040 */
041public class Output<T extends Buffer> extends Event<Void> {
042
043        private ManagedBuffer<T> buffer;
044        private boolean eor;
045
046        /**
047         * Create a new output event with the given buffer and optionally flips
048         * it. Used internally for constructor ("super(...)") invocations that 
049         * don't flip the buffer.
050         * 
051         * @param buffer the buffer with the data
052         * @param flip if the buffer should be flipped
053         * @param endOfRecord if the event ends a data record
054         */
055        private Output(ManagedBuffer<T> buffer, boolean flip, boolean endOfRecord) {
056                this.buffer = buffer;
057                this.eor = endOfRecord;
058                if (flip) {
059                        buffer.flip();
060                }
061        }
062        
063        /**
064         * Create a new event from an existing event. This constructor
065         * is useful if the data is to be forwarded to another channel
066         * by a new event.
067         * 
068         * The buffer is reused in the new event (the lock count is 
069         * incremented).
070         * 
071         * @param event the existing event
072         */
073        public Output(Output<T> event) {
074                this(event.buffer(), false, event.isEndOfRecord());
075                event.buffer().lockBuffer();
076        }
077        
078        /**
079         * Create a new event with the given buffer. The buffer must
080         * have been prepared for invoking `get`-methods.
081         * 
082         * @param buffer the buffer with the data
083         * @param endOfRecord if the event ends a data record
084         */
085        public static <B extends Buffer> Output<B> fromSource(
086                        ManagedBuffer<B> buffer, boolean endOfRecord) {
087                return new Output<>(buffer, false, endOfRecord);
088        }
089
090        /**
091         * Create a new event with the given buffer. Creating the event
092         * flips the buffer, which is assumed to have been used for
093         * collecting data up to now.
094         * 
095         * @param buffer the buffer with the data
096         * @param endOfRecord if the event ends a data record
097         */
098        public static <B extends Buffer> Output<B> fromSink(
099                        ManagedBuffer<B> buffer, boolean endOfRecord) {
100                return new Output<>(buffer, true, endOfRecord);
101        }
102
103        /**
104         * Convenience method that creates a 
105         * {@code Output<CharBuffer} event from a {@link String}.
106         * 
107         * @param data the string to wrap
108         * @param endOfRecord if the event ends a data record
109         * @return the event
110         */
111        public static Output<CharBuffer> 
112                from(String data, boolean endOfRecord) {
113                return new Output<>(new ManagedBuffer<>(
114                                CharBuffer.wrap(data), BufferCollector.NOOP_COLLECTOR),
115                                false, endOfRecord);
116        }
117        
118        /**
119         * Convenience method that creates a 
120         * {@code Output<ManagedCharBuffer} event from a `byte[]`.
121         * 
122         * @param data the array to wrap
123         * @param endOfRecord if the event ends a data record
124         * @return the event
125         */
126        public static Output<ByteBuffer> 
127                from(byte[] data, boolean endOfRecord) {
128                return new Output<>(new ManagedBuffer<ByteBuffer>(
129                                ByteBuffer.wrap(data), BufferCollector.NOOP_COLLECTOR),
130                                false, endOfRecord);
131        }
132        
133        /**
134         * Get the buffer with the data from this event.
135         * 
136         * @return the buffer
137         */
138        public ManagedBuffer<T> buffer() {
139                return buffer;
140        }
141
142        /**
143         * Return the end of record flag passed to the constructor.
144         * The precise interpretation of a record depends on the data
145         * handled. 
146         * 
147         * @return the end of record flag
148         */
149        public boolean isEndOfRecord() {
150                return eor;
151        }
152        
153        /**
154         * Releases the buffer, unless locked.
155         */
156        @Override
157        protected synchronized void handled() {
158                buffer.unlockBuffer();
159                buffer = null;
160        }
161        
162        /* (non-Javadoc)
163         * @see java.lang.Object#toString()
164         */
165        @Override
166        public String toString() {
167                StringBuilder builder = new StringBuilder();
168                builder.append(Components.objectName(this));
169                builder.append(" [");
170                if (channels != null) {
171                        builder.append("channels=");
172                        builder.append(Channel.toString(channels));
173                }
174                builder.append(",eor=");
175                builder.append(eor);
176                builder.append("]");
177                return builder.toString();
178        }
179        
180        /**
181         * Return the backing buffer of the managed buffer.
182         * Short for `buffer().backingBuffer()`.
183         *
184         * @return the buffer
185         */
186        public Buffer backingBuffer() {
187                return buffer.backingBuffer();
188        }
189        
190        /**
191         * @return the backing array
192         * @see java.nio.Buffer#array()
193         */
194        public Object array() {
195                return buffer.array();
196        }
197
198        /**
199         * @return the backing array offset
200         * @see java.nio.Buffer#arrayOffset()
201         */
202        public int arrayOffset() {
203                return buffer.arrayOffset();
204        }
205
206        /**
207         * @return the capacity
208         * @see java.nio.Buffer#capacity()
209         */
210        public final int capacity() {
211                return buffer.capacity();
212        }
213
214        /**
215         * @return the buffer
216         * @see java.nio.Buffer#clear()
217         */
218        public final Buffer clear() {
219                return buffer.clear();
220        }
221
222        /**
223         * @return the buffer
224         * @see java.nio.Buffer#flip()
225         */
226        public final Buffer flip() {
227                return buffer.flip();
228        }
229
230        /**
231         * @return the result
232         * @see java.nio.Buffer#hasArray()
233         */
234        public boolean hasArray() {
235                return buffer.hasArray();
236        }
237
238        /**
239         * @return the result
240         * @see java.nio.Buffer#hasRemaining()
241         */
242        public final boolean hasRemaining() {
243                return buffer.hasRemaining();
244        }
245
246        /**
247         * @return the result
248         * @see java.nio.Buffer#isDirect()
249         */
250        public boolean isDirect() {
251                return buffer.isDirect();
252        }
253
254        /**
255         * @return the result
256         * @see java.nio.Buffer#isReadOnly()
257         */
258        public boolean isReadOnly() {
259                return buffer.isReadOnly();
260        }
261
262        /**
263         * @return the result
264         * @see java.nio.Buffer#limit()
265         */
266        public final int limit() {
267                return buffer.limit();
268        }
269
270        /**
271         * @param newLimit the new limit
272         * @return the result
273         * @see java.nio.Buffer#limit(int)
274         */
275        public final Buffer limit(int newLimit) {
276                return buffer.limit(newLimit);
277        }
278
279        /**
280         * @return the buffer
281         * @see java.nio.Buffer#mark()
282         */
283        public final Buffer mark() {
284                return buffer.mark();
285        }
286
287        /**
288         * @return the result
289         * @see java.nio.Buffer#position()
290         */
291        public final int position() {
292                return buffer.position();
293        }
294
295        /**
296         * @param newPosition the new position
297         * @return the buffer
298         * @see java.nio.Buffer#position(int)
299         */
300        public final Buffer position(int newPosition) {
301                return buffer.position(newPosition);
302        }
303
304        /**
305         * @return the result
306         * @see java.nio.Buffer#remaining()
307         */
308        public final int remaining() {
309                return buffer.remaining();
310        }
311
312        /**
313         * @return the Buffer
314         * @see java.nio.Buffer#reset()
315         */
316        public final Buffer reset() {
317                return buffer.reset();
318        }
319
320        /**
321         * @return the Buffer
322         * @see java.nio.Buffer#rewind()
323         */
324        public final Buffer rewind() {
325                return buffer.rewind();
326        }
327}
