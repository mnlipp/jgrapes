019package org.jgrapes.io.events;
020
021import java.nio.Buffer;
022import java.nio.ByteBuffer;
023import java.nio.CharBuffer;
024
025import org.jgrapes.io.util.BufferCollector;
026import org.jgrapes.io.util.ManagedBuffer;
027
028/**
029 * This event signals that a new chunk of internally generated data is to be
030 * forwarded to some destination. This type of
031 * event is commonly used for data flowing out of the application.
032 */
033public class Output<T extends Buffer> extends IOEvent<T> {
034
035        /**
036         * Create a new output event with the given buffer and optionally flips
037         * it. Used internally for constructor ("super(...)") invocations that 
038         * don't flip the buffer.
039         * 
040         * @param buffer the buffer with the data
041         * @param flip if the buffer should be flipped
042         * @param endOfRecord if the event ends a data record
043         */
044        private Output(ManagedBuffer<T> buffer, boolean flip, boolean endOfRecord) {
045                super (buffer, endOfRecord);
046                if (flip) {
047                        buffer.flip();
048                }
049        }
050        
051        /**
052         * Create a new event from an existing event. This constructor
053         * is useful if the data is to be forwarded to another channel
054         * by a new event.
055         * 
056         * The buffer is reused in the new event (the lock count is 
057         * incremented).
058         * 
059         * @param event the existing event
060         */
061        public Output(Output<T> event) {
062                this(event.buffer(), false, event.isEndOfRecord());
063                event.buffer().lockBuffer();
064        }
065        
066        /**
067         * Create a new event with the given buffer. The buffer must
068         * have been prepared for invoking `get`-methods.
069         * 
070         * @param buffer the buffer with the data
071         * @param endOfRecord if the event ends a data record
072         */
073        public static <B extends Buffer> Output<B> fromSource(
074                        ManagedBuffer<B> buffer, boolean endOfRecord) {
075                return new Output<>(buffer, false, endOfRecord);
076        }
077
078        /**
079         * Create a new event with the given buffer. Creating the event
080         * flips the buffer, which is assumed to have been used for
081         * collecting data up to now.
082         * 
083         * @param buffer the buffer with the data
084         * @param endOfRecord if the event ends a data record
085         */
086        public static <B extends Buffer> Output<B> fromSink(
087                        ManagedBuffer<B> buffer, boolean endOfRecord) {
088                return new Output<>(buffer, true, endOfRecord);
089        }
090
091        /**
092         * Convenience method that creates a 
093         * {@code Output<CharBuffer} event from a {@link String}.
094         * 
095         * @param data the string to wrap
096         * @param endOfRecord if the event ends a data record
097         * @return the event
098         */
099        public static Output<CharBuffer> 
100                from(String data, boolean endOfRecord) {
101                return new Output<>(new ManagedBuffer<>(
102                                CharBuffer.wrap(data), BufferCollector.NOOP_COLLECTOR),
103                                false, endOfRecord);
104        }
105        
106        /**
107         * Convenience method that creates a 
108         * {@code Output<ByteBuffer} event from a `byte[]`.
109         * 
110         * @param data the array to wrap
111         * @param endOfRecord if the event ends a data record
112         * @return the event
113         */
114        public static Output<ByteBuffer> 
115                from(byte[] data, boolean endOfRecord) {
116                return new Output<>(new ManagedBuffer<ByteBuffer>(
117                                ByteBuffer.wrap(data), BufferCollector.NOOP_COLLECTOR),
118                                false, endOfRecord);
119        }
120}
