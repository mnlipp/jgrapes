019package org.jgrapes.io.events;
020
021import java.nio.Buffer;
022import java.util.Optional;
023
024import org.jgrapes.core.Channel;
025import org.jgrapes.core.Components;
026import org.jgrapes.core.Event;
027import org.jgrapes.io.util.ManagedBuffer;
028
029/**
030 * Events of this type signal that a new chunk of data is available
031 * for processing. From a consumer's point of view, the data is
032 * kept in a NIO {@link Buffer}. However, when creating the event
033 * the data has to be provided as a {@link ManagedBuffer}. 
034 * This buffer is returned to the pool upon successful processing 
035 * of the event.
036 * 
037 * As a convenience, the class provides the methods known
038 * from {@link Buffer} as short-cuts for invoking
039 * `data().`*method()*.
040 * 
041 * @param <T> the type of data used in this event
042 */
043public abstract class IOEvent<T extends Buffer> extends Event<Void> {
044
045        private ManagedBuffer<T> buffer;
046        private boolean eor;
047        
048        protected IOEvent(ManagedBuffer<T> buffer, boolean endOfRecord) {
049                this.buffer = buffer;
050                this.eor = endOfRecord;
051        }
052
053        /**
054         * Get the managed buffer with the data from this event.
055         * 
056         * @return the buffer
057         */
058        public ManagedBuffer<T> buffer() {
059                return buffer;
060        }
061
062        /**
063         * Return the data associated with this event 
064         * as {@link Buffer}. This is short for
065         * `buffer().backingBuffer()`.
066         *
067         * @return the data
068         */
069        public T data() {
070                return buffer.backingBuffer();
071        }
072        
073        /**
074         * Return the end of record flag passed to the constructor.
075         * The precise interpretation of a record depends on the data
076         * handled. 
077         * 
078         * @return the end of record flag
079         */
080        public boolean isEndOfRecord() {
081                return eor;
082        }
083        
084        /* (non-Javadoc)
085         * @see org.jgrapes.core.internal.EventBase#stopped()
086         */
087        @Override
088        protected void handled() {
089                buffer.unlockBuffer();
090                buffer = null;
091        }
092        
093        /* (non-Javadoc)
094         * @see java.lang.Object#toString()
095         */
096        @Override
097        public String toString() {
098                StringBuilder builder = new StringBuilder();
099                builder.append(Components.objectName(this));
100                builder.append(" [");
101                if (channels != null) {
102                        builder.append("channels=");
103                        builder.append(Channel.toString(channels));
104                }
105                builder.append(",size=");
106                Optional.ofNullable(buffer).map(
107                                b -> b.backingBuffer().position()).orElse(0);
108                builder.append(",eor=");
109                builder.append(eor);
110                builder.append("]");
111                return builder.toString();
112        }
113        
114        /**
115         * @return the result
116         * @see java.nio.Buffer#hasRemaining()
117         */
118        public final boolean hasRemaining() {
119                return buffer.hasRemaining();
120        }
121
122        /**
123         * @return the result
124         * @see java.nio.Buffer#remaining()
125         */
126        public final int remaining() {
127                return buffer.remaining();
128        }
129}
