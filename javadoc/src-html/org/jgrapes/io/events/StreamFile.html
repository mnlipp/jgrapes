019package org.jgrapes.io.events;
020
021import java.nio.file.OpenOption;
022import java.nio.file.Path;
023import java.util.Arrays;
024import org.jgrapes.core.Event;
025
026/**
027 * Causes the content of a file to be streamed as a sequence of {@link Output}
028 * events (terminated by an event with the end of record flag set) on the 
029 * channel that this event is fired on.
030 */
031public class StreamFile extends Event<Void> {
032
033    private final Path path;
034    private final OpenOption[] options;
035
036    /**
037     * Creates a new instance.
038     * 
039     * @param path the file's path
040     * @param options open options
041     */
042    public StreamFile(Path path, OpenOption... options) {
043        this.path = path;
044        this.options = Arrays.copyOf(options, options.length);
045    }
046
047    /**
048     * Returns the event's path.
049     * 
050     * @return the path
051     */
052    public Path path() {
053        return path;
054    }
055
056    /**
057     * Returns the event's options.
058     * 
059     * @return the options
060     */
061    public OpenOption[] options() {
062        return Arrays.copyOf(options, options.length);
063    }
064
065}
