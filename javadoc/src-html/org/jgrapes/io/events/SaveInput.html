019package org.jgrapes.io.events;
020
021import java.nio.file.OpenOption;
022import java.nio.file.Path;
023import java.util.Arrays;
024import org.jgrapes.core.Event;
025import org.jgrapes.io.FileStorage;
026
027/**
028 * Causes the {@link FileStorage} component to write the data from all
029 * {@link Input} events on the channel that this event is fired on to a file
030 * until an event with the end of record flag set is sent on the channel.
031 */
032public class SaveInput extends Event<Void> {
033
034    private final Path path;
035    private final OpenOption[] options;
036
037    /**
038     * Creates a new instance.
039     * 
040     * @param path the file's path
041     * @param options open options
042     */
043    public SaveInput(Path path, OpenOption... options) {
044        this.path = path;
045        this.options = Arrays.copyOf(options, options.length);
046    }
047
048    /**
049     * Returns the event's path.
050     * 
051     * @return the path
052     */
053    public Path path() {
054        return path;
055    }
056
057    /**
058     * Returns the event's open options
059     * 
060     * @return the options
061     */
062    public OpenOption[] options() {
063        return Arrays.copyOf(options, options.length);
064    }
065
066}
