060/**
061 * A queue based buffer pool. Using buffers from a pool is an important
062 * feature for limiting the computational resources for an {@link IOSubchannel}.
063 * A producer of {@link Output} events that simply creates its own buffers
064 * may produce and enqueue a large number of events that are not consumed
065 * as fast as they are produced. 
066 * 
067 * Using a buffer pool with a typical size of two synchronizes the 
068 * producer and the consumers of events nicely. The producer
069 * (thread) holds one buffer and fills it, the consumer (thread) holds 
070 * the other buffer and works with its content. If the producer finishes
071 * before the consumer, it has to stop until the consumer has processed 
072 * previous event and releases the buffer. The consumer can continue
073 * without delay, because the data has already been prepared and enqueued
074 * as the next event.
075 * 
076 * One of the biggest problems when using a pool can be to identify 
077 * leaking buffers, i.e. buffers that are not properly returned to the pool.
078 * This implementation therefore tracks all created buffers 
079 * (with a small overhead) and logs a warning if a buffer is no longer
080 * used (referenced) but has not been returned to the pool. If the
081 * log level for {@link ManagedBufferPool} is set to {@link Level#FINE},
082 * the warning also includes a stack trace of the call to {@link #acquire()}
083 * that handed out the buffer. Providing this information in addition 
084 * obviously requires a larger overhead and is therefore limited to the
085 * finer log levels.  
086 */
087public class ManagedBufferPool<W extends ManagedBuffer<T>, T extends Buffer>
088        implements BufferCollector {
089
090        protected static final Logger logger 
091                = Logger.getLogger(ManagedBufferPool.class.getName());
092        
093        private static long defaultDrainDelay = 1000;
094        
095        private String name = Components.objectName(this);

098        private Map<Integer,List<BufferMonitor>> monitoredBuffers = new HashMap<>();
099        private BlockingQueue<W> queue;
100        private int bufferSize = -1;
101        private int preservedBufs;
102        private int maximumBufs;
103        private AtomicInteger createdBufs;
104        private long drainDelay = -1;
105        private AtomicReference<Timer> idleTimer = new AtomicReference<>(null);

126        /**
127         * Create a pool that contains a varying number of (wrapped) buffers.
128         * The pool is initially empty. When buffers are requested and none 
129         * are left in the pool, new buffers are created up to the given 
130         * upper limit. Recollected buffers are put in the pool until it holds
131         * the number specified by the lower threshold. Any additional 
132         * recollected buffers are discarded. 
133         * 
134         * @param wrapper the function that converts buffers to managed buffers
135         * @param bufferFactory a function that creates a new buffer
136         * @param lowerThreshold the number of buffers kept in the pool
137         * @param upperLimit the maximum number of buffers
138         */
139        public ManagedBufferPool(BiFunction<T,BufferCollector, W> wrapper, 
140                        Supplier<T> bufferFactory, int lowerThreshold, int upperLimit) {
141                this.wrapper = wrapper;
142                this.bufferFactory = bufferFactory;
143                preservedBufs = lowerThreshold;
144                maximumBufs = upperLimit;
145                createdBufs = new AtomicInteger();
146                queue = new ArrayBlockingQueue<W>(lowerThreshold);
147                MBeanView.addPool(this);
148        }
149
150        /**
151         * Create a pool that keeps up to the given number of (wrapped) buffers
152         * in the pool and also uses that number as upper limit.
153         * 
154         * @param wrapper the function that converts buffers to managed buffers
155         * @param bufferFactory a function that creates a new buffer
156         * @param buffers the number of buffers
157         */
158        public ManagedBufferPool(BiFunction<T,BufferCollector, W> wrapper, 
159                        Supplier<T> bufferFactory, int buffers) {
160                this(wrapper, bufferFactory, buffers, buffers);
161        }

195        private W createBuffer() {
196                createdBufs.incrementAndGet();
197                W buffer = wrapper.apply(this.bufferFactory.get(), this);
198                BufferMonitor monitor = new BufferMonitor(buffer);
199                synchronized (monitoredBuffers) {
200                        monitoredBuffers.computeIfAbsent(System.identityHashCode(buffer), 
201                                        key -> new LinkedList<>()).add(monitor);
202                }
203                bufferSize = buffer.capacity();
204                return buffer;
205        }
206
207        private void removeBuffer(ManagedBuffer<?> buffer) {
208                createdBufs.decrementAndGet();
209                synchronized (monitoredBuffers) {
210                        int hashCode = System.identityHashCode(buffer);
211                        List<BufferMonitor> candidates = monitoredBuffers.get(hashCode);
212                        boolean found = false;
213                        if (candidates != null) {
214                                for (Iterator<BufferMonitor> itr = candidates.iterator(); 
215                                        itr.hasNext(); ) {
216                                        BufferMonitor monitor = itr.next();
217                                        if (buffer == monitor.get()) {
218                                                itr.remove();
219                                                found = true;
220                                                break;
221                                        }
222                                }
223                                if (candidates.isEmpty()) {
224                                        monitoredBuffers.remove(hashCode);
225                                }
226                        }
227                        if (!found) {
228                                logger.warning("Attempt to remove unknown buffer from pool.");
229                        }
230                }
231        }
232        
233        /**
234         * Returns the size of the buffers managed by this pool.
235         * 
236         * @return the buffer size
237         */
238        public int bufferSize() {
239                if (bufferSize < 0) {
240                        createBuffer().unlockBuffer();
241                }
242                return bufferSize;
243        }
244
245        /**
246         * Acquires a managed buffer from the pool. If the pool is empty,
247         * waits for a buffer to become available. The buffer has a lock count
248         * of one.
249         * 
250         * @return the acquired buffer
251         * @throws InterruptedException if the current thread is interrupted
252         */
253        public W acquire() throws InterruptedException {
254                if (createdBufs.get() < maximumBufs) {
255                        W buffer = queue.poll();
256                        if (buffer != null) {
257                                return buffer;
258                        }
259                        return createBuffer();
260                }
261                return queue.take();
262        }
263        
264        /**
265         * Re-adds the buffer to the pool. The buffer is cleared.
266         * 
267         * @see org.jgrapes.io.util.BufferCollector#recollect(org.jgrapes.io.util.ManagedBuffer)
268         */
269        @Override
270        public void recollect(ManagedBuffer<?> buffer) {
271                if (queue.size() < preservedBufs) {
272                        long effectiveDrainDelay 
273                                = drainDelay > 0 ? drainDelay : defaultDrainDelay;
274                        if (effectiveDrainDelay > 0) {
275                                // Enqueue
276                                buffer.clear();
277                                buffer.lockBuffer();
278                                @SuppressWarnings("unchecked")
279                                W buf = (W)buffer;
280                                queue.add(buf);
281                                Timer old = idleTimer.getAndSet(Components.schedule(this::drain, 
282                                                Duration.ofMillis(effectiveDrainDelay)));
283                                if (old != null) {
284                                        old.cancel();
285                                }
286                                return;
287                        }
288                }
289                // Discard
290                removeBuffer(buffer);
291        }
292        
293        private void drain(Timer timer) {
294                idleTimer.set(null);
295                while(true) {
296                        ManagedBuffer<?> buffer = queue.poll();
297                        if (buffer == null) {
298                                break;
299                        }
300                        removeBuffer(buffer);
301                }
302        }

311                if (queue != null) {

319        /**
320         * To identify buffers not returned to the pool, 
321         * all buffers handed out are tracked using this class, 
322         * essentially a weak reference. Instances are created for each 
323         * buffer and removed when the buffer is returned to the pool.
324         * 
325         * If the weak reference is collected by the garbage collector,
326         * this means that the buffer has not been returned. This condition
327         * is watched for by the {@link OrphanedCollector} and produces
328         * a warning.
329         */
330        private class BufferMonitor extends WeakReference<ManagedBuffer<?>> {
331
332                private StackTraceElement[] createdBy;
333                
334                public BufferMonitor(ManagedBuffer<?> referent) {
335                        super(referent, OrphanedCollector.orphanedBuffers);
336                        if (logger.isLoggable(Level.FINE)) {
337                                createdBy = Thread.currentThread().getStackTrace();
338                        }
339                }
340
341                public ManagedBufferPool<?, ?> manager() {
342                        return ManagedBufferPool.this;
343                }
344                
345                public StackTraceElement[] createdBy() {
346                        return createdBy;
347                }
348        }

350        private static class OrphanedCollector extends Thread {
351                
352                private static ReferenceQueue<ManagedBuffer<?>> orphanedBuffers 
353                        = new ReferenceQueue<>();
354                
355                public OrphanedCollector() {
356                        setName(ManagedBufferPool.class.getSimpleName() + ".Collector");
357                        setDaemon(true);
358                        start();
359                }
360
361                @Override
362                public void run() {
363                        while(true) {
364                                try {
365                                        @SuppressWarnings("rawtypes")
366                                        ManagedBufferPool.BufferMonitor monitor 
367                                                = (ManagedBufferPool.BufferMonitor)orphanedBuffers
368                                                        .remove();
369                                        ManagedBufferPool<?,?> mbp = monitor.manager();
370                                        // Managed buffer has not been properly recollected, fix.
371                                        mbp.monitoredBuffers.remove(monitor);
372                                        mbp.createdBufs.decrementAndGet();
373                                        // Create warning
374                                        if (logger.isLoggable(Level.WARNING)) {
375                                                final StringBuilder msg = new StringBuilder(
376                                                                "Orphaned buffer from pool " + mbp.name());
377                                                StackTraceElement[] st = monitor.createdBy();
378                                                if (st != null) {
379                                                        msg.append(", created");
380                                                        for (StackTraceElement e: st) {
381                                                                msg.append(System.lineSeparator());
382                                                                msg.append("\tat ");
383                                                                msg.append(e.toString());
384                                                        }
385                                                }
386                                                logger.warning(msg.toString());
387                                        }
388                                } catch (InterruptedException e) {
389                                        break;
390                                }
391                        }
392                }
393        }

399        /**
400         * An MBean interface for getting information about the managed
401         * buffer pools. Note that created buffer pools are tracked using
402         * weak references. Therefore, the MBean may report more pools than
403         * are really in use. 
404         */
405        public static interface ManagedBufferPoolMXBean {
406
407                /**
408                 * Information about a single managed pool.
409                 */
410                public static class PoolInfo {
411                        private int created;
412                        private int pooled;
413                        private int bufferSize;
414                        
415                        @ConstructorProperties({"created", "pooled", "bufferSize"})
416                        public PoolInfo(int created, int pooled, int bufferSize) {
417                                this.created = created;
418                                this.pooled = pooled;
419                                this.bufferSize = bufferSize;
420                        }
421
422                        /**
423                         * The number of buffers created by this pool.
424                         * 
425                         * @return the value
426                         */
427                        public int getCreated() {
428                                return created;
429                        }
430
431                        /**
432                         * The number of buffers pooled (ready to be acquired).
433                         * 
434                         * @return the value
435                         */
436                        public int getPooled() {
437                                return pooled;
438                        }
439
440                        /**
441                         * The size of the buffers in items.
442                         * 
443                         * @return the buffer size
444                         */
445                        public int getBufferSize() {
446                                return bufferSize;
447                        }
448                }
449
450                /**
451                 * Three views on the existing pool.
452                 */
453                public static class PoolInfos {
454                        private SortedMap<String,PoolInfo> allPools;
455                        private SortedMap<String,PoolInfo> nonEmptyPools;
456                        private SortedMap<String,PoolInfo> usedPools;
457
458                        public PoolInfos(Set<ManagedBufferPool<?, ?>> pools) {
459                                allPools = new TreeMap<>();
460                                nonEmptyPools = new TreeMap<>();
461                                usedPools = new TreeMap<>();
462                                
463                                Map<String, Integer> dupsNext = new HashMap<>();
464                                for (ManagedBufferPool<?,?> mbp: pools) {
465                                        String key = mbp.name();
466                                        PoolInfo qi = new PoolInfo(
467                                                        mbp.createdBufs.get(), mbp.queue.size(), mbp.bufferSize());
468                                        if (allPools.containsKey(key) || dupsNext.containsKey(key)) {
469                                                if (allPools.containsKey(key)) {
470                                                        // Found first duplicate, rename
471                                                        allPools.put(key + "#1", allPools.get(key));
472                                                        allPools.remove(key);
473                                                        dupsNext.put(key, 2);
474                                                }
475                                                allPools.put(key + "#"
476                                                                + (dupsNext.put(key, dupsNext.get(key) + 1)), qi);
477                                        } else {
478                                                allPools.put(key, qi);
479                                        }
480                                }
481                                for (Map.Entry<String,PoolInfo> e: allPools.entrySet()) {
482                                        PoolInfo qi = e.getValue();
483                                        if (qi.getPooled() > 0) {
484                                                nonEmptyPools.put(e.getKey(), qi);
485                                        }
486                                        if (qi.getCreated() > 0) {
487                                                usedPools.put(e.getKey(), qi);
488                                        }
489                                }
490                        }
491                        
492                        /**
493                         * All pools.
494                         */
495                        public SortedMap<String, PoolInfo> getAllPools() {
496                                return allPools;
497                        }
498
499                        /**
500                         * Pools that have at least managed buffer enqueued
501                         * (ready to be acquired).
502                         */
503                        public SortedMap<String, PoolInfo> getNonEmptyPools() {
504                                return nonEmptyPools;
505                        }
506
507                        /**
508                         * Pools that have at least one associated buffer
509                         * (in pool or in use).
510                         */
511                        public SortedMap<String, PoolInfo> getUsedPools() {
512                                return usedPools;
513                        }
514                }

552        private static class MBeanView implements ManagedBufferPoolMXBean {
553
554                private static Set<ManagedBufferPool<?,?>> allPools
555                        = Collections.synchronizedSet(
556                                Collections.newSetFromMap(
557                                                new WeakHashMap<ManagedBufferPool<?, ?>, Boolean>()));
558
559                public static void addPool(ManagedBufferPool<?, ?> pool) {
560                        allPools.add(pool);
561                }
562                
563                @Override
564                public void setDefaultDrainDelay(long millis) {
565                        ManagedBufferPool.setDefaultDrainDelay(millis);
566                }
567
568                @Override
569                public long getDefaultDrainDelay() {
570                        return ManagedBufferPool.defaultDrainDelay();
571                }
572
573                
574                @Override
575                public PoolInfos getPoolInfos() {
576                        return new PoolInfos(allPools);
577                }
578
579                @Override
580                public IntSummaryStatistics getPooledPerPoolStatistics() {
581                        return allPools.stream().collect(
582                                        Collectors.summarizingInt(mbp -> mbp.queue.size()));
583                }
584                
585                @Override
586                public IntSummaryStatistics getCreatedPerPoolStatistics() {
587                        return allPools.stream().collect(
588                                        Collectors.summarizingInt(mbp -> mbp.createdBufs.get()));
589                }
590        }
