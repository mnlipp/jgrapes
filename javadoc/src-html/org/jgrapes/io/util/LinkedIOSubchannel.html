019package org.jgrapes.io.util;
020
021import java.lang.ref.WeakReference;
022import java.nio.ByteBuffer;
023import java.util.Collections;
024import java.util.Map;
025import java.util.WeakHashMap;
026
027import org.jgrapes.core.Channel;
028import org.jgrapes.core.EventPipeline;
029import org.jgrapes.core.Manager;
030import org.jgrapes.core.internal.Common;
031import org.jgrapes.io.IOSubchannel;
032
033/**
034 * Provides an I/O subchannel that is linked to another I/O subchannel. A
035 * typical use case for this class is a protocol converter.
036 * 
037 * Protocol converters receive events related to an I/O resource from upstream,
038 * and while processing them usually generate new events to other components
039 * downstream (and vice versa). The events received are associated with a
040 * particular resource by the subchannel that is used to relay them. The
041 * association with the resource must be maintained for the newly generated
042 * events as well. It is, however, not possible to use the same subchannel for
043 * receiving from upstream and sending downstream because it wouldn't be
044 * possible to distinguish between e.g. an {@code Input} event from upstream to
045 * the converter and an {@code Input} event (conveying the converted data) from
046 * the converter to the downstream components.
047 * 
048 * Therefore, the converter must provide and manage independent subchannels for
049 * the data streams on its downstream side with a one-to-one relationship to the
050 * upstream subchannels. The {@code LinkedIOSubchannel} class simplifies this
051 * task. It provides a new subchannel with its own pipeline and a reference to
052 * an existing subchannel. This makes it easy to find the upstream subchannel
053 * for a given downstream ({@code LinkedIOSubchannel}) when processing response
054 * events. For finding the downstream {@code IOSubchannel} for a given upstream
055 * connection, the class maintains a mapping in a {@link WeakHashMap}.
056 * 
057 * Note that memory management is a bit special here. The 
058 * {@code LinkedIOSubchannel} should exist as long as the subchannel that
059 * it is linked to exists. This subchannel, however, does not have a reference
060 * to the {@code LinkedIOSubchannel}. Unless other references exist, the
061 * existence of the {@code LinkedIOSubchannel} is ensured only by the reference
062 * as value in the {@code WeakHashMap}. This reference will go away when
063 * the upstream subchannel (used as key) goes away.
064 * 
065 */
066public class LinkedIOSubchannel implements IOSubchannel {
067
068        private static final Map<IOSubchannel, LinkedIOSubchannel> 
069                reverseMap = Collections.synchronizedMap(new WeakHashMap<>());
070
071        private final Manager converterComponent;
072        // Must be weak, else there will always be a reference to the 
073        // upstream channel and, through the reverseMap, to this object.
074        private final WeakReference<IOSubchannel> upstreamChannel;
075        private final EventPipeline responsePipeline;
076
077        /**
078         * Creates a new {@code LinkedIOSubchannel} that links to the give I/O
079         * subchannel. Using this constructor is similar to invoking
080         * {@link #LinkedIOSubchannel(Manager, IOSubchannel, boolean)} with
081         * {@code true} as last parameter.
082         * 
083         * @param converterComponent
084         *            the converter component; used to get the main channel and the
085         *            new event pipeline
086         * @param upstreamChannel
087         *            the upstream channel
088         */
089        public LinkedIOSubchannel(Manager converterComponent,
090                IOSubchannel upstreamChannel) {
091                this(converterComponent, upstreamChannel, true);
092        }
093
094        /**
095         * Creates a new {@code LinkedIOSubchannel} for a given I/O subchannel.
096         * Using this constructor with {@code false} as last parameter prevents the
097         * addition of the linked I/O subchannel to the mapping from (upstream)
098         * subchannel to linked subchannel. This can save some space if a converter
099         * component has some other means to maintain that information. Addition to
100         * the map is thread safe.
101         * 
102         * @param converterComponent
103         *            the converter component; used to get the main channel and the
104         *            new event pipeline
105         * @param upstreamChannel
106         *            the upstream channel
107         * @param addToMap
108         *            add an entry in the map
109         */
110        public LinkedIOSubchannel(Manager converterComponent,
111                IOSubchannel upstreamChannel, boolean addToMap) {
112                super();
113                this.converterComponent = converterComponent;
114                this.upstreamChannel = new WeakReference<>(upstreamChannel);
115                responsePipeline = converterComponent.newEventPipeline();
116                if (addToMap) {
117                        reverseMap.put(upstreamChannel, this);
118                }
119        }
120
121        /**
122         * @return the converterComponent
123         */
124        public Manager converterComponent() {
125                return converterComponent;
126        }
127
128        /* (non-Javadoc)
129         * @see org.jgrapes.io.IOSubchannel#getMainChannel()
130         */
131        @Override
132        public Channel mainChannel() {
133                return converterComponent.channel();
134        }
135
136        /* (non-Javadoc)
137         * @see org.jgrapes.io.IOSubchannel#getResponsePipeline()
138         */
139        @Override
140        public EventPipeline responsePipeline() {
141                return responsePipeline;
142        }
143
144        /**
145         * @return the upstream channel
146         */
147        public IOSubchannel upstreamChannel() {
148                return upstreamChannel.get();
149        }
150
151        /**
152         * Delegates the invocation to the upstream channel.
153         * 
154         * @see org.jgrapes.io.IOSubchannel#bufferPool()
155         */
156        @Override
157        public ManagedBufferQueue<ManagedByteBuffer, ByteBuffer> bufferPool() {
158                IOSubchannel up = upstreamChannel.get();
159                return up == null ? null : up.bufferPool();
160        }
161
162        /* (non-Javadoc)
163         * @see java.lang.Object#toString()
164         */
165        public String toString() {
166                StringBuilder builder = new StringBuilder();
167                builder.append(IOSubchannel.toString(this));
168                builder.append(" (");
169                if (upstreamChannel != null) {
170                        builder.append("â€•>");
171                        builder.append(Common.channelToString(upstreamChannel.get()));
172                }
173                builder.append(")");
174                return builder.toString();
175        }
176        
177        /**
178         * Returns the linked subchannel that has been created for the given
179         * subchannel. If more than one linked subchannel has been created for a
180         * subchannel, the linked subchannel created last is returned.
181         * 
182         * @param channel
183         *            the channel
184         * @return the linked subchannel created for the given subchannel or
185         *         {@code null} if no such linked subchannel exists
186         */
187        public static LinkedIOSubchannel lookupLinked(IOSubchannel channel) {
188                return reverseMap.get(channel);
189        }
190}
