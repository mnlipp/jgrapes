019package org.jgrapes.io.util;
020
021import java.lang.ref.WeakReference;
022import java.nio.ByteBuffer;
023import java.util.Collections;
024import java.util.Map;
025import java.util.WeakHashMap;
026
027import org.jgrapes.core.Channel;
028import org.jgrapes.core.Components;
029import org.jgrapes.core.EventPipeline;
030import org.jgrapes.core.Manager;
031import org.jgrapes.core.internal.Common;
032import org.jgrapes.io.IOSubchannel;
033
034/**
035 * Provides an I/O subchannel that is linked to another I/O subchannel. A
036 * typical use case for this class is a protocol converter.
037 * <P>
038 * Protocol converters receive events related to an I/O resource from upstream,
039 * and while processing them usually generate new events to other components
040 * downstream (and vice versa). The events received are associated with a
041 * particular resource by the subchannel that is used to relay them. The
042 * association with the resource must be maintained for the newly generated
043 * events as well. It is, however, not possible to use the same subchannel for
044 * receiving from upstream and sending downstream because it wouldn't be
045 * possible to distinguish between e.g. an {@code Input} event from upstream to
046 * the converter and an {@code Input} event (conveying the converted data) from
047 * the converter to the downstream components.
048 * <P>
049 * Therefore, the converter must provide and manage independent subchannels for
050 * the data streams on its downstream side with a one-to-one relationship to the
051 * upstream subchannels. The {@code LinkedIOSubchannel} class simplifies this
052 * task. It provides a new subchannel with its own pipeline and a reference to
053 * an existing subchannel. This makes it easy to find the upstream subchannel
054 * for a given downstream ({@code LinkedIOSubchannel}) when processing response
055 * events. For finding the downstream {@code IOSubchannel} for a given upstream
056 * connection, the class maintains a mapping in a {@link WeakHashMap}.
057 * <P>
058 * Note that memory management is a bit special here. The 
059 * {@code LinkedIOSubchannel} should exist as long as the subchannel that
060 * it is linked to exists. This subchannel, however, does not have a reference
061 * to the {@code LinkedIOSubchannel}. Unless other references exist, the
062 * existence of the {@code LinkedIOSubchannel} is ensured only by the reference
063 * as value in the {@code WeakHashMap}. This reference will go away when
064 * the upstream subchannel (used as key) goes away.
065 * 
066 * @author Michael N. Lipp
067 */
068public class LinkedIOSubchannel implements IOSubchannel {
069
070        private static final Map<IOSubchannel, LinkedIOSubchannel> 
071                reverseMap = Collections.synchronizedMap(new WeakHashMap<>());
072
073        private final Manager converterComponent;
074        // Must be weak, else there will always be a reference to the 
075        // upstream channel and, through the reverseMap, to this object.
076        private final WeakReference<IOSubchannel> upstreamChannel;
077        private final EventPipeline responsePipeline;
078
079        /**
080         * Creates a new {@code LinkedIOSubchannel} that links to the give I/O
081         * subchannel. Using this constructor is similar to invoking
082         * {@link #LinkedIOSubchannel(Manager, IOSubchannel, boolean)} with
083         * {@code true} as last parameter.
084         * 
085         * @param converterComponent
086         *            the converter component; used to get the main channel and the
087         *            new event pipeline
088         * @param upstreamChannel
089         *            the upstream channel
090         */
091        public LinkedIOSubchannel(Manager converterComponent,
092                IOSubchannel upstreamChannel) {
093                this(converterComponent, upstreamChannel, true);
094        }
095
096        /**
097         * Creates a new {@code LinkedIOSubchannel} for a given I/O subchannel.
098         * Using this constructor with {@code false} as last parameter prevents the
099         * addition of the linked I/O subchannel to the mapping from (upstream)
100         * subchannel to linked subchannel. This can save some space if a converter
101         * component has some other means to maintain that information. Addition to
102         * the map is thread safe.
103         * 
104         * @param converterComponent
105         *            the converter component; used to get the main channel and the
106         *            new event pipeline
107         * @param upstreamChannel
108         *            the upstream channel
109         * @param addToMap
110         *            add an entry in the map
111         */
112        public LinkedIOSubchannel(Manager converterComponent,
113                IOSubchannel upstreamChannel, boolean addToMap) {
114                super();
115                this.converterComponent = converterComponent;
116                this.upstreamChannel = new WeakReference<>(upstreamChannel);
117                responsePipeline = converterComponent.newEventPipeline();
118                if (addToMap) {
119                        reverseMap.put(upstreamChannel, this);
120                }
121        }
122
123        /* (non-Javadoc)
124         * @see org.jgrapes.io.IOSubchannel#getMainChannel()
125         */
126        @Override
127        public Channel mainChannel() {
128                return converterComponent.channel();
129        }
130
131        /* (non-Javadoc)
132         * @see org.jgrapes.io.IOSubchannel#getResponsePipeline()
133         */
134        @Override
135        public EventPipeline responsePipeline() {
136                return responsePipeline;
137        }
138
139        /**
140         * @return the upstream channel
141         */
142        public IOSubchannel upstreamChannel() {
143                return upstreamChannel.get();
144        }
145
146        /**
147         * Delegates the invocation to the upstream channel.
148         * 
149         * @see org.jgrapes.io.IOSubchannel#bufferPool()
150         */
151        @Override
152        public ManagedBufferQueue<ManagedByteBuffer, ByteBuffer> bufferPool() {
153                IOSubchannel up = upstreamChannel.get();
154                return up == null ? null : up.bufferPool();
155        }
156
157        /* (non-Javadoc)
158         * @see java.lang.Object#toString()
159         */
160        public String toString() {
161                StringBuilder builder = new StringBuilder();
162                builder.append(Components.objectName(this));
163                builder.append("(");
164                        builder.append(Common.channelToString(mainChannel()));
165                builder.append(")");
166                builder.append(" [");
167                if (upstreamChannel != null) {
168                        builder.append("â€•>");
169                        builder.append(Common.channelToString(upstreamChannel.get()));
170                }
171                builder.append("]");
172                return builder.toString();
173        }
174        
175        /**
176         * Returns the linked subchannel that has been created for the given
177         * subchannel. If more than one linked subchannel has been created for a
178         * subchannel, the linked subchannel created last is returned.
179         * 
180         * @param channel
181         *            the channel
182         * @return the linked subchannel created for the given subchannel or
183         *         {@code null} if no such linked subchannel exists
184         */
185        public static LinkedIOSubchannel lookupLinked(IOSubchannel channel) {
186                return reverseMap.get(channel);
187        }
188}
