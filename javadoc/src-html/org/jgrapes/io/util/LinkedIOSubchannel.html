019package org.jgrapes.io.util;
020
021import java.lang.ref.WeakReference;
022import java.util.Optional;
023
024import org.jgrapes.core.Channel;
025import org.jgrapes.core.Manager;
026import org.jgrapes.io.IOSubchannel;
027import org.jgrapes.io.IOSubchannel.DefaultSubchannel;
028
029/**
030 * Provides an I/O subchannel that is linked to another I/O subchannel. A
031 * typical use case for this class is a protocol converter.
032 * 
033 * Protocol converters receive events related to an I/O resource from upstream,
034 * and while processing them usually generate new events to other components
035 * downstream (and vice versa). The events received are associated with a
036 * particular resource by the subchannel that is used to relay them. The
037 * association with the resource must be maintained for the newly generated
038 * events as well. It is, however, not possible to use the same subchannel for
039 * receiving from upstream and sending downstream because it wouldn't be
040 * possible to distinguish between e.g. an {@code Input} event from upstream to
041 * the converter and an {@code Input} event (conveying the converted data) from
042 * the converter to the downstream components.
043 * 
044 * Therefore, the converter must provide and manage independent subchannels for
045 * the data streams on its downstream side with a one-to-one relationship to the
046 * upstream subchannels. The {@code LinkedIOSubchannel} class simplifies this
047 * task. It provides a new subchannel with its own pipeline and a reference to
048 * an existing subchannel. This makes it easy to find the upstream subchannel
049 * for a given downstream ({@code LinkedIOSubchannel}) when processing response
050 * events. For finding the downstream {@code IOSubchannel} for a given upstream
051 * connection, instances associate themselves with the upstream channel using
052 * a special key.
053 * 
054 */
055public class LinkedIOSubchannel extends DefaultSubchannel {
056
057        private final Manager converterComponent;
058        // Must be weak, else there will always be a reference to the 
059        // upstream channel and, through the reverseMap, to this object.
060        private final WeakReference<IOSubchannel> upstreamChannel;
061        
062        /**
063         * Creates a new {@code LinkedIOSubchannel} that links to the give I/O
064         * subchannel. Using this constructor is similar to invoking
065         * {@link #LinkedIOSubchannel(Manager, IOSubchannel, boolean)} with
066         * {@code true} as last parameter.
067         * 
068         * @param converterComponent
069         *            the converter component; used to get the main channel and the
070         *            new event pipeline
071         * @param upstreamChannel
072         *            the upstream channel
073         */
074        public LinkedIOSubchannel(Manager converterComponent,
075                IOSubchannel upstreamChannel) {
076                this(converterComponent, upstreamChannel, true);
077        }
078
079        /**
080         * Creates a new {@code LinkedIOSubchannel} for a given I/O subchannel.
081         * Using this constructor with {@code false} as last parameter prevents the
082         * addition of the back link from the upstream channel to the downstream
083         * channel (see {@link #downstreamChannel(Manager, IOSubchannel)}). 
084         * This can save some space if a converter component has some other 
085         * means to maintain that information. Addition to
086         * the map is thread safe.
087         * 
088         * @param converterComponent
089         *            the converter component; used to get the main channel and the
090         *            new event pipeline
091         * @param upstreamChannel
092         *            the upstream channel
093         * @param linkBack
094         *            create the link from upstream to downstream
095         */
096        public LinkedIOSubchannel(Manager converterComponent,
097                IOSubchannel upstreamChannel, boolean linkBack) {
098                super(converterComponent);
099                this.converterComponent = converterComponent;
100                this.upstreamChannel = new WeakReference<>(upstreamChannel);
101                if (linkBack) {
102                        upstreamChannel.setAssociated(
103                                        new KeyWrapper(converterComponent), this);
104                }
105        }
106
107        /**
108         * @return the converterComponent
109         */
110        public Manager converterComponent() {
111                return converterComponent;
112        }
113
114        /**
115         * @return the upstream channel
116         */
117        public IOSubchannel upstreamChannel() {
118                return upstreamChannel.get();
119        }
120        
121        /**
122         * Delegates the invocation to the upstream channel 
123         * if no associated data is found for this channel. 
124         */
125        @Override
126        public <V> Optional<V> associated(Object by, Class<V> type) {
127                Optional<V> result = super.associated(by, type);
128                if (!result.isPresent()) {
129                        IOSubchannel up = upstreamChannel();
130                        if (up != null) {
131                                return up.associated(by, type);
132                        }
133                }
134                return result;
135        }
136
137        /* (non-Javadoc)
138         * @see java.lang.Object#toString()
139         */
140        public String toString() {
141                StringBuilder builder = new StringBuilder();
142                builder.append(IOSubchannel.toString(this));
143                builder.append(" (");
144                if (upstreamChannel != null) {
145                        builder.append("â€•>");
146                        builder.append(Channel.toString(upstreamChannel.get()));
147                }
148                builder.append(")");
149                return builder.toString();
150        }
151        
152        /**
153         * Returns the linked downstream channel that has been created for the 
154         * given converter and (upstream) subchannel. If more than one linked 
155         * subchannel has been created for a given converter and subchannel, 
156         * the linked subchannel created last is returned.
157         * 
158         * @param upstreamChannel the (upstream) channel
159         * @return the linked downstream subchannel created for the 
160         * given cobverter and (upstream) subchannel if it exists
161         */
162        public static Optional<? extends LinkedIOSubchannel> downstreamChannel(
163                        Manager converterComponent, IOSubchannel upstreamChannel) {
164                return upstreamChannel.associated(
165                                new KeyWrapper(converterComponent), LinkedIOSubchannel.class);
166        }
167        
168        private static class KeyWrapper {
169
170                private Manager converterComponent;
171
172                /**
173                 * @param converterComponent
174                 */
175                public KeyWrapper(Manager converterComponent) {
176                        super();
177                        this.converterComponent = converterComponent;
178                }
179
180                /* (non-Javadoc)
181                 * @see java.lang.Object#hashCode()
182                 */
183                @Override
184                public int hashCode() {
185                        final int prime = 31;
186                        int result = 1;
187                        result = prime * result + ((converterComponent == null) ? 0
188                                : converterComponent.hashCode());
189                        return result;
190                }
191
192                /* (non-Javadoc)
193                 * @see java.lang.Object#equals(java.lang.Object)
194                 */
195                @Override
196                public boolean equals(Object obj) {
197                        if (this == obj) {
198                                return true;
199                        }
200                        if (obj == null) {
201                                return false;
202                        }
203                        if (getClass() != obj.getClass()) {
204                                return false;
205                        }
206                        KeyWrapper other = (KeyWrapper) obj;
207                        if (converterComponent == null) {
208                                if (other.converterComponent != null) {
209                                        return false;
210                                }
211                        } else if (!converterComponent.equals(other.converterComponent)) {
212                                return false;
213                        }
214                        return true;
215                }
216        }
217}
