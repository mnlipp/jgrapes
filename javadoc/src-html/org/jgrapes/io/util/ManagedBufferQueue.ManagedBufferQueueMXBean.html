058/**
059 * A queue based buffer pool.
060 */
061public class ManagedBufferQueue<W extends ManagedBuffer<T>, T extends Buffer>
062        implements BufferCollector {
063
064        protected static final Logger logger 
065                = Logger.getLogger(ManagedBufferQueue.class.getName());
066        
067        private static long defaultDrainDelay = 1000;
068        private static Set<ManagedBufferQueue<?,?>> allQueues
069                = Collections.synchronizedSet(
070                                Collections.newSetFromMap(
071                                                new WeakHashMap<ManagedBufferQueue<?, ?>, Boolean>()));
072        private static ReferenceQueue<ManagedBuffer<?>> orphanedBuffers 
073                = new ReferenceQueue<>();
074        
075        private String name = Components.objectName(this);
076        private BiFunction<T, BufferCollector,W> wrapper = null;
077        private Supplier<T> bufferFactory = null;
078        private List<BufferMonitor> monitoredBuffers
079                = Collections.synchronizedList(new LinkedList<>());
080        private BlockingQueue<W> queue;
081        private int bufferSize = -1;
082        private int preservedBufs;
083        private int maximumBufs;
084        private AtomicInteger createdBufs;
085        private long drainDelay = -1;
086        private AtomicReference<Timer> idleTimer = new AtomicReference<>(null);

107        /**
108         * Create a pool that contains a varying number of (wrapped) buffers.
109         * The queue is initially empty. When buffers are requested and none 
110         * are left in the queue, new buffers are created up to the given 
111         * upper limit. Recollected buffers are put in the queue until it holds
112         * the number specified by the lower threshold. Any additional 
113         * recollected buffers are discarded. 
114         * 
115         * @param wrapper the function that converts buffers to managed buffers
116         * @param bufferFactory a function that creates a new buffer
117         * @param lowerThreshold the number of buffers kept in the queue
118         * @param upperLimit the maximum number of buffers
119         */
120        public ManagedBufferQueue(BiFunction<T,BufferCollector, W> wrapper, 
121                        Supplier<T> bufferFactory, int lowerThreshold, int upperLimit) {
122                this.wrapper = wrapper;
123                this.bufferFactory = bufferFactory;
124                preservedBufs = lowerThreshold;
125                maximumBufs = upperLimit;
126                createdBufs = new AtomicInteger();
127                queue = new ArrayBlockingQueue<W>(lowerThreshold);
128                allQueues.add(this);
129        }
130
131        /**
132         * Create a pool that keeps up to the given number of (wrapped) buffers
133         * in the queue and also uses that number as upper limit.
134         * 
135         * @param wrapper the function that converts buffers to managed buffers
136         * @param bufferFactory a function that creates a new buffer
137         * @param buffers the number of buffers
138         */
139        public ManagedBufferQueue(BiFunction<T,BufferCollector, W> wrapper, 
140                        Supplier<T> bufferFactory, int buffers) {
141                this(wrapper, bufferFactory, buffers, buffers);
142        }

166         * the queue.
167         * 
168         * @param delay the delay

171        public ManagedBufferQueue<W, T> setDrainDelay(long delay) {
172                this.drainDelay = delay;
173                return this;
174        }

176        private W createBuffer() {
177                createdBufs.incrementAndGet();
178                W buffer = wrapper.apply(this.bufferFactory.get(), this);
179                monitoredBuffers.add(new BufferMonitor(buffer));
180                bufferSize = buffer.capacity();
181                return buffer;
182        }
183
184        private void removeBuffer(ManagedBuffer<?> buffer) {
185                createdBufs.decrementAndGet();
186                for (Iterator<BufferMonitor> itr = monitoredBuffers.iterator(); 
187                                itr.hasNext(); ) {
188                        BufferMonitor monitor = itr.next();
189                        if (buffer.equals(monitor.get())) {
190                                itr.remove();
191                                break;
192                        }
193                }
194        }
195        
196        /**
197         * Returns the size of the buffers managed by this queue.
198         * 
199         * @return the buffer size
200         */
201        public int bufferSize() {
202                if (bufferSize < 0) {
203                        createBuffer().unlockBuffer();
204                }
205                return bufferSize;
206        }
207
208        /**
209         * Acquires a managed buffer from the pool. If the pool is empty,
210         * waits for a buffer to become available. The buffer has a lock count
211         * of one.
212         * 
213         * @return the acquired buffer
214         * @throws InterruptedException if the current thread is interrupted
215         */
216        public W acquire() throws InterruptedException {
217                if (createdBufs.get() < maximumBufs) {
218                        W buffer = queue.poll();
219                        if (buffer != null) {
220                                return buffer;
221                        }
222                        return createBuffer();
223                }
224                return queue.take();
225        }
226        
227        /**
228         * Re-adds the buffer to the pool. The buffer is cleared.
229         * 
230         * @see org.jgrapes.io.util.BufferCollector#recollect(org.jgrapes.io.util.ManagedBuffer)
231         */
232        @Override
233        public void recollect(ManagedBuffer<?> buffer) {
234                if (queue.size() < preservedBufs) {
235                        long effectiveDrainDelay 
236                                = drainDelay > 0 ? drainDelay : defaultDrainDelay;
237                        if (effectiveDrainDelay > 0) {
238                                // Enqueue
239                                buffer.clear();
240                                buffer.lockBuffer();
241                                @SuppressWarnings("unchecked")
242                                W buf = (W)buffer;
243                                queue.add(buf);
244                                Timer old = idleTimer.getAndSet(Components.schedule(this::drain, 
245                                                Duration.ofMillis(effectiveDrainDelay)));
246                                if (old != null) {
247                                        old.cancel();
248                                }
249                                return;
250                        }
251                }
252                // Discard
253                removeBuffer(buffer);
254        }
255        
256        private void drain(Timer timer) {
257                idleTimer.set(null);
258                while(true) {
259                        ManagedBuffer<?> buffer = queue.poll();
260                        if (buffer == null) {
261                                break;
262                        }
263                        removeBuffer(buffer);
264                }
265        }

267        /* (non-Javadoc)
268         * @see java.lang.Object#toString()
269         */
270        @Override
271        public String toString() {
272                StringBuilder builder = new StringBuilder();
273                builder.append("ManagedBufferQueue [");
274                if (queue != null) {
275                        builder.append("queue=");
276                        builder.append(queue);
277                }
278                builder.append("]");
279                return builder.toString();
280        }

282        private class BufferMonitor extends WeakReference<ManagedBuffer<?>> {
283
284                private StackTraceElement[] createdBy;
285                
286                public BufferMonitor(ManagedBuffer<?> referent) {
287                        super(referent, orphanedBuffers);
288                        if (logger.isLoggable(Level.FINE)) {
289                                createdBy = Thread.currentThread().getStackTrace();
290                        }
291                }
292
293                public ManagedBufferQueue<?, ?> manager() {
294                        return ManagedBufferQueue.this;
295                }
296                
297                public StackTraceElement[] createdBy() {
298                        return createdBy;
299                }
300        }

302        private static class OrphanedCollector extends Thread {
303                
304                public OrphanedCollector() {
305                        setName(ManagedBufferQueue.class.getSimpleName() + ".Collector");
306                        setDaemon(true);
307                        start();
308                }
309
310                @Override
311                public void run() {
312                        while(true) {
313                                try {
314                                        @SuppressWarnings("rawtypes")
315                                        ManagedBufferQueue.BufferMonitor monitor 
316                                                = (ManagedBufferQueue.BufferMonitor)orphanedBuffers
317                                                        .remove();
318                                        ManagedBufferQueue<?,?> mbq = monitor.manager();
319                                        // Managed buffer has not been properly recollected, fix.
320                                        mbq.monitoredBuffers.remove(monitor);
321                                        mbq.createdBufs.decrementAndGet();
322                                        // Create warning
323                                        if (logger.isLoggable(Level.WARNING)) {
324                                                final StringBuilder msg = new StringBuilder(
325                                                                "Orphaned buffer from queue " + mbq.name());
326                                                StackTraceElement[] st = monitor.createdBy();
327                                                if (st != null) {
328                                                        msg.append(", created");
329                                                        for (StackTraceElement e: st) {
330                                                                msg.append(System.lineSeparator());
331                                                                msg.append("\tat ");
332                                                                msg.append(e.toString());
333                                                        }
334                                                }
335                                                logger.warning(msg.toString());
336                                        }
337                                } catch (InterruptedException e) {
338                                        break;
339                                }
340                        }
341                }
342        }

358                        private int created;
359                        private int queued;
360                        private int bufferSize;
361                        
362                        @ConstructorProperties({"created", "queued", "bufferSize"})
363                        public QueueInfo(int created, int queued, int bufferSize) {
364                                this.created = created;
365                                this.queued = queued;
366                                this.bufferSize = bufferSize;
367                        }

380                         * 
381                         * @return the value

385                        }
386

390                         * @return

397                /**
398                 * Three views on the existing queues.
399                 */
400                public static class QueueInfos {
401                        private SortedMap<String,QueueInfo> allQueues;
402                        private SortedMap<String,QueueInfo> queuingQueues;
403                        private SortedMap<String,QueueInfo> nonEmptyQueues;
404
405                        public QueueInfos(Set<ManagedBufferQueue<?, ?>> queues) {
406                                allQueues = new TreeMap<>();
407                                queuingQueues = new TreeMap<>();
408                                nonEmptyQueues = new TreeMap<>();
409                                
410                                Map<String, Integer> dupsNext = new HashMap<>();
411                                for (ManagedBufferQueue<?,?> mbq: queues) {
412                                        String key = mbq.name();
413                                        QueueInfo qi = new QueueInfo(
414                                                        mbq.createdBufs.get(), mbq.queue.size(), mbq.bufferSize());
415                                        if (allQueues.containsKey(key) || dupsNext.containsKey(key)) {
416                                                if (allQueues.containsKey(key)) {
417                                                        // Found first duplicate, rename
418                                                        allQueues.put(key + "#1", allQueues.get(key));
419                                                        allQueues.remove(key);
420                                                        dupsNext.put(key, 2);
421                                                }
422                                                allQueues.put(key + "#"
423                                                                + (dupsNext.put(key, dupsNext.get(key) + 1)), qi);
424                                        } else {
425                                                allQueues.put(key, qi);
426                                        }
427                                }
428                                for (Map.Entry<String,QueueInfo> e: allQueues.entrySet()) {
429                                        QueueInfo qi = e.getValue();
430                                        if (qi.getQueued() > 0) {
431                                                queuingQueues.put(e.getKey(), qi);
432                                        }
433                                        if (qi.getCreated() > 0) {
434                                                nonEmptyQueues.put(e.getKey(), qi);
435                                        }
436                                }
437                        }
438                        
439                        /**
440                         * All queues.
441                         */
442                        public SortedMap<String, QueueInfo> getAllQueues() {
443                                return allQueues;
444                        }
445
446                        /**
447                         * Queues that have at least managed buffer enqueued
448                         * (ready to be acquired).
449                         */
450                        public SortedMap<String, QueueInfo> getQueuingQueues() {
451                                return queuingQueues;
452                        }
453
454                        /**
455                         * Queues that have at least one associated buffer
456                         * (enqueued or in use).
457                         */
458                        public SortedMap<String, QueueInfo> getNonEmptyQueues() {
459                                return nonEmptyQueues;
460                        }
461                }
