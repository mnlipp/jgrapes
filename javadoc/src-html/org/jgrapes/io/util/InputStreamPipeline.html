019package org.jgrapes.io.util;
020
021import java.io.IOException;
022import java.io.InputStream;
023import java.nio.ByteBuffer;
024import java.nio.channels.Channels;
025import java.nio.channels.ReadableByteChannel;
026import org.jgrapes.core.EventPipeline;
027import org.jgrapes.io.IOSubchannel;
028import org.jgrapes.io.events.Close;
029import org.jgrapes.io.events.IOError;
030import org.jgrapes.io.events.Output;
031
032/**
033 * Forwards the content of an input stream as a sequence of {@link Output}
034 * events.
035 */
036@SuppressWarnings("PMD.DataflowAnomalyAnalysis")
037public class InputStreamPipeline implements Runnable {
038
039    private InputStream inStream;
040    private IOSubchannel channel;
041    private EventPipeline eventPipeline;
042    private boolean sendClose = true;
043
044    /**
045     * Creates a new pipeline that sends the data from the given input stream
046     * as events on the given channel, using the given event pipeline.
047     * 
048     * @param in the input stream to read from
049     * @param channel the channel to send to
050     * @param eventPipeline
051     *            the event pipeline used for firing events
052     */
053    @SuppressWarnings("PMD.ShortVariable")
054    public InputStreamPipeline(InputStream in, IOSubchannel channel,
055            EventPipeline eventPipeline) {
056        this.inStream = in;
057        this.channel = channel;
058        this.eventPipeline = eventPipeline;
059    }
060
061    /**
062     * Creates a new pipeline that sends the data from the given input stream
063     * as events on the given channel, using the channel's response pipeline.
064     * 
065     * @param in the input stream to read from
066     * @param channel the channel to send to
067     */
068    @SuppressWarnings("PMD.ShortVariable")
069    public InputStreamPipeline(InputStream in, IOSubchannel channel) {
070        this(in, channel, channel.responsePipeline());
071    }
072
073    /**
074     * Suppresses the sending of a close event when the stream is closed. 
075     * 
076     * @return the stream for easy chaining
077     */
078    public InputStreamPipeline suppressClose() {
079        sendClose = false;
080        return this;
081    }
082
083    @Override
084    public void run() {
085        try (ReadableByteChannel inChannel = Channels.newChannel(inStream)) {
086            ManagedBuffer<ByteBuffer> lookAhead = ManagedBuffer.wrap(
087                ByteBuffer.allocate(channel.byteBufferPool().bufferSize()));
088            // First attempt
089            if (lookAhead.fillFromChannel(inChannel) == -1) {
090                ManagedBuffer<ByteBuffer> buffer
091                    = channel.byteBufferPool().acquire();
092                eventPipeline.fire(Output.fromSink(buffer, true), channel);
093            } else {
094                while (true) {
095                    // Save data read so far
096                    ManagedBuffer<ByteBuffer> buffer
097                        = channel.byteBufferPool().acquire();
098                    buffer.linkBackingBuffer(lookAhead);
099                    // Get new look ahead
100                    lookAhead = ManagedBuffer.wrap(ByteBuffer.allocate(
101                        channel.byteBufferPool().bufferSize()));
102                    // Next read attempt
103                    boolean eof;
104                    try {
105                        eof = lookAhead.fillFromChannel(inChannel) == -1;
106                    } catch (IOException e) {
107                        buffer.unlockBuffer();
108                        throw e;
109                    }
110                    // Fire "old" data with up-to-date end of record flag.
111                    eventPipeline.fire(Output.fromSink(buffer, eof), channel);
112                    if (eof) {
113                        break;
114                    }
115                }
116            }
117            if (sendClose) {
118                eventPipeline.fire(new Close(), channel);
119            }
120        } catch (InterruptedException e) {
121            // Just stop
122        } catch (IOException e) {
123            eventPipeline.fire(new IOError(null, e), channel);
124        }
125    }
126}
