001package org.jgrapes.io.util;
002
003import java.io.IOException;
004import java.io.InputStream;
005import java.nio.channels.Channels;
006import java.nio.channels.ReadableByteChannel;
007
008import org.jgrapes.core.EventPipeline;
009import org.jgrapes.io.IOSubchannel;
010import org.jgrapes.io.events.Closed;
011import org.jgrapes.io.events.IOError;
012import org.jgrapes.io.events.Output;
013
014/**
015 * Forwards the content of an input stream as a sequence of {@link Output}
016 * events.
017 */
018public class InputStreamPipeline implements Runnable {
019        
020        private InputStream inStream;
021        private IOSubchannel channel;
022        private EventPipeline eventPipeline;
023        private boolean sendClose = true;
024
025        /**
026         * Creates a new pipeline that sends the data from the given input stream
027         * as events on the given channel, using the given event pipeline.
028         * 
029         * @param in the input stream to read from
030         * @param channel the channel to send to
031         * @param eventPipeline
032         *            the event pipeline used for firing events
033         */
034        public InputStreamPipeline(InputStream in, IOSubchannel channel,
035                        EventPipeline eventPipeline) {
036                this.inStream = in;
037                this.channel = channel;
038                this.eventPipeline = eventPipeline;
039        }
040
041        /**
042         * Creates a new pipeline that sends the data from the given input stream
043         * as events on the given channel, using the channel's response pipeline.
044         * 
045         * @param in the input stream to read from
046         * @param channel the channel to send to
047         */
048        public InputStreamPipeline(InputStream in, IOSubchannel channel) {
049                this(in, channel, channel.responsePipeline());
050        }
051        
052        /**
053         * Suppresses the sending of a close event when the stream is closed. 
054         * 
055         * @return the stream for easy chaining
056         */
057        public InputStreamPipeline suppressClose() {
058                sendClose = false;
059                return this;
060        }
061        
062        
063        @Override
064        public void run() {
065                // Reading from stream
066                try (ReadableByteChannel inChannel = Channels.newChannel(inStream)) {
067                        while (true) {
068                                ManagedByteBuffer buffer 
069                                        = channel.byteBufferPool().acquire();
070                                int read;
071                                try {
072                                        read = inChannel.read(buffer.backingBuffer());
073                                } catch (IOException e) {
074                                        // Read failed, release buffer
075                                        buffer.unlockBuffer();
076                                        throw e;
077                                }
078                                boolean eof = (read == -1);
079                                eventPipeline.fire(new Output<>(buffer, eof), channel);
080                                if (eof) {
081                                        break;
082                                }
083                        }
084                        if (sendClose) {
085                                eventPipeline.fire(new Closed(), channel);
086                        }
087                } catch (InterruptedException e) {
088                        // Just stop
089                } catch (IOException e) {
090                        eventPipeline.fire(new IOError(null, e), channel);
091                }
092        }
093}
