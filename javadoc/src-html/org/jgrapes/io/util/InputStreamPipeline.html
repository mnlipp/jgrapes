001package org.jgrapes.io.util;
002
003import java.io.IOException;
004import java.io.InputStream;
005import java.nio.channels.Channels;
006import java.nio.channels.ReadableByteChannel;
007
008import org.jgrapes.io.IOSubchannel;
009import org.jgrapes.io.events.Closed;
010import org.jgrapes.io.events.IOError;
011import org.jgrapes.io.events.Output;
012
013/**
014 * Forwards the content of an input stream as a sequence of {@link Output}
015 * events.
016 */
017public class InputStreamPipeline implements Runnable {
018        
019        private InputStream inStream;
020        private IOSubchannel channel;
021
022        /**
023         * Creates a new pipeline that sends the data from the given input stream
024         * as events on thegiven channel.
025         * 
026         * @param in the input stream to read from
027         * @param channel the channel to send to
028         */
029        public InputStreamPipeline(InputStream in, IOSubchannel channel) {
030                this.inStream = in;
031                this.channel = channel;
032        }
033
034        @Override
035        public void run() {
036                // Reading from stream
037                try (ReadableByteChannel inChannel = Channels.newChannel(inStream)) {
038                        while (true) {
039                                ManagedByteBuffer buffer 
040                                        = channel.byteBufferPool().acquire();
041                                int read = inChannel.read(buffer.backingBuffer());
042                                boolean eof = (read == -1);
043                                channel.respond(new Output<>(buffer, eof));
044                                if (eof) {
045                                        break;
046                                }
047                        }
048                        channel.respond(new Closed());
049                } catch (InterruptedException e) {
050                        // Just stop
051                } catch (IOException e) {
052                        channel.respond(new IOError(null, e));
053                }
054        }
055}
