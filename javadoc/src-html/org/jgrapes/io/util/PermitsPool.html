019package org.jgrapes.io.util;
020
021import java.lang.ref.WeakReference;
022import java.util.ArrayList;
023import java.util.Iterator;
024import java.util.LinkedList;
025import java.util.List;
026
027/**
028 * A class that manages a set of permits and notifies listeners
029 * about changes of availability.
030 * 
031 * Listeners are added using {@link WeakReference}, so removing
032 * them isn't strictly necessary.
033 */
034public class PermitsPool {
035
036        private int permits;
037        private List<WeakReference<AvailabilityListener>> 
038                listeners = new LinkedList<>();
039        private boolean lastNotification = true;
040        
041        /**
042         * @param permits
043         */
044        public PermitsPool(int permits) {
045                this.permits = permits;
046        }
047
048        /**
049         * Adds an AvailabilityListener.
050         * 
051         * @param listener the AvailabilityListener
052         */
053        public void addListener(AvailabilityListener listener) {
054                synchronized (listeners) {
055                        listeners.add(new WeakReference<>(listener));
056                }
057        }
058
059        /**
060         * @param listener the AvailabilityListener
061         */
062        public void removeListener(AvailabilityListener listener) {
063                synchronized (listeners) {
064                        for (Iterator<WeakReference<AvailabilityListener>> 
065                                iter = listeners.iterator(); iter.hasNext();) {
066                                WeakReference<AvailabilityListener> item = iter.next();
067                                if (item.get() == null || item.get() == listener) {
068                                        iter.remove();
069                                }
070                        }
071                }
072        }
073
074        private void notifyAvailabilityListeners() {
075                boolean available = (permits > 0);
076                if (available == lastNotification) {
077                        return;
078                }
079                lastNotification = available;
080                List<AvailabilityListener> copy = new ArrayList<>();
081                synchronized (listeners) {
082                        for (Iterator<WeakReference<AvailabilityListener>> 
083                                iter = listeners.iterator(); iter.hasNext();) {
084                                WeakReference<AvailabilityListener> item = iter.next();
085                                AvailabilityListener listener = item.get();
086                                if (listener == null) {
087                                        iter.remove();
088                                        continue;
089                                }
090                                copy.add(listener);
091                        }
092                }
093                for (AvailabilityListener l: copy) {
094                        l.availabilityChanged(this, available);
095                }
096        }
097
098        /**
099         * Release a previously obtained permit.
100         */
101        public synchronized void release() {
102                permits += 1;
103                notifyAvailabilityListeners();
104        }
105
106        /**
107         * Try to acquire a permit
108         * 
109         * @return `true` if successful
110         */
111        public synchronized boolean tryAcquire() {
112                if (permits > 0) {
113                        permits -= 1;
114                        notifyAvailabilityListeners();
115                        return true;
116                }
117                return false;
118        }
119
120}
