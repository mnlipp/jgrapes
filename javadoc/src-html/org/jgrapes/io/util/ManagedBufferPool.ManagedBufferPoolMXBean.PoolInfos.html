064/**
065 * A queue based buffer pool. Using buffers from a pool is an important
066 * feature for limiting the computational resources for an {@link IOSubchannel}.
067 * A producer of {@link Output} events that simply creates its own buffers
068 * may produce and enqueue a large number of events that are not consumed
069 * as fast as they are produced. 
070 * 
071 * Using a buffer pool with a typical size of two synchronizes the 
072 * producer and the consumers of events nicely. The producer
073 * (thread) holds one buffer and fills it, the consumer (thread) holds 
074 * the other buffer and works with its content. If the producer finishes
075 * before the consumer, it has to stop until the consumer has processed 
076 * previous event and releases the buffer. The consumer can continue
077 * without delay, because the data has already been prepared and enqueued
078 * as the next event.
079 * 
080 * One of the biggest problems when using a pool can be to identify 
081 * leaking buffers, i.e. buffers that are not properly returned to the pool.
082 * This implementation therefore tracks all created buffers 
083 * (with a small overhead) and logs a warning if a buffer is no longer
084 * used (referenced) but has not been returned to the pool. If the
085 * log level for {@link ManagedBufferPool} is set to {@link Level#FINE},
086 * the warning also includes a stack trace of the call to {@link #acquire()}
087 * that handed out the buffer. Providing this information in addition 
088 * obviously requires a larger overhead and is therefore limited to the
089 * finer log levels.
090 *
091 * @param <W> the type of the wrapped (managed) buffer
092 * @param <T> the type of the content buffer that is wrapped
093 */
094public class ManagedBufferPool<W extends ManagedBuffer<T>, T extends Buffer>
095        implements BufferCollector<W> {
096

134        /**
135         * Create a pool that contains a varying number of (wrapped) buffers.
136         * The pool is initially empty. When buffers are requested and none 
137         * are left in the pool, new buffers are created up to the given 
138         * upper limit. Recollected buffers are put in the pool until it holds
139         * the number specified by the lower threshold. Any additional 
140         * recollected buffers are discarded. 
141         * 
142         * @param wrapper the function that converts buffers to managed buffers
143         * @param bufferFactory a function that creates a new buffer
144         * @param lowerThreshold the number of buffers kept in the pool
145         * @param upperLimit the maximum number of buffers
146         */
147        public ManagedBufferPool(BiFunction<T,BufferCollector<W>, W> wrapper, 
148                        Supplier<T> bufferFactory, int lowerThreshold, int upperLimit) {
149                this.wrapper = wrapper;
150                this.bufferFactory = bufferFactory;
151                preservedBufs = lowerThreshold;
152                maximumBufs = upperLimit;
153                createdBufs = new AtomicInteger();
154                queue = new ArrayBlockingQueue<W>(lowerThreshold);
155                bufferMonitor = new BufferMonitor(upperLimit);
156                MBeanView.addPool(this);
157        }
158
159        /**
160         * Create a pool that keeps up to the given number of (wrapped) buffers
161         * in the pool and also uses that number as upper limit.
162         * 
163         * @param wrapper the function that converts buffers to managed buffers
164         * @param bufferFactory a function that creates a new buffer
165         * @param buffers the number of buffers
166         */
167        public ManagedBufferPool(BiFunction<T,BufferCollector<W>, W> wrapper, 
168                        Supplier<T> bufferFactory, int buffers) {
169                this(wrapper, bufferFactory, buffers, buffers);
170        }

204        private W createBuffer() {
205                createdBufs.incrementAndGet();
206                W buffer = wrapper.apply(this.bufferFactory.get(), this);
207                bufferMonitor.put(buffer, new BufferProperties());
208                bufferSize = buffer.capacity();
209                return buffer;
210        }
211
212        /**
213         * Removes the buffer from the pool.
214         * 
215         * @param buffer the buffer to remove
216         */
217        private void removeBuffer(W buffer) {
218                createdBufs.decrementAndGet();
219                if (bufferMonitor.remove(buffer) == null) {
220                        if (logger.isLoggable(Level.FINE)) {
221                                logger.log(Level.WARNING, 
222                                                "Attempt to remove unknown buffer from pool.",
223                                                new Throwable());
224                        } else {
225                                logger.warning("Attempt to remove unknown buffer from pool.");
226                        }
227                }
228        }
229        
230        /**
231         * Returns the size of the buffers managed by this pool.
232         * 
233         * @return the buffer size
234         */
235        public int bufferSize() {
236                if (bufferSize < 0) {
237                        createBuffer().unlockBuffer();
238                }
239                return bufferSize;
240        }
241
242        /**
243         * Acquires a managed buffer from the pool. If the pool is empty,
244         * waits for a buffer to become available. The acquired buffer has 
245         * a lock count of one.
246         * 
247         * @return the acquired buffer
248         * @throws InterruptedException if the current thread is interrupted
249         */
250        public W acquire() throws InterruptedException {
251                // Stop draining, because we obviously need this kind of buffers 
252                Optional.ofNullable(idleTimer.getAndSet(null)).ifPresent(
253                                timer -> timer.cancel());
254                if (createdBufs.get() < maximumBufs) {
255                        // Haven't reached maximum, so if no buffer is queued, create one.
256                        W buffer = queue.poll();
257                        if (buffer != null) {
258                                buffer.lockBuffer();
259                                return buffer;
260                        }
261                        return createBuffer();
262                }
263                // Wait for buffer to become available.
264                if (logger.isLoggable(Level.FINE)) {
265                        // If configured, log message after waiting some time.
266                        W buffer = queue.poll(acquireWarningLimit, TimeUnit.MILLISECONDS);
267                        if (buffer != null) {
268                                buffer.lockBuffer();
269                                return buffer;
270                        }
271                        logger.fine("Waiting > " + acquireWarningLimit + "ms for buffer: " 
272                                                + Thread.currentThread().getName());
273                }
274                W buffer = queue.take();
275                buffer.lockBuffer();
276                return buffer;
277        }
278        
279        /**
280         * Re-adds the buffer to the pool. The buffer is cleared.
281         * 
282         * @see org.jgrapes.io.util.BufferCollector#recollect(org.jgrapes.io.util.ManagedBuffer)
283         */
284        @Override
285        public void recollect(W buffer) {
286                if (queue.size() < preservedBufs) {
287                        long effectiveDrainDelay 
288                                = drainDelay > 0 ? drainDelay : defaultDrainDelay;
289                        if (effectiveDrainDelay > 0) {
290                                // Enqueue
291                                buffer.clear();
292                                queue.add(buffer);
293                                Timer old = idleTimer.getAndSet(Components.schedule(this::drain, 
294                                                Duration.ofMillis(effectiveDrainDelay)));
295                                if (old != null) {
296                                        old.cancel();
297                                }
298                                return;
299                        }
300                }
301                // Discard
302                removeBuffer(buffer);
303        }
304        
305        private void drain(Timer timer) {
306                idleTimer.set(null);
307                while(true) {
308                        W buffer = queue.poll();
309                        if (buffer == null) {
310                                break;
311                        }
312                        removeBuffer(buffer);
313                }
314        }

346        private class BufferMonitor {
347                
348                private List<Map.Entry<WeakReference<W>,
349                        BufferProperties>>[] data;
350                private int indexMask = 0;
351                private ReferenceQueue<ManagedBuffer<?>> orphanedBuffers 
352                        = new ReferenceQueue<>();
353        
354                /**
355                 * @param data
356                 */
357                @SuppressWarnings("unchecked")
358                public BufferMonitor(int maxBuffers) {
359                        int lists = 1;
360                        while (lists < maxBuffers) {
361                                lists <<= 1;
362                                indexMask = (indexMask << 1) + 1;
363                        }
364                        data = (List<Map.Entry<WeakReference<W>,BufferProperties>>[])
365                                        new List[maxBuffers];
366                }
367
368                public BufferProperties put(W buffer, BufferProperties bufferMonitor) {
369                        check();
370                        int index = buffer.hashCode() & indexMask;
371                        List<Map.Entry<WeakReference<W>,BufferProperties>> list;
372                        synchronized(data) {
373                                list = data[index];
374                                if (list == null) {
375                                        list = data[index] = new LinkedList<>();
376                                }
377                        }
378                        synchronized(list) {
379                                for (Map.Entry<WeakReference<W>,BufferProperties> entry: list) {
380                                        if (entry.getKey().get() == buffer) {
381                                                BufferProperties old = entry.getValue();
382                                                entry.setValue(bufferMonitor);
383                                                return old;
384                                        }
385                                }
386                                list.add(new AbstractMap.SimpleEntry<>(
387                                                new WeakReference<>(buffer, orphanedBuffers), bufferMonitor));
388                        }
389                        return null;
390                }
391                
392                @SuppressWarnings("unused")
393                public BufferProperties get(ManagedBuffer<?> buffer) {
394                        check();
395                        int index = buffer.hashCode() & indexMask;
396                        List<Map.Entry<WeakReference<W>,BufferProperties>> list = data[index];
397                        if (list == null) {
398                                return null;
399                        }
400                        synchronized(list) {
401                                for (Iterator<Map.Entry<WeakReference<W>,BufferProperties>> 
402                                        itr = list.iterator(); itr.hasNext();) {
403                                        Map.Entry<WeakReference<W>,BufferProperties> entry = itr.next();
404                                        if (entry.getKey().get() == buffer) {
405                                                return entry.getValue();
406                                        }
407                                }
408                        }
409                        return null;
410                }
411                
412                public BufferProperties remove(ManagedBuffer<?> buffer) {
413                        check();
414                        int index = buffer.hashCode() & indexMask;
415                        List<Map.Entry<WeakReference<W>,BufferProperties>> list = data[index];
416                        if (list == null) {
417                                return null;
418                        }
419                        synchronized(list) {
420                                for (Iterator<Map.Entry<WeakReference<W>,BufferProperties>> 
421                                        itr = list.iterator(); itr.hasNext();) {
422                                        Map.Entry<WeakReference<W>,BufferProperties> entry = itr.next();
423                                        if (entry.getKey().get() == buffer) {
424                                                BufferProperties value = entry.getValue();
425                                                itr.remove();
426                                                return value;
427                                        }
428                                }
429                        }
430                        return null;
431                }
432                
433                private void check() {
434                        while (true) {
435                                Reference<? extends ManagedBuffer<?>> ref = orphanedBuffers.poll();
436                                if (ref == null) {
437                                        return;
438                                }
439                                ManagedBuffer<?> buffer = ref.get();
440                                // Managed buffer has not been properly recollected, fix.
441                                createdBufs.decrementAndGet();
442                                BufferProperties props = remove(buffer);
443                                // Create warning
444                                if (logger.isLoggable(Level.WARNING)) {
445                                        final StringBuilder msg = new StringBuilder(
446                                                        "Orphaned buffer from pool " + name());
447                                        StackTraceElement[] st = props.createdBy();
448                                        if (st != null) {
449                                                msg.append(", created");
450                                                for (StackTraceElement e: st) {
451                                                        msg.append(System.lineSeparator());
452                                                        msg.append("\tat ");
453                                                        msg.append(e.toString());
454                                                }
455                                        }
456                                        logger.warning(msg.toString());
457                                }
458                        }
459                }
460        }

513                /**
514                 * Three views on the existing pool.
515                 */
516                public static class PoolInfos {
517                        private SortedMap<String,PoolInfo> allPools;
518                        private SortedMap<String,PoolInfo> nonEmptyPools;
519                        private SortedMap<String,PoolInfo> usedPools;
520
521                        public PoolInfos(Set<ManagedBufferPool<?, ?>> pools) {
522                                allPools = new TreeMap<>();
523                                nonEmptyPools = new TreeMap<>();
524                                usedPools = new TreeMap<>();
525                                
526                                Map<String, Integer> dupsNext = new HashMap<>();
527                                for (ManagedBufferPool<?,?> mbp: pools) {
528                                        String key = mbp.name();
529                                        PoolInfo qi = new PoolInfo(
530                                                        mbp.createdBufs.get(), mbp.queue.size(), mbp.bufferSize());
531                                        if (allPools.containsKey(key) || dupsNext.containsKey(key)) {
532                                                if (allPools.containsKey(key)) {
533                                                        // Found first duplicate, rename
534                                                        allPools.put(key + "#1", allPools.get(key));
535                                                        allPools.remove(key);
536                                                        dupsNext.put(key, 2);
537                                                }
538                                                allPools.put(key + "#"
539                                                                + (dupsNext.put(key, dupsNext.get(key) + 1)), qi);
540                                        } else {
541                                                allPools.put(key, qi);
542                                        }
543                                }
544                                for (Map.Entry<String,PoolInfo> e: allPools.entrySet()) {
545                                        PoolInfo qi = e.getValue();
546                                        if (qi.getPooled() > 0) {
547                                                nonEmptyPools.put(e.getKey(), qi);
548                                        }
549                                        if (qi.getCreated() > 0) {
550                                                usedPools.put(e.getKey(), qi);
551                                        }
552                                }
553                        }
554                        
555                        /**
556                         * All pools.
557                         */
558                        public SortedMap<String, PoolInfo> getAllPools() {
559                                return allPools;
560                        }
561
562                        /**
563                         * Pools that have at least managed buffer enqueued
564                         * (ready to be acquired).
565                         */
566                        public SortedMap<String, PoolInfo> getNonEmptyPools() {
567                                return nonEmptyPools;
568                        }
569
570                        /**
571                         * Pools that have at least one associated buffer
572                         * (in pool or in use).
573                         */
574                        public SortedMap<String, PoolInfo> getUsedPools() {
575                                return usedPools;
576                        }
577                }

652                        ManagedBufferPool.acquireWarningLimit = millis;

657                        return ManagedBufferPool.acquireWarningLimit;
