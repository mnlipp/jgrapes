019package org.jgrapes.io;
020
021import java.nio.ByteBuffer;
022
023import org.jgrapes.core.Channel;
024import org.jgrapes.core.Component;
025import org.jgrapes.core.Components;
026import org.jgrapes.core.Event;
027import org.jgrapes.core.EventPipeline;
028import org.jgrapes.core.internal.Common;
029import org.jgrapes.io.util.ManagedBufferQueue;
030import org.jgrapes.io.util.ManagedByteBuffer;
031
032/**
033 * Represents an I/O subchannel. Subchannels delegate the invocations of a
034 * {@link Channel}'s methods to their respective main channel. Events fired on a
035 * subchannel are therefore handled by the framework as if they were fired on
036 * the main channel. Firing events on a subchannel instance instead of on the
037 * main channel is a means to group related events. In particular, I/O
038 * subchannels are used to group events that relate to an I/O resource such as
039 * an opened file or a network connection.
040 * 
041 * A subchannel has an initiator that creates and manages the subchannel. Events
042 * fired by the initiator are said to flow downstream on the channel. Events
043 * fired by components in response are said to flow upstream.
044 * 
045 * Upstream and downstream events are usually handled by two different pipelines
046 * managed by the initiator. One pipeline, accessible only to the initiator,
047 * handles the downstream events. The other, made available as a property of the
048 * I/O subchannel, handles the upstream events. Of course, any pipeline can be
049 * used to send events upstream to the initiator component. However, using
050 * arbitrary pipelines holds the risk that events aren't delivered in the
051 * intended order.
052 * <P>
053 * An I/O subchannel also has an associated buffer pool. Buffers from this pool
054 * may be used for read and write events.
055 * 
056 * @author Michael N. Lipp
057 */
058public interface IOSubchannel extends Channel {
059
060        /**
061         * Returns the main channel.
062         * 
063         * @return the mainChannel
064         */
065        Channel mainChannel();
066
067        /**
068         * Returns the main channel's match value.
069         * 
070         * @see Channel#defaultCriterion()
071         */
072        @Override
073        default Object defaultCriterion() {
074                return mainChannel().defaultCriterion();
075        }
076
077        /**
078         * Delegates to main channel.
079         * 
080         * @see Channel#isEligibleFor(Object)
081         */
082        @Override
083        default boolean isEligibleFor(Object value) {
084                return mainChannel().isEligibleFor(value);
085        }
086
087        /**
088         * Gets the {@link EventPipeline} that can be used for events going back to
089         * the initiator of this connection. Consistently using this event pipeline
090         * for response events ensures that the events are written in proper
091         * sequence.
092         * 
093         * @return the event pipeline
094         */
095        public abstract EventPipeline responsePipeline();
096
097        /**
098         * Get the subchannel's buffer pool.
099         * 
100         * @return the buffer pool
101         */
102        public abstract 
103                ManagedBufferQueue<ManagedByteBuffer, ByteBuffer> bufferPool();
104        
105        /**
106         * Fires the given event on this subchannel using the subchannel's response
107         * pipeline. Effectively, {@code fire(someEvent)} is a shortcut for
108         * {@code getResponsePipeline.add(someEvent, this)}.
109         * 
110         * @param <T> the event's type
111         * @param event
112         *            the event to fire
113         * @return the event (for easy chaining)
114         */
115        default <T extends Event<?>> T fire(T event) {
116                return responsePipeline().fire(event, this);
117        }
118
119        /* (non-Javadoc)
120         * @see java.lang.Object#toString()
121         */
122        static String toString(IOSubchannel subchannel) {
123                StringBuilder builder = new StringBuilder();
124                builder.append(Common.channelToString(subchannel.mainChannel()));
125                builder.append("{");
126                builder.append(Components.objectName(subchannel));
127                builder.append("}");
128                return builder.toString();
129        }
130
131        /**
132         * Creates a new subchannel of the given component's channel with a new
133         * event pipeline and a buffer pool with two buffers sized 4096.
134         * 
135         * @param component
136         *            the component used to get the main channel and the event
137         *            pipeline
138         * @return the subchannel
139         */
140        public static IOSubchannel defaultInstance(Component component) {
141                return new DefaultSubchannel(component);
142        }
143        
144        public static class DefaultSubchannel implements IOSubchannel {
145
146                private Channel mainChannel;
147                private EventPipeline eventPipeline;
148                private ManagedBufferQueue<ManagedByteBuffer, ByteBuffer> bufferPool;
149
150                private DefaultSubchannel(Component component) {
151                        mainChannel = component.channel();
152                        eventPipeline = component.newEventPipeline();
153                }
154                
155                /* (non-Javadoc)
156                 * @see org.jgrapes.io.IOSubchannel#getMainChannel()
157                 */
158                @Override
159                public Channel mainChannel() {
160                        return mainChannel;
161                }
162
163                @Override
164                public EventPipeline responsePipeline() {
165                        return eventPipeline;
166                }
167
168                @Override
169                public ManagedBufferQueue<ManagedByteBuffer, ByteBuffer> bufferPool() {
170                        if (bufferPool == null) {
171                                bufferPool = new ManagedBufferQueue<>(ManagedByteBuffer.class,
172                                                ByteBuffer.allocate(4096), 
173                                                ByteBuffer.allocate(4096));
174                        }
175                        return bufferPool;
176                }
177
178                /* (non-Javadoc)
179                 * @see java.lang.Object#toString()
180                 */
181                @Override
182                public String toString() {
183                        return IOSubchannel.toString(this);
184                }
185                
186        }
187}
