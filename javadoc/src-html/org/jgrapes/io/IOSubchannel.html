019package org.jgrapes.io;
020
021import java.nio.ByteBuffer;
022import java.nio.CharBuffer;
023import java.util.Map;
024import java.util.Optional;
025import java.util.concurrent.ConcurrentHashMap;
026
027import org.jgrapes.core.Associator;
028import org.jgrapes.core.Channel;
029import org.jgrapes.core.Component;
030import org.jgrapes.core.Components;
031import org.jgrapes.core.Event;
032import org.jgrapes.core.EventPipeline;
033import org.jgrapes.io.util.ManagedBuffer;
034import org.jgrapes.io.util.ManagedBufferPool;
035
036/**
037 * Represents an I/O subchannel. Subchannels delegate the invocations of a
038 * {@link Channel}'s methods to their respective main channel. Events fired on a
039 * subchannel are therefore handled by the framework as if they were fired on
040 * the main channel. Firing events on a subchannel instance instead of on the
041 * main channel is a means to group related events. In particular, I/O
042 * subchannels are used to group events that relate to an I/O resource such as
043 * an opened file or a network connection.
044 * 
045 * A subchannel has an initiator that creates and manages the subchannel. Events
046 * fired by the initiator are said to flow downstream on the channel. Events
047 * fired by components in response are said to flow upstream.
048 * 
049 * Upstream and downstream events are usually handled by two different pipelines
050 * managed by the initiator. One pipeline, accessible only to the initiator,
051 * handles the downstream events. The other, made available as a property of the
052 * I/O subchannel, handles the upstream events. Of course, any pipeline can be
053 * used to send events upstream to the initiator component. However, using
054 * arbitrary pipelines holds the risk that events aren't delivered in the
055 * intended order.
056 * 
057 * As a convenience, an I/O subchannel also has an associated buffer pool. 
058 * Buffers from this pool should only be acquired for upstream events.
059 */
060public interface IOSubchannel extends Channel, Associator {
061
062    /**
063     * Returns the main channel.
064     * 
065     * @return the mainChannel
066     */
067    Channel mainChannel();
068
069    /**
070     * Returns the main channel's match value.
071     * 
072     * @see Channel#defaultCriterion()
073     */
074    @Override
075    default Object defaultCriterion() {
076        return mainChannel().defaultCriterion();
077    }
078
079    /**
080     * Delegates to main channel.
081     * 
082     * @see Channel#isEligibleFor(Object)
083     */
084    @Override
085    default boolean isEligibleFor(Object value) {
086        return mainChannel().isEligibleFor(value);
087    }
088
089    /**
090     * Gets the {@link EventPipeline} that can be used for events going back to
091     * the initiator of this connection. Consistently using this event pipeline
092     * for response events ensures that the events are written in proper
093     * sequence.
094     * 
095     * @return the event pipeline
096     */
097    EventPipeline responsePipeline();
098
099    /**
100     * Get the subchannel's byte buffer pool.
101     * 
102     * @return the buffer pool
103     */
104    ManagedBufferPool<ManagedBuffer<ByteBuffer>, ByteBuffer>
105            byteBufferPool();
106
107    /**
108     * Get the subchannel's char buffer pool.
109     * 
110     * @return the buffer pool
111     */
112    ManagedBufferPool<ManagedBuffer<CharBuffer>, CharBuffer>
113            charBufferPool();
114
115    /**
116     * Fires the given event on this subchannel using the subchannel's response
117     * pipeline. Effectively, {@code fire(someEvent)} is a shortcut for
118     * {@code getResponsePipeline.add(someEvent, this)}.
119     * 
120     * @param <T> the event's type
121     * @param event
122     *            the event to fire
123     * @return the event (for easy chaining)
124     */
125    default <T extends Event<?>> T respond(T event) {
126        return responsePipeline().fire(event, this);
127    }
128
129    /**
130     * Returns a string representation of the channel.
131     *
132     * @param subchannel the subchannel
133     * @return the string
134     */
135    static String toString(IOSubchannel subchannel) {
136        StringBuilder builder = new StringBuilder();
137        builder.append(Channel.toString(subchannel.mainChannel()))
138            .append('{')
139            .append(Components.objectName(subchannel))
140            .append('}');
141        return builder.toString();
142    }
143
144    /**
145     * Creates a new subchannel of the given component's channel with the
146     * given event pipeline and a buffer pool with two buffers sized 4096.
147     *
148     * @param component the component used to get the main channel
149     * @param responsePipeline the response pipeline
150     * @return the subchannel
151     */
152    static IOSubchannel create(
153            Component component, EventPipeline responsePipeline) {
154        return new DefaultSubchannel(component.channel(), responsePipeline);
155    }
156
157    /**
158     * A simple implementation of {@link IOSubchannel}.
159     */
160    class DefaultSubchannel implements IOSubchannel {
161        private final Channel mainChannel;
162        private final EventPipeline responsePipeline;
163        private ManagedBufferPool<ManagedBuffer<ByteBuffer>,
164                ByteBuffer> byteBufferPool;
165        private ManagedBufferPool<ManagedBuffer<CharBuffer>,
166                CharBuffer> charBufferPool;
167        private Map<Object, Object> contextData;
168
169        /**
170         * Creates a new instance with the given main channel and response
171         * pipeline.  
172         * 
173         * @param mainChannel the main channel
174         * @param responsePipeline the response pipeline to use
175         * 
176         */
177        public DefaultSubchannel(
178                Channel mainChannel, EventPipeline responsePipeline) {
179            super();
180            this.mainChannel = mainChannel;
181            this.responsePipeline = responsePipeline;
182        }
183
184        protected void setByteBufferPool(
185                ManagedBufferPool<ManagedBuffer<ByteBuffer>,
186                        ByteBuffer> bufferPool) {
187            this.byteBufferPool = bufferPool;
188        }
189
190        protected void setCharBufferPool(
191                ManagedBufferPool<ManagedBuffer<CharBuffer>,
192                        CharBuffer> bufferPool) {
193            this.charBufferPool = bufferPool;
194        }
195
196        /*
197         * (non-Javadoc)
198         * 
199         * @see org.jgrapes.io.IOSubchannel#getMainChannel()
200         */
201        @Override
202        public Channel mainChannel() {
203            return mainChannel;
204        }
205
206        /*
207         * (non-Javadoc)
208         * 
209         * @see org.jgrapes.io.IOSubchannel#responsePipeline()
210         */
211        @Override
212        public EventPipeline responsePipeline() {
213            return responsePipeline;
214        }
215
216        /**
217         * Returns the buffer pool set. If no buffer pool has been set, a
218         * buffer pool with with two buffers of size 4096 is created.
219         */
220        public ManagedBufferPool<ManagedBuffer<ByteBuffer>, ByteBuffer>
221                byteBufferPool() {
222            if (byteBufferPool == null) {
223                byteBufferPool = new ManagedBufferPool<>(ManagedBuffer::new,
224                    () -> {
225                        return ByteBuffer.allocate(4096);
226                    }, 2)
227                        .setName(Components.objectName(this)
228                            + ".upstream.byteBuffers");
229            }
230            return byteBufferPool;
231        }
232
233        /**
234         * Returns the buffer pool set. If no buffer pool has been set, a
235         * buffer pool with with two buffers of size 4096 is created.
236         */
237        public ManagedBufferPool<ManagedBuffer<CharBuffer>, CharBuffer>
238                charBufferPool() {
239            if (charBufferPool == null) {
240                charBufferPool = new ManagedBufferPool<>(ManagedBuffer::new,
241                    () -> {
242                        return CharBuffer.allocate(4096);
243                    }, 2)
244                        .setName(Components.objectName(this)
245                            + ".upstream.charBuffers");
246            }
247            return charBufferPool;
248        }
249
250        /**
251         * Establishes a "named" association to an associated object. Note that 
252         * anything that represents an id can be used as value for 
253         * parameter `name`, it does not necessarily have to be a string.
254         * 
255         * @param by the "name"
256         * @param with the object to be associated
257         */
258        @SuppressWarnings("PMD.ShortVariable")
259        public DefaultSubchannel setAssociated(Object by, Object with) {
260            if (contextData == null) {
261                contextData = new ConcurrentHashMap<>();
262            }
263            if (with == null) {
264                contextData.remove(by);
265            } else {
266                contextData.put(by, with);
267            }
268            return this;
269        }
270
271        /**
272         * Retrieves the associated object following the association 
273         * with the given "name". This general version of the method
274         * supports the retrieval of values of arbitrary types
275         * associated by any "name" types. 
276         * 
277         * @param by the "name"
278         * @param type the tape of the value to be retrieved
279         * @param <V> the type of the value to be retrieved
280         * @return the associate, if any
281         */
282        @SuppressWarnings("PMD.ShortVariable")
283        public <V> Optional<V> associated(Object by, Class<V> type) {
284            if (contextData == null) {
285                return Optional.empty();
286            }
287            return Optional.ofNullable(contextData.get(by))
288                .filter(found -> type.isAssignableFrom(found.getClass()))
289                .map(match -> type.cast(match));
290        }
291
292        /*
293         * (non-Javadoc)
294         * 
295         * @see java.lang.Object#toString()
296         */
297        @Override
298        public String toString() {
299            return IOSubchannel.toString(this);
300        }
301
302    }
303}
