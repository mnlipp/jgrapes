019package org.jgrapes.portal.events;
020
021import java.net.URI;
022import java.util.ArrayList;
023import java.util.List;
024
025import org.jgrapes.core.Event;
026
027/**
028 * Adds a portlet type with its global resources (JavaScript and/or CSS) 
029 * to the portal page. Specifying global resources result in the respective
030 * `&lt;link .../&gt;` or `&lt;script ...&gt;&lt/script&gt;` nodes
031 * being added to the page's `&lt;head&gt;` section.
032 * 
033 * This in turn causes the browser to issue send `GET` request that
034 * (usually) refer to the portlet's resources. These requests are
035 * converted to {@link PortletResourceRequest}s by the portal and
036 * sent to the portlets, which must respond to theses requests.
037 * 
038 * The complete sequence of events is shown in the diagram.
039 * 
040 * ![Portal Ready Event Sequence](AddPortletTypeSeq.svg)
041 * 
042 * Of course, due to internal buffering, the "Response Header" data
043 * and the "Response body" data may collapse in a single message
044 * that is sent to the browser (in case of a small resource).
045 * 
046 * @startuml AddPortletTypeSeq.svg
047 * hide footbox
048 * 
049 * activate PortletX
050 * PortletX -> Portal: AddPortletType 
051 * deactivate PortletX
052 * activate Portal
053 * Portal -> Browser: "addPortletType"
054 * activate Browser
055 * deactivate Portal
056 * deactivate Portal
057 * Browser -> Portal: "GET <portlet resource1 URL>"
058 * activate Portal
059 * Portal -> PortletX: PortletResourceRequest
060 * activate PortletX
061 * PortletX -> Portal: PortletResourceResponse
062 * Portal -> Browser: "Response Header"
063 * deactivate Portal
064 * loop until end of data
065 *     PortletX -> Portal: Output
066 *     activate Portal
067 *     Portal -> Browser: "Response body"
068 *     deactivate Portal
069 * end loop
070 * deactivate PortletX
071 * deactivate Browser
072 * 
073 * @enduml
074 */
075public class AddPortletType extends Event<Void> {
076
077        private String portletType;
078        private String displayName = "";
079        private boolean instantiable = false;
080        private List<URI> scriptUris = new ArrayList<>();
081        private List<URI> cssUris = new ArrayList<>();
082        
083        /**
084         * Create a new event for the given portlet type.
085         * 
086         * @param portletType a unique id for the portklet type (usually
087         * the class name)
088         */
089        public AddPortletType(String portletType) {
090                this.portletType = portletType;
091        }
092        
093        /**
094         * Return the portlet type.
095         * 
096         * @return the portlet type
097         */
098        public String portletType() {
099                return portletType;
100        }
101
102        /**
103         * Sets the display name.
104         * 
105         * @param displayName the display name
106         * @return the event for easy chaining
107         */
108        public AddPortletType setDisplayName(String displayName) {
109                this.displayName = displayName;
110                return this;
111        }
112        
113        /**
114         * Return the display name.
115         * 
116         * @return the displayName
117         */
118        public String displayName() {
119                return displayName;
120        }
121
122        /**
123         * Mark the portlet type as instantiable.
124         * 
125         * @return the event for easy chaining
126         */
127        public AddPortletType setInstantiable() {
128                instantiable = true;
129                return this;
130        }
131        
132        /**
133         * Return if the portelt is instantiable.
134         * 
135         * @return the result
136         */
137        public boolean isInstantiable() {
138                return instantiable;
139        }
140
141        /**
142         * Add the URI of a JavaScript resource that is to be added to the
143         * header section of the portal page.
144         * 
145         * @param uri the URI
146         * @return the event for easy chaining
147         */
148        public AddPortletType addScript(URI uri) {
149                scriptUris.add(uri);
150                return this;
151        }
152
153        /**
154         * Add the URI of a CSS resource that is to be added to the
155         * header section of the portal page.
156         * 
157         * @param uri the URI
158         * @return the event for easy chaining
159         */
160        public AddPortletType addCss(URI uri) {
161                cssUris.add(uri);
162                return this;
163        }
164
165        /**
166         * Return all script URIs
167         * 
168         * @return the result
169         */
170        public URI[] scriptUris() {
171                return scriptUris.toArray(new URI[scriptUris.size()]);
172        }
173
174        /**
175         * Return all CSS URIs.
176         * 
177         * @return the result
178         */
179        public URI[] cssUris() {
180                return cssUris.toArray(new URI[cssUris.size()]);
181        }
182}
