019package org.jgrapes.portal.freemarker;
020
021import freemarker.template.Configuration;
022import freemarker.template.SimpleScalar;
023import freemarker.template.Template;
024import freemarker.template.TemplateException;
025import freemarker.template.TemplateExceptionHandler;
026import freemarker.template.TemplateMethodModelEx;
027import freemarker.template.TemplateModel;
028import freemarker.template.TemplateModelException;
029
030import java.io.IOException;
031import java.io.Writer;
032import java.util.Collections;
033import java.util.HashMap;
034import java.util.List;
035import java.util.Locale;
036import java.util.Map;
037import java.util.MissingResourceException;
038import java.util.ResourceBundle;
039
040import org.jgrapes.core.Channel;
041import org.jgrapes.http.LanguageSelector.Selection;
042import org.jgrapes.io.IOSubchannel;
043import org.jgrapes.portal.AbstractPortlet;
044import org.jgrapes.portal.PortalView;
045import org.jgrapes.portal.RenderSupport;
046import org.jgrapes.portal.events.RenderPortletFromProvider.ContentProvider;
047
048/**
049 * 
050 */
051public abstract class FreeMarkerPortlet extends AbstractPortlet {
052
053        private Configuration fmConfig = null;
054        private Map<String,Object> fmModel = null;
055        
056        /**
057         * @param componentChannel
058         */
059        public FreeMarkerPortlet(Channel componentChannel) {
060                super(componentChannel);
061        }
062
063        protected Configuration freemarkerConfig() {
064                if (fmConfig == null) {
065                        fmConfig = new Configuration(Configuration.VERSION_2_3_26);
066                        fmConfig.setClassLoaderForTemplateLoading(
067                                        getClass().getClassLoader(), getClass().getPackage()
068                                        .getName().replace('.', '/'));
069                        fmConfig.setDefaultEncoding("utf-8");
070                        fmConfig.setTemplateExceptionHandler(
071                                        TemplateExceptionHandler.RETHROW_HANDLER);
072                fmConfig.setLogTemplateExceptions(false);
073                }
074                return fmConfig;
075        }
076        
077        /**
078         * Creates the request independant part of the freemarker model. The
079         * result is cached as unmodifiable map as it can safely be assumed
080         * that the render support does not change for a given portal.
081         * 
082         * @param renderSupport the render support from the portal
083         * @return the result
084         */
085        protected Map<String,Object> freemarkerBaseModel(
086                        RenderSupport renderSupport) {
087                if (fmModel == null) {
088                        fmModel = new HashMap<>();
089                        fmModel.put("portletResource", new TemplateMethodModelEx() {
090                                @Override
091                                public Object exec(@SuppressWarnings("rawtypes") List arguments)
092                                                throws TemplateModelException {
093                                        @SuppressWarnings("unchecked")
094                                        List<TemplateModel> args = (List<TemplateModel>)arguments;
095                                        if (!(args.get(0) instanceof SimpleScalar)) {
096                                                throw new TemplateModelException("Not a string.");
097                                        }
098                                        return renderSupport.portletResource(
099                                                        FreeMarkerPortlet.this.getClass().getName(),
100                                                        PortalView.uriFromPath(
101                                                                        ((SimpleScalar)args.get(0)).getAsString()))
102                                                        .getRawPath();
103                                }
104                        });
105                        fmModel = Collections.unmodifiableMap(fmModel);
106                }
107                return fmModel;
108        }
109
110        /**
111         * Build a freemarker model from the given base model, portlet model
112         * and the information associated with the channel.
113         * 
114         * @param baseModel the base model
115         * @param portletModel the portlet model
116         * @param channel the channel
117         * @return the model
118         */
119        protected Map<String,Object> freemarkerModel(Map<String,Object> baseModel,
120                        PortletBaseModel portletModel, IOSubchannel channel) {
121                final Map<String,Object> model = new HashMap<>(baseModel);
122                model.put("portlet", portletModel);
123                Locale locale = channel.associated(Selection.class)
124                                .map(s -> s.get()[0]).orElse(Locale.getDefault());
125                model.put("locale", locale);
126                final ResourceBundle resourceBundle = resourceSupplier().apply(locale);
127                model.put("_", new TemplateMethodModelEx() {
128                        @Override
129                        public Object exec(@SuppressWarnings("rawtypes") List arguments)
130                                        throws TemplateModelException {
131                                @SuppressWarnings("unchecked")
132                                List<TemplateModel> args = (List<TemplateModel>)arguments;
133                                if (!(args.get(0) instanceof SimpleScalar)) {
134                                        throw new TemplateModelException("Not a string.");
135                                }
136                                String key = ((SimpleScalar)args.get(0)).getAsString();
137                                try {
138                                        return resourceBundle.getString(key);
139                                } catch (MissingResourceException e) {
140                                        // no luck
141                                }
142                                return key;
143                        }
144                });
145                return model;
146        }
147        
148        protected ContentProvider newContentProvider(
149                        Template template, Map<String,Object> dataModel) {
150                return new ContentProvider() {
151                        
152                        @Override
153                        public void writeTo(Writer out) throws IOException {
154                                try {
155                                        template.process(dataModel, out);
156                                } catch (TemplateException e) {
157                                        throw new IOException(e);
158                                }
159                        }
160                };
161        }
162        
163}
