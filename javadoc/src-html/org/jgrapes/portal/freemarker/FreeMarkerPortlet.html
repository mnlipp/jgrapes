019package org.jgrapes.portal.freemarker;
020
021import freemarker.template.Configuration;
022import freemarker.template.SimpleScalar;
023import freemarker.template.Template;
024import freemarker.template.TemplateException;
025import freemarker.template.TemplateExceptionHandler;
026import freemarker.template.TemplateMethodModelEx;
027import freemarker.template.TemplateModel;
028import freemarker.template.TemplateModelException;
029
030import java.io.IOException;
031import java.io.PipedReader;
032import java.io.PipedWriter;
033import java.io.Reader;
034import java.io.Writer;
035import java.util.Collections;
036import java.util.HashMap;
037import java.util.List;
038import java.util.Locale;
039import java.util.Map;
040import java.util.MissingResourceException;
041import java.util.ResourceBundle;
042import java.util.regex.Pattern;
043
044import org.jgrapes.core.Channel;
045import org.jgrapes.http.Session;
046import org.jgrapes.io.IOSubchannel;
047import org.jgrapes.io.util.CharBufferWriter;
048import org.jgrapes.portal.AbstractPortlet;
049import org.jgrapes.portal.PortalView;
050import org.jgrapes.portal.RenderSupport;
051import org.jgrapes.portal.events.PortletResourceRequest;
052import org.jgrapes.portal.events.PortletResourceResponse;
053import org.jgrapes.portal.events.RenderPortletRequest;
054import org.jgrapes.portal.events.RenderPortletRequestBase;
055
056/**
057 * 
058 */
059public abstract class FreeMarkerPortlet extends AbstractPortlet {
060
061        static final Pattern templatePattern 
062                = Pattern.compile(".*\\.ftl\\.[a-z]+$");
063        
064        private Configuration fmConfig = null;
065        private Map<String,Object> fmModel = null;
066        
067        /**
068         * @param componentChannel
069         */
070        public FreeMarkerPortlet(Channel componentChannel) {
071                super(componentChannel);
072        }
073
074        protected Configuration freemarkerConfig() {
075                if (fmConfig == null) {
076                        fmConfig = new Configuration(Configuration.VERSION_2_3_26);
077                        fmConfig.setClassLoaderForTemplateLoading(
078                                        getClass().getClassLoader(), getClass().getPackage()
079                                        .getName().replace('.', '/'));
080                        fmConfig.setDefaultEncoding("utf-8");
081                        fmConfig.setTemplateExceptionHandler(
082                                        TemplateExceptionHandler.RETHROW_HANDLER);
083                fmConfig.setLogTemplateExceptions(false);
084                }
085                return fmConfig;
086        }
087        
088        /**
089         * Creates the request independent part of the freemarker model. The
090         * result is cached as unmodifiable map as it can safely be assumed
091         * that the render support does not change for a given portal.
092         * 
093         * This model provides:
094         *  * The function `portletResource` that makes 
095         *    {@link RenderSupport#portletResource(String, java.net.URI)}
096         *    available in the template. The first argument is set to the name
097         *    of the portlet, only the second must be supplied when the function
098         *    is invoked in a template.
099         * 
100         * @param renderSupport the render support from the portal
101         * @return the result
102         */
103        protected Map<String,Object> fmTypeModel(
104                        RenderSupport renderSupport) {
105                if (fmModel == null) {
106                        fmModel = new HashMap<>();
107                        fmModel.put("portletResource", new TemplateMethodModelEx() {
108                                @Override
109                                public Object exec(@SuppressWarnings("rawtypes") List arguments)
110                                                throws TemplateModelException {
111                                        @SuppressWarnings("unchecked")
112                                        List<TemplateModel> args = (List<TemplateModel>)arguments;
113                                        if (!(args.get(0) instanceof SimpleScalar)) {
114                                                throw new TemplateModelException("Not a string.");
115                                        }
116                                        return renderSupport.portletResource(
117                                                        FreeMarkerPortlet.this.getClass().getName(),
118                                                        PortalView.uriFromPath(
119                                                                        ((SimpleScalar)args.get(0)).getAsString()))
120                                                        .getRawPath();
121                                }
122                        });
123                        fmModel = Collections.unmodifiableMap(fmModel);
124                }
125                return fmModel;
126        }
127
128        /**
129         * Build a freemarker model holdiung the information associated 
130         * with the session.
131         * 
132         * This model provides:
133         *  * The `locale` (of type {@link Locale}).
134         *  * The `resourceBundle` (of type {@link ResourceBundle}).
135         *  * A function `_` that looks up the given key in the portlet's
136         *    resource bundle.
137         *    
138         * @param session the session
139         * @return the model
140         */
141        protected Map<String,Object> fmSessionModel(Session session) {
142                final Map<String,Object> model = new HashMap<>();
143                Locale locale = session.locale();
144                model.put("locale", locale);
145                final ResourceBundle resourceBundle = resourceBundle(locale);
146                model.put("resourceBundle", resourceBundle);
147                model.put("_", new TemplateMethodModelEx() {
148                        @Override
149                        public Object exec(@SuppressWarnings("rawtypes") List arguments)
150                                        throws TemplateModelException {
151                                @SuppressWarnings("unchecked")
152                                List<TemplateModel> args = (List<TemplateModel>)arguments;
153                                if (!(args.get(0) instanceof SimpleScalar)) {
154                                        throw new TemplateModelException("Not a string.");
155                                }
156                                String key = ((SimpleScalar)args.get(0)).getAsString();
157                                try {
158                                        return resourceBundle.getString(key);
159                                } catch (MissingResourceException e) {
160                                        // no luck
161                                }
162                                return key;
163                        }
164                });
165                return model;
166        }
167        
168        /**
169         * Build a freemarker model for the current request.
170         * 
171         * This model provides:
172         *  * The `event` property (of type {@link RenderPortletRequest}).
173         *  * The `portlet` property (of type {@link PortletBaseModel}).
174         *    
175         * @param event the event
176         * @param channel the channel
177         * @param portletModel the portlet model
178         * @return the model
179         */
180        protected Map<String,Object> fmPortletModel(RenderPortletRequestBase event, 
181                        IOSubchannel channel, PortletBaseModel portletModel) {
182                final Map<String,Object> model = new HashMap<>();
183                model.put("event", event);
184                model.put("portlet", portletModel);
185                return model;
186        }
187        
188        /**
189         * Build a freemarker model that combines {@link #fmTypeModel},
190         * {@link #fmSessionModel} and {@link #fmPortletModel}.
191         * 
192         * @param event the event
193         * @param channel the channel
194         * @param portletModel the portlet model
195         * @return the model
196         */
197        protected Map<String,Object> fmModel(RenderPortletRequestBase event,
198                        IOSubchannel channel, PortletBaseModel portletModel) {
199                final Map<String,Object> model = fmSessionModel(session(channel));
200                model.putAll(fmTypeModel(event.renderSupport()));
201                model.putAll(fmPortletModel(event, channel, portletModel));
202                return model;
203        }
204
205        /**
206         * Checks if the path of the requested resource ends with
207         * `*.ftl.*`. If so, processes the template with the
208         * {@link #fmTypeModel(RenderSupport)} and 
209         * {@link #fmSessionModel(Session)} and
210         * sends the result. Else, invoke the super class' method. 
211         * 
212         * @param event the event. The result will be set to
213         * `true` on success
214         * @param channel the channel
215         */
216        @SuppressWarnings("resource")
217        @Override
218        protected void doGetResource(PortletResourceRequest event, IOSubchannel channel) {
219                if (!templatePattern.matcher(event.resourceUri().getPath()).matches()) {
220                        super.doGetResource(event, channel);
221                        return;
222                }
223                try {
224                        Template tpl = freemarkerConfig().getTemplate(event.resourceUri().getPath());
225                        Map<String, Object> model = fmSessionModel(
226                                        event.associated(Session.class).get());
227                        model.putAll(fmTypeModel(event.renderSupport()));
228                        channel.respond(new PortletResourceResponse(event, true));
229                        Writer out = new CharBufferWriter(channel).suppressClose();
230                        tpl.process(model, out);
231                        out.close();
232                        event.setResult(true);
233                } catch (TemplateException | IOException e) {
234                        throw new IllegalArgumentException(e);
235                }
236        }
237
238        
239        /**
240         * Creates a reader that delivers the result of processing the given
241         * template with the given data.
242         * 
243         * @param template the template
244         * @param dataModel the data
245         * @return the stream with the resulting data
246         */
247        public Reader templateProcessor(Template template, Object dataModel) {
248                try {
249                        PipedReader reader = new PipedReader();
250                        Writer writer = new PipedWriter(reader);
251                        new Thread(new Runnable() {
252
253                                /* (non-Javadoc)
254                                 * @see java.lang.Runnable#run()
255                                 */
256                                @Override
257                                public void run() {
258                                        try (Writer out = writer) {
259                                                template.process(dataModel, out);
260                                        } catch (TemplateException | IOException e) {
261                                                e.printStackTrace();
262                                        }
263                                }
264                        }).start();
265                        return reader;
266                } catch (IOException e) {
267                        throw new IllegalStateException(e);
268                }
269        }
270}
