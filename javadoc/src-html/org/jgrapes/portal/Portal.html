019package org.jgrapes.portal;
020
021import java.io.IOException;
022import java.io.StringWriter;
023import java.net.URI;
024import java.util.Arrays;
025import java.util.Locale;
026import java.util.ResourceBundle;
027import java.util.function.Function;
028
029import javax.json.JsonArray;
030import javax.json.JsonString;
031import javax.json.JsonValue;
032
033import org.jgrapes.core.Channel;
034import org.jgrapes.core.Component;
035import org.jgrapes.core.annotation.Handler;
036import org.jgrapes.io.util.LinkedIOSubchannel;
037import org.jgrapes.portal.Portlet.RenderMode;
038import org.jgrapes.portal.events.AddPortletRequest;
039import org.jgrapes.portal.events.AddPortletType;
040import org.jgrapes.portal.events.DeletePortlet;
041import org.jgrapes.portal.events.DeletePortletRequest;
042import org.jgrapes.portal.events.JsonInput;
043import org.jgrapes.portal.events.JsonOutput;
044import org.jgrapes.portal.events.LastPortalLayout;
045import org.jgrapes.portal.events.NotifyPortletModel;
046import org.jgrapes.portal.events.NotifyPortletView;
047import org.jgrapes.portal.events.PortalConfigured;
048import org.jgrapes.portal.events.PortalLayoutChanged;
049import org.jgrapes.portal.events.PortalReady;
050import org.jgrapes.portal.events.RenderPortletFromProvider;
051import org.jgrapes.portal.events.RenderPortletFromString;
052import org.jgrapes.portal.events.RenderPortletRequest;
053import org.jgrapes.portal.events.SetLocale;
054import org.jgrapes.portal.events.SetTheme;
055
056/**
057 * 
058 */
059public class Portal extends Component {
060
061        private URI prefix;
062        private PortalView view;
063        
064        /**
065         * 
066         */
067        public Portal(URI prefix) {
068                this(Channel.SELF, prefix);
069        }
070
071        /**
072         * @param componentChannel
073         */
074        public Portal(Channel componentChannel, URI prefix) {
075                this(componentChannel, componentChannel, prefix);
076        }
077
078        /**
079         * @param componentChannel
080         */
081        public Portal(Channel componentChannel, Channel viewChannel, URI prefix) {
082                super(componentChannel);
083                this.prefix = URI.create(prefix.getPath().endsWith("/") 
084                                ? prefix.getPath() : (prefix.getPath() + "/"));
085                view = attach(new PortalView(this, viewChannel));
086        }
087
088        /**
089         * @return the prefix
090         */
091        public URI prefix() {
092                return prefix;
093        }
094
095        public Portal setResourceSupplier(
096                        Function<Locale,ResourceBundle> supplier) {
097                view.setResourceSupplier(supplier);
098                return this;
099        }
100        
101        @Handler
102        public void onRenderPortlet(RenderPortletFromString event,
103                        LinkedIOSubchannel channel) 
104                                        throws InterruptedException, IOException {
105                fire(new JsonOutput("updatePortlet",
106                                event.portletId(), event.renderMode().name(),
107                                event.supportedRenderModes().stream().map(RenderMode::name)
108                                .toArray(size -> new String[size]),
109                                event.content()), channel);
110        }
111        
112        @Handler
113        public void onRenderPortlet(RenderPortletFromProvider event,
114                        LinkedIOSubchannel channel) 
115                                        throws InterruptedException, IOException {
116                StringWriter content = new StringWriter();
117                event.provider().writeTo(content);
118                fire(new JsonOutput("updatePortlet",
119                                event.portletId(), event.renderMode().name(),
120                                event.supportedRenderModes().stream().map(RenderMode::name)
121                                .toArray(size -> new String[size]),
122                                content.toString(), event.isForeground()), channel);
123        }
124        
125        @Handler
126        public void onAddPortletType(
127                        AddPortletType event, LinkedIOSubchannel channel)
128                                        throws InterruptedException, IOException {
129                fire(new JsonOutput("addPortletType",
130                                event.portletType(),
131                                event.displayName(),
132                                Arrays.stream(event.cssUris()).map(uri -> 
133                                        view.renderSupport().portletResource(
134                                                        event.portletType(), uri).toString())
135                                .toArray(String[]::new),
136                                Arrays.stream(event.scriptUris()).map(uri -> 
137                                        view.renderSupport().portletResource(
138                                                        event.portletType(), uri).toString())
139                                .toArray(String[]::new),
140                                event.isInstantiable()), channel);
141        }
142        
143        @Handler
144        public void onDeletePortlet(
145                        DeletePortlet event, LinkedIOSubchannel channel) 
146                                        throws InterruptedException, IOException {
147                fire(new JsonOutput("deletePortlet", event.portletId()), channel);
148        }
149        
150        @Handler 
151        public void onNotifyPortletView(
152                        NotifyPortletView event, LinkedIOSubchannel channel) 
153                                        throws InterruptedException, IOException {
154                fire(new JsonOutput("invokePortletMethod",
155                                event.portletClass(), event.portletId(), 
156                                event.method(), event.params()), channel);
157        }
158
159        @Handler
160        public void onJsonInput(JsonInput event, LinkedIOSubchannel channel) 
161                        throws InterruptedException, IOException {
162                // Send events to portlets on portal's channel
163                JsonArray params = (JsonArray)event.params();
164                switch (event.method()) {
165                case "portalReady": {
166                        fire(new PortalReady(view.renderSupport()), channel);
167                        break;
168                }
169                case "addPortlet": {
170                        fire(new AddPortletRequest(view.renderSupport(), params.getString(0),
171                                        RenderMode.valueOf(params.getString(1))), channel);
172                        break;
173                }
174                case "deletePortlet": {
175                        fire(new DeletePortletRequest(
176                                        view.renderSupport(), params.getString(0)), channel);
177                        break;
178                }
179                case "portalLayout": {
180                        String[][] previewLayout = params.getJsonArray(0)
181                                        .getValuesAs(column -> ((JsonArray)column)
182                                                        .getValuesAs(JsonString::getString)
183                                                        .stream().toArray(s -> new String[s]))
184                                        .stream().toArray(s -> new String[s][]);
185                        String[] tabsLayout = params.getJsonArray(1)
186                                        .getValuesAs(JsonString::getString)
187                                        .stream().toArray(s -> new String[s]);
188                        fire(new PortalLayoutChanged(
189                                        previewLayout, tabsLayout), channel);
190                        break;
191                }
192                case "renderPortlet": {
193                        fire(new RenderPortletRequest(view.renderSupport(), params.getString(0),
194                                        RenderMode.valueOf(params.getString(1)),
195                                        params.getBoolean(2)), channel);
196                        break;
197                }
198                case "setLocale": {
199                        fire(new SetLocale(Locale.forLanguageTag(params.getString(0))),
200                                        channel);
201                        break;
202                }
203                case "setTheme": {
204                        fire(new SetTheme(params.getString(0)), channel);
205                        break;
206                }
207                case "sendToPortlet": {
208                        fire(new NotifyPortletModel(view.renderSupport(), params.getString(0),
209                                        params.getString(1), params.size() <= 2
210                                        ? JsonValue.EMPTY_JSON_ARRAY : params.getJsonArray(2)),
211                                        channel);
212                        break;
213                }
214                }               
215        }
216        
217        @Handler
218        public void onPortalConfigured(
219                        PortalConfigured event, LinkedIOSubchannel channel) 
220                                        throws InterruptedException, IOException {
221                fire(new JsonOutput("portalConfigured"), channel);
222        }
223        
224        @Handler
225        public void onLastPortalLayout(
226                        LastPortalLayout event, LinkedIOSubchannel channel) 
227                                        throws InterruptedException, IOException {
228                fire(new JsonOutput("lastPortalLayout",
229                                event.previewLayout(), event.tabsLayout()), channel);
230        }
231        
232}
