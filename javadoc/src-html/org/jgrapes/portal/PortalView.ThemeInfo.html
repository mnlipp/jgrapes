120        public PortalView(Portal portal, Channel componentChannel) {
121                super(componentChannel);
122                this.portal = portal;
123                if (fmConfig == null) {
124                        fmConfig = new Configuration(Configuration.VERSION_2_3_26);
125                        fmConfig.setClassLoaderForTemplateLoading(
126                                        getClass().getClassLoader(), "org/jgrapes/portal");
127                        fmConfig.setDefaultEncoding("utf-8");
128                        fmConfig.setTemplateExceptionHandler(
129                                        TemplateExceptionHandler.RETHROW_HANDLER);
130                fmConfig.setLogTemplateExceptions(false);
131                }
132                baseTheme = new Provider();
133                
134                supportedLocales = new HashSet<>();
135                for (Locale locale: Locale.getAvailableLocales()) {
136                        if (locale.getLanguage().equals("")) {
137                                continue;
138                        }
139                        if (resourceSupplier != null) {
140                                ResourceBundle rb = resourceSupplier.apply(locale);
141                                if (rb.getLocale().equals(locale)) {
142                                        supportedLocales.add(locale);
143                                }
144                        }
145                        ResourceBundle rb = ResourceBundle.getBundle(getClass()
146                                        .getPackage().getName() + ".l10n", locale);
147                        if (rb.getLocale().equals(locale)) {
148                                supportedLocales.add(locale);
149                        }
150                }
151                
152                RequestHandler.Evaluator.add(this, "onGet", portal.prefix() + "/**");
153                
154                // Create portal model
155                portalBaseModel.put("resourceUrl", new TemplateMethodModelEx() {
156                        @Override
157                        public Object exec(@SuppressWarnings("rawtypes") List arguments)
158                                        throws TemplateModelException {
159                                @SuppressWarnings("unchecked")
160                                List<TemplateModel> args = (List<TemplateModel>)arguments;
161                                if (!(args.get(0) instanceof SimpleScalar)) {
162                                        throw new TemplateModelException("Not a string.");
163                                }
164                                return portal.prefix().resolve(
165                                                ((SimpleScalar)args.get(0)).getAsString()).getRawPath();
166                        }
167                });
168                
169                portalBaseModel.put("themeInfos", 
170                                StreamSupport.stream(themeLoader.spliterator(), false)
171                                .map(t -> new ThemeInfo(t.themeId(), t.themeName()))
172                                .sorted().toArray(size -> new ThemeInfo[size]));
173                
174                portalBaseModel = Collections.unmodifiableMap(portalBaseModel);
175        }

186        private LinkedIOSubchannel portalChannel(IOSubchannel channel) {
187                Optional<? extends LinkedIOSubchannel> portalChannel
188                        = LinkedIOSubchannel.downstreamChannel(portal, channel);
189                if (portalChannel.isPresent()) {
190                        return portalChannel.get();
191                }
192                LinkedIOSubchannel newPortalChannel
193                        = new LinkedIOSubchannel(portal, channel);
194                return newPortalChannel;
195        }

197        @RequestHandler(dynamic=true)
198        public void onGet(GetRequest event, IOSubchannel channel) 
199                        throws InterruptedException, IOException {
200                URI requestUri = event.requestUri();
201                // Append trailing slash, if missing
202                if ((requestUri.getRawPath() + "/").equals(
203                                portal.prefix().getRawPath())) {
204                        requestUri = portal.prefix();
205                }
206                
207                // Request for portal?
208                if (!requestUri.getRawPath().startsWith(portal.prefix().getRawPath())) {
209                        return;
210                }
211                
212                // Normalize and evaluate
213                requestUri = portal.prefix().relativize(
214                                URI.create(requestUri.getRawPath()));
215                if (requestUri.getRawPath().isEmpty()) {
216                        if (event.httpRequest().findField(
217                                        HttpField.UPGRADE, Converters.STRING_LIST)
218                                        .map(f -> f.value().containsIgnoreCase("websocket"))
219                                        .orElse(false)) {
220                                channel.setAssociated(this, new PortalInfo());
221                                channel.respond(new WebSocketAccepted(event));
222                                event.stop();
223                                return;
224                        }
225                        
226                        renderPortal(event, channel);
227                        return;
228                }
229                URI subUri = uriFromPath("portal-resource/").relativize(requestUri);
230                if (!subUri.equals(requestUri)) {
231                        sendPortalResource(event, channel, subUri.getPath());
232                        return;
233                }
234                subUri = uriFromPath("theme-resource/").relativize(requestUri);
235                if (!subUri.equals(requestUri)) {
236                        sendThemeResource(event, channel, subUri.getPath());
237                        return;
238                }
239                subUri = uriFromPath("portlet-resource/").relativize(requestUri);
240                if (!subUri.equals(requestUri)) {
241                        requestPortletResource(event, channel, subUri);
242                        return;
243                }
244        }

246        private void renderPortal(GetRequest event, IOSubchannel channel)
247                throws IOException, InterruptedException {
248                event.stop();
249                
250                // Because language is changed via websocket, locale cookie 
251                // may be out-dated
252                event.associated(Selection.class)
253                        .ifPresent(s -> s.prefer(s.get()[0]));
254                
255                // Prepare response
256                HttpResponse response = event.httpRequest().response().get();
257                MediaType mediaType = MediaType.builder().setType("text", "html")
258                                .setParameter("charset", "utf-8").build();
259                response.setField(HttpField.CONTENT_TYPE, mediaType);
260                response.setStatus(HttpStatus.OK);
261                response.setHasPayload(true);
262                channel.respond(new Response(response));
263                try (Writer out = new OutputStreamWriter(new ByteBufferOutputStream(
264                                channel, channel.responsePipeline()), "utf-8")) {
265                        Map<String,Object> portalModel = new HashMap<>(portalBaseModel);
266
267                        // Add locale
268                        final Locale locale = event.associated(Selection.class).map(
269                                        s -> s.get()[0]).orElse(Locale.getDefault());
270                        portalModel.put("locale", locale);
271
272                        // Add supported locales
273                        final Collator coll = Collator.getInstance(locale);
274                        final Comparator<LanguageInfo> comp 
275                                = new Comparator<PortalView.LanguageInfo>() {
276                                @Override
277                                public int compare(LanguageInfo o1,  LanguageInfo o2) {
278                                        return coll.compare(o1.getLabel(), o2.getLabel());
279                                }
280                        };
281                        LanguageInfo[] languages = supportedLocales.stream()
282                                        .map(l -> new LanguageInfo(l))
283                                        .sorted(comp).toArray(size -> new LanguageInfo[size]);
284                        portalModel.put("supportedLanguages", languages);
285
286                        // Add localization
287                        final ResourceBundle additionalResources = resourceSupplier == null
288                                        ? null : resourceSupplier.apply(locale);
289                        final ResourceBundle baseResources = ResourceBundle.getBundle(
290                                        getClass().getPackage().getName() + ".l10n", locale,
291                                        ResourceBundle.Control.getNoFallbackControl(
292                                                        ResourceBundle.Control.FORMAT_DEFAULT));
293                        portalModel.put("_", new TemplateMethodModelEx() {
294                                @Override
295                                public Object exec(@SuppressWarnings("rawtypes") List arguments)
296                                                throws TemplateModelException {
297                                        @SuppressWarnings("unchecked")
298                                        List<TemplateModel> args = (List<TemplateModel>)arguments;
299                                        if (!(args.get(0) instanceof SimpleScalar)) {
300                                                throw new TemplateModelException("Not a string.");
301                                        }
302                                        String key = ((SimpleScalar)args.get(0)).getAsString();
303                                        try {
304                                                return additionalResources.getString(key);
305                                        } catch (MissingResourceException e) {
306                                                // try base resources
307                                        }
308                                        try {
309                                                return baseResources.getString(key);
310                                        } catch (MissingResourceException e) {
311                                                // no luck
312                                        }
313                                        return key;
314                                }
315                        });
316                        Template tpl = fmConfig.getTemplate("portal.ftlh");
317                        tpl.process(portalModel, out);
318                } catch (TemplateException e) {
319                        throw new IOException(e);
320                }
321        }

323        private void sendPortalResource(GetRequest event, IOSubchannel channel,
324                        String resource) {
325                // Look for content
326                InputStream in = this.getClass().getResourceAsStream(resource);
327                if (in == null) {
328                        return;
329                }
330                
331                // Send header
332                HttpResponse response = event.httpRequest().response().get();
333                prepareResourceResponse(response, event.requestUri());
334                channel.respond(new Response(response));
335                
336                // Send content
337                activeEventPipeline().executorService()
338                        .submit(new InputStreamPipeline(in, channel));
339                
340                // Done
341                event.stop();
342        }

344        private void sendThemeResource(GetRequest event, IOSubchannel channel,
345                        String resource) {
346                try {
347                        // Get resource
348                        ThemeProvider themeProvider = event.associated(Session.class)
349                                        .map(session -> (ThemeProvider)session.get("themeProvider"))
350                                        .orElse(baseTheme);
351                        InputStream resIn;
352                        try {
353                                resIn = themeProvider.getResourceAsStream(resource);
354                        } catch (ResourceNotFoundException e) {
355                                resIn = baseTheme.getResourceAsStream(resource);
356                        }
357                        
358                        // Send header
359                        HttpResponse response = event.httpRequest().response().get();
360                        prepareResourceResponse(response, event.requestUri());
361                        channel.respond(new Response(response));
362                
363                        // Send content
364                        activeEventPipeline().executorService()
365                                .submit(new InputStreamPipeline(resIn, channel));
366                
367                        // Done
368                        event.stop();
369                } catch (ResourceNotFoundException e) {
370                        return;
371                }
372        }

374        public static void prepareResourceResponse(
375                        HttpResponse response, URI request) {
376                response.setContentType(request);
377                // Set max age in cache-control header
378                List<Directive> directives = new ArrayList<>();
379                directives.add(new Directive("max-age", 600));
380                response.setField(HttpField.CACHE_CONTROL, directives);
381                response.setField(HttpField.LAST_MODIFIED, Instant.now());
382
383                response.setStatus(HttpStatus.OK);
384        }

386        private void requestPortletResource(GetRequest event, IOSubchannel channel,
387                        URI resource) throws InterruptedException {
388                String resPath = resource.getPath();
389                int sep = resPath.indexOf('/');
390                // Send events to portlets on portal's channel
391                if (Boolean.TRUE.equals(newEventPipeline().fire(
392                                new PortletResourceRequest(resPath.substring(0, sep), 
393                                                uriFromPath(resPath.substring(sep + 1)),
394                                                event.httpRequest(), channel), portalChannel(channel))
395                                .get())) {
396                        event.stop();
397                }
398        }

400        @Handler
401        public void onInput(Input<ManagedCharBuffer> event, IOSubchannel channel)
402                        throws IOException {
403                Optional<PortalInfo> optPortalInfo 
404                        = channel.associated(this, PortalInfo.class);
405                if (!optPortalInfo.isPresent()) {
406                        return;
407                }
408                optPortalInfo.get().toEvent(portalChannel(channel),
409                                event.buffer().backingBuffer(), event.isEndOfRecord());
410        }

412        void setLocale(IOSubchannel channel, String locale) {
413                supportedLocales.stream()
414                        .filter(l -> l.toString().equals(locale)).findFirst()
415                        .ifPresent(l -> channel.associated(Selection.class)
416                                        .map(s -> s.prefer(l)));
417        }

419        void setTheme(IOSubchannel channel, String theme) {
420                StreamSupport.stream(themeLoader.spliterator(), false)
421                        .filter(t -> t.themeId().equals(theme)).findFirst()
422                        .ifPresent(themeProvider ->  
423                                channel.associated(Session.class).map(session ->
424                                        session.put("themeProvider", themeProvider)));
425        }

427        void onAddPortletType(
428                        AddPortletType event, LinkedIOSubchannel channel)
429                                        throws InterruptedException, IOException {
430                sendNotificationResponse(channel, "addPortletType",
431                                event.portletType(),
432                                event.displayName(),
433                                Arrays.stream(event.cssUris()).map(uri -> 
434                                renderSupport.portletResource(
435                                                event.portletType(), uri).toString())
436                                .toArray(String[]::new),
437                                Arrays.stream(event.scriptUris()).map(uri -> 
438                                renderSupport.portletResource(event.portletType(), uri).toString())
439                                .toArray(String[]::new),
440                                event.isInstantiable());
441        }

472        public void onPortletResourceResponse(PortletResourceResponse event,
473                LinkedIOSubchannel channel) {
474                HttpRequest request = event.request().httpRequest();
475                // Send header
476                HttpResponse response = request.response().get();
477                prepareResourceResponse(response, request.requestUri());
478                channel.upstreamChannel().respond(new Response(response));
479                
480                // Send content
481                activeEventPipeline().executorService().submit(
482                                new InputStreamPipeline(
483                                                event.stream(), channel.upstreamChannel()));
484        }

494        void sendNotificationResponse(LinkedIOSubchannel channel,
495                String method, Object... params)
496                                throws InterruptedException, IOException {
497                JsonBuilderFactory factory = Json.createBuilderFactory(null);
498                JsonObjectBuilder notification = factory.createObjectBuilder()
499                                .add("jsonrpc", "2.0").add("method", method);
500                if (params.length > 0) {
501                        notification.add("params", JsonUtil.toJsonArray(factory, params));
502                }
503                IOSubchannel upstream = channel.upstreamChannel();
504                @SuppressWarnings("resource")
505                CharBufferWriter out = new CharBufferWriter(upstream, 
506                                upstream.responsePipeline()).suppressClose();
507                Json.createWriter(out).write(notification.build());
508                out.close();
509        }

511        private class PortalInfo {
512
513                private PipedWriter decodeWriter;
514                
515                public void toEvent(IOSubchannel channel, CharBuffer buffer,
516                                boolean last) throws IOException {
517                        if (decodeWriter == null) {
518                                decodeWriter = new PipedWriter();
519                                PipedReader reader = new PipedReader(
520                                                decodeWriter, buffer.capacity());
521                                activeEventPipeline().executorService()
522                                        .submit(new DecodeTask(reader, channel));
523                        }
524                        decodeWriter.append(buffer);
525                        if (last) {
526                                decodeWriter.close();
527                                decodeWriter = null;
528                        }
529                }
530                
531                private class DecodeTask implements Runnable {
532
533                        IOSubchannel channel;
534                        private Reader reader;
535                        
536                        public DecodeTask(Reader reader, IOSubchannel channel) {
537                                this.reader = reader;
538                                this.channel = channel;
539                        }
540
541                        /* (non-Javadoc)
542                         * @see java.lang.Runnable#run()
543                         */
544                        @Override
545                        public void run() {
546                                try (Reader in = reader) {
547                                        JsonReader reader = Json.createReader(in);
548                                        fire(new JsonRequest(reader.readObject()), channel);
549                                } catch (IOException e) {
550                                        // Shouldn't happen
551                                }
552                        }
553                }
554        }

556        public static class LanguageInfo {
557                private Locale locale;
558
559                /**
560                 * @param locale
561                 */
562                public LanguageInfo(Locale locale) {
563                        this.locale = locale;
564                }
565
566                /**
567                 * @return the locale
568                 */
569                public Locale getLocale() {
570                        return locale;
571                }

573                public String getLabel() {
574                        String str = locale.getDisplayName(locale);
575                        return Character.toUpperCase(str.charAt(0)) + str.substring(1);
576                }

616        /**
617         * Create a {@link URI} from a path. This is similar to calling
618         * `new URI(null, null, path, null)` with the {@link URISyntaxException}
619         * converted to a {@link IllegalArgumentException}.
620         * 
621         * @param path the path
622         * @return the uri
623         * @throws IllegalArgumentException if the string violates 
624         * RFC 2396
625         */
626        public static URI uriFromPath(String path) throws IllegalArgumentException {
627                try {
628                        return new URI(null, null, path, null);
629                } catch (URISyntaxException e) {
630                        throw new IllegalArgumentException(e);
631                }
632        }
