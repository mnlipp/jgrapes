019package org.jgrapes.portal;
020
021import java.io.ByteArrayInputStream;
022import java.io.ByteArrayOutputStream;
023import java.io.IOException;
024import java.io.ObjectInputStream;
025import java.io.ObjectOutputStream;
026import java.io.UnsupportedEncodingException;
027import java.util.Base64;
028import java.util.HashMap;
029import java.util.Map;
030import java.util.Optional;
031
032import org.jgrapes.core.Channel;
033import org.jgrapes.core.CompletionLock;
034import org.jgrapes.core.Component;
035import org.jgrapes.core.annotation.Handler;
036import org.jgrapes.http.Session;
037import org.jgrapes.io.IOSubchannel;
038import org.jgrapes.io.util.LinkedIOSubchannel;
039import org.jgrapes.portal.events.LastPortalLayout;
040import org.jgrapes.portal.events.PortalLayoutChanged;
041import org.jgrapes.portal.events.PortalPrepared;
042import org.jgrapes.portal.events.PortalReady;
043import org.jgrapes.portal.events.RenderPortletRequest;
044import org.jgrapes.util.events.KeyValueStoreData;
045import org.jgrapes.util.events.KeyValueStoreQuery;
046import org.jgrapes.util.events.KeyValueStoreUpdate;
047
048/**
049 * 
050 */
051public class BrowserBasedPortalPolicy extends Component {
052
053        private static final String DATA_KEY 
054                = "/" + BrowserBasedPortalPolicy.class.getName();
055
056        /**
057         * Creates a new component with its channel set to
058         * itself.
059         */
060        public BrowserBasedPortalPolicy() {
061        }
062
063        /**
064         * Creates a new component with its channel set to the given channel.
065         * 
066         * @param componentChannel
067         */
068        public BrowserBasedPortalPolicy(Channel componentChannel) {
069                super(componentChannel);
070        }
071
072        /**
073         * Intercept the {@link PortalReady} event. Request the 
074         * session data from the portal and resume.
075         * 
076         * @param event
077         * @param channel
078         * @throws InterruptedException
079         */
080        @Handler
081        public void onPortalReady(PortalReady event, IOSubchannel channel) 
082                        throws InterruptedException {
083                Optional<PortalSession> session = lookupPortalSession(channel);
084                if(session.isPresent()) {
085                        session.get().onPortalReady(event, channel);
086                }
087        }
088
089        @Handler
090        public void onKeyValueStoreData(
091                        KeyValueStoreData event, IOSubchannel channel) {
092                lookupPortalSession(channel).ifPresent(
093                                ps -> ps.onKeyValueStoreData(event, channel));
094        }
095
096        @Handler
097        public void onPortalPrepared(PortalPrepared event, IOSubchannel channel) {
098                lookupPortalSession(channel).ifPresent(
099                                ps -> ps.onPortalPrepared(event, channel));
100        }
101
102        @Handler
103        public void onPortalLayoutChanged(PortalLayoutChanged event, 
104                        LinkedIOSubchannel channel) {
105                lookupPortalSession(channel).ifPresent(
106                                ps -> ps.onPortalLayoutChanged(event, channel));
107        }
108        
109        private Optional<PortalSession> lookupPortalSession(IOSubchannel channel) {
110                return channel.associated(Session.class)
111                        .map(session -> (PortalSession)session.computeIfAbsent(
112                                        BrowserBasedPortalPolicy.class, 
113                                        k -> new PortalSession()));
114        }
115        
116        private class PortalSession {
117
118                private CompletionLock readyLock;
119                private Map<String,Object> persisted = null;
120                
121                public void onPortalReady(PortalReady event, IOSubchannel channel) 
122                                throws InterruptedException {
123                        if (persisted != null) {
124                                return;
125                        }
126                        KeyValueStoreQuery query = new KeyValueStoreQuery(DATA_KEY, true);
127                        readyLock = new CompletionLock(event, 3000);
128                        fire(query, channel);
129                }
130
131                public void onKeyValueStoreData(
132                                KeyValueStoreData event, IOSubchannel channel) {
133                        if (!event.event().query().equals(DATA_KEY)) {
134                                return;
135                        }
136                        String data = event.data().get(DATA_KEY);
137                        if (data != null) {
138                                try (ObjectInputStream in = new ObjectInputStream(
139                                                Base64.getDecoder().wrap(
140                                                                new ByteArrayInputStream(
141                                                                                data.getBytes("utf-8"))))) {
142                                        @SuppressWarnings("unchecked")
143                                        Map<String,Object> restored 
144                                                = (Map<String,Object>)in.readObject();
145                                        persisted = restored;
146                                } catch (IOException | ClassNotFoundException e) {
147                                        // cannot happen
148                                }
149                        }
150                        readyLock.remove();
151                        readyLock = null;
152                }
153                
154                public void onPortalPrepared(
155                                PortalPrepared event, IOSubchannel channel) {
156                        if (persisted == null) {
157                                // Retrieval was not successful
158                                persisted = new HashMap<>();
159                        }
160                        // Make sure data is consistent
161                        String[][] previewLayout = (String[][])persisted.computeIfAbsent(
162                                        "previewLayout", k -> { return new String[0][0]; });
163                        String[] tabsLayout = (String[])persisted.computeIfAbsent(
164                                        "tabsLayout", k -> { return new String[0]; });
165
166                        // Update layout
167                        channel.respond(new LastPortalLayout(previewLayout, tabsLayout));
168                        
169                        // Restore portlets
170                        for (String portletId: tabsLayout) {
171                                fire(new RenderPortletRequest(
172                                                event.event().renderSupport(), portletId,
173                                                Portlet.RenderMode.View, false), channel);
174                        }
175                        for (String[] column: previewLayout) {
176                                for (String portletId: column) {
177                                        fire(new RenderPortletRequest(
178                                                        event.event().renderSupport(), portletId,
179                                                        Portlet.RenderMode.Preview, true), channel);
180                                }
181                        }
182                }
183                
184                public void onPortalLayoutChanged(
185                                PortalLayoutChanged event, IOSubchannel channel) {
186                        persisted.put("previewLayout", event.previewLayout());
187                        persisted.put("tabsLayout", event.tabsLayout());
188                        storeState(channel);
189                }
190                
191                private void storeState(IOSubchannel channel) {
192                        ByteArrayOutputStream encoded = new ByteArrayOutputStream();
193                        try (ObjectOutputStream out = new ObjectOutputStream(
194                                        Base64.getEncoder().wrap(encoded))) {
195                                out.writeObject(persisted);
196                        } catch (IOException e) {
197                                // cannot happen
198                        }
199                        try {
200                                String data = encoded.toString("utf-8");
201                                fire(new KeyValueStoreUpdate()
202                                                .update(DATA_KEY, data), channel);
203                        } catch (UnsupportedEncodingException e) {
204                                // cannot happen
205                        }
206                }
207
208        }
209}
