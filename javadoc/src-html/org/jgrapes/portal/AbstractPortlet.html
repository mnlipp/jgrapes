048/**
049 * Provides a base class for implementing portlet components that
050 * maintain the portlet's state in the session.
051 * 
052 * Method {@link #putInSession(Session, String, Serializable)} puts
053 * arbitrary state information for the portlet type (as derived from 
054 * {@link #type()}) in the session associated with the channel.
055 * Method {@link #putInSession(Session, PortletBaseModel)} handles
056 * state information that is derived from {@link PortletBaseModel}.
057 * Deriving the portlet's state representation from this class
058 * saves passing the portlet id as parameter for some invocatiosn. 
059 * 
060 * Method {@link #stateFromSession(Session, String, Class)} retrieves 
061 * the portelt's state from the session, 
062 * {@link #stateFromSession(Session, String, Function)} may also create
063 * the state information if none exists.
064 * 
065 * {@link #removeFromSession(Session, String)} removes
066 * the state information.
067 * 
068 * Using these methods, this class also provides basic event handlers that
069 * implement e.g. the necessary lookup of a session required by every portlet
070 * that uses portlet state information.
071 */
072public abstract class AbstractPortlet extends Component {       
073        
074        /**
075         * Creates a new component that listens for new events
076         * on the given channel.
077         * 
078         * @param channel the channel to listen on
079         */
080        public AbstractPortlet(Channel channel) {
081                super(channel);
082        }
083
084        /**
085         * Returns the portlet type. The default implementation
086         * returns the class' name.
087         * 
088         * @return the type
089         */
090        protected String type() {
091                return getClass().getName();
092        }
093        
094        /**
095         * Generates a new unique portlet id.
096         * 
097         * @return the portlet id
098         */
099        protected String generatePortletId() {
100                return UUID.randomUUID().toString();
101        }
102        
103        /**
104         * Returns the locale associated with the channel or the default
105         * locale if none is associated.
106         * 
107         * @param channel the channel
108         * @return the locale
109         */
110        protected Locale locale(IOSubchannel channel) {
111                return channel.associated(Selection.class)
112                                .map(s -> s.get()[0]).orElse(Locale.getDefault());
113        }
114        
115        /**
116         * A default handler for resource requests. Checks that the request
117         * is directed at this portlet, and calls {@link #doGetResource}.
118         * 
119         * @param event the resource request event
120         * @param channel the channel that the request was recived on
121         */
122        @Handler
123        public void onResourceRequest(
124                        PortletResourceRequest event, IOSubchannel channel) {
125                // For me?
126                if (!event.portletClass().equals(type())) {
127                        return;
128                }
129                
130                doGetResource(event, channel);
131        }
132
133        /**
134         * Generates the reponse event for a resource request
135         * (a {@link PortletResourceResponse} and at least one
136         * {@link Output} event.
137         * 
138         * The default implementation searches for
139         * a file with the requested URI in the portlets class path
140         * and sends its content if found.
141         * 
142         * @param event the event. The result will be set to
143         * `true` on success
144         * @param channel the channel
145         */
146        protected void doGetResource(PortletResourceRequest event,
147                        IOSubchannel channel) {
148                InputStream stream = this.getClass().getResourceAsStream(
149                                event.resourceUri().getPath());
150                if (stream == null) {
151                        return;
152                }
153                
154                // Found resource, send.
155                channel.respond(new PortletResourceResponse(event, false));
156                new InputStreamPipeline(stream, channel).suppressClose().run();
157                event.setResult(true);
158        }
159        
160        /**
161         * Provides resource bundle for localization.
162         * The default implementation looks up a bundle using the
163         * package name plus "l10n" as base name.
164         * 
165         * @return the resource bundle
166         */
167        protected ResourceBundle resourceBundle(Locale locale) {
168                return ResourceBundle.getBundle(
169                        getClass().getPackage().getName() + ".l10n", locale, 
170                        getClass().getClassLoader(),
171                                ResourceBundle.Control.getNoFallbackControl(
172                                                ResourceBundle.Control.FORMAT_DEFAULT));
173        }
174
175        /**
176         * Puts the given portlet state in the session using the 
177         * {@link #type()} and the given portlet id as keys.
178         * 
179         * @param session the session to use
180         * @param portletId the portlet id
181         * @param portletState the portlet state
182         * @return the portlet state
183         */
184        @SuppressWarnings("unchecked")
185        protected <T extends Serializable> T putInSession(
186                        Session session, String portletId, T portletState) {
187                ((Map<Serializable,Map<Serializable,Map<String,Serializable>>>)
188                                (Map<Serializable,?>)session)
189                        .computeIfAbsent(AbstractPortlet.class, ac -> new HashMap<>())
190                        .computeIfAbsent(type(), t -> new HashMap<>())
191                        .put(portletId, portletState);
192                return portletState;
193        }
194        
195        /**
196         * Puts the given portlet model in the session using the 
197         * {@link #type()} and the model's id as keys.
198         * 
199         * @param session the session to use
200         * @param portletModel the portlet model
201         * @return the portlet model
202         */
203        protected <T extends PortletBaseModel> T putInSession(
204                        Session session, T portletModel) {
205                        return putInSession(
206                                        session, portletModel.getPortletId(), portletModel);
207        }
208        
209        /**
210         * Returns all portlet states of this portlet's type from the
211         * session.
212         * 
213         * @param channel the channel, used to access the session
214         * @param type the states' type
215         * @return the states
216         */
217        @SuppressWarnings("unchecked")
218        protected <T extends Serializable> Collection<T> statesFromSession(
219                        IOSubchannel channel, Class<T> type) {
220                return channel.associated(Session.class).map(session ->
221                                ((Map<Serializable,Map<Serializable,Map<String,T>>>)
222                                                (Map<Serializable,?>)session)
223                                .computeIfAbsent(AbstractPortlet.class, ac -> new HashMap<>())
224                                .computeIfAbsent(type(), t -> new HashMap<>()).values())
225                        .orElseThrow(() -> new IllegalStateException("Session is missing."));
226        }
227        
228        /**
229         * Returns the portlet state of this portlet's type with the given id
230         * from the session.
231         * 
232         * @param session the session to use
233         * @param portletId the portlet id
234         * @param type the state's type
235         * @return the portlet state
236         */
237        @SuppressWarnings("unchecked")
238        protected <T extends Serializable> Optional<T> stateFromSession(
239                        Session session, String portletId, Class<T> type) {
240                return Optional.ofNullable(
241                                ((Map<Serializable,Map<Serializable,Map<String,T>>>)
242                                                (Map<Serializable,?>)session)
243                                .computeIfAbsent(AbstractPortlet.class, ac -> new HashMap<>())
244                                .computeIfAbsent(type(), t -> new HashMap<>())
245                                .get(portletId));
246        }
247
248        /**
249         * Returns the portlet state of this portlet's type with the given id
250         * from the session, creating it if it doesn't exist. The function
251         * creating the state is invoked with the portlet id.
252         * 
253         * This method uses {@link #stateFromSession(Session, String, Class)}
254         * to find out if the state information exists already.
255         * 
256         * @param session the session to use
257         * @param portletId the portlet id
258         * @param supplier the function that creates a new state
259         * @return the portlet state
260         */
261        @SuppressWarnings("unchecked")
262        protected <T extends Serializable> T stateFromSession(
263                        Session session, String portletId, Function<String,T> supplier) {
264                return ((Optional<T>)stateFromSession(
265                                session, portletId, Serializable.class))
266                                .orElse(putInSession(
267                                                session, portletId, supplier.apply(portletId)));
268        }
269
270        /**
271         * Removes the portlet state of the portlet with the given id
272         * from the session.
273         * 
274         * @param session the session to use
275         * @param portletId the portlet id
276         */
277        @SuppressWarnings("unchecked")
278        protected void removeFromSession(
279                        Session session, String portletId) {
280                ((Map<Serializable,Map<Serializable,Map<String,Serializable>>>)
281                                (Map<Serializable,?>)session)
282                        .computeIfAbsent(AbstractPortlet.class, ac -> new HashMap<>())
283                        .computeIfAbsent(type(), t -> new HashMap<>())
284                        .remove(portletId);
285        }
286
287        /**
288         * Removes the portlet with the given model
289         * from the session.
290         * 
291         * @param session the session to use
292         * @param portletModel the portlet model
293         * @return the portlet model
294         */
295        @SuppressWarnings("unchecked")
296        protected <T extends PortletBaseModel> T removeFromSession(
297                        Session session, T portletModel) {
298                ((Map<Serializable,Map<Serializable,Map<String,Serializable>>>)
299                                (Map<Serializable,?>)session)
300                        .computeIfAbsent(AbstractPortlet.class, ac -> new HashMap<>())
301                        .computeIfAbsent(type(), t -> new HashMap<>())
302                        .remove(portletModel.getPortletId());
303                return portletModel;
304        }
305
306        /**
307         * Remove all models belonging to this type of portlet
308         * from the session.
309         * 
310         * @param session the session
311         */
312        @SuppressWarnings("unchecked")
313        protected void removeAllModelsFromSession(Session session) {
314                ((Map<Serializable,Map<Serializable,Map<String,PortletBaseModel>>>)
315                                (Map<Serializable,?>)session)
316                        .computeIfAbsent(AbstractPortlet.class, ac -> new HashMap<>())
317                        .computeIfAbsent(type(), t -> new HashMap<>()).clear();
318        }
319        
320        /**
321         * Checks if the request applies to this component. If so, stops the event,
322         * and calls {@link #doAddPortlet}. 
323         * 
324         * @param event the event
325         * @param channel the channel
326         */
327        @Handler
328        public void onAddPortletRequest(AddPortletRequest event,
329                        IOSubchannel channel) throws Exception {
330                if (!event.portletType().equals(type())) {
331                        return;
332                }
333                event.stop();
334                Session session = session(channel);
335                doAddPortlet(event, channel, session);
336        }
337
338        /**
339         * Called by {@link #onAddPortletRequest} to complete adding the portlet.
340         * If the portlet has associated state, the implementation should
341         * call {@link #stateFromSession(Session, String, Function)} to create
342         * the state and put it in the session.
343         * 
344         * @param event the event
345         * @param channel the channel
346         * @param session the session
347         */
348        protected abstract void doAddPortlet(AddPortletRequest event, 
349                        IOSubchannel channel, Session session) throws Exception;
350        
351        /**
352         * Checks if the request applies to this component. If so, stops the event,
353         * removes the portlet model bean from the session
354         * and calls {@link #doDeletePortlet}
355         * with the model. 
356         * 
357         * @param event the event
358         * @param channel the channel
359         */
360        @Handler
361        public void onDeletePortletRequest(DeletePortletRequest event, 
362                        IOSubchannel channel) throws Exception {
363                Session session = session(channel);
364                Optional<? extends Serializable> optPortletState 
365                        = stateFromSession(session, event.portletId(), Serializable.class);
366                if (!optPortletState.isPresent()) {
367                        return;
368                }
369                event.stop();
370                Serializable portletState = optPortletState.get();
371                removeFromSession(session, event.portletId());
372                doDeletePortlet(event, channel, session, event.portletId(), portletState);
373        }
374        
375        /**
376         * Called by {@link #onDeletePortletRequest} to complete deleting
377         * the portlet.
378         * 
379         * @param event the event
380         * @param channel the channel
381         * @param session the session
382         * @param portletId the portlet id
383         * @param portletState the portlet state
384         */
385        protected abstract void doDeletePortlet(DeletePortletRequest event, 
386                        IOSubchannel channel, Session session, String portletId, 
387                        Serializable portletState) throws Exception;
388        
389        /**
390         * Checks if the request applies to this component by calling
391         * {@link #stateFromSession(Session, String, Class)}. If a model
392         * is found, stops the event, and calls 
393         * {@link #doRenderPortlet} with the state information. 
394         * 
395         * Some portlets that do not persist their models between sessions
396         * (e.g. because the model only references data maintained elsewhere)
397         * should override {@link #stateFromSession(Session, String, Class)}
398         * in such a way that it creates the requested model if it doesn't 
399         * exist yet.
400         * 
401         * @param event the event
402         * @param channel the channel
403         */
404        @Handler
405        public void onRenderPortlet(RenderPortletRequest event,
406                        IOSubchannel channel) throws Exception {
407                Session session = session(channel);
408                Optional<? extends Serializable> optPortletState 
409                        = stateFromSession(session, event.portletId(), Serializable.class);
410                if (!optPortletState.isPresent()) {
411                        return;
412                }
413                event.stop();
414                doRenderPortlet(event, channel, session, 
415                                event.portletId(), optPortletState.get());
416        }
417
418        /**
419         * Called by {@link #onRenderPortlet} to complete rendering
420         * the portlet.
421         * 
422         * @param event the event
423         * @param channel the channel
424         * @param portletId the portlet id
425         * @param portletState the portletState
426         */
427        protected abstract void doRenderPortlet(RenderPortletRequest event, 
428                        IOSubchannel channel, Session session, String portletId, 
429                        Serializable portletState) throws Exception;
430        
431        /**
432         * Convenience method that returns the session associated with
433         * the channel.
434         * 
435         * @return the session
436         * @throws IllegalStateException if no session is found
437         */
438        public Session session(IOSubchannel channel) {
439                return channel.associated(Session.class).orElseThrow(
440                                () -> new IllegalStateException("Session is missing."));
441        }
442        
443        /**
444         * Checks if the request applies to this component by calling
445         * {@link #stateFromSession(Session, String, Class)}. If a model
446         * is found, calls {@link #doNotifyPortletModel} with the state 
447         * information. 
448         * 
449         * @param event the event
450         * @param channel the channel
451         */
452        @Handler
453        public void onNotifyPortletModel(NotifyPortletModel event,
454                        IOSubchannel channel) throws Exception {
455                Session session = session(channel);
456                Optional<? extends Serializable> optPortletState 
457                        = stateFromSession(session, event.portletId(), Serializable.class);
458                if (!optPortletState.isPresent()) {
459                        return;
460                }
461                
462                doNotifyPortletModel(event, channel, session, optPortletState.get());
463        }
464
465        /**
466         * Called by {@link #onNotifyPortletModel} to complete handling
467         * the notification. The default implementation does nothing.
468         * 
469         * @param event the event
470         * @param channel the channel
471         * @param session the session
472         * @param portletState the portletState
473         */
474        protected void doNotifyPortletModel(NotifyPortletModel event, 
475                        IOSubchannel channel, Session session,  
476                        Serializable portletState) throws Exception {
477        }
478        
479        /**
480         * Defines the portlet model following the JavaBean conventions.
481         * 
482         * Portlet models should follow these conventions because
483         * many template engines rely on them and to support serialization
484         * to portable formats. 
485         */
486        @SuppressWarnings("serial")
487        public static class PortletBaseModel implements Serializable {
488
489                protected String portletId;
490
491                /**
492                 * Creates a new model with the given type and id.
493                 * 
494                 * @param portletId the portlet id
495                 */
496                @ConstructorProperties({"portletId"})
497                public PortletBaseModel(String portletId) {
498                        this.portletId = portletId;
499                }
500
501                /**
502                 * Returns the portlet id.
503                 * 
504                 * @return the portlet id
505                 */
506                public String getPortletId() {
507                        return portletId;
508                }
509
510                /* (non-Javadoc)
511                 * @see java.lang.Object#hashCode()
512                 */
513                @Override
514                public int hashCode() {
515                        final int prime = 31;
516                        int result = 1;
517                        result = prime * result
518                                + ((portletId == null) ? 0 : portletId.hashCode());
519                        return result;
520                }
521
522                /**
523                 * Two objects are equal if they have equal portlet ids.
524                 * 
525                 * @param obj the other object
526                 * @return the result
527                 */
528                @Override
529                public boolean equals(Object obj) {
530                        if (this == obj) {
531                                return true;
532                        }
533                        if (obj == null) {
534                                return false;
535                        }
536                        if (getClass() != obj.getClass()) {
537                                return false;
538                        }
539                        PortletBaseModel other = (PortletBaseModel) obj;
540                        if (portletId == null) {
541                                if (other.portletId != null) {
542                                        return false;
543                                }
544                        } else if (!portletId.equals(other.portletId)) {
545                                return false;
546                        }
547                        return true;
548                }
549        }
