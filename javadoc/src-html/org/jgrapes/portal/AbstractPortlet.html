019package org.jgrapes.portal;
020
021import java.io.InputStream;
022import java.util.Collection;
023import java.util.HashMap;
024import java.util.Locale;
025import java.util.Map;
026import java.util.Optional;
027import java.util.ResourceBundle;
028import java.util.UUID;
029import java.util.function.Function;
030
031import org.jgrapes.core.Channel;
032import org.jgrapes.core.Component;
033import org.jgrapes.core.annotation.Handler;
034import org.jgrapes.http.Session;
035import org.jgrapes.io.IOSubchannel;
036import org.jgrapes.portal.events.PortletResourceRequest;
037import org.jgrapes.portal.events.PortletResourceResponse;
038
039/**
040 * Provides a base class for implementing portlet components. 
041 */
042public abstract class AbstractPortlet extends Component {       
043        
044        /**
045         * Creates a new component that listens for new events
046         * on the given channel.
047         * 
048         * @param channel the channel to listen on
049         */
050        public AbstractPortlet(Channel channel) {
051                super(channel);
052        }
053
054        /**
055         * A default handler for resource requests. Searches for
056         * a file with the requested URI in the portlets class path. 
057         * 
058         * @param event the resource request event
059         * @param channel the channel that the request was recived on
060         */
061        @Handler
062        public void onResourceRequest(
063                        PortletResourceRequest event, IOSubchannel channel) {
064                // For me?
065                if (!event.portletClass().equals(getClass().getName())) {
066                        return;
067                }
068                
069                // Look for content
070                InputStream in = this.getClass().getResourceAsStream(
071                                event.resourceUri().getPath());
072                if (in == null) {
073                        return;
074                }
075
076                // Respond
077                channel.respond(new PortletResourceResponse(event, in));
078                
079                // Done
080                event.setResult(true);
081        }
082
083        /**
084         * Returns all portlet models of this portlet's class from the
085         * session.
086         * 
087         * @param channel the channel, used to access the session
088         * @return the models
089         */
090        @SuppressWarnings("unchecked")
091        protected Collection<PortletModelBean> modelsFromSession(
092                        IOSubchannel channel) {
093                return channel.associated(Session.class).map(session ->
094                        ((Map<String,PortletModelBean>)session.computeIfAbsent(getClass(),
095                                        k -> new HashMap<>())).values())
096                        .orElseThrow(() -> new IllegalStateException("Session is missing."));
097        }
098        
099        /**
100         * Returns the portlet model of this portlet's class with the given id
101         * from the session.
102         * 
103         * @param channel the channel, used to access the session
104         * @param portletId the portel id
105         * @return the models
106         */
107        @SuppressWarnings("unchecked")
108        protected Optional<PortletModelBean> modelFromSession(
109                        IOSubchannel channel, String portletId) {
110                return channel.associated(Session.class).map(session ->
111                        Optional.ofNullable(
112                                        ((Map<String,PortletModelBean>)session.computeIfAbsent(
113                                                        getClass(), k -> new HashMap<>()))
114                                        .get(portletId)))
115                        .orElseThrow(() -> new IllegalStateException("Session is missing."));
116        }
117
118        /**
119         * Adds the given portlet model to the session.
120         * 
121         * @param channel the channel, used to access the session
122         * @param model the model
123         * @return the model
124         */
125        @SuppressWarnings("unchecked")
126        protected <T extends PortletModelBean> T addToSession(
127                        IOSubchannel channel, T model) {
128                Optional<Session> optSession = channel.associated(Session.class);
129                if (!optSession.isPresent()) {
130                        throw new IllegalStateException("Session is missing.");
131                }
132                ((Map<String,PortletModelBean>)optSession.get()
133                                .computeIfAbsent(getClass(), k -> new HashMap<>()))
134                        .put(model.getPortletId(), model);
135                return model;
136        }
137        
138        /**
139         * Adds the given portlet model to the session.
140         * 
141         * @param channel the channel, used to access the session
142         * @param model the model
143         * @return the model
144         */
145        @SuppressWarnings("unchecked")
146        protected <T extends PortletModelBean> T removeFromSession(
147                        IOSubchannel channel, T model) {
148                Optional<Session> optSession = channel.associated(Session.class);
149                if (!optSession.isPresent()) {
150                        throw new IllegalStateException("Session is missing.");
151                }
152                ((Map<String,PortletModelBean>)optSession.get()
153                                .computeIfAbsent(getClass(), k -> new HashMap<>()))
154                        .remove(model.getPortletId());
155                return model;
156        }
157        
158        /**
159         * Provides a supplier for a resource bundle for localization.
160         * The default implementation looks up a bundle using the
161         * package name plus "l10n" as base name.
162         * 
163         * @return the supplier
164         */
165        protected Function<Locale,ResourceBundle> resourceSupplier() {
166                return locale -> ResourceBundle.getBundle(
167                        getClass().getPackage().getName() + ".l10n", locale, 
168                                ResourceBundle.Control.getNoFallbackControl(
169                                                ResourceBundle.Control.FORMAT_DEFAULT));
170                
171        }
172        
173        /**
174         * Defines the portlet model following the JavaBean conventions.
175         * Portlet models should follow these cpnventions because
176         * many template engines rely on this. 
177         */
178        public abstract static class PortletModelBean {
179
180                private String portletId;
181
182                /**
183                 * Creates a new model with a new unique portlet id.
184                 */
185                public PortletModelBean() {
186                        portletId = UUID.randomUUID().toString();
187                }
188
189                /**
190                 * Creates a new model with the given portlet id.
191                 * 
192                 * @param portletId the portlet id
193                 */
194                public PortletModelBean(String portletId) {
195                        this.portletId = portletId;
196                }
197
198                /**
199                 * Returns the portlet id.
200                 * 
201                 * @return the portlet id
202                 */
203                public String getPortletId() {
204                        return portletId;
205                }
206        }
207}
