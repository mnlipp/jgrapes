019package org.jgrapes.portal;
020
021import java.beans.ConstructorProperties;
022import java.io.InputStream;
023import java.io.Serializable;
024import java.util.Collection;
025import java.util.HashMap;
026import java.util.Locale;
027import java.util.Map;
028import java.util.Optional;
029import java.util.ResourceBundle;
030import java.util.UUID;
031import java.util.function.Function;
032
033import org.jgrapes.core.Channel;
034import org.jgrapes.core.Component;
035import org.jgrapes.core.annotation.Handler;
036import org.jgrapes.http.Session;
037import org.jgrapes.io.IOSubchannel;
038import org.jgrapes.portal.events.AddPortletRequest;
039import org.jgrapes.portal.events.DeletePortletRequest;
040import org.jgrapes.portal.events.PortletResourceRequest;
041import org.jgrapes.portal.events.PortletResourceResponse;
042import org.jgrapes.portal.events.RenderPortletRequest;
043
044/**
045 * Provides a base class for implementing portlet components that
046 * maintain a model for each portlet as a JavaBean in the session.
047 * 
048 * Method {@link #addToSession} adds
049 * a model for the portlet type (as derived from {@link #type()}) 
050 * to the session associated with the channel. Method
051 * {@link #modelFromSession} retrieves a model,
052 * {@link AbstractPortlet#removeFromSession}
053 * removes it.
054 * 
055 * Using these methods, this class also provides basic event handlers that
056 * implement e.g. the necessary lookup of a session required by every portlet
057 * that uses a portlet model.
058 * 
059 */
060public abstract class AbstractPortlet extends Component {       
061        
062        /**
063         * Creates a new component that listens for new events
064         * on the given channel.
065         * 
066         * @param channel the channel to listen on
067         */
068        public AbstractPortlet(Channel channel) {
069                super(channel);
070        }
071
072        /**
073         * Returns the portlet type. The default implementation
074         * returns the class' name.
075         * 
076         * @return the type
077         */
078        protected String type() {
079                return getClass().getName();
080        }
081        
082        /**
083         * Generates a new unique portlet id.
084         * 
085         * @return the portlet id
086         */
087        protected String generatePortletId() {
088                return UUID.randomUUID().toString();
089        }
090        
091        /**
092         * Creates new portlet model bean. Should be 
093         * overridden by sub classes that need a derived model with more 
094         * information than the base model.
095         * 
096         * The default implementation creates a model with
097         * `new PortletBaseModel(generatePortletId())`.
098         * 
099         * @return the model bean
100         */
101        protected PortletBaseModel createPortletModel() {
102                PortletBaseModel portletModel = new PortletBaseModel(
103                                generatePortletId());
104                return portletModel;
105        }
106        
107        /**
108         * A default handler for resource requests. Searches for
109         * a file with the requested URI in the portlets class path. 
110         * 
111         * @param event the resource request event
112         * @param channel the channel that the request was recived on
113         */
114        @Handler
115        public void onResourceRequest(
116                        PortletResourceRequest event, IOSubchannel channel) {
117                // For me?
118                if (!event.portletClass().equals(type())) {
119                        return;
120                }
121                
122                // Look for content
123                InputStream in = this.getClass().getResourceAsStream(
124                                event.resourceUri().getPath());
125                if (in == null) {
126                        return;
127                }
128
129                // Respond
130                channel.respond(new PortletResourceResponse(event, in));
131                
132                // Done
133                event.setResult(true);
134        }
135
136        /**
137         * Returns all portlet models of this portlet's type from the
138         * session.
139         * 
140         * @param channel the channel, used to access the session
141         * @return the models
142         */
143        @SuppressWarnings("unchecked")
144        protected Collection<PortletBaseModel> modelsFromSession(
145                        IOSubchannel channel) {
146                return channel.associated(Session.class).map(session ->
147                                ((Map<Object,Map<Object,Map<String,PortletBaseModel>>>)
148                                                (Map<Object,?>)session)
149                                .computeIfAbsent(AbstractPortlet.class, ac -> new HashMap<>())
150                                .computeIfAbsent(type(), t -> new HashMap<>()).values())
151                        .orElseThrow(() -> new IllegalStateException("Session is missing."));
152        }
153        
154        /**
155         * Returns the portlet model of this portlet's type with the given id
156         * from the session.
157         * 
158         * @param session the session to use
159         * @param portletId the portel id
160         * @return the models
161         */
162        @SuppressWarnings("unchecked")
163        protected Optional<PortletBaseModel> modelFromSession(
164                        Session session, String portletId) {
165                return Optional.ofNullable(
166                                ((Map<Object,Map<Object,Map<String,PortletBaseModel>>>)
167                                                (Map<Object,?>)session)
168                                .computeIfAbsent(AbstractPortlet.class, ac -> new HashMap<>())
169                                .computeIfAbsent(type(), t -> new HashMap<>())
170                                .get(portletId));
171        }
172
173        /**
174         * Adds the given portlet model to the session using the {@link #type()}
175         * and the model's {@link PortletBaseModel#getPortletId()} as keys.
176         * 
177         * @param session the session to use
178         * @param model the model
179         * @return the model
180         */
181        @SuppressWarnings("unchecked")
182        protected <T extends PortletBaseModel> T addToSession(
183                        Session session, T model) {
184                ((Map<Object,Map<Object,Map<String,PortletBaseModel>>>)
185                                (Map<Object,?>)session)
186                        .computeIfAbsent(AbstractPortlet.class, ac -> new HashMap<>())
187                        .computeIfAbsent(type(), t -> new HashMap<>())
188                        .put(model.getPortletId(), model);
189                return model;
190        }
191        
192        /**
193         * Removes the given portlet model from the session.
194         * 
195         * @param session the session to use
196         * @param model the model
197         * @return the model
198         */
199        @SuppressWarnings("unchecked")
200        protected <T extends PortletBaseModel> T removeFromSession(
201                        Session session, T model) {
202                ((Map<Object,Map<Object,Map<String,PortletBaseModel>>>)
203                                (Map<Object,?>)session)
204                        .computeIfAbsent(AbstractPortlet.class, ac -> new HashMap<>())
205                        .computeIfAbsent(type(), t -> new HashMap<>())
206                        .remove(model.getPortletId());
207                return model;
208        }
209        
210        /**
211         * Provides a supplier for a resource bundle for localization.
212         * The default implementation looks up a bundle using the
213         * package name plus "l10n" as base name.
214         * 
215         * @return the supplier
216         */
217        protected Function<Locale,ResourceBundle> resourceSupplier() {
218                return locale -> ResourceBundle.getBundle(
219                        getClass().getPackage().getName() + ".l10n", locale, 
220                        getClass().getClassLoader(),
221                                ResourceBundle.Control.getNoFallbackControl(
222                                                ResourceBundle.Control.FORMAT_DEFAULT));
223                
224        }
225
226        /**
227         * Checks if the request applies to this component. If so, stops the event,
228         * calls {@link #createPortletModel()} to create a new model bean, adds it
229         * to the session and call {@link #doAddPortlet}
230         * with the newly created model bean. 
231         * 
232         * @param event the event
233         * @param channel the channel
234         */
235        @Handler
236        public void onAddPortletRequest(AddPortletRequest event,
237                        IOSubchannel channel) throws Exception {
238                if (!event.portletType().equals(type())) {
239                        return;
240                }
241                event.stop();
242                Session session = session(channel);
243                PortletBaseModel portletModel= addToSession(
244                                session, createPortletModel());
245                doAddPortlet(event, channel, session, portletModel);
246        }
247
248        /**
249         * Called by {@link #onAddPortletRequest} to complete adding the portlet.
250         * 
251         * @param event the event
252         * @param channel the channel
253         * @param session the session
254         * @param portletModel the model bean
255         */
256        protected abstract void doAddPortlet(AddPortletRequest event, 
257                        IOSubchannel channel, Session session, 
258                        PortletBaseModel portletModel) throws Exception;
259        
260        /**
261         * Checks if the request applies to this component. If so, stops the event,
262         * removes the portlet model bean from the session
263         * and calls {@link #doDeletePortlet}
264         * with the model. 
265         * 
266         * @param event the event
267         * @param channel the channel
268         */
269        @Handler
270        public void onDeletePortletRequest(DeletePortletRequest event, 
271                        IOSubchannel channel) throws Exception {
272                Session session = session(channel);
273                Optional<PortletBaseModel> optPortletModel 
274                        = modelFromSession(session, event.portletId());
275                if (!optPortletModel.isPresent()) {
276                        return;
277                }
278                event.stop();
279                PortletBaseModel portletModel = optPortletModel.get();
280                removeFromSession(session, portletModel);
281                doDeletePortlet(event, channel, session, portletModel);
282        }
283        
284        /**
285         * Called by {@link #onDeletePortletRequest} to complete deleting
286         * the portlet.
287         * 
288         * @param event the event
289         * @param channel the channel
290         * @param session the session
291         * @param portletModel the model bean
292         */
293        protected abstract void doDeletePortlet(DeletePortletRequest event, 
294                        IOSubchannel channel, Session session, 
295                        PortletBaseModel portletModel) throws Exception;
296        
297        /**
298         * Checks if the request applies to this component by calling
299         * {@link #modelFromSession}. If a model
300         * is found, stops the event, and calls 
301         * {@link #doRenderPortlet} with the model. 
302         * 
303         * Some portlets that do not persist their models between sessions
304         * (e.g. because the model only references data maintained elsewhere)
305         * should override {@link #modelFromSession}
306         * in such a way that it creates the requested model if it doesn't 
307         * exist yet.
308         * 
309         * @param event the event
310         * @param channel the channel
311         */
312        @Handler
313        public void onRenderPortlet(RenderPortletRequest event,
314                        IOSubchannel channel) throws Exception {
315                Session session = session(channel);
316                Optional<PortletBaseModel> optPortletModel 
317                        = modelFromSession(session, event.portletId());
318                if (!optPortletModel.isPresent()) {
319                        return;
320                }
321                event.stop();
322                doRenderPortlet(event, channel, session, optPortletModel.get());
323        }
324
325        /**
326         * Called by {@link #onRenderPortlet} to complete rendering
327         * the portlet.
328         * 
329         * @param event the event
330         * @param channel the channel
331         * @param portletModel the model bean
332         */
333        protected abstract void doRenderPortlet(RenderPortletRequest event, 
334                        IOSubchannel channel, Session session, 
335                        PortletBaseModel portletModel) throws Exception;
336        
337        /**
338         * Convenience method that returns the session associated with
339         * the channel.
340         * 
341         * @return the session
342         * @throws IllegalStateException if no session is found
343         */
344        public Session session(IOSubchannel channel) {
345                return channel.associated(Session.class).orElseThrow(
346                                () -> new IllegalStateException("Session is missing."));
347        }
348        
349        /**
350         * Defines the portlet model following the JavaBean conventions.
351         * 
352         * Portlet models should follow these conventions because
353         * many template engines rely on them and to support serialization
354         * to portable formats. 
355         */
356        @SuppressWarnings("serial")
357        public static class PortletBaseModel implements Serializable {
358
359                protected String portletId;
360
361                /**
362                 * Creates a new model with the given type and id.
363                 * 
364                 * @param portletId the portlet id
365                 */
366                @ConstructorProperties({"portletId"})
367                public PortletBaseModel(String portletId) {
368                        this.portletId = portletId;
369                }
370
371                /**
372                 * Returns the portlet id.
373                 * 
374                 * @return the portlet id
375                 */
376                public String getPortletId() {
377                        return portletId;
378                }
379        }
