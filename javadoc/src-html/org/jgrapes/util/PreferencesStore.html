019package org.jgrapes.util;
020
021import java.util.Map;
022import java.util.prefs.BackingStoreException;
023import java.util.prefs.Preferences;
024
025import org.jgrapes.core.Channel;
026import org.jgrapes.core.Component;
027import org.jgrapes.core.annotation.Handler;
028import org.jgrapes.core.events.Start;
029import org.jgrapes.core.events.Stop;
030import org.jgrapes.util.events.InitialPreferences;
031import org.jgrapes.util.events.UpdatePreferences;
032
033/**
034 * This component provides a store for application preferences.
035 * 
036 * The component reads the initial values from a Java {@link Preferences}
037 * node. During application bootstrap, it intercepts the {@link Start} 
038 * event using a handler with  priority 999999. When receiving this event, 
039 * it fires all known preferences values (skipping nodes whose name
040 * starts with a dot) on its channel as an {@link InitialPreferences} 
041 * event. Then, it re-fires the intercepted {@link Start} event. 
042 * 
043 * Components that depend on preference values define handlers
044 * for {@link UpdatePreferences} events and adapt themselves to the values 
045 * received. Note that due to the intercepted {@link Start} event, the initial
046 * preferences values are received before the {@link Start} event, so
047 * components' configurations can be rearranged before they actually
048 * start doing something.
049 *
050 * Besides initially publishing the stored preferences values,
051 * the component also listens for {@link UpdatePreferences} events
052 * fired by other components and updates the preferences store.
053 */
054public class PreferencesStore extends Component {
055
056        private Preferences preferences;
057        private boolean started = false;
058        
059        /**
060         * Creates a new component base with its channel set to the given 
061         * channel.
062         *  
063         * @param componentChannel the channel 
064         * @param node the preferences node, formed by replacing
065         * each dot in the class's full name with a slash and
066         * prepending a slash
067         */
068        public PreferencesStore(Channel componentChannel, Class<?> node) {
069                super(componentChannel);
070                preferences = Preferences.userNodeForPackage(node)
071                                .node(node.getSimpleName());
072        }
073
074        @Handler(priority=999999)
075        public void onStart(Start event) throws BackingStoreException {
076                if (started) {
077                        return;
078                }
079                started = true;
080                event.cancel(false);
081                InitialPreferences updEvt 
082                        = new InitialPreferences(preferences.absolutePath());
083                addPrefs(updEvt, preferences.absolutePath(), preferences);
084                fire(updEvt);
085                fire(new Start(), event.channels());
086        }
087
088        private void addPrefs(
089                        UpdatePreferences updEvt, String appPath, Preferences node) 
090                                        throws BackingStoreException {
091                String nodePath = node.absolutePath();
092                String relPath = nodePath.substring(Math.min(
093                                appPath.length() + 1, nodePath.length()));
094                for (String key: node.keys()) {
095                        updEvt.add(relPath, key, node.get(key, null));
096                }
097                for (String child: node.childrenNames()) {
098                        if (child.startsWith(".")) {
099                                continue;
100                        }
101                        addPrefs(updEvt, appPath, node.node(child));
102                }
103        }
104        
105        @Handler
106        public void onStop(Stop event) {
107                started = false;
108        }
109        
110        @Handler
111        public void onUpdatePreferences(UpdatePreferences event) 
112                        throws BackingStoreException {
113                if (event instanceof InitialPreferences) {
114                        return;
115                }
116                for (String path: event.paths()) {
117                        for (Map.Entry<String, String> e: 
118                                event.preferences(path).entrySet()) {
119                                preferences.node(path).put(e.getKey(), e.getValue());
120                        }
121                }
122                preferences.flush();
123        }
124}
