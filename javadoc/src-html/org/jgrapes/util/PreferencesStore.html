019package org.jgrapes.util;
020
021import java.io.IOException;
022import java.util.Map;
023import java.util.Optional;
024import java.util.prefs.BackingStoreException;
025import java.util.prefs.Preferences;
026import org.jgrapes.core.Channel;
027import org.jgrapes.core.Component;
028import org.jgrapes.core.EventPipeline;
029import org.jgrapes.core.annotation.Handler;
030import org.jgrapes.core.events.Start;
031import org.jgrapes.util.events.ConfigurationUpdate;
032import org.jgrapes.util.events.InitialPreferences;
033
034/**
035 * This component provides a store for an application's configuration
036 * backed by the Java {@link Preferences}. Preferences
037 * are maps of key value pairs that are associated with a path. A common
038 * base path is passed to the component on creation. The application's
039 * configuration information is stored using paths relative to that 
040 * base path.
041 * 
042 * The component reads the initial values from the Java {@link Preferences}
043 * tree denoted by the base path. During application bootstrap, it 
044 * intercepts the {@link Start} event using a handler with  priority 
045 * 999999. When receiving this event, it fires all known preferences 
046 * values on the channels of the start event as a 
047 * {@link InitialPreferences} event, using a new {@link EventPipeline}
048 * and waiting for its completion. Then, allows the intercepted 
049 * {@link Start} event to continue. 
050 * 
051 * Components that depend on configuration values define handlers
052 * for {@link ConfigurationUpdate} events and adapt themselves to the values 
053 * received. Note that due to the intercepted {@link Start} event, the initial
054 * preferences values are received before the {@link Start} event, so
055 * components' configurations can be rearranged before they actually
056 * start doing something.
057 *
058 * Besides initially publishing the stored preferences values,
059 * the component also listens for {@link ConfigurationUpdate} events
060 * on its channel and updates the preferences store (may be suppressed).
061 */
062public class PreferencesStore extends Component {
063
064    private Preferences preferences;
065
066    /**
067     * Creates a new component with its channel set to the given 
068     * channel and a base path derived from the given class.
069     * 
070     * @param componentChannel the channel 
071     * @param appClass the application class; the base path
072     * is formed by replacing each dot in the class's package's full 
073     * name with a slash, prepending a slash, and appending 
074     * "`/PreferencesStore`".
075     */
076    public PreferencesStore(Channel componentChannel, Class<?> appClass) {
077        this(componentChannel, appClass, true);
078    }
079
080    /**
081     * Allows the creation of a "read-only" store.
082     * 
083     * @param componentChannel the channel 
084     * @param appClass the application class; the base path
085     * is formed by replacing each dot in the class's package's full 
086     * name with a slash, prepending a slash, and appending 
087     * "`/PreferencesStore`".
088     * @param update whether to update the store when 
089     * {@link ConfigurationUpdate} events are received
090     * 
091     * @see #PreferencesStore(Channel, Class)
092     */
093    public PreferencesStore(
094            Channel componentChannel, Class<?> appClass, boolean update) {
095        super(componentChannel);
096        if (update) {
097            Handler.Evaluator.add(this, "onConfigurationUpdate",
098                channel().defaultCriterion());
099        }
100        preferences = Preferences.userNodeForPackage(appClass)
101            .node("PreferencesStore");
102    }
103
104    /**
105     * Intercepts the {@link Start} event and fires a
106     * {@link ConfigurationUpdate} event.
107     *
108     * @param event the event
109     * @throws BackingStoreException the backing store exception
110     * @throws InterruptedException the interrupted exception
111     */
112    @SuppressWarnings("PMD.DataflowAnomalyAnalysis")
113    @Handler(priority = 999999, channels = Channel.class)
114    public void onStart(Start event)
115            throws BackingStoreException, InterruptedException {
116        InitialPreferences updEvt
117            = new InitialPreferences(preferences.parent().absolutePath());
118        addPrefs(updEvt, preferences.absolutePath(), preferences);
119        newEventPipeline().fire(updEvt, event.channels()).get();
120    }
121
122    @SuppressWarnings("PMD.DataflowAnomalyAnalysis")
123    private void addPrefs(
124            InitialPreferences updEvt, String rootPath, Preferences node)
125            throws BackingStoreException {
126        String nodePath = node.absolutePath();
127        String relPath = "/" + nodePath.substring(Math.min(
128            rootPath.length() + 1, nodePath.length()));
129        for (String key : node.keys()) {
130            updEvt.add(relPath, key, node.get(key, null));
131        }
132        for (String child : node.childrenNames()) {
133            addPrefs(updEvt, rootPath, node.node(child));
134        }
135    }
136
137    /**
138     * Merges and saves configuration updates.
139     *
140     * @param event the event
141     * @throws IOException Signals that an I/O exception has occurred.
142     */
143    @Handler(dynamic = true)
144    public void onConfigurationUpdate(ConfigurationUpdate event)
145            throws BackingStoreException {
146        if (event instanceof InitialPreferences) {
147            return;
148        }
149        for (String path : event.paths()) {
150            Optional<Map<String, String>> prefs = event.values(path);
151            path = path.substring(1); // Remove leading slash
152            if (!prefs.isPresent()) {
153                preferences.node(path).removeNode();
154                continue;
155            }
156            for (Map.Entry<String, String> e : prefs.get().entrySet()) {
157                preferences.node(path).put(e.getKey(), e.getValue());
158            }
159        }
160        preferences.flush();
161    }
162}
