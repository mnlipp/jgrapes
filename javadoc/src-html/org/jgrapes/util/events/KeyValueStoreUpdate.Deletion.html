019package org.jgrapes.util.events;
020
021import java.util.ArrayList;
022import java.util.Collections;
023import java.util.List;
024import org.jgrapes.core.Event;
025
026// TODO: Auto-generated Javadoc
027/**
028 * An event that triggers updates or deletions in a key/value store.
029 */
030public class KeyValueStoreUpdate extends Event<Void> {
031
032    private final List<Action> actions = new ArrayList<>();
033
034    /**
035     * Adds a new update action to the event.
036     * 
037     * @param key the key
038     * @param value the value
039     * @return the event for easy chaining
040     */
041    public KeyValueStoreUpdate update(String key, String value) {
042        actions.add(new Update(key, value));
043        return this;
044    }
045
046    /**
047     * Adds a new update action to the event that stores the given value
048     * on the path formed by the path segments.
049     * 
050     * @param value the value
051     * @param segments the path segments
052     * @return the event for easy chaining
053     */
054    public KeyValueStoreUpdate storeAs(String value, String... segments) {
055        actions.add(new Update("/" + String.join("/", segments), value));
056        return this;
057    }
058
059    /**
060     * Adds a new deletion action to the event.
061     * 
062     * @param key the key
063     * @return the event for easy chaining
064     */
065    public KeyValueStoreUpdate delete(String key) {
066        actions.add(new Deletion(key));
067        return this;
068    }
069
070    /**
071     * Adds a new deletion action that clears all keys with the given
072     * path prefix.
073     * 
074     * @param segments the path segments
075     * @return the event for easy chaining
076     */
077    public KeyValueStoreUpdate clearAll(String... segments) {
078        actions.add(new Deletion("/" + String.join("/", segments)));
079        return this;
080    }
081
082    /**
083     * Returns the actions.
084     * 
085     * @return the actions
086     */
087    public List<Action> actions() {
088        return Collections.unmodifiableList(actions);
089    }
090
091    /**
092     * The base class for all actions.
093     */
094    @SuppressWarnings("PMD.AbstractClassWithoutAbstractMethod")
095    public abstract static class Action {
096
097        private final String key;
098
099        /**
100         * Instantiates a new action.
101         *
102         * @param key the key
103         */
104        public Action(String key) {
105            this.key = key;
106        }
107
108        /**
109         * Key.
110         *
111         * @return the string
112         */
113        public String key() {
114            return key;
115        }
116    }
117
118    /**
119     * A key (and value) deletion.
120     */
121    public static class Deletion extends Action {
122
123        /**
124         * Instantiates a new deletion.
125         *
126         * @param key the key
127         */
128        public Deletion(String key) {
129            super(key);
130        }
131    }
132
133    /**
134     * A value update.
135     */
136    public static class Update extends Action {
137        private final String value;
138
139        /**
140         * Instantiates a new update.
141         *
142         * @param key the key
143         * @param value the value
144         */
145        public Update(String key, String value) {
146            super(key);
147            this.value = value;
148        }
149
150        /**
151         * Value.
152         *
153         * @return the string
154         */
155        public String value() {
156            return value;
157        }
158    }
159}
