019package org.jgrapes.util.events;
020
021import java.util.Collections;
022import java.util.HashMap;
023import java.util.Map;
024import java.util.Set;
025
026import org.jgrapes.core.Event;
027import org.jgrapes.core.Manager;
028
029/**
030 * An event to indicate that configuration information has been
031 * updated.
032 * 
033 * Configurability based on this type of event can optionally be 
034 * supported by components. If supported, a component implements a 
035 * handler that checks whether the paths of this event include the 
036 * component's path in the application's component tree (see 
037 * {@link Manager#path()}). If so, the component adapts itself to the 
038 * information propagated.
039 */
040public class ConfigurationUpdate extends Event<Void> {
041
042        private Map<String,Map<String,String>> paths = new HashMap<>();
043
044        /**
045         * Add new (updated) preferences to the event.
046         * 
047         * @param path the preference's path
048         * @param key the key of the preference
049         * @param value the value of the preference
050         * @return the event for easy chaining
051         */
052        public ConfigurationUpdate add(String path, String key, String value) {
053                if (path == null || !path.startsWith("/")) {
054                        throw new IllegalArgumentException("Path must start with \"/\".");
055                }
056                Map<String,String> scoped = paths
057                                .computeIfAbsent(path, s -> new HashMap<String,String>());
058                scoped.put(key, value);
059                return this;
060        }
061
062        /**
063         * Remove a path from the configuration.
064         * 
065         * @param path the path to be removed
066         * @return the event for easy chaining
067         */
068        public ConfigurationUpdate removePath(String path) {
069                if (path == null || !path.startsWith("/")) {
070                        throw new IllegalArgumentException("Path must start with \"/\".");
071                }
072                paths.put(path, null);
073                return this;
074        }
075        
076        /**
077         * Return all paths affected by this event.
078         * 
079         * @return the paths
080         */
081        public Set<String> paths() {
082                return Collections.unmodifiableSet(paths.keySet());
083        }
084        
085        /**
086         * Return the preferences for a given path.
087         * 
088         * @param path the path
089         * @return the updated preferences or `null` if the path has been
090         * removed (implies the removal of all preferences for that path).
091         */
092        public Map<String,String> preferences(String path) {
093                if (!paths.containsKey(path)) {
094                        return Collections.emptyMap();
095                }
096                Map<String,String> result = paths.get(path);
097                if (result == null) {
098                        return result;
099                }
100                return Collections.unmodifiableMap(result);
101        }
102}
