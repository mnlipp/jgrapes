019package org.jgrapes.util.events;
020
021import java.util.Collections;
022import java.util.HashMap;
023import java.util.Map;
024import java.util.Optional;
025import java.util.Set;
026
027import org.jgrapes.core.Event;
028import org.jgrapes.core.Manager;
029
030/**
031 * An event to indicate that configuration information has been
032 * updated.
033 * 
034 * Configurability based on this type of event can optionally be 
035 * supported by components. If supported, a component implements a 
036 * handler that checks whether the paths of this event include the 
037 * component's path in the application's component tree (see 
038 * {@link Manager#path()}). If so, the component adapts itself to the 
039 * information propagated.
040 */
041public class ConfigurationUpdate extends Event<Void> {
042
043        private Map<String,Map<String,String>> paths = new HashMap<>();
044
045        /**
046         * Add new (updated) configuration value to the event.
047         * 
048         * @param path the value's path
049         * @param key the key of the value
050         * @param value the value
051         * @return the event for easy chaining
052         */
053        public ConfigurationUpdate add(String path, String key, String value) {
054                if (path == null || !path.startsWith("/")) {
055                        throw new IllegalArgumentException("Path must start with \"/\".");
056                }
057                Map<String,String> scoped = paths
058                                .computeIfAbsent(path, s -> new HashMap<String,String>());
059                scoped.put(key, value);
060                return this;
061        }
062
063        /**
064         * Remove a path from the configuration.
065         * 
066         * @param path the path to be removed
067         * @return the event for easy chaining
068         */
069        public ConfigurationUpdate removePath(String path) {
070                if (path == null || !path.startsWith("/")) {
071                        throw new IllegalArgumentException("Path must start with \"/\".");
072                }
073                paths.put(path, null);
074                return this;
075        }
076        
077        /**
078         * Return all paths affected by this event.
079         * 
080         * @return the paths
081         */
082        public Set<String> paths() {
083                return Collections.unmodifiableSet(paths.keySet());
084        }
085        
086        /**
087         * Return the values for a given path if they exists.
088         * 
089         * @param path the path
090         * @return the updated values or `null` if the path has been
091         * removed (implies the removal of all values for that path).
092         */
093        public Optional<Map<String,String>> values(String path) {
094                Map<String,String> result = paths.get(path);
095                if (result == null) {
096                        return Optional.empty();
097                }
098                return Optional.of(Collections.unmodifiableMap(result));
099        }
100
101        /**
102         * Return the value with the given path and key if it exists.
103         * 
104         * @param path the path
105         * @param key the key
106         * @return the value
107         */
108        public Optional<String> value(String path, String key) {
109                return Optional.ofNullable(paths.get(path))
110                                .flatMap(map -> Optional.ofNullable(map.get(key)));
111        }
112}
